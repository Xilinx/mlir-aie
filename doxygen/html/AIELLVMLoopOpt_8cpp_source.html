40/// Information about a candidate index-to-pointer transformation
41struct LoopIndexCandidate {
42 BlockArgument indexArg; // The index argument in loop header
43 Operation *gepOp; // The getelementptr using this index
44 Operation *incrementOp; // The add operation incrementing the index
45 Value basePtr; // The base pointer (loop-invariant)
46 Value stride; // The increment stride (loop-invariant)
47 unsigned argIndex; // Position in block arguments
48 Type gepResultType; // Result type of the GEP
49 Type gepElemType; // Element type for the GEP
50};

52/// Information about a candidate bitcast fusion transformation
53struct BitcastFusionCandidate {
54 BlockArgument accArg; // The accumulator argument
55 Type currentType; // Current type (e.g., vector<64xi32>)
56 Type nativeType; // Native type for operations (e.g., vector<32xi64>)
57 unsigned argIndex; // Position in block arguments
58 SmallVector<LLVM::BitcastOp> inputCasts; // Bitcasts before operations
59 SmallVector<LLVM::BitcastOp> outputCasts; // Bitcasts after operations
60};

71/// Analyze a block argument to determine if it's a candidate for
72/// index-to-pointer transformation
73static std::optional<LoopIndexCandidate>
74analyzeIndexArgument(BlockArgument arg, Block *loopHeader) {
75 // Must be an integer type (i64, i32, etc.)
76 if (!arg.getType().isInteger())
77 return std::nullopt;
78
79 LoopIndexCandidate candidate;
80 candidate.indexArg = arg;
81 candidate.argIndex = arg.getArgNumber();
82
83 // Find the GEPOp that uses this index
84 LLVM::GEPOp foundGEP = nullptr;
85 for (Operation *user : arg.getUsers()) {
86 if (auto gepOp = dyn_cast<LLVM::GEPOp>(user)) {
87 // Check if this is a simple GEP: base[index]
88 // The index should be used as one of the dynamic indices
89 bool indexUsedInGEP = false;
90 for (Value dynamicIndex : gepOp.getDynamicIndices()) {
91 if (dynamicIndex == arg) {
92 indexUsedInGEP = true;
93 break;
94 }
95 }
96
97 if (indexUsedInGEP) {
98 if (foundGEP) {
99 // Multiple GEPs use this index - not a simple pattern
100 return std::nullopt;
101 }
102 foundGEP = gepOp;
103 candidate.gepOp = gepOp.getOperation();
104 candidate.basePtr = gepOp.getBase();
105 candidate.gepResultType = gepOp.getType();
106 candidate.gepElemType = gepOp.getElemType();
107 }
108 }
109 }
110
111 if (!foundGEP)
112 return std::nullopt;
113
114 // Check that the GEP result is only used for loads
115 for (Operation *user : foundGEP.getResult().getUsers()) {
116 if (!isa<LLVM::LoadOp>(user)) {
117 // GEP result used for something other than load - skip
118 return std::nullopt;
119 }
120 }
121
122 // Find the AddOp that increments this index
123 LLVM::AddOp foundAdd = nullptr;
124 for (Operation *user : arg.getUsers()) {
125 if (auto addOp = dyn_cast<LLVM::AddOp>(user)) {
126 // Check if one operand is the index
127 if (addOp.getLhs() == arg || addOp.getRhs() == arg) {
128 if (foundAdd) {
129 // Multiple adds - not a simple pattern
130 return std::nullopt;
131 }
132 foundAdd = addOp;
133 candidate.incrementOp = addOp;
134 // Get the stride (the other operand)
135 candidate.stride =
136 (addOp.getLhs() == arg) ? addOp.getRhs() : addOp.getLhs();
137 }
138 }
139 }
140
141 if (!foundAdd)
142 return std::nullopt;
143
144 // Verify the incremented value flows back to the loop header
145 // (This is a simplified check - a full implementation would use dominance
146 // analysis)
147 bool flowsToBackEdge = false;
148 for (Operation *user : foundAdd.getResult().getUsers()) {
149 if (auto brOp = dyn_cast<LLVM::BrOp>(user)) {
150 // Check if this branch goes to our loop header
151 if (brOp->getSuccessor(0) == loopHeader) {
152 // Check if our add result is passed as the corresponding argument
153 if (brOp.getDestOperands()[candidate.argIndex] ==
154 foundAdd.getResult()) {
155 flowsToBackEdge = true;
156 break;
157 }
158 }
159 }
160 }
161
162 if (!flowsToBackEdge)
163 return std::nullopt;
164
165 return candidate;
166}

168/// Check if a value is loop-invariant (defined outside the loop)
169static bool isLoopInvariant(Value val, Block *loopHeader) {
170 Operation *defOp = val.getDefiningOp();
171 if (!defOp)
172 return true; // Block argument or constant
173 return defOp->getBlock() != loopHeader;
174}

176/// Find loop headers by looking for blocks with back-edges
177/// A block is a loop header if it has a back-edge pointing to it
178static SmallVector<Block *> findLoopHeaders(LLVM::LLVMFuncOp func) {
179 SmallVector<Block *> loopHeaders;
180 DenseSet<Block *> visited;
181
182 // A block is a loop header if:
183 // - It has block arguments (phi nodes)
184 // - One of its successors branches back to it (creating a cycle)
185 for (Block &block : func.getBlocks()) {
186 if (block.getNumArguments() == 0)
187 continue; // Loop headers have arguments (loop-carried values)
188
189 // Check if any successor eventually branches back to this block
190 for (Block *succ : block.getSuccessors()) {
191 // Look for back-edge: does this successor branch back to us?
192 for (Block *succOfSucc : succ->getSuccessors()) {
193 if (succOfSucc == &block) {
194 // Found a back-edge: succ -> block
195 if (llvm::find(loopHeaders, &block) == loopHeaders.end()) {
196 loopHeaders.push_back(&block);
197 LLVM_DEBUG(llvm::dbgs() << "Found loop header block\n");
198 }
199 break;
200 }
201 }
202 }
203 }
204
205 return loopHeaders;
206}

208/// Transform a loop by converting index arguments to pointer arguments
209static void transformLoop(Block *loopHeader,
210 ArrayRef<LoopIndexCandidate> candidates,
211 OpBuilder &builder) {
212 if (candidates.empty())
213 return;
214
215 LLVM_DEBUG(llvm::dbgs() << "Transforming loop with " << candidates.size()
216 << " candidates\n");
217
218 // Step 1: Find the loop preheader (predecessor that enters the loop)
219 Block *preheader = nullptr;
220 for (Block *pred : loopHeader->getPredecessors()) {
221 // Check if this is not a back-edge
222 bool isBackEdge = false;
223 for (Block *succ : loopHeader->getSuccessors()) {
224 if (succ == pred) {
225 isBackEdge = true;
226 break;
227 }
228 }
229 if (!isBackEdge && !preheader) {
230 preheader = pred;
231 }
232 }
233
234 if (!preheader) {
235 LLVM_DEBUG(llvm::dbgs() << "Could not find preheader\n");
236 return; // Can't find preheader
237 }
238
239 // Step 2: For each candidate, compute initial pointer in preheader
240 DenseMap<unsigned, Value> argIndexToInitPtr;
241
242 builder.setInsertionPoint(preheader->getTerminator());
243
244 for (const auto &candidate : candidates) {
245 // Get the initial index value from preheader's branch
246 Value initialIndex = nullptr;
247 for (Operation &op : preheader->getOperations()) {
248 if (auto brOp = dyn_cast<LLVM::BrOp>(&op)) {
249 if (brOp->getSuccessor(0) == loopHeader) {
250 initialIndex = brOp.getDestOperands()[candidate.argIndex];
251 break;
252 }
253 }
254 }
255
256 if (!initialIndex)
257 continue;
258
259 // Create initial GEP: base[initial_index]
260 auto initGEP = builder.create<LLVM::GEPOp>(
261 preheader->getTerminator()->getLoc(), candidate.gepResultType,
262 candidate.gepElemType, candidate.basePtr, ValueRange{initialIndex});
263
264 argIndexToInitPtr[candidate.argIndex] = initGEP.getResult();
265 }
266
267 // Step 3: Change block argument types from i64 to !llvm.ptr
268 for (const auto &candidate : candidates) {
269 // Change the type of the block argument
270 loopHeader->getArgument(candidate.argIndex)
271 .setType(candidate.gepResultType);
272 }
273
274 // Step 4: Update preheader branch to pass pointers instead of indices
275 builder.setInsertionPoint(preheader->getTerminator());
276 auto preheaderBr = cast<LLVM::BrOp>(preheader->getTerminator());
277 SmallVector<Value> newOperands;
278 for (unsigned i = 0; i < preheaderBr.getDestOperands().size(); ++i) {
279 if (argIndexToInitPtr.count(i)) {
280 newOperands.push_back(argIndexToInitPtr[i]);
281 } else {
282 newOperands.push_back(preheaderBr.getDestOperands()[i]);
283 }
284 }
285 builder.create<LLVM::BrOp>(preheaderBr.getLoc(), newOperands, loopHeader);
286 preheaderBr.erase();
287
288 // Step 5: Update loop body - replace GEP and Add operations
289 for (const auto &candidate : candidates) {
290 BlockArgument ptrArg = loopHeader->getArgument(candidate.argIndex);
291
292 // Replace uses of GEP result with direct pointer
293 candidate.gepOp->getResult(0).replaceAllUsesWith(ptrArg);
294
295 // Replace Add with GEP for pointer arithmetic
296 builder.setInsertionPoint(candidate.incrementOp);
297 auto ptrIncrement = builder.create<LLVM::GEPOp>(
298 candidate.incrementOp->getLoc(), ptrArg.getType(),
299 candidate.gepElemType, ptrArg, ValueRange{candidate.stride});
300
301 candidate.incrementOp->getResult(0).replaceAllUsesWith(
302 ptrIncrement.getResult());
303
304 // Clean up old operations
305 candidate.incrementOp->erase();
306 candidate.gepOp->erase();
307 }
308}

310/// Fuse load-bitcast pairs in a block
311static void fuseLoadBitcasts(Block *block, OpBuilder &builder) {
312 SmallVector<std::pair<LLVM::LoadOp, LLVM::BitcastOp>> fusePairs;
313
314 // Find all load-bitcast pairs
315 for (Operation &op : block->getOperations()) {
316 if (auto loadOp = dyn_cast<LLVM::LoadOp>(&op)) {
317 // Check if this load has a single bitcast user
318 if (loadOp.getResult().hasOneUse()) {
319 if (auto bitcastOp = dyn_cast<LLVM::BitcastOp>(
320 *loadOp.getResult().getUsers().begin())) {
321 fusePairs.push_back({loadOp, bitcastOp});
322 }
323 }
324 }
325 }
326
327 // Fuse pairs by replacing load result type and removing bitcast
328 for (auto [loadOp, bitcastOp] : fusePairs) {
329 builder.setInsertionPoint(loadOp);
330
331 // Create new load with bitcast result type using the correct builder
332 // signature
333 auto newLoad = builder.create<LLVM::LoadOp>(
334 loadOp.getLoc(), bitcastOp.getType(), loadOp.getAddr());
335
336 // Copy attributes if present
337 if (loadOp.getAlignmentAttr())
338 newLoad.setAlignmentAttr(loadOp.getAlignmentAttr());
339 if (loadOp.getVolatile_Attr())
340 newLoad.setVolatile_Attr(loadOp.getVolatile_Attr());
341 if (loadOp.getNontemporalAttr())
342 newLoad.setNontemporalAttr(loadOp.getNontemporalAttr());
343
344 // Replace bitcast uses with new load
345 bitcastOp.getResult().replaceAllUsesWith(newLoad.getResult());
346
347 // Erase old operations
348 bitcastOp.erase();
349 loadOp.erase();
350
351 LLVM_DEBUG(llvm::dbgs() << "Fused load-bitcast pair\n");
352 }
353}

355/// Optimize loop-carried accumulator bitcasts by changing block argument types
356/// This handles the pattern:
357/// arg (vector<64xi32>) -> bitcast to vector<32xi64> -> op -> bitcast back to
358/// vector<64xi32> -> pass to next iteration
359static void optimizeAccumulatorBitcasts(Block *loopHeader, OpBuilder &builder) {
360 SmallVector<std::tuple<BlockArgument, Type, unsigned>> transformations;
361
362 // Find block arguments that follow the bitcast pattern
363 for (BlockArgument arg : loopHeader->getArguments()) {
364 auto vecType = dyn_cast<VectorType>(arg.getType());
365 if (!vecType)
366 continue;
367
368 // Check pattern: has bitcasts to same type (but may have other uses like
369 // shufflevector)
370 Type targetType = nullptr;
371 SmallVector<LLVM::BitcastOp> directBitcasts;
372
373 for (Operation *user : arg.getUsers()) {
374 if (auto bitcastOp = dyn_cast<LLVM::BitcastOp>(user)) {
375 if (!targetType) {
376 targetType = bitcastOp.getType();
377 } else if (targetType != bitcastOp.getType()) {
378 // Bitcasts to different types - skip
379 targetType = nullptr;
380 break;
381 }
382 directBitcasts.push_back(bitcastOp);
383 } else {
384 // Allow other uses (e.g., shufflevector in exit blocks)
385 }
386 }
387
388 // We need at least some bitcasts to a consistent type
389 if (!targetType || directBitcasts.empty())
390 continue;
391
392 LLVM_DEBUG(llvm::dbgs() << "Checking arg " << arg.getArgNumber()
393 << " has bitcasts to " << targetType << "\n");
394
395 // Verify that values passed back to loop header are bitcast from targetType
396 // Look at all branches to this header
397 bool validPattern = true;
398 for (auto it = loopHeader->pred_begin(); it != loopHeader->pred_end();
399 ++it) {
400 Block *pred = *it;
401 // Skip preheader (entry to loop)
402 if (pred == loopHeader->getSinglePredecessor()) {
403 LLVM_DEBUG(llvm::dbgs() << "Skipping preheader\n");
404 continue;
405 }
406
407 if (auto brOp = dyn_cast_or_null<LLVM::BrOp>(pred->getTerminator())) {
408 if (brOp->getSuccessor(0) == loopHeader) {
409 Value passedValue = brOp.getDestOperands()[arg.getArgNumber()];
410
411 LLVM_DEBUG(llvm::dbgs() << "Checking branch operand type: "
412 << passedValue.getType() << "\n");
413
414 // The passed value should have the same type as the current arg type
415 // (it will be bitcast to targetType after we transform)
416 if (passedValue.getType() != arg.getType()) {
417 LLVM_DEBUG(
418 llvm::dbgs()
419 << "Invalid pattern: operand type doesn't match arg type\n");
420 validPattern = false;
421 break;
422 }
423
424 // Optionally check if it's derived from an operation that produces
425 // targetType This is a sanity check but not strictly required
426 if (auto bitcastOp = passedValue.getDefiningOp<LLVM::BitcastOp>()) {
427 if (bitcastOp.getArg().getType() == targetType) {
428 LLVM_DEBUG(llvm::dbgs()
429 << "Operand is bitcast from targetType - good!\n");
430 }
431 }
432 }
433 }
434 }
435
436 if (validPattern) {
437 transformations.push_back({arg, targetType, arg.getArgNumber()});
438 LLVM_DEBUG(llvm::dbgs() << "Found accumulator bitcast candidate at arg "
439 << arg.getArgNumber() << " from " << arg.getType()
440 << " to " << targetType << "\n");
441 }
442 }
443
444 if (transformations.empty())
445 return;
446
447 // Apply transformations
448 for (auto [arg, targetType, argNum] : transformations) {
449 LLVM_DEBUG(llvm::dbgs() << "Transforming argument " << argNum << " from "
450 << arg.getType() << " to " << targetType << "\n");
451
452 Type originalType = arg.getType();
453
454 // Step 1: Change block argument type to targetType (e.g., vector<64xi32> ->
455 // vector<32xi64>)
456 arg.setType(targetType);
457
458 // Step 2: Insert bitcast at start of loop to convert back to original type
459 // This hoisted bitcast replaces all the scattered bitcasts
460 builder.setInsertionPointToStart(loopHeader);
461 auto hoistedBitcast = builder.create<LLVM::BitcastOp>(
462 loopHeader->front().getLoc(), originalType, arg);
463
464 LLVM_DEBUG(llvm::dbgs() << "Created hoisted bitcast from " << targetType
465 << " to " << originalType << "\n");
466
467 // Step 3: Replace all uses of arg with the hoisted bitcast
468 // (except the hoisted bitcast itself which uses arg)
469 arg.replaceAllUsesExcept(hoistedBitcast.getResult(), hoistedBitcast);
470
471 LLVM_DEBUG(llvm::dbgs() << "Replaced arg uses with hoisted bitcast\n");
472
473 // Step 4: Update ALL branches (including preheader) to pass targetType
474 for (auto it = loopHeader->pred_begin(); it != loopHeader->pred_end();
475 ++it) {
476 Block *pred = *it;
477
478 if (auto brOp = dyn_cast<LLVM::BrOp>(pred->getTerminator())) {
479 if (brOp->getSuccessor(0) == loopHeader) {
480 Value operand = brOp.getDestOperands()[argNum];
481 Value newOperand = operand;
482
483 // If operand is bitcast from targetType to originalType, use the
484 // source directly
485 if (auto bitcastOp = operand.getDefiningOp<LLVM::BitcastOp>()) {
486 if (bitcastOp.getArg().getType() == targetType) {
487 newOperand = bitcastOp.getArg();
488 LLVM_DEBUG(llvm::dbgs() << "Using bitcast source directly\n");
489 }
490 }
491
492 // If operand type doesn't match target, insert bitcast
493 if (newOperand.getType() != targetType) {
494 builder.setInsertionPoint(brOp);
495 auto bitcast = builder.create<LLVM::BitcastOp>(
496 brOp.getLoc(), targetType, newOperand);
497 newOperand = bitcast.getResult();
498 LLVM_DEBUG(llvm::dbgs()
499 << "Inserted bitcast to targetType in branch\n");
500 }
501
502 // Update branch with new operand
503 if (newOperand != operand) {
504 builder.setInsertionPoint(brOp);
505 SmallVector<Value> newOperands(brOp.getDestOperands());
506 newOperands[argNum] = newOperand;
507 builder.create<LLVM::BrOp>(brOp.getLoc(), newOperands, loopHeader);
508 brOp.erase();
509 }
510 }
511 }
512 }

516struct AIELLVMLoopOptPass : AIELLVMLoopOptBase<AIELLVMLoopOptPass> {
517 void getDependentDialects(DialectRegistry &registry) const override {
518 registry.insert<LLVM::LLVMDialect>();
519 }
520
521 void runOnOperation() override {
522 LLVM::LLVMFuncOp func = getOperation();
523 OpBuilder builder(&getContext());
524
525 SmallVector<Block *> loopHeaders = findLoopHeaders(func);
526
527 LLVM_DEBUG(llvm::dbgs()
528 << "Found " << loopHeaders.size() << " loop headers\n");
529
530 // Phase 1: Index-to-pointer transformation
531 for (Block *header : loopHeaders) {
532 LLVM_DEBUG(llvm::dbgs() << "Analyzing loop header with "
533 << header->getNumArguments() << " arguments\n");
534
535 SmallVector<LoopIndexCandidate> candidates;
536
537 for (BlockArgument arg : header->getArguments()) {
538 if (auto candidate = analyzeIndexArgument(arg, header)) {
539 if (isLoopInvariant(candidate->basePtr, header)) {
540 LLVM_DEBUG(llvm::dbgs() << "Found candidate at argument "
541 << candidate->argIndex << "\n");
542 candidates.push_back(*candidate);
543 }
544 }
545 }
546
547 if (!candidates.empty()) {
548 LLVM_DEBUG(llvm::dbgs()
549 << "Transforming " << candidates.size() << " candidates\n");
550 transformLoop(header, candidates, builder);
551 }
552 }
553
554 // Phase 2: Fuse load-bitcast pairs
555 LLVM_DEBUG(llvm::dbgs() << "Phase 2: Fusing load-bitcast pairs\n");
556 for (Block &block : func.getBlocks()) {
557 fuseLoadBitcasts(&block, builder);
558 }
559
560 // Phase 3: Optimize loop-carried accumulator bitcasts
561 LLVM_DEBUG(llvm::dbgs() << "Phase 3: Optimizing accumulator bitcasts\n");
562 for (Block *header : loopHeaders) {
563 optimizeAccumulatorBitcasts(header, builder);
564 }
565 }
566};
