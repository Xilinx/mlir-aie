36class BfpToIntegerConverter : public mlir::TypeConverter {
37public:
38 BfpToIntegerConverter(const AIETargetModel &targetModel) {
39 addTypeAttributeConversion([&](Type type, Attribute attr) {
40 auto newType = convertType(type);
41 if (!newType) {
42 llvm::errs() << "Failed to convert type: " << type << "\n";
43 return AttributeConversionResult::abort();
44 }
45 return AttributeConversionResult(TypeAttr::get(newType));
46 });
47
48 // Note that the most recently added conversions will be invoked first
49
50 // Leave other types unchanged
51 addConversion([](Type type) -> std::optional<Type> { return type; });
52
53 // Add a conversion for bfpTypes to an integer type
54 addConversion([&](BlockFloatType blockType) -> std::optional<IntegerType> {
55 bool isSupported =
56 targetModel.isSupportedBlockFormat(blockType.getBlockType().str());
57 if (!isSupported) {
58 llvm::errs() << "Block type " << blockType.getBlockType()
59 << " is not supported in the specified model\n";
60 // Note that returning a nullptr here will stop the conversion while
61 // returning a std::nullopt will allow the converter to keep trying the
62 // remaining conversions (thus reaching the default one in this case)
63 return nullptr;
64 }
65
66 return mlir::IntegerType::get(blockType.getContext(),
67 blockType.getTotalSizeInBits());
68 });
69
70 // Add a conversion for MemRefType
71 addConversion([&](MemRefType memRefType) -> std::optional<MemRefType> {
72 auto newElementType = convertType(memRefType.getElementType());
73 if (!newElementType) {
74 llvm::errs() << "Failed to convert memref element type\n";
75 return nullptr;
76 }
77 return MemRefType::get(memRefType.getShape(), newElementType,
78 memRefType.getLayout(),
79 memRefType.getMemorySpace());
80 });
81
82 // Add a conversion for ObjectFifoType
83 addConversion([&](AIEObjectFifoType objectFifoType)
84 -> std::optional<AIEObjectFifoType> {
85 auto newElementType = convertType(objectFifoType.getElementType());
86 if (!newElementType) {
87 llvm::errs() << "Failed to convert ObjectFifoType element type\n";
88 return nullptr;
89 }
90
91 if (auto newMemRef = dyn_cast<MemRefType>(newElementType))
92 return AIEObjectFifoType::get(objectFifoType.getContext(), newMemRef);
93
94 llvm::errs()
95 << "ObjectFifoType converted element type is not a MemRefType\n";
96 return nullptr;
97 });
98
99 // Add a conversion for ObjectFifoSubviewType
100 addConversion([&](AIEObjectFifoSubviewType objectFifoSubviewType)
101 -> std::optional<AIEObjectFifoSubviewType> {
102 auto newElementType = convertType(objectFifoSubviewType.getElementType());
103 if (!newElementType) {
104 llvm::errs()
105 << "Failed to convert ObjectFifoSubviewType element type\n";
106 return nullptr;
107 }
108
109 if (auto newMemRef = dyn_cast<MemRefType>(newElementType))
110 return AIEObjectFifoSubviewType::get(objectFifoSubviewType.getContext(),
111 newMemRef);
112
113 llvm::errs()
114 << "ObjectFifoSubviewType element type is not a MemRefType\n";
115 return nullptr;
116 });
117
118 // Add a conversion for FunctionType
119 addConversion([&](FunctionType funcType) -> std::optional<FunctionType> {
120 llvm::SmallVector<Type> newInputTypes;
121 auto check = convertTypes(funcType.getInputs(), newInputTypes);
122 if (check.failed()) {
123 llvm::errs() << "Failed to convert function input types\n";
124 return nullptr;
125 }
126
127 llvm::SmallVector<Type> newOutputTypes;
128 check = convertTypes(funcType.getResults(), newOutputTypes);
129 if (check.failed()) {
130 llvm::errs() << "Failed to convert function output types\n";
131 return nullptr;
132 }
133
134 return FunctionType::get(funcType.getContext(), newInputTypes,
135 newOutputTypes);
136 });
137
138 // Add conversions for other types as needed (llvm arrays?)
139 }
140};

142class BfpToIntegerConversionPattern : public ConversionPattern {
143public:
144 BfpToIntegerConversionPattern(TypeConverter &typeConverter,
145 MLIRContext *context, bool &conversionFailed)
146 : ConversionPattern(typeConverter, MatchAnyOpTypeTag(), 1, context),
147 conversionFailed(conversionFailed) {}
148
149 LogicalResult
150 matchAndRewrite(Operation *op, ArrayRef<Value> operands,
151 ConversionPatternRewriter &rewriter) const override {
152
153 // The objective is to replace all bfp operations by an integer of the
154 // appropriate width. This pass currently does not have any other
155 // functionality.
156
157 // Operation results
158 for (auto result : op->getResults()) {
159 auto conversion = typeConverter->convertType(result.getType());
160 if (!conversion) {
161 conversionFailed = true;
162 return op->emitError()
163 << "Failed to convert result type: " << result.getType();
164 }
165 result.setType(conversion);
166 }
167
168 // Operation operands
169 for (auto operand : op->getOperands()) {
170 auto conversion = typeConverter->convertType(operand.getType());
171 if (!conversion) {
172 conversionFailed = true;
173 return op->emitError()
174 << "Failed to convert operand type: " << operand.getType();
175 }
176 operand.setType(conversion);
177 }
178
179 // Operation attributes
180 // Note that the attribute list is immutable and
181 // needs to be recreated from scratch. Also note that type attributes cannot
182 // access their type and must therefore be managed through
183 // the convertTypeAttribute conversion instead
184 SmallVector<NamedAttribute> newAttrs;
185 for (auto attr : op->getAttrs()) {
186 if (auto typeAttr = dyn_cast<TypeAttr>(attr.getValue())) {
187 auto conversion = typeConverter->convertTypeAttribute(
188 typeAttr.getValue(), attr.getValue());
189 if (!conversion) {
190 conversionFailed = true;
191 return op->emitError()
192 << "Failed to convert attribute type: " << typeAttr.getValue();
193 }
194 newAttrs.push_back(NamedAttribute(attr.getName(), conversion.value()));
195 } else {
196 newAttrs.push_back(attr);
197 }
198 }
199 op->setAttrs(DictionaryAttr::get(op->getContext(), newAttrs));
200
201 return success();
202 }
203
204private:
205 bool &conversionFailed;
206};

208class AIETransformBfpTypesPass
209 : public AIETransformBfpTypesBase<AIETransformBfpTypesPass> {
210public:
211 void runOnOperation() override {
212 DeviceOp device = getOperation();
213 MLIRContext *context = device.getContext();
214
215 BfpToIntegerConverter typeConverter(device.getTargetModel());
216
217 // Set up an empty conversion target, since we have to iterate over all ops
218 ConversionTarget target(*context);
219
220 RewritePatternSet patterns(context);
221 bool conversionFailed = false;
222 patterns.add<BfpToIntegerConversionPattern>(typeConverter, context,
223 conversionFailed);
224
225 if (failed(applyPartialConversion(device, target, std::move(patterns))) ||
226 conversionFailed) {
227 signalPassFailure();
228 }
229 }
230};
