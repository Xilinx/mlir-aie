169 typedef struct Port {
170  WireBundle bundle;
171  int channel;
172 
173  bool operator==(const Port &rhs) const {
174  return std::tie(bundle, channel) == std::tie(rhs.bundle, rhs.channel);
175  }
176 
177  bool operator!=(const Port &rhs) const { return !(*this == rhs); }
178 
179  bool operator<(const Port &rhs) const {
180  return std::tie(bundle, channel) < std::tie(rhs.bundle, rhs.channel);
181  }
182 
183  friend std::ostream &operator<<(std::ostream &os, const Port &port) {
184  os << "(";
185  switch (port.bundle) {
186  case WireBundle::Core:
187  os << "Core";
188  break;
189  case WireBundle::DMA:
190  os << "DMA";
191  break;
192  case WireBundle::North:
193  os << "N";
194  break;
195  case WireBundle::East:
196  os << "E";
197  break;
198  case WireBundle::South:
199  os << "S";
200  break;
201  case WireBundle::West:
202  os << "W";
203  break;
204  default:
205  os << "X";
206  break;
207  }
208  os << ": " << std::to_string(port.channel) << ")";
209  return os;
210  }
