173typedef struct Port {
174 WireBundle bundle;
175 int channel;
176
177 bool operator==(const Port &rhs) const {
178 return std::tie(bundle, channel) == std::tie(rhs.bundle, rhs.channel);
179 }
180
181 bool operator!=(const Port &rhs) const { return !(*this == rhs); }
182
183 bool operator<(const Port &rhs) const {
184 return std::tie(bundle, channel) < std::tie(rhs.bundle, rhs.channel);
185 }
186
187 friend std::ostream &operator<<(std::ostream &os, const Port &port) {
188 os << "(";
189 switch (port.bundle) {
190 case WireBundle::Core:
191 os << "Core";
192 break;
193 case WireBundle::DMA:
194 os << "DMA";
195 break;
196 case WireBundle::North:
197 os << "N";
198 break;
199 case WireBundle::East:
200 os << "E";
201 break;
202 case WireBundle::South:
203 os << "S";
204 break;
205 case WireBundle::West:
206 os << "W";
207 break;
208 default:
209 os << "X";
210 break;
211 }
212 os << ": " << std::to_string(port.channel) << ")";
213 return os;
214 }
