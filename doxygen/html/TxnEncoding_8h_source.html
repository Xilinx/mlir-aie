25// Transaction opcodes for the firmware TXN format the compiler currently
26// targets.
27//
28// These DO NOT match the third_party/aie-rt xaie_txn.h enum, which is an older
29// layout (CONFIG_SHIMDMA_BD=5, no NOOP/PREEMPT/LOADPDI block). They match the
30// newer firmware opcodes the compiler emits (e.g. CREATE_SCRATCHPAD=0x0A,
31// UPDATE_REG=0x0C).
32//
33// DRIFT WARNING: when the firmware TXN format changes, update these values
34// here. Do not "correct" them against the submodule's xaie_txn.h — it tracks a
35// different (older) numbering and reconciling to it would break the emitted
36// transactions.
37enum TxnOpcode : uint32_t {
38 TXN_OPC_WRITE = 0,
39 TXN_OPC_BLOCKWRITE = 1,
40 TXN_OPC_BLOCKSET = 2,
41 TXN_OPC_MASKWRITE = 3,
42 TXN_OPC_MASKPOLL = 4,
43 TXN_OPC_NOOP = 5,
44 TXN_OPC_PREEMPT = 6,
45 TXN_OPC_MASKPOLL_BUSY = 7,
46 TXN_OPC_LOADPDI = 8,
47 TXN_OPC_LOAD_PM_START = 9,
48 TXN_OPC_CREATE_SCRATCHPAD = 10,
49 TXN_OPC_UPDATE_STATE_TABLE = 11,
50 TXN_OPC_UPDATE_REG = 12,
51 TXN_OPC_UPDATE_SCRATCH = 13,
52 TXN_OPC_CONFIG_SHIMDMA_BD = 14,
53 TXN_OPC_CONFIG_SHIMDMA_DMABUF_BD = 15,
54 TXN_OPC_CUSTOM_OP_BEGIN = 1U << 7U,
55 TXN_OPC_TCT = TXN_OPC_CUSTOM_OP_BEGIN,
56 TXN_OPC_DDR_PATCH = TXN_OPC_CUSTOM_OP_BEGIN + 1,
57 TXN_OPC_READ_REGS = TXN_OPC_CUSTOM_OP_BEGIN + 2,
58 TXN_OPC_RECORD_TIMER = TXN_OPC_CUSTOM_OP_BEGIN + 3,
59 TXN_OPC_MERGE_SYNC = TXN_OPC_CUSTOM_OP_BEGIN + 4,
60 TXN_OPC_CUSTOM_OP_NEXT = TXN_OPC_CUSTOM_OP_BEGIN + 5,
61 TXN_OPC_LOAD_PM_END_INTERNAL = 200,
62 TXN_OPC_CUSTOM_OP_MAX = 255,
63};

65// Device information for the TXN header.
66struct TxnDeviceInfo {
67 uint8_t major = 0;
68 uint8_t minor = 1;
69 uint8_t devGen = 3; // 3 = NPU (PHX/HWK), 4 = NPU2 (STX/KRK)
70 uint8_t numRows = 6;
71 uint8_t numCols = 5;
72 uint8_t numMemTileRows = 1;

75// Append a 6-word write32 instruction.
76inline void txn_append_write32(std::vector<uint32_t> &txn, uint32_t addr,
77 uint32_t val) {
78 size_t pos = txn.size();
79 txn.resize(pos + 6, 0);
80 txn[pos + 0] = TXN_OPC_WRITE;
81 // txn[pos + 1] is reserved (0)
82 txn[pos + 2] = addr;
83 txn[pos + 3] = 0; // extra bits for reg offset
84 txn[pos + 4] = val;
85 txn[pos + 5] = 6 * sizeof(uint32_t); // operation size
86}

95 txn[pos + 2] = addr;

102// Append a 4-word sync (TCT) instruction.
103inline void txn_append_sync(std::vector<uint32_t> &txn, uint32_t col,
104 uint32_t row, uint32_t dir, uint32_t chan,
105 uint32_t ncol, uint32_t nrow) {
106 size_t pos = txn.size();
107 txn.resize(pos + 4, 0);
108 txn[pos + 0] = TXN_OPC_TCT;
109 txn[pos + 1] = 4 * sizeof(uint32_t); // operation size
110 txn[pos + 2] = (dir & 0xff) | ((row & 0xff) << 8) | ((col & 0xff) << 16);
111 txn[pos + 3] =
112 ((nrow & 0xff) << 8) | ((ncol & 0xff) << 16) | ((chan & 0xff) << 24);
113}

115// Append a variable-length blockwrite instruction.
116// `data` points to `count` uint32_t words of payload.
117//
118// The destination tile is encoded twice in the firmware blockwrite format: once
119// in the upper bits of the absolute `addr` (column/row folded in by the
120// compiler before this point) and once in the dedicated col/row field at
121// word[1]. Callers that build an absolute address from a tile pass that same
122// (col, row) here so both representations agree; callers with a purely flat
123// address pass 0/0.
124inline void txn_append_blockwrite(std::vector<uint32_t> &txn, uint32_t addr,
125 const uint32_t *data, size_t count,
126 uint32_t col = 0, uint32_t row = 0) {
127 const unsigned headerSize = 4;
128 size_t pos = txn.size();
129 txn.resize(pos + headerSize + count, 0);
130 txn[pos + 0] = TXN_OPC_BLOCKWRITE;
131 txn[pos + 1] = (col & 0xff) | ((row & 0xff) << 8);
132 txn[pos + 2] = addr;
133 txn[pos + 3] = static_cast<uint32_t>((headerSize + count) * sizeof(uint32_t));
134 for (size_t i = 0; i < count; ++i)
135 txn[pos + headerSize + i] = data[i];
136}

138// Append a 12-word address_patch (DDR_PATCH) instruction.
139inline void txn_append_address_patch(std::vector<uint32_t> &txn, uint32_t addr,
140 int32_t arg_idx, uint32_t arg_plus) {
141 size_t pos = txn.size();
142 txn.resize(pos + 12, 0);
143 txn[pos + 0] = TXN_OPC_DDR_PATCH; // opcode
144 txn[pos + 1] = 12 * sizeof(uint32_t); // operation size
145 // pos+2..4 are reserved (zero)
146 txn[pos + 5] = 0; // action (0 = patch)
147 txn[pos + 6] = addr; // register address to patch
148 // pos+7 is reserved (zero)
149 txn[pos + 8] = static_cast<uint32_t>(arg_idx); // buffer argument index
150 // pos+9 is reserved (zero)
151 txn[pos + 10] = arg_plus; // byte offset into buffer
152 // pos+11 is reserved (zero)
153}

155// Append a 4-word loadpdi instruction.
156inline void txn_append_loadpdi(std::vector<uint32_t> &txn, uint32_t id,
157 uint32_t size, uint64_t addr) {
158 size_t pos = txn.size();
159 txn.resize(pos + 4, 0);
160 txn[pos + 0] = TXN_OPC_LOADPDI | (id << 16);
161 txn[pos + 1] = size;
162 txn[pos + 2] = static_cast<uint32_t>(addr);
163 txn[pos + 3] = static_cast<uint32_t>(addr >> 32);
164}

166// Append a 1-word preempt instruction.
167inline void txn_append_preempt(std::vector<uint32_t> &txn, uint32_t level) {
168 txn.push_back(TXN_OPC_PREEMPT | (level << 8));
169}

171// Reserve 4 placeholder words for the TXN header. Call this BEFORE appending
172// any instructions, so that the header space is already allocated.
173inline void txn_init(std::vector<uint32_t> &txn) { txn.resize(4, 0); }
174
175// Finalize the 4-word TXN header in-place. Call this AFTER all instructions
176// are appended. The first 4 words must have been reserved by txn_init().
177// `op_count` is the number of operations appended.
178inline void txn_prepend_header(std::vector<uint32_t> &txn, uint32_t op_count,
179 TxnDeviceInfo info = {}) {
180 if (txn.size() < 4)
181 txn.resize(4, 0);
182 txn[0] = (static_cast<uint32_t>(info.numRows) << 24) |
183 (static_cast<uint32_t>(info.devGen) << 16) |
184 (static_cast<uint32_t>(info.minor) << 8) |
185 static_cast<uint32_t>(info.major);
186 txn[1] = (static_cast<uint32_t>(info.numMemTileRows) << 8) |
187 static_cast<uint32_t>(info.numCols);
188 txn[2] = op_count;
189 txn[3] = static_cast<uint32_t>(txn.size() * sizeof(uint32_t));
190}
