57AIERTControl::AIERTControl(const AIE::BaseNPUTargetModel &tm)
58 : targetModel(tm) {
59 // The first column in the NPU lacks a shim tile. AIE-RT exposes some of
60 // the internals about how this is modeled in a somewhat awkward way.
61 size_t partitionStartCol =
62 tm.hasProperty(AIETargetModel::IsVirtualized) ? 1 : 0;
63 size_t partitionNumCols = tm.columns();
64 size_t deviceRows = tm.rows();
65 size_t deviceCols = tm.columns() + partitionStartCol;
66
67 // Don't put this in the target model, because it's XAIE specific.
68 unsigned char devGen;
69 switch (tm.getTargetArch()) {
70 case AIEArch::AIE1: // probably unreachable.
71 devGen = XAIE_DEV_GEN_AIE;
72 break;
73 case AIEArch::AIE2:
74 // FIXME: What if we don't have an IPU? aie-rt
75 // models non-IPU devices differently.
76 devGen = XAIE_DEV_GEN_AIE2IPU;
77 break;
78 case AIEArch::AIE2p:
79 devGen = XAIE_DEV_GEN_AIE2P_STRIX_B0;
80 break;
81 default:
82 assert(false);
83 }
84 configPtr = XAie_Config{
85 /*AieGen*/ devGen,
86 /*BaseAddr*/ XAIE_BASE_ADDR,
87 /*ColShift*/ static_cast<uint8_t>(tm.getColumnShift()),
88 /*RowShift*/ static_cast<uint8_t>(tm.getRowShift()),
89 /*NumRows*/ static_cast<uint8_t>(deviceRows),
90 /*NumCols*/ static_cast<uint8_t>(deviceCols),
91 /*ShimRowNum*/ XAIE_SHIM_ROW,
92 /*MemTileRowStart*/ XAIE_MEM_TILE_ROW_START,
93 /*MemTileNumRows*/ static_cast<uint8_t>(tm.getNumMemTileRows()),
94 /*AieTileRowStart*/
95 static_cast<uint8_t>(XAIE_MEM_TILE_ROW_START + tm.getNumMemTileRows()),
96 /*AieTileNumRows*/
97 static_cast<uint8_t>(tm.rows() - tm.getNumMemTileRows() - 1),
98 /*PartProp*/ {},
99 /*Backend*/ XAIE_IO_BACKEND_CDO};
100 XAie_InstDeclare(_devInst, &configPtr);
101 devInst = _devInst;
102 TRY_XAIE_API_FATAL_ERROR(XAie_SetupPartitionConfig, &devInst,
103 XAIE_PARTITION_BASE_ADDR, partitionStartCol,
104 partitionNumCols);
105 TRY_XAIE_API_FATAL_ERROR(XAie_CfgInitialize, &devInst, &configPtr);
106 TRY_XAIE_API_FATAL_ERROR(XAie_UpdateNpiAddr, &devInst, NPI_ADDR);
107}

109LogicalResult AIERTControl::setIOBackend(bool aieSim, bool xaieDebug) {
110 // Quoting: The instance of a device must be always declared using this
111 // macro. In the future, the same macro will be expanded to
112 // allocate more memory from the user application for resource
113 // management.
114 if (aieSim) {
115 TRY_XAIE_API_FATAL_ERROR(XAie_SetIOBackend, &devInst, XAIE_IO_BACKEND_SIM);
116 } else if (xaieDebug)
117 TRY_XAIE_API_FATAL_ERROR(XAie_SetIOBackend, &devInst,
118 XAIE_IO_BACKEND_DEBUG);
119 else
120 TRY_XAIE_API_FATAL_ERROR(XAie_SetIOBackend, &devInst, XAIE_IO_BACKEND_CDO);
121 return success();
122}

124LogicalResult AIERTControl::configureLocksInBdBlock(XAie_DmaDesc &dmaTileBd,
125 Block &block,
126 XAie_LocType &tileLoc) {
127 LLVM_DEBUG(llvm::dbgs() << "\nstart configuring bds\n");
128 std::optional<int> acqValue, relValue, acqLockId, relLockId;
129 bool acqEn = false;
130
131 // switch (lock->getAc)
132 LockOp lock;
133 for (auto op : block.getOps<UseLockOp>()) {
134 // Only dyn_cast if you are going to check if it was of the type
135 // expected; if you aren't checking use cast instead as it will at
136 // least assert in debug mode with an easier to understand error than
137 // dereferencing.
138 lock = cast<LockOp>(op.getLock().getDefiningOp());
139 switch (op.getAction()) {
140 case LockAction::Acquire:
141 case LockAction::AcquireGreaterEqual:
142 acqEn = op.getAcqEn();
143 acqLockId = lock.getLockIDValue();
144 acqValue = op.getLockValue();
145 if (op.acquireGE())
146 acqValue.value() = -acqValue.value();
147 break;
148 case LockAction::Release:
149 relLockId = lock.getLockIDValue();
150 relValue = op.getLockValue();
151 break;
152 }
153 }
154
155 assert(acqValue && relValue && acqLockId && relLockId &&
156 "expected both use_lock(acquire) and use_lock(release) with bd");
157
158 if (targetModel.isMemTile(tileLoc.Col, tileLoc.Row)) {
159 // check if buffer is allocated on the same memtile, the west, or the east
160 // one
161 int increaseValue = 0;
162 auto lockRow = lock.rowIndex();
163 auto lockCol = lock.colIndex();
164 bool isWestLock =
165 targetModel.isWest(tileLoc.Col, tileLoc.Row, lockCol, lockRow);
166 bool isEastLock =
167 targetModel.isEast(tileLoc.Col, tileLoc.Row, lockCol, lockRow);
168 if (isWestLock) {
169 increaseValue = MEM_TILE_LOCK_ID_INCR_WEST;
170 } else if (isEastLock) {
171 increaseValue = MEM_TILE_LOCK_ID_INCR_EAST;
172 } else {
173 increaseValue = MEM_TILE_LOCK_ID_INCR;
174 }
175 if (acqLockId)
176 acqLockId.value() += increaseValue;
177 if (relLockId)
178 relLockId.value() += increaseValue;
179 }
180
181 // no RelEn in the arch spec even though the API requires you to set it?
182 bool relEn = false;
183 XAie_Lock acqLock = XAie_LockInit(acqLockId.value(), acqValue.value());
184 XAie_Lock relLock = XAie_LockInit(relLockId.value(), relValue.value());
185 TRY_XAIE_API_EMIT_ERROR((*block.getOps<UseLockOp>().begin()),
186 dmaTileBd.DmaMod->SetLock, &dmaTileBd, acqLock,
187 relLock, acqEn, relEn);
188 return success();
189}

191LogicalResult AIERTControl::configureBdInBlock(XAie_DmaDesc &dmaTileBd,
192 Block &block,
193 XAie_LocType &tileLoc, int bdId,
194 std::optional<int> nextBdId) {
195 std::optional<int> packetType;
196 std::optional<int> packetID;
197
198 // Below should go
199 auto maybePacketOps = block.getOps<DMABDPACKETOp>();
200 if (!maybePacketOps.empty()) {
201 assert(llvm::range_size(maybePacketOps) == 1 &&
202 "expected only one dma_bd_packet");
203 auto packetOp = *maybePacketOps.begin();
204 packetType = packetOp.getPacketType();
205 packetID = packetOp.getPacketID();
206 }
207
208 auto bdOp = *block.getOps<DMABDOp>().begin();
209
210 if (targetModel.isShimNOCTile(tileLoc.Col, tileLoc.Row)) {
211 // write them out like this so they show up with names in debug prints
212 uint8_t smid = 0;
213 uint32_t burstLen =
214 getShimBurstLengthBytes(targetModel, bdOp.getBurstLength());
215 uint8_t qOs = 0;
216 uint8_t cache = 0;
217 uint8_t secure = 0;
218 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetAxi, &dmaTileBd, smid,
219 burstLen / 16, qOs, cache, secure);
220 }
221
222 // get address from BufferOp (core,mem) or ExternalBufferOp (shim)
223 uint64_t baseAddr = 0;
224 if (targetModel.isShimNOCTile(tileLoc.Col, tileLoc.Row)) {
225 auto bufferOp =
226 cast<AIE::ExternalBufferOp>(bdOp.getBuffer().getDefiningOp());
227 // external buffers aren't required to have an address here because the
228 // address might get patched later or the default of zero might be a valid
229 // address.
230 if (bufferOp.getAddress())
231 baseAddr = bufferOp.getAddress().value();
232 } else {
233 auto bufferOp = cast<AIE::BufferOp>(bdOp.getBuffer().getDefiningOp());
234 if (!bufferOp.getAddress())
235 return bufferOp.emitError("buffer must have address assigned");
236 baseAddr = bufferOp.getAddress().value();
237 }
238
239 if (targetModel.isMemTile(tileLoc.Col, tileLoc.Row)) {
240 // check if buffer is allocated on the same memtile, the west, or the east
241 // one
242 auto bufferOp = cast<AIE::BufferOp>(bdOp.getBuffer().getDefiningOp());
243 auto bufferRow = bufferOp.getTileOp().getRow();
244 auto bufferCol = bufferOp.getTileOp().getCol();
245 bool isWestBuff =
246 targetModel.isWest(tileLoc.Col, tileLoc.Row, bufferCol, bufferRow);
247 bool isEastBuff =
248 targetModel.isEast(tileLoc.Col, tileLoc.Row, bufferCol, bufferRow);
249 if (isWestBuff) {
250 baseAddr += BASE_ADDR_A_INCR_WEST;
251 } else if (isEastBuff) {
252 baseAddr += BASE_ADDR_A_INCR_EAST;
253 } else {
254 baseAddr += BASE_ADDR_A_INCR;
255 }
256 }
257
258 std::optional<llvm::ArrayRef<BDDimLayoutAttr>> dims = bdOp.getDimensions();
259 uint64_t lenInBytes = bdOp.getLenInBytes();
260 uint64_t basePlusOffsetInBytes = baseAddr + bdOp.getOffsetInBytes();
261 if (!dims) {
262 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetAddrLen, &dmaTileBd,
263 basePlusOffsetInBytes, lenInBytes);
264 } else {
265 XAie_DmaTensor dmaTileBdTensor = {};
266 dmaTileBdTensor.NumDim = dims->size();
267 dmaTileBdTensor.Dim = static_cast<XAie_DmaDimDesc *>(
268 calloc(dmaTileBdTensor.NumDim, sizeof(XAie_DmaDimDesc)));
269 if (!dmaTileBdTensor.Dim)
270 return bdOp.emitError("couldn't allocate array of XAie_DmaDimDesc");
271 // libxaie requires stride in multiples of 32b
272 double elementWidthIn32bWords =
273 static_cast<double>(bdOp.getBufferElementTypeWidthInBytes()) / 4.0;
274 for (size_t i = 0; i < dims->size(); i++) {
275 // Pass down dimensions in reverse order; in the MLIR, this allows
276 // us to specify step sizes/wraps in the same order as we would
277 // access a multi-dim C array, with the highest dimension first.
278 int j = dims->size() - i - 1;
279 uint16_t size;
280 uint32_t stride;
281 if (j > 0) {
282 stride = static_cast<uint32_t>(dims.value()[i].getStride() *
283 elementWidthIn32bWords);
284 size = dims.value()[i].getSize();
285 } else {
286 stride = dims.value()[i].getStride();
287 size = static_cast<uint16_t>(dims.value()[i].getSize() *
288 elementWidthIn32bWords);
289 }
290 stride = stride > 0 ? stride : 1;
291 // Assume AIE-ML architecture (ie use AieMlDimDesc instead of AieDimDesc);
292 // asserted in AIETranslateToCDODirect).
293 dmaTileBdTensor.Dim[j].AieMlDimDesc = {stride, size};
294 }
295 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetMultiDimAddr, &dmaTileBd,
296 &dmaTileBdTensor, basePlusOffsetInBytes,
297 lenInBytes);
298 }
299
300 // ND zero padding.
301 std::optional<llvm::ArrayRef<BDPadLayoutAttr>> padDims =
302 bdOp.getPadDimensions();
303
304 if (padDims) {
305 XAie_DmaPadTensor dmaPadTensor = {};
306 dmaPadTensor.NumDim = padDims->size();
307 dmaPadTensor.PadDesc = static_cast<XAie_PadDesc *>(
308 calloc(dmaPadTensor.NumDim, sizeof(XAie_PadDesc)));
309 if (!dmaPadTensor.PadDesc)
310 return bdOp.emitError("couldn't allocate array of XAie_PadDesc");
311 // libxaie requires stride in multiples of 32b
312 double elementWidthIn32bWords =
313 static_cast<double>(bdOp.getBufferElementTypeWidthInBytes()) / 4.0;
314 for (size_t i = 0; i < padDims->size(); i++) {
315 // Pass down dimensions in reverse order.
316 int j = padDims->size() - i - 1;
317 uint8_t before;
318 uint8_t after;
319 if (j > 0) {
320 before = static_cast<uint8_t>(padDims.value()[i].getConstPadBefore());
321 after = static_cast<uint8_t>(padDims.value()[i].getConstPadAfter());
322 } else {
323 before = static_cast<uint8_t>(padDims.value()[i].getConstPadBefore() *
324 elementWidthIn32bWords);
325 after = static_cast<uint8_t>(padDims.value()[i].getConstPadAfter() *
326 elementWidthIn32bWords);
327 }
328 dmaPadTensor.PadDesc[j] = {before, after};
329 }
330 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetPadding, &dmaTileBd,
331 &dmaPadTensor);
332 }
333 if (nextBdId) {
334 auto enableNextBd = 1;
335 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetNextBd, &dmaTileBd,
336 nextBdId.value(), enableNextBd);
337 }
338
339 if (auto packetInfo = bdOp.getPacket()) {
340 packetType = packetInfo->getPktType();
341 packetID = packetInfo->getPktId();
342 }
343
344 if (packetID) {
345 if (!packetType)
346 bdOp.emitError("must have packetType with packetID");
347 if (bdOp.getLen() == 0)
348 return bdOp.emitOpError(
349 "For MM2S channels, if Buffer_Length=0 then Enable_Packet must be "
350 "set to 0, otherwise behavior is undefined (3.7.8 arch spec)");
351 TRY_XAIE_API_EMIT_ERROR(
352 bdOp, XAie_DmaSetPkt, &dmaTileBd,
353 XAie_PacketInit(packetID.value(), packetType.value()));
354 }
355 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaEnableBd, &dmaTileBd);
356 TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaWriteBd, &devInst, &dmaTileBd, tileLoc,
357 bdId);
358 LLVM_DEBUG(llvm::dbgs() << "\nend configuring bds\n");
359 return success();
360};

362LogicalResult
363AIERTControl::pushToBdQueueAndEnable(Operation &op, XAie_LocType &tileLoc,
364 int chNum, const DMAChannelDir &channelDir,
365 int bdId, int repeatCount) {
366 XAie_DmaDirection direction =
367 channelDir == DMAChannelDir::S2MM ? DMA_S2MM : DMA_MM2S;
368 auto enTokenIssue = tileLoc.Row == 0 && direction == DMA_S2MM;
369 // in english repeat_count==0 means "do it once" and don't repeat but
370 // libxaie treats repeat_count=1 as do it once.
371 repeatCount += 1;
372 TRY_XAIE_API_EMIT_ERROR(op, XAie_DmaChannelSetStartQueue, &devInst, tileLoc,
373 chNum, direction, bdId, repeatCount, enTokenIssue);
374 TRY_XAIE_API_EMIT_ERROR(op, XAie_DmaChannelEnable, &devInst, tileLoc, chNum,
375 direction);
376 return success();
377};

379LogicalResult AIERTControl::configureLocksAndBd(Block &block,
380 XAie_LocType tileLoc) {
381 DMABDOp bd = *block.getOps<DMABDOp>().begin();
382 assert(bd.getBdId().has_value() &&
383 "DMABDOp must have assigned bd_id; did you forget to run "
384 "aie-assign-bd-ids?");
385 XAie_DmaDesc dmaTileBd;
386 TRY_XAIE_API_EMIT_ERROR(bd, XAie_DmaDescInit, &devInst, &dmaTileBd, tileLoc);
387 if (!block.getOps<UseLockOp>().empty() &&
388 failed(configureLocksInBdBlock(dmaTileBd, block, tileLoc)))
389 return failure();
390 if (!block.getOps<DMABDOp>().empty() &&
391 failed(configureBdInBlock(dmaTileBd, block, tileLoc, bd.getBdId().value(),
392 bd.getNextBdId())))
393 return failure();
394 return success();
395}

397LogicalResult AIERTControl::initLocks(DeviceOp &targetOp) {
398 for (auto tileOp : targetOp.getOps<TileOp>()) {
399 auto tileLoc = XAie_TileLoc(tileOp.colIndex(), tileOp.rowIndex());
400 if (!tileOp.isShimTile() && tileOp.getCoreOp()) {
401 TRY_XAIE_API_EMIT_ERROR(tileOp, XAie_CoreReset, &devInst, tileLoc);
402 TRY_XAIE_API_EMIT_ERROR(tileOp, XAie_CoreUnreset, &devInst, tileLoc);
403 // Set locks to zero
404 for (uint8_t l = 0; l < NUM_LOCKS; l++) {
405 auto locInit = XAie_LockInit(l, 0);
406 TRY_XAIE_API_EMIT_ERROR(tileOp, XAie_LockSetValue, &devInst, tileLoc,
407 locInit);
408 }
409 }
410 }
411
412 // Set locks with explicit initializers
413 targetOp.walk<WalkOrder::PreOrder>([&](LockOp lockOp) {
414 if (lockOp.getLockID() && lockOp.getInit()) {
415 auto tileLoc = XAie_TileLoc(lockOp.getTileOp().colIndex(),
416 lockOp.getTileOp().rowIndex());
417 auto locInit = XAie_LockInit(*lockOp.getLockID(), *lockOp.getInit());
418 TRY_XAIE_API_FATAL_ERROR(XAie_LockSetValue, &devInst, tileLoc, locInit);
419 } else
420 LLVM_DEBUG(llvm::dbgs()
421 << "lock op missing either id or init" << lockOp << "\n");
422 });
423 return success();
424}

426LogicalResult AIERTControl::initBuffers(DeviceOp &targetOp) {
427 // Set buffers with explicit initializers
428 targetOp.walk<WalkOrder::PreOrder>([&](BufferOp bufferOp) {
429 auto initialValue = bufferOp.getInitialValue();
430 if (!initialValue)
431 return;
432 mlir::DenseElementsAttr denseInit =
433 dyn_cast<mlir::DenseElementsAttr>(initialValue.value());
434 if (!denseInit)
435 return;
436 auto tileLoc = XAie_TileLoc(bufferOp.getTileOp().colIndex(),
437 bufferOp.getTileOp().rowIndex());
438 std::vector<char> byteVec;
439 if (denseInit.getElementType().isIntOrIndex()) {
440 for (auto intVal : denseInit.getValues<APInt>()) {
441 // Get the size in bytes
442 size_t byteSize = (intVal.getBitWidth() + 7) / 8;
443 // Create a buffer for the integer bytes and copy
444 std::vector<char> bytes(byteSize);
445 std::copy(
446 static_cast<const char *>(static_cast<const void *>(&intVal)),
447 static_cast<const char *>(static_cast<const void *>(&intVal)) +
448 byteSize,
449 bytes.begin());
450 byteVec.insert(byteVec.end(), bytes.begin(), bytes.end());
451 }
452 } else if (isa<FloatType>(denseInit.getElementType())) {
453 for (auto floatVal : denseInit.getValues<APFloat>()) {
454 APInt floatInt = floatVal.bitcastToAPInt();
455 // Get the size in bytes
456 size_t byteSize = (floatInt.getBitWidth() + 7) / 8;
457 // Create a buffer for the float bytes and copy
458 std::vector<char> bytes(byteSize);
459 std::copy(
460 static_cast<const char *>(static_cast<const void *>(&floatInt)),
461 static_cast<const char *>(static_cast<const void *>(&floatInt)) +
462 byteSize,
463 bytes.begin());
464 byteVec.insert(byteVec.end(), bytes.begin(), bytes.end());
465 }
466 } else {
467 llvm::outs() << "buffer op type not supported for initialization "
468 << bufferOp << "\n";
469 return;
470 }
471 TRY_XAIE_API_FATAL_ERROR(XAie_DataMemBlockWrite, &devInst, tileLoc,
472 bufferOp.getAddress().value(), byteVec.data(),
473 byteVec.size());
474 });
475 return success();
476}

478LogicalResult AIERTControl::configureSwitches(DeviceOp &targetOp) {
479
480 // StreamSwitch (switchbox) configuration
481 for (auto switchboxOp : targetOp.getOps<SwitchboxOp>()) {
482 int32_t col = switchboxOp.colIndex();
483 int32_t row = switchboxOp.rowIndex();
484 XAie_LocType tileLoc = XAie_TileLoc(col, row);
485 assert(targetModel.hasProperty(AIETargetModel::IsNPU) &&
486 "Only NPU currently supported");
487
488 Block &b = switchboxOp.getConnections().front();
489 for (auto connectOp : b.getOps<ConnectOp>())
490 TRY_XAIE_API_EMIT_ERROR(
491 switchboxOp, XAie_StrmConnCctEnable, &devInst, tileLoc,
492 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getSourceBundle()),
493 connectOp.sourceIndex(),
494 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getDestBundle()),
495 connectOp.destIndex());
496
497 for (auto masterSetOp : b.getOps<MasterSetOp>()) {
498 int mask = 0;
499 int arbiter = -1;
500
501 for (auto val : masterSetOp.getAmsels()) {
502 AMSelOp amsel = cast<AMSelOp>(val.getDefiningOp());
503 arbiter = amsel.arbiterIndex();
504 int msel = amsel.getMselValue();
505 mask |= (1 << msel);
506 }
507
508 // the default is to keep header
509 bool keepHeader = true;
510 // the default for dma destinations is to drop the header
511 if (masterSetOp.getDestBundle() == WireBundle::DMA)
512 keepHeader = false;
513 // assume a connection going south from row zero gets wired to shimdma
514 // by a shimmux.
515 if (switchboxOp.rowIndex() == 0 &&
516 masterSetOp.getDestBundle() == WireBundle::South)
517 keepHeader = false;
518
519 // "keep_pkt_header" attribute overrides the above defaults, if set
520 if (auto keep = masterSetOp.getKeepPktHeader())
521 keepHeader = *keep;
522
523 auto dropHeader =
524 keepHeader ? XAIE_SS_PKT_DONOT_DROP_HEADER : XAIE_SS_PKT_DROP_HEADER;
525 TRY_XAIE_API_EMIT_ERROR(
526 masterSetOp, XAie_StrmPktSwMstrPortEnable, &devInst, tileLoc,
527 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(masterSetOp.getDestBundle()),
528 masterSetOp.destIndex(), dropHeader, arbiter, mask);
529 }
530
531 for (auto packetRulesOp : b.getOps<PacketRulesOp>()) {
532 int slot = 0;
533 Block &block = packetRulesOp.getRules().front();
534 for (auto slotOp : block.getOps<PacketRuleOp>()) {
535 AMSelOp amselOp = cast<AMSelOp>(slotOp.getAmsel().getDefiningOp());
536 int arbiter = amselOp.arbiterIndex();
537 int msel = amselOp.getMselValue();
538 TRY_XAIE_API_EMIT_ERROR(packetRulesOp, XAie_StrmPktSwSlavePortEnable,
539 &devInst, tileLoc,
540 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
541 packetRulesOp.getSourceBundle()),
542 packetRulesOp.sourceIndex());
543 auto packetInit = XAie_PacketInit(slotOp.valueInt(), /*PktType*/ 0);
544 // TODO Need to better define packet id,type used here
545 TRY_XAIE_API_EMIT_ERROR(packetRulesOp, XAie_StrmPktSwSlaveSlotEnable,
546 &devInst, tileLoc,
547 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
548 packetRulesOp.getSourceBundle()),
549 packetRulesOp.sourceIndex(), slot, packetInit,
550 slotOp.maskInt(), msel, arbiter);
551 slot++;
552 }
553 }
554 }
555
556 for (auto muxOp : targetOp.getOps<ShimMuxOp>()) {
557 // NOTE ShimMux always connects from the south as directions are
558 // defined relative to the tile stream switch.
559 auto tileLoc =
560 XAie_TileLoc(muxOp.getTileOp().getCol(), muxOp.getTileOp().getRow());
561 Block &b = muxOp.getConnections().front();
562 for (auto connectOp : b.getOps<ConnectOp>()) {
563 // demux!
564 if (connectOp.getSourceBundle() == WireBundle::North)
565 TRY_XAIE_API_EMIT_ERROR(muxOp, XAie_EnableAieToShimDmaStrmPort,
566 &devInst, tileLoc, connectOp.sourceIndex());
567 // mux
568 if (connectOp.getDestBundle() == WireBundle::North)
569 TRY_XAIE_API_EMIT_ERROR(muxOp, XAie_EnableShimDmaToAieStrmPort,
570 &devInst, tileLoc, connectOp.destIndex());
571 }
572 }
573
574 for (auto switchboxOp : targetOp.getOps<ShimSwitchboxOp>()) {
575 Block &b = switchboxOp.getConnections().front();
576 auto tileLoc = XAie_TileLoc(switchboxOp.getCol(), 0);
577 for (auto connectOp : b.getOps<ConnectOp>())
578 TRY_XAIE_API_EMIT_ERROR(
579 switchboxOp, XAie_StrmConnCctEnable, &devInst, tileLoc,
580 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getSourceBundle()),
581 connectOp.sourceIndex(),
582 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getDestBundle()),
583 connectOp.destIndex());
584 }
585
586 // Cascade configuration
587 if (isa<AIE2TargetModel>(targetModel)) {
588 for (auto configOp : targetOp.getOps<ConfigureCascadeOp>()) {
589 TileOp tile = cast<TileOp>(configOp.getTile().getDefiningOp());
590 auto tileLoc = XAie_TileLoc(tile.getCol(), tile.getRow());
591 TRY_XAIE_API_EMIT_ERROR(
592 targetOp, XAie_CoreConfigAccumulatorControl, &devInst, tileLoc,
593 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
594 static_cast<WireBundle>(configOp.getInputDir())),
595 WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
596 static_cast<WireBundle>(configOp.getOutputDir())));
597 }
598 }
599
600 return success();
601}

603LogicalResult AIERTControl::addInitConfig(DeviceOp &targetOp) {
604
605 if (failed(initLocks(targetOp))) {
606 return failure();
607 }
608
609 if (failed(initBuffers(targetOp))) {
610 return failure();
611 }
612
613 auto memOps = llvm::to_vector_of<TileElement>(targetOp.getOps<MemOp>());
614 llvm::append_range(memOps, targetOp.getOps<MemTileDMAOp>());
615 llvm::append_range(memOps, targetOp.getOps<ShimDMAOp>());
616 for (TileElement memOp : memOps) {
617 int col = memOp.getTileID().col;
618 int row = memOp.getTileID().row;
619 XAie_LocType tileLoc = XAie_TileLoc(col, row);
620
621 // Get the region's entry block, then start traversing through the chain of
622 // blocks.
623 llvm::SetVector<Block *> blockVector =
624 getOrderedChainOfBlocks(&memOp.getOperation()->getRegion(0));
625
626 // handle DMA ops separately
627 auto dmaOps = llvm::to_vector_of<DMAOp>(
628 memOp.getOperation()->getRegion(0).getOps<DMAOp>());
629 if (!dmaOps.empty()) {
630 for (auto dmaOp : dmaOps)
631 for (auto &bdRegion : dmaOp.getBds()) {
632 Block &block = bdRegion.getBlocks().front();
633 if (failed(configureLocksAndBd(block, tileLoc)))
634 return failure();
635 }
636 } else {
637 for (Block *block : blockVector) {
638 if (block->getOps<DMABDOp>().empty())
639 continue;
640 if (failed(configureLocksAndBd(*block, tileLoc)))
641 return failure();
642 }
643 }
644
645 if (!dmaOps.empty())
646 for (auto dmaOp : dmaOps) {
647 auto &block = dmaOp.getBds().front().getBlocks().front();
648 DMABDOp bd = *block.getOps<DMABDOp>().begin();
649 if (failed(pushToBdQueueAndEnable(
650 *dmaOp.getOperation(), tileLoc, dmaOp.getChannelIndex(),
651 dmaOp.getChannelDir(), bd.getBdId().value(),
652 dmaOp.getRepeatCount())))
653 return failure();
654 }
655 else
656 for (Block *block : blockVector) {
657 for (auto op : block->getOps<DMAStartOp>()) {
658 DMABDOp bd = *op.getDest()->getOps<DMABDOp>().begin();
659 int chNum = op.getChannelIndex();
660 auto channelDir = op.getChannelDir();
661 if (failed(pushToBdQueueAndEnable(*bd.getOperation(), tileLoc, chNum,
662 channelDir, bd.getBdId().value(),
663 op.getRepeatCount())))
664 return failure();
665 }
666 }
667 }
668
669 if (failed(configureSwitches(targetOp))) {
670 return failure();
671 }
672
673 return success();
674}
