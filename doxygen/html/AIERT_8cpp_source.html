57 AIERTControl::AIERTControl(const AIE::BaseNPUTargetModel &tm)
58  : targetModel(tm) {
59  // The first column in the NPU lacks a shim tile. AIE-RT exposes some of
60  // the internals about how this is modeled in a somewhat awkward way.
61  size_t partitionStartCol =
62  tm.hasProperty(AIETargetModel::IsVirtualized) ? 1 : 0;
63  size_t partitionNumCols = tm.columns();
64  size_t deviceRows = tm.rows();
65  size_t deviceCols = tm.columns() + partitionStartCol;
66 
67  // Don't put this in the target model, because it's XAIE specific.
68  unsigned char devGen;
69  switch (tm.getTargetArch()) {
70  case AIEArch::AIE1: // probably unreachable.
71  devGen = XAIE_DEV_GEN_AIE;
72  break;
73  case AIEArch::AIE2:
74  // FIXME: What if we don't have an IPU? aie-rt
75  // models non-IPU devices differently.
76  devGen = XAIE_DEV_GEN_AIE2IPU;
77  break;
78  case AIEArch::AIE2p:
79  devGen = XAIE_DEV_GEN_AIE2P_STRIX_B0;
80  break;
81  default:
82  assert(false);
83  }
84  configPtr = XAie_Config{
85  /*AieGen*/ devGen,
86  /*BaseAddr*/ XAIE_BASE_ADDR,
87  /*ColShift*/ static_cast<uint8_t>(tm.getColumnShift()),
88  /*RowShift*/ static_cast<uint8_t>(tm.getRowShift()),
89  /*NumRows*/ static_cast<uint8_t>(deviceRows),
90  /*NumCols*/ static_cast<uint8_t>(deviceCols),
91  /*ShimRowNum*/ XAIE_SHIM_ROW,
92  /*MemTileRowStart*/ XAIE_MEM_TILE_ROW_START,
93  /*MemTileNumRows*/ static_cast<uint8_t>(tm.getNumMemTileRows()),
94  /*AieTileRowStart*/
95  static_cast<uint8_t>(XAIE_MEM_TILE_ROW_START + tm.getNumMemTileRows()),
96  /*AieTileNumRows*/
97  static_cast<uint8_t>(tm.rows() - tm.getNumMemTileRows() - 1),
98  /*PartProp*/ {},
99  /*Backend*/ XAIE_IO_BACKEND_CDO};
100  XAie_InstDeclare(_devInst, &configPtr);
101  devInst = _devInst;
102  TRY_XAIE_API_FATAL_ERROR(XAie_SetupPartitionConfig, &devInst,
103  XAIE_PARTITION_BASE_ADDR, partitionStartCol,
104  partitionNumCols);
105  TRY_XAIE_API_FATAL_ERROR(XAie_CfgInitialize, &devInst, &configPtr);
106  TRY_XAIE_API_FATAL_ERROR(XAie_UpdateNpiAddr, &devInst, NPI_ADDR);
107 }

124 LogicalResult AIERTControl::configureLocksInBdBlock(XAie_DmaDesc &dmaTileBd,
125  Block &block,
126  XAie_LocType &tileLoc) {
127  LLVM_DEBUG(llvm::dbgs() << "\nstart configuring bds\n");
128  std::optional<int> acqValue, relValue, acqLockId, relLockId;
129  bool acqEn = false;
130 
131  // switch (lock->getAc)
132  LockOp lock;
133  for (auto op : block.getOps<UseLockOp>()) {
134  // Only dyn_cast if you are going to check if it was of the type
135  // expected; if you aren't checking use cast instead as it will at
136  // least assert in debug mode with an easier to understand error than
137  // dereferencing.
138  lock = cast<LockOp>(op.getLock().getDefiningOp());
139  switch (op.getAction()) {
140  case LockAction::Acquire:
141  case LockAction::AcquireGreaterEqual:
142  acqEn = op.getAcqEn();
143  acqLockId = lock.getLockIDValue();
144  acqValue = op.getLockValue();
145  if (op.acquireGE())
146  acqValue.value() = -acqValue.value();
147  break;
148  case LockAction::Release:
149  relLockId = lock.getLockIDValue();
150  relValue = op.getLockValue();
151  break;
152  }
153  }
154 
155  assert(acqValue && relValue && acqLockId && relLockId &&
156  "expected both use_lock(acquire) and use_lock(release) with bd");
157 
158  if (targetModel.isMemTile(tileLoc.Col, tileLoc.Row)) {
159  // check if buffer is allocated on the same memtile, the west, or the east
160  // one
161  int increaseValue = 0;
162  auto lockRow = lock.rowIndex();
163  auto lockCol = lock.colIndex();
164  bool isWestLock =
165  targetModel.isWest(tileLoc.Col, tileLoc.Row, lockCol, lockRow);
166  bool isEastLock =
167  targetModel.isEast(tileLoc.Col, tileLoc.Row, lockCol, lockRow);
168  if (isWestLock) {
169  increaseValue = MEM_TILE_LOCK_ID_INCR_WEST;
170  } else if (isEastLock) {
171  increaseValue = MEM_TILE_LOCK_ID_INCR_EAST;
172  } else {
173  increaseValue = MEM_TILE_LOCK_ID_INCR;
174  }
175  if (acqLockId)
176  acqLockId.value() += increaseValue;
177  if (relLockId)
178  relLockId.value() += increaseValue;
179  }
180 
181  // no RelEn in the arch spec even though the API requires you to set it?
182  bool relEn = false;
183  XAie_Lock acqLock = XAie_LockInit(acqLockId.value(), acqValue.value());
184  XAie_Lock relLock = XAie_LockInit(relLockId.value(), relValue.value());
185  TRY_XAIE_API_EMIT_ERROR((*block.getOps<UseLockOp>().begin()),
186  dmaTileBd.DmaMod->SetLock, &dmaTileBd, acqLock,
187  relLock, acqEn, relEn);
188  return success();
189 }

191 LogicalResult AIERTControl::configureBdInBlock(XAie_DmaDesc &dmaTileBd,
192  Block &block,
193  XAie_LocType &tileLoc, int bdId,
194  std::optional<int> nextBdId) {
195  std::optional<int> packetType;
196  std::optional<int> packetID;
197 
198  // Below should go
199  auto maybePacketOps = block.getOps<DMABDPACKETOp>();
200  if (!maybePacketOps.empty()) {
201  assert(llvm::range_size(maybePacketOps) == 1 &&
202  "expected only one dma_bd_packet");
203  auto packetOp = *maybePacketOps.begin();
204  packetType = packetOp.getPacketType();
205  packetID = packetOp.getPacketID();
206  }
207 
208  auto bdOp = *block.getOps<DMABDOp>().begin();
209 
210  if (targetModel.isShimNOCTile(tileLoc.Col, tileLoc.Row)) {
211  // write them out like this so they show up with names in debug prints
212  size_t smid = 0;
213  size_t burstLen = 16; // (10):BLEN=16 (256Byte) (corresponds to
214  // 0x800000000 from target)
215  size_t qOs = 0;
216  size_t cache = 0;
217  size_t secure = 0;
218  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetAxi, &dmaTileBd, smid, burstLen,
219  qOs, cache, secure);
220  }
221 
222  // StringRef FifoMode = disable; // FIXME: when to enable FIFO mode?
223  int baseAddr = 0;
224  if (!targetModel.isShimNOCTile(tileLoc.Col, tileLoc.Row)) {
225  auto bufferOp = cast<AIE::BufferOp>(bdOp.getBuffer().getDefiningOp());
226  if (!bufferOp.getAddress())
227  return bufferOp.emitError("buffer must have address assigned");
228  baseAddr = bufferOp.getAddress().value();
229  if (targetModel.isMemTile(tileLoc.Col, tileLoc.Row)) {
230  // check if buffer is allocated on the same memtile, the west, or the east
231  // one
232  auto bufferRow = bufferOp.getTileOp().getRow();
233  auto bufferCol = bufferOp.getTileOp().getCol();
234  bool isWestBuff =
235  targetModel.isWest(tileLoc.Col, tileLoc.Row, bufferCol, bufferRow);
236  bool isEastBuff =
237  targetModel.isEast(tileLoc.Col, tileLoc.Row, bufferCol, bufferRow);
238  if (isWestBuff) {
239  baseAddr += BASE_ADDR_A_INCR_WEST;
240  } else if (isEastBuff) {
241  baseAddr += BASE_ADDR_A_INCR_EAST;
242  } else {
243  baseAddr += BASE_ADDR_A_INCR;
244  }
245  }
246  }
247 
248  std::optional<llvm::ArrayRef<BDDimLayoutAttr>> dims = bdOp.getDimensions();
249  int lenInBytes = bdOp.getLenInBytes();
250  int basePlusOffsetInBytes = baseAddr + bdOp.getOffsetInBytes();
251  if (!dims) {
252  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetAddrLen, &dmaTileBd,
253  basePlusOffsetInBytes, lenInBytes);
254  } else {
255  XAie_DmaTensor dmaTileBdTensor = {};
256  dmaTileBdTensor.NumDim = dims->size();
257  dmaTileBdTensor.Dim = static_cast<XAie_DmaDimDesc *>(
258  calloc(dmaTileBdTensor.NumDim, sizeof(XAie_DmaDimDesc)));
259  if (!dmaTileBdTensor.Dim)
260  return bdOp.emitError("couldn't allocate array of XAie_DmaDimDesc");
261  // libxaie requires stride in multiples of 32b
262  double elementWidthIn32bWords =
263  static_cast<double>(bdOp.getBufferElementTypeWidthInBytes()) / 4.0;
264  for (size_t i = 0; i < dims->size(); i++) {
265  // Pass down dimensions in reverse order; in the MLIR, this allows
266  // us to specify step sizes/wraps in the same order as we would
267  // access a multi-dim C array, with the highest dimension first.
268  int j = dims->size() - i - 1;
269  uint16_t size;
270  uint32_t stride;
271  if (j > 0) {
272  stride = static_cast<uint32_t>(dims.value()[i].getStride() *
273  elementWidthIn32bWords);
274  size = dims.value()[i].getSize();
275  } else {
276  stride = dims.value()[i].getStride();
277  size = static_cast<uint16_t>(dims.value()[i].getSize() *
278  elementWidthIn32bWords);
279  }
280  stride = stride > 0 ? stride : 1;
281  // Assume AIE-ML architecture (ie use AieMlDimDesc instead of AieDimDesc);
282  // asserted in AIETranslateToCDODirect).
283  dmaTileBdTensor.Dim[j].AieMlDimDesc = {stride, size};
284  }
285  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetMultiDimAddr, &dmaTileBd,
286  &dmaTileBdTensor, basePlusOffsetInBytes,
287  lenInBytes);
288  }
289 
290  // ND zero padding.
291  std::optional<llvm::ArrayRef<BDPadLayoutAttr>> padDims =
292  bdOp.getPadDimensions();
293 
294  if (padDims) {
295  XAie_DmaPadTensor dmaPadTensor = {};
296  dmaPadTensor.NumDim = padDims->size();
297  dmaPadTensor.PadDesc = static_cast<XAie_PadDesc *>(
298  calloc(dmaPadTensor.NumDim, sizeof(XAie_PadDesc)));
299  if (!dmaPadTensor.PadDesc)
300  return bdOp.emitError("couldn't allocate array of XAie_PadDesc");
301  // libxaie requires stride in multiples of 32b
302  double elementWidthIn32bWords =
303  static_cast<double>(bdOp.getBufferElementTypeWidthInBytes()) / 4.0;
304  for (size_t i = 0; i < padDims->size(); i++) {
305  // Pass down dimensions in reverse order.
306  int j = padDims->size() - i - 1;
307  uint8_t before;
308  uint8_t after;
309  if (j > 0) {
310  before = static_cast<uint8_t>(padDims.value()[i].getConstPadBefore());
311  after = static_cast<uint8_t>(padDims.value()[i].getConstPadAfter());
312  } else {
313  before = static_cast<uint8_t>(padDims.value()[i].getConstPadBefore() *
314  elementWidthIn32bWords);
315  after = static_cast<uint8_t>(padDims.value()[i].getConstPadAfter() *
316  elementWidthIn32bWords);
317  }
318  dmaPadTensor.PadDesc[j] = {before, after};
319  }
320  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetPadding, &dmaTileBd,
321  &dmaPadTensor);
322  }
323  if (nextBdId) {
324  auto enableNextBd = 1;
325  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaSetNextBd, &dmaTileBd,
326  nextBdId.value(), enableNextBd);
327  }
328 
329  if (auto packetInfo = bdOp.getPacket()) {
330  packetType = packetInfo->getPktType();
331  packetID = packetInfo->getPktId();
332  }
333 
334  if (packetID) {
335  if (!packetType)
336  bdOp.emitError("must have packetType with packetID");
337  if (bdOp.getLen() == 0)
338  return bdOp.emitOpError(
339  "For MM2S channels, if Buffer_Length=0 then Enable_Packet must be "
340  "set to 0, otherwise behavior is undefined (3.7.8 arch spec)");
341  TRY_XAIE_API_EMIT_ERROR(
342  bdOp, XAie_DmaSetPkt, &dmaTileBd,
343  XAie_PacketInit(packetID.value(), packetType.value()));
344  }
345  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaEnableBd, &dmaTileBd);
346  TRY_XAIE_API_EMIT_ERROR(bdOp, XAie_DmaWriteBd, &devInst, &dmaTileBd, tileLoc,
347  bdId);
348  LLVM_DEBUG(llvm::dbgs() << "\nend configuring bds\n");
349  return success();
350 };

365  direction);

369 LogicalResult AIERTControl::configureLocksAndBd(Block &block,
370  XAie_LocType tileLoc) {
371  DMABDOp bd = *block.getOps<DMABDOp>().begin();
372  assert(bd.getBdId().has_value() &&
373  "DMABDOp must have assigned bd_id; did you forget to run "
374  "aie-assign-bd-ids?");

376  TRY_XAIE_API_EMIT_ERROR(bd, XAie_DmaDescInit, &devInst, &dmaTileBd, tileLoc);

382  bd.getNextBdId())))
383  return failure();
384  return success();
385 }

387 LogicalResult AIERTControl::initLocks(DeviceOp &targetOp) {
388  for (auto tileOp : targetOp.getOps<TileOp>()) {
389  auto tileLoc = XAie_TileLoc(tileOp.colIndex(), tileOp.rowIndex());
390  if (!tileOp.isShimTile() && tileOp.getCoreOp()) {
391  TRY_XAIE_API_EMIT_ERROR(tileOp, XAie_CoreReset, &devInst, tileLoc);
392  TRY_XAIE_API_EMIT_ERROR(tileOp, XAie_CoreUnreset, &devInst, tileLoc);
393  // Set locks to zero
394  for (uint8_t l = 0; l < NUM_LOCKS; l++) {
395  auto locInit = XAie_LockInit(l, 0);
396  TRY_XAIE_API_EMIT_ERROR(tileOp, XAie_LockSetValue, &devInst, tileLoc,
397  locInit);
398  }
399  }
400  }
401 
402  // Set locks with explicit initializers
403  targetOp.walk<WalkOrder::PreOrder>([&](LockOp lockOp) {
404  if (lockOp.getLockID() && lockOp.getInit()) {
405  auto tileLoc = XAie_TileLoc(lockOp.getTileOp().colIndex(),
406  lockOp.getTileOp().rowIndex());
407  auto locInit = XAie_LockInit(*lockOp.getLockID(), *lockOp.getInit());
408  TRY_XAIE_API_FATAL_ERROR(XAie_LockSetValue, &devInst, tileLoc, locInit);
409  } else
410  LLVM_DEBUG(llvm::dbgs()
411  << "lock op missing either id or init" << lockOp << "\n");
412  });
413  return success();
414 }

416 LogicalResult AIERTControl::initBuffers(DeviceOp &targetOp) {
417  // Set buffers with explicit initializers
418  targetOp.walk<WalkOrder::PreOrder>([&](BufferOp bufferOp) {
419  auto initialValue = bufferOp.getInitialValue();
420  if (!initialValue)
421  return;
422  mlir::DenseElementsAttr denseInit =
423  dyn_cast<mlir::DenseElementsAttr>(initialValue.value());
424  if (!denseInit)
425  return;
426  auto tileLoc = XAie_TileLoc(bufferOp.getTileOp().colIndex(),
427  bufferOp.getTileOp().rowIndex());
428  std::vector<char> byteVec;
429  if (denseInit.getElementType().isIntOrIndex()) {
430  for (auto intVal : denseInit.getValues<APInt>()) {
431  // Get the size in bytes
432  size_t byteSize = (intVal.getBitWidth() + 7) / 8;
433  // Create a buffer for the integer bytes and copy
434  std::vector<char> bytes(byteSize);
435  std::copy(
436  static_cast<const char *>(static_cast<const void *>(&intVal)),
437  static_cast<const char *>(static_cast<const void *>(&intVal)) +
438  byteSize,
439  bytes.begin());
440  byteVec.insert(byteVec.end(), bytes.begin(), bytes.end());
441  }
442  } else if (isa<FloatType>(denseInit.getElementType())) {
443  for (auto floatVal : denseInit.getValues<APFloat>()) {
444  APInt floatInt = floatVal.bitcastToAPInt();
445  // Get the size in bytes
446  size_t byteSize = (floatInt.getBitWidth() + 7) / 8;
447  // Create a buffer for the float bytes and copy
448  std::vector<char> bytes(byteSize);
449  std::copy(
450  static_cast<const char *>(static_cast<const void *>(&floatInt)),
451  static_cast<const char *>(static_cast<const void *>(&floatInt)) +
452  byteSize,
453  bytes.begin());
454  byteVec.insert(byteVec.end(), bytes.begin(), bytes.end());
455  }
456  } else {
457  llvm::outs() << "buffer op type not supported for initialization "
458  << bufferOp << "\n";
459  return;
460  }
461  TRY_XAIE_API_FATAL_ERROR(XAie_DataMemBlockWrite, &devInst, tileLoc,
462  bufferOp.getAddress().value(), byteVec.data(),
463  byteVec.size());
464  });
465  return success();
466 }

468 LogicalResult AIERTControl::configureSwitches(DeviceOp &targetOp) {
469 
470  // StreamSwitch (switchbox) configuration
471  for (auto switchboxOp : targetOp.getOps<SwitchboxOp>()) {
472  int32_t col = switchboxOp.colIndex();
473  int32_t row = switchboxOp.rowIndex();
474  XAie_LocType tileLoc = XAie_TileLoc(col, row);
475  assert(targetModel.hasProperty(AIETargetModel::IsNPU) &&
476  "Only NPU currently supported");
477 
478  Block &b = switchboxOp.getConnections().front();
479  for (auto connectOp : b.getOps<ConnectOp>())
480  TRY_XAIE_API_EMIT_ERROR(
481  switchboxOp, XAie_StrmConnCctEnable, &devInst, tileLoc,
482  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getSourceBundle()),
483  connectOp.sourceIndex(),
484  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getDestBundle()),
485  connectOp.destIndex());
486 
487  for (auto masterSetOp : b.getOps<MasterSetOp>()) {
488  int mask = 0;
489  int arbiter = -1;
490 
491  for (auto val : masterSetOp.getAmsels()) {
492  AMSelOp amsel = cast<AMSelOp>(val.getDefiningOp());
493  arbiter = amsel.arbiterIndex();
494  int msel = amsel.getMselValue();
495  mask |= (1 << msel);
496  }
497 
498  // the default is to keep header
499  bool keepHeader = true;
500  // the default for dma destinations is to drop the header
501  if (masterSetOp.getDestBundle() == WireBundle::DMA)
502  keepHeader = false;
503  // assume a connection going south from row zero gets wired to shimdma
504  // by a shimmux.
505  if (switchboxOp.rowIndex() == 0 &&
506  masterSetOp.getDestBundle() == WireBundle::South)
507  keepHeader = false;
508 
509  // "keep_pkt_header" attribute overrides the above defaults, if set
510  if (auto keep = masterSetOp.getKeepPktHeader())
511  keepHeader = *keep;
512 
513  auto dropHeader =
514  keepHeader ? XAIE_SS_PKT_DONOT_DROP_HEADER : XAIE_SS_PKT_DROP_HEADER;
515  TRY_XAIE_API_EMIT_ERROR(
516  masterSetOp, XAie_StrmPktSwMstrPortEnable, &devInst, tileLoc,
517  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(masterSetOp.getDestBundle()),
518  masterSetOp.destIndex(), dropHeader, arbiter, mask);
519  }
520 
521  for (auto packetRulesOp : b.getOps<PacketRulesOp>()) {
522  int slot = 0;
523  Block &block = packetRulesOp.getRules().front();
524  for (auto slotOp : block.getOps<PacketRuleOp>()) {
525  AMSelOp amselOp = cast<AMSelOp>(slotOp.getAmsel().getDefiningOp());
526  int arbiter = amselOp.arbiterIndex();
527  int msel = amselOp.getMselValue();
528  TRY_XAIE_API_EMIT_ERROR(packetRulesOp, XAie_StrmPktSwSlavePortEnable,
529  &devInst, tileLoc,
530  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
531  packetRulesOp.getSourceBundle()),
532  packetRulesOp.sourceIndex());
533  auto packetInit = XAie_PacketInit(slotOp.valueInt(), /*PktType*/ 0);
534  // TODO Need to better define packet id,type used here
535  TRY_XAIE_API_EMIT_ERROR(packetRulesOp, XAie_StrmPktSwSlaveSlotEnable,
536  &devInst, tileLoc,
537  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
538  packetRulesOp.getSourceBundle()),
539  packetRulesOp.sourceIndex(), slot, packetInit,
540  slotOp.maskInt(), msel, arbiter);
541  slot++;
542  }
543  }
544  }
545 
546  for (auto muxOp : targetOp.getOps<ShimMuxOp>()) {
547  // NOTE ShimMux always connects from the south as directions are
548  // defined relative to the tile stream switch.
549  auto tileLoc =
550  XAie_TileLoc(muxOp.getTileOp().getCol(), muxOp.getTileOp().getRow());
551  Block &b = muxOp.getConnections().front();
552  for (auto connectOp : b.getOps<ConnectOp>()) {
553  // demux!
554  if (connectOp.getSourceBundle() == WireBundle::North)
555  TRY_XAIE_API_EMIT_ERROR(muxOp, XAie_EnableAieToShimDmaStrmPort,
556  &devInst, tileLoc, connectOp.sourceIndex());
557  // mux
558  if (connectOp.getDestBundle() == WireBundle::North)
559  TRY_XAIE_API_EMIT_ERROR(muxOp, XAie_EnableShimDmaToAieStrmPort,
560  &devInst, tileLoc, connectOp.destIndex());
561  }
562  }
563 
564  for (auto switchboxOp : targetOp.getOps<ShimSwitchboxOp>()) {
565  Block &b = switchboxOp.getConnections().front();
566  auto tileLoc = XAie_TileLoc(switchboxOp.getCol(), 0);
567  for (auto connectOp : b.getOps<ConnectOp>())
568  TRY_XAIE_API_EMIT_ERROR(
569  switchboxOp, XAie_StrmConnCctEnable, &devInst, tileLoc,
570  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getSourceBundle()),
571  connectOp.sourceIndex(),
572  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(connectOp.getDestBundle()),
573  connectOp.destIndex());
574  }
575 
576  // Cascade configuration
577  if (isa<AIE2TargetModel>(targetModel)) {
578  for (auto configOp : targetOp.getOps<ConfigureCascadeOp>()) {
579  TileOp tile = cast<TileOp>(configOp.getTile().getDefiningOp());
580  auto tileLoc = XAie_TileLoc(tile.getCol(), tile.getRow());
581  TRY_XAIE_API_EMIT_ERROR(
582  targetOp, XAie_CoreConfigAccumulatorControl, &devInst, tileLoc,
583  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
584  static_cast<WireBundle>(configOp.getInputDir())),
585  WIRE_BUNDLE_TO_STRM_SW_PORT_TYPE.at(
586  static_cast<WireBundle>(configOp.getOutputDir())));
587  }
588  }
589 
590  return success();

593 LogicalResult AIERTControl::addInitConfig(DeviceOp &targetOp) {
594 
595  if (failed(initLocks(targetOp))) {
596  return failure();
597  }
598 
599  if (failed(initBuffers(targetOp))) {
600  return failure();
601  }
602 
603  auto memOps = llvm::to_vector_of<TileElement>(targetOp.getOps<MemOp>());
604  llvm::append_range(memOps, targetOp.getOps<MemTileDMAOp>());
605  llvm::append_range(memOps, targetOp.getOps<ShimDMAOp>());
606  for (TileElement memOp : memOps) {
607  int col = memOp.getTileID().col;
608  int row = memOp.getTileID().row;
609  XAie_LocType tileLoc = XAie_TileLoc(col, row);
610 
611  // Get the region's entry block, then start traversing through the chain of
612  // blocks.
613  llvm::SetVector<Block *> blockVector =
614  getOrderedChainOfBlocks(&memOp.getOperation()->getRegion(0));
615 
616  // handle DMA ops separately
617  auto dmaOps = llvm::to_vector_of<DMAOp>(
618  memOp.getOperation()->getRegion(0).getOps<DMAOp>());
619  if (!dmaOps.empty()) {
620  for (auto dmaOp : dmaOps)
621  for (auto &bdRegion : dmaOp.getBds()) {
622  Block &block = bdRegion.getBlocks().front();
623  if (failed(configureLocksAndBd(block, tileLoc)))
624  return failure();
625  }
626  } else {
627  for (Block *block : blockVector) {
628  if (block->getOps<DMABDOp>().empty())
629  continue;
630  if (failed(configureLocksAndBd(*block, tileLoc)))
631  return failure();
632  }
633  }
634 
635  if (!dmaOps.empty())
636  for (auto dmaOp : dmaOps) {
637  auto &block = dmaOp.getBds().front().getBlocks().front();
638  DMABDOp bd = *block.getOps<DMABDOp>().begin();
639  if (failed(pushToBdQueueAndEnable(
640  *dmaOp.getOperation(), tileLoc, dmaOp.getChannelIndex(),
641  dmaOp.getChannelDir(), bd.getBdId().value(),
642  dmaOp.getRepeatCount())))
643  return failure();
644  }
645  else
646  for (Block *block : blockVector) {
647  for (auto op : block->getOps<DMAStartOp>()) {
648  DMABDOp bd = *op.getDest()->getOps<DMABDOp>().begin();
649  int chNum = op.getChannelIndex();
650  auto channelDir = op.getChannelDir();
651  if (failed(pushToBdQueueAndEnable(*bd.getOperation(), tileLoc, chNum,
652  channelDir, bd.getBdId().value(),
653  op.getRepeatCount())))
654  return failure();
655  }
656  }
657  }
658 
659  if (failed(configureSwitches(targetOp))) {
660  return failure();
661  }
662 
663  return success();
664 }
