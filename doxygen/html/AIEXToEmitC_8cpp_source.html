8// Converts the straight-line npu transaction ops in an aie.runtime_sequence
9// into EmitC dialect calls naming the functions in aie/Runtime/TxnEncoding.h.
10// translateToCpp() on the result produces a standalone C++ function that
11// assembles the same TXN words the compile-time binary emitter
12// (AIETargetNPU.cpp) produces -- this is the runtime-parameterizable mirror of
13// that path. This pass handles straight-line sequences only; control flow is
14// rejected with a diagnostic and added by later dynamic-sequences passes.
15//
16//===----------------------------------------------------------------------===//

51emitc::OpaqueType getU32Type(MLIRContext *ctx) {

55// A uint32_t literal operand (e.g. `42u`) for a txn_append_* argument. Used for
56// the compile-time-resolved address fields (buffer/col/row are folded in).
57Value u32Literal(OpBuilder &b, Location loc, uint32_t val) {
58 return emitc::ConstantOp::create(
59 b, loc, getU32Type(b.getContext()),
60 emitc::OpaqueAttr::get(b.getContext(), std::to_string(val) + "u"));
61}

63// Emit a call to one of TxnEncoding.h's free functions on the txn vector.
64// Scalar operands are passed through as-is: an arith.constant or a runtime
65// runtime-sequence value lowers to the matching C++ literal or function
66// parameter by the later convert-arith-to-emitc step. TxnEncoding.h's uint32_t
67// parameters take the implicit conversion, byte-identical to the static path.
68void emitTxnCall(OpBuilder &b, Location loc, StringRef fn, Value txnVec,
69 ValueRange args) {
70 SmallVector<Value> callArgs;
71 callArgs.push_back(txnVec);
72 callArgs.append(args.begin(), args.end());
73 emitc::CallOpaqueOp::create(b, loc, TypeRange{}, ("aie_runtime::" + fn).str(),
74 callArgs);
75}

77// Device-relative values resolved on the original in-device ops, keyed by their
78// clone in the emitc.func. getAbsoluteAddress()/getDataWords() resolve
79// buffer/column/row and memref.global symbols against the parent aie.device, so
80// they must be computed before the ops are cloned out of it.
81struct DeviceResolved {
82 llvm::DenseMap<Operation *, uint32_t> absoluteAddr;
83 llvm::DenseMap<Operation *, DenseIntElementsAttr> blockWriteData;
84};

86class AIEXToEmitCConverter {
87public:
88 AIEXToEmitCConverter(emitc::FuncOp funcOp, Value txnVec,
89 const DeviceResolved &resolved)
90 : funcOp(funcOp), txnVec(txnVec), resolved(resolved) {}
91
92 // Convert every straight-line op cloned into the function body. Returns the
93 // number of txn ops appended (the op_count for the header), or nullopt on a
94 // conversion error already diagnosed.
95 std::optional<uint32_t> run() {
96 Block &body = funcOp.getBlocks().front();
97 SmallVector<Operation *> work;
98 for (Operation &op : body)
99 work.push_back(&op);
100
101 uint32_t count = 0;
102 SmallVector<Operation *> consumed;
103 for (Operation *op : work) {
104 // Structural emitc ops (txn vector decl, txn_init) are already done.
105 // arith ops (constant scalar fields and runtime-value arithmetic) are
106 // left in place for the later convert-arith-to-emitc step -- passing
107 // operands through uniformly is what lets runtime values flow into the
108 // C++.
109 if (isa<emitc::EmitCDialect, arith::ArithDialect>(op->getDialect()))
110 continue;
111 convertOne(op, count);
112 if (!ok)
113 return std::nullopt;
114 // The npu op is replaced by its emitc call; mark it for erase. A
115 // memref.get_global feeding a blockwrite is now dead, but a later
116 // blockwrite may still reference it, so defer its erase to the dead-op
117 // sweep below.
118 if (!isa<memref::GetGlobalOp>(op))
119 consumed.push_back(op);
120 }
121
122 // Erase the converted npu ops, then sweep now-dead clones back to front so
123 // a def outlives its uses (e.g. get_global feeding a blockwrite, or the
124 // arith.constant that defined an address we replaced with a folded
125 // literal). Live arith feeding runtime value/mask/sync fields has uses and
126 // is kept for the arith-to-emitc step.
127 for (Operation *op : llvm::reverse(consumed))
128 op->erase();
129 for (Operation *op : llvm::reverse(work))
130 if (op->getBlock() &&
131 isa<memref::MemRefDialect, arith::ArithDialect>(op->getDialect()) &&
132 op->use_empty())
133 op->erase();
134 return count;
135 }
136
137private:
138 // Convert one op, appending to the txn vector. Increments `count` for each
139 // txn instruction emitted. Scalar operands of the (already-cloned) op are
140 // passed through directly -- a constant or a runtime value both flow as SSA,
141 // lowered later by convert-arith-to-emitc. Only the address field is replaced
142 // by its device-resolved literal (which folds in buffer/col/row). Unsupported
143 // ops are diagnosed (sets `ok` false).
144 void convertOne(Operation *op, uint32_t &count) {
145 OpBuilder b(op);
146 Location loc = op->getLoc();
147
148 llvm::TypeSwitch<Operation *, void>(op)
149 .Case<AIEX::NpuWrite32Op>([&](auto w) {
150 Value addrV = resolvedAddr(b, loc, w);
151 if (!addrV)
152 return fail(w, "cannot convert a symbolic/unresolved write32 "
153 "address to the C++ TXN target");
154 emitTxnCall(b, loc, "txn_append_write32", txnVec,
155 {addrV, w.getValue()});
156 ++count;
157 })
158 .Case<AIEX::NpuMaskWrite32Op>([&](auto mw) {
159 Value addrV = resolvedAddr(b, loc, mw);
160 if (!addrV)
161 return fail(mw, "cannot convert a symbolic/unresolved maskwrite32 "
162 "address to the C++ TXN target");
163 emitTxnCall(b, loc, "txn_append_maskwrite32", txnVec,
164 {addrV, mw.getValue(), mw.getMask()});
165 ++count;
166 })
167 .Case<AIEX::NpuSyncOp>([&](auto s) {
168 emitTxnCall(b, loc, "txn_append_sync", txnVec,
169 {s.getColumn(), s.getRow(), s.getDirection(),
170 s.getChannel(), s.getColumnNum(), s.getRowNum()});
171 ++count;
172 })
173 .Case<AIEX::NpuAddressPatchOp>([&](auto ap) {
174 Value addrV = u32Literal(b, loc, ap.getAddr());
175 Value idxV = emitc::ConstantOp::create(
176 b, loc, emitc::OpaqueType::get(b.getContext(), "int32_t"),
177 emitc::OpaqueAttr::get(b.getContext(),
178 std::to_string(ap.getArgIdx())));
179 emitTxnCall(b, loc, "txn_append_address_patch", txnVec,
180 {addrV, idxV, ap.getArgPlus()});
181 ++count;
182 })
183 .Case<AIEX::NpuBlockWriteOp>([&](auto bw) {
184 convertBlockWrite(b, loc, bw);
185 ++count;
186 })
187 // memref.get_global feeding a blockwrite is consumed by
188 // convertBlockWrite (data inlined); the now-dead op is erased later.
189 .Case<memref::GetGlobalOp>([&](auto) {})
190 .Default([&](Operation *unsupported) {
191 if (isa<scf::SCFDialect>(unsupported->getDialect()))
192 fail(unsupported,
193 "control flow in runtime sequences is not yet supported by "
194 "the C++ TXN target");
195 else
196 fail(unsupported, "is not supported by the C++ TXN target");
197 });
198 }
199
200 // The compile-time-resolved absolute address (folding buffer/col/row) as a
201 // uint32_t literal, or null if it could not be resolved statically. Address
202 // resolution depends on the parent device, so it is precomputed before the op
203 // is cloned out (see recordDeviceResolved).
204 Value resolvedAddr(OpBuilder &b, Location loc, Operation *clone) {
205 auto it = resolved.absoluteAddr.find(clone);
206 if (it == resolved.absoluteAddr.end())
207 return {};
208 return u32Literal(b, loc, it->second);
209 }
210
211 // Emit a diagnostic on `op` and mark the conversion failed.
212 void fail(Operation *op, const llvm::Twine &msg) {
213 op->emitOpError(msg);
214 ok = false;
215 }
216
217 void convertBlockWrite(OpBuilder &b, Location loc, AIEX::NpuBlockWriteOp bw) {
218 auto addrIt = resolved.absoluteAddr.find(bw);
219 auto dataIt = resolved.blockWriteData.find(bw);
220 if (addrIt == resolved.absoluteAddr.end() ||
221 dataIt == resolved.blockWriteData.end() || !dataIt->second) {
222 bw.emitOpError("cannot convert blockwrite without a constant address and "
223 "data to the C++ TXN target");
224 ok = false;
225 return;
226 }
227 uint32_t addr = addrIt->second;
228 DenseIntElementsAttr data = dataIt->second;
229
230 // Emit the payload as a typed EmitC array variable rather than a verbatim
231 // C++ string, so the generated code carries real types.
232 MLIRContext *ctx = b.getContext();
233 int64_t n = data.size();
234 auto arrTy =
235 emitc::ArrayType::get(ctx, SmallVector<int64_t>{n}, getU32Type(ctx));
236 std::string init = "{";
237 llvm::raw_string_ostream ss(init);
238 bool first = true;
239 for (APInt word : data.getValues<APInt>()) {
240 if (!first)
241 ss << ", ";
242 ss << llvm::format("0x%08xu", static_cast<unsigned>(word.getZExtValue()));
243 first = false;
244 }
245 ss << "}";
246 auto arrVar = emitc::VariableOp::create(b, loc, arrTy,
247 emitc::OpaqueAttr::get(ctx, init));
248
249 uint32_t colVal = 0, rowVal = 0;
250 if (bw.getColumn() && bw.getRow()) {
251 colVal = *bw.getColumn();
252 rowVal = *bw.getRow();
253 }
254 emitTxnCall(b, loc, "txn_append_blockwrite", txnVec,
255 {u32Literal(b, loc, addr), arrVar.getResult(),
256 u32Literal(b, loc, static_cast<uint32_t>(n)),
257 u32Literal(b, loc, colVal), u32Literal(b, loc, rowVal)});
258 }
259
260 emitc::FuncOp funcOp;
261 Value txnVec;
262 const DeviceResolved &resolved;
263 bool ok = true;
264};

266struct ConvertAIEXToEmitCPass
267 : xilinx::impl::ConvertAIEXToEmitCBase<ConvertAIEXToEmitCPass> {
268 void runOnOperation() override {
269 ModuleOp moduleOp = getOperation();
270 MLIRContext *ctx = &getContext();
271
272 struct SeqInfo {
273 AIE::RuntimeSequenceOp seq;
274 AIE::DeviceOp device;
275 };
276 SmallVector<SeqInfo> sequences;
277 moduleOp.walk([&](AIE::RuntimeSequenceOp seq) {
278 sequences.push_back({seq, seq->getParentOfType<AIE::DeviceOp>()});
279 });
280 if (sequences.empty())
281 return;
282
283 OpBuilder builder(ctx);
284 for (auto &[seqOp, deviceOp] : sequences) {
285 if (!deviceOp) {
286 seqOp.emitOpError("must be nested inside an aie.device");
287 return signalPassFailure();
288 }
289 if (failed(emitFunction(builder, moduleOp, seqOp, deviceOp)))
290 return signalPassFailure();
291 }
292
293 // Replace the module body with just the generated emitc funcs + includes.
294 SmallVector<Operation *> toErase;
295 for (Operation &op : *moduleOp.getBody())
296 if (!isa<emitc::FuncOp, emitc::IncludeOp>(op))
297 toErase.push_back(&op);
298 for (Operation *op : llvm::reverse(toErase))
299 op->erase();
300
301 // Lower the arith ops left behind in the function bodies (constant scalar
302 // fields and any runtime-value arithmetic feeding npu ops) to emitc. scf
303 // was already rejected, so only arith needs lowering here.
304 if (failed(lowerArithToEmitC(moduleOp)))
305 return signalPassFailure();
306
307 builder.setInsertionPointToStart(moduleOp.getBody());
308 emitc::IncludeOp::create(builder, moduleOp.getLoc(),
309 "aie/Runtime/TxnEncoding.h",
310 /*is_standard=*/false);
311 emitc::IncludeOp::create(builder, moduleOp.getLoc(), "cstdint",
312 /*is_standard=*/true);
313 emitc::IncludeOp::create(builder, moduleOp.getLoc(), "vector",
314 /*is_standard=*/true);
315 }
316
317 // Run the upstream arith-to-emitc conversion over the generated functions.
318 LogicalResult lowerArithToEmitC(ModuleOp moduleOp) {
319 TypeConverter typeConverter;
320 typeConverter.addConversion([](Type t) { return t; });
321 populateEmitCSizeTTypeConversions(typeConverter);
322
323 RewritePatternSet patterns(moduleOp.getContext());
324 populateArithToEmitCPatterns(typeConverter, patterns);
325
326 ConversionTarget target(*moduleOp.getContext());
327 target.addLegalDialect<emitc::EmitCDialect>();
328 target.addIllegalDialect<arith::ArithDialect>();
329 return applyPartialConversion(moduleOp, target, std::move(patterns));
330 }
331
332private:
333 // Resolve `orig`'s device-dependent values (still under the device) keyed by
334 // its detached `clone`.
335 static void recordDeviceResolved(Operation *orig, Operation *clone,
336 DeviceResolved &resolved) {
337 if (auto w = dyn_cast<AIEX::NpuWrite32Op>(orig)) {
338 if (auto a = w.getAbsoluteAddress())
339 resolved.absoluteAddr[clone] = *a;
340 } else if (auto mw = dyn_cast<AIEX::NpuMaskWrite32Op>(orig)) {
341 if (auto a = mw.getAbsoluteAddress())
342 resolved.absoluteAddr[clone] = *a;
343 } else if (auto bw = dyn_cast<AIEX::NpuBlockWriteOp>(orig)) {
344 if (auto a = bw.getAbsoluteAddress())
345 resolved.absoluteAddr[clone] = *a;
346 if (auto d = bw.getDataWords())
347 resolved.blockWriteData[clone] = d;
348 }
349 }
350
351 LogicalResult emitFunction(OpBuilder &builder, ModuleOp moduleOp,
352 AIE::RuntimeSequenceOp seqOp,
353 AIE::DeviceOp deviceOp) {
354 Location loc = seqOp.getLoc();
355 Block &entry = seqOp.getBody().front();
356
357 // The generated function takes the sequence's non-memref args (the runtime
358 // scalars) and returns the assembled txn vector. Memref args address host
359 // buffers resolved by address_patch; they are not function parameters here.
360 SmallVector<Type> paramTypes;
361 auto txnVecType =
362 emitc::OpaqueType::get(moduleOp.getContext(), kTxnVecType);
363 for (BlockArgument arg : entry.getArguments())
364 if (!isa<BaseMemRefType>(arg.getType()))
365 paramTypes.push_back(arg.getType());
366
367 builder.setInsertionPointToEnd(moduleOp.getBody());
368 std::string funcName = "generate_txn_" + deviceOp.getSymName().str() + "_" +
369 seqOp.getSymName().str();
370 auto funcOp = emitc::FuncOp::create(
371 builder, loc, funcName,
372 FunctionType::get(moduleOp.getContext(), paramTypes,
373 TypeRange{txnVecType}));
374 funcOp.setSpecifiersAttr(builder.getStrArrayAttr({"inline"}));
375 Block *funcBlock = funcOp.addEntryBlock();
376 OpBuilder fb = OpBuilder::atBlockBegin(funcBlock);
377
378 // EmitC has no native std::vector construction, so declare the accumulator
379 // with a verbatim statement and reference it by name via an emitc.literal
380 // (which translates inline, no extra copy). All txn_append_* calls take
381 // this value, so the generated C++ operates on the single `txn` vector.
382 emitc::VerbatimOp::create(fb, loc, "std::vector<uint32_t> txn;");
383 Value txnVec = emitc::LiteralOp::create(fb, loc, txnVecType, "txn");
384 emitTxnCall(fb, loc, "txn_init", txnVec, {});
385
386 // Clone the straight-line sequence body into the function, mapping the
387 // non-memref block args to the function parameters. While the originals are
388 // still under the device, resolve their device-relative values (absolute
389 // address, blockwrite data) keyed by the clone -- the converter reads these
390 // instead of calling device-dependent accessors on the detached clones.
391 IRMapping mapping;
392 unsigned p = 0;
393 for (BlockArgument arg : entry.getArguments())
394 if (!isa<BaseMemRefType>(arg.getType()))
395 mapping.map(arg, funcBlock->getArgument(p++));
396 DeviceResolved resolved;
397 for (Operation &op : entry.without_terminator()) {
398 Operation *clone = fb.clone(op, mapping);
399 recordDeviceResolved(&op, clone, resolved);
400 }
401
402 AIEXToEmitCConverter conv(funcOp, txnVec, resolved);
403 std::optional<uint32_t> count = conv.run();
404 if (!count)
405 return failure();
406
407 // Finalize the header with the statically-known op count + device info.
408 const AIE::AIETargetModel &tm = deviceOp.getTargetModel();
409 uint8_t devGen = isa<AIE::BaseNPU2TargetModel>(tm) ? 4 : 3;
410 OpBuilder eb(funcBlock, funcBlock->end());
411 std::string header =
412 "aie_runtime::txn_prepend_header(txn, " + std::to_string(*count) +
413 "u, {0, 1, " + std::to_string(devGen) + ", " +
414 std::to_string(tm.rows()) + ", " + std::to_string(tm.columns()) + ", " +
415 std::to_string(tm.getNumMemTileRows()) + "});";
416 emitc::VerbatimOp::create(eb, loc, header);
417 emitc::ReturnOp::create(eb, loc, txnVec);
418 return success();
419 }
420};
