47class ParameterScratchpad {
48public:
49#ifdef TEST_UTILS_USE_XRT
50 /// Construct from an XRT run handle (C++ usage).
51 ParameterScratchpad(xrt::run &run, const std::string &paramsPath) {
52 parseParams(paramsPath);
53 scratchpadBo = run.get_ctrl_scratchpad_bo();
54 boMap = scratchpadBo.map<uint32_t *>();
55 if (scratchpadBo.size() < scratchpadSizeBytes)
56 throw std::runtime_error("ParameterScratchpad: BO size (" +
57 std::to_string(scratchpadBo.size()) +
58 ") < required scratchpad size (" +
59 std::to_string(scratchpadSizeBytes) + ")");
60 clear();
61 }
62#endif
63
64 /// Construct from a raw buffer pointer (Python bindings / testing).
65 ParameterScratchpad(uint32_t *buffer, const std::string &paramsPath)
66 : boMap(buffer) {
67 parseParams(paramsPath);
68 clear();
69 }
70
71 /// Write raw bytes (up to 4) by name, interpreted as a little-endian
72 /// uint32. For core-kind parameters, the bits are left-shifted by 2
73 /// (firmware requirement).
74 /// For addr-kind parameters, the value is written raw (no shift).
75 void writeBytes(const std::string &name, const void *data, size_t len) {
76 uint32_t bits = 0;
77 std::memcpy(&bits, data, std::min(len, sizeof(bits)));
78 writeBits(name, bits);
79 }
80
81 /// Write a raw 32-bit value by name. For core-kind parameters, the bits
82 /// are left-shifted by 2 (firmware requirement). For addr-kind parameters,
83 /// the value is written directly (no shift).
84 void writeBits(const std::string &name, uint32_t bits) {
85 auto it = paramMap.find(name);
86 if (it == paramMap.end()) {
87 throw std::runtime_error("ParameterScratchpad: unknown parameter '" +
88 name + "'");
89 }
90 uint8_t idx = it->second;
91 uint32_t encoded = bits;
92 if (coreParams.count(name)) {
93 // core parameters require shift-2 to survive masking of lowest bits by
94 // firmware op
95 encoded = bits << 2;
96 }
97 boMap[idx] = encoded;
98 }
99
100 /// Write a typed parameter value. For core-kind parameters, the raw bits
101 /// are left-shifted by 2 as required by the firmware's UPDATE_REG Incr mode,
102 /// and the core right-shifts (unsigned) by 2 after reading. The round trip
103 /// therefore zeroes the top 2 bits of the value (the bottom 2 bits are
104 /// preserved). Types that fit in 30 bits (e.g. uint16_t, int16_t,
105 /// std::bfloat16_t, and uint32_t values < 2^30) round-trip losslessly.
106 /// `float` is not supported as a core-kind parameter (the verifier in
107 /// `--aie-lower-scratchpad-parameters` rejects it).
108 template <typename T>
109 void write(const std::string &name, T value) {
110 static_assert(sizeof(T) <= 4, "Parameter values must be at most 32 bits");
111 uint32_t bits = 0;
112 std::memcpy(&bits, &value, sizeof(T));
113 writeBits(name, bits);
114 }
115
116#ifdef TEST_UTILS_USE_XRT
117 /// Sync the scratchpad buffer to device. Call after all writes for this run.
118 void sync() { scratchpadBo.sync(XCL_BO_SYNC_BO_TO_DEVICE); }
119#endif
120
121 /// Read back a parameter's current encoded value (for debugging).
122 uint32_t read(const std::string &name) const {
123 auto it = paramMap.find(name);
124 if (it == paramMap.end()) {
125 throw std::runtime_error("ParameterScratchpad: unknown parameter '" +
126 name + "'");
127 }
128 return boMap[it->second];
129 }
130
131private:
132#ifdef TEST_UTILS_USE_XRT
133 xrt::bo scratchpadBo;
134#endif
135 uint32_t *boMap = nullptr;
136 size_t scratchpadSizeBytes = 0;
137 std::unordered_map<std::string, uint8_t> paramMap;
138 std::unordered_set<std::string> coreParams; // params with kind="core"
139
140 void clear() {
141 for (size_t i = 0; i < scratchpadSizeBytes / 4; i++) {
142 boMap[i] = 0;
143 }
144 }
145
146 void parseParams(const std::string &path) {
147 std::ifstream file(path);
148 if (!file.is_open()) {
149 throw std::runtime_error("ParameterScratchpad: cannot open '" + path +
150 "'");
151 }
152
153 // Format:
154 // <num_parameters>
155 // <name> <state_table_idx> <type> <kind>
156 // ...
157 // where kind is "core" or "addr".
158 unsigned numParams = 0;
159 file >> numParams;
160 scratchpadSizeBytes = numParams * 4;
161
162 for (unsigned i = 0; i < numParams; i++) {
163 std::string name, type, kind;
164 unsigned idx;
165 file >> name >> idx >> type >> kind;
166 if (idx > 255)
167 throw std::runtime_error("ParameterScratchpad: state_table_idx " +
168 std::to_string(idx) + " for '" + name +
169 "' exceeds uint8_t range");
170 if (paramMap.count(name))
171 throw std::runtime_error(
172 "ParameterScratchpad: duplicate parameter name '" + name + "'");
173 paramMap[name] = static_cast<uint8_t>(idx);
174 if (kind != "core" && kind != "addr") {
175 throw std::runtime_error("ParameterScratchpad: invalid kind '" + kind +
176 "' for parameter '" + name + "'");
177 } else if (kind == "core") {
178 coreParams.insert(name);
179 }
180 }
181 }
182};
