39#else

45mlir::LogicalResult AIETranslateToHSA(ModuleOp module, raw_ostream &output,
46 llvm::StringRef deviceName) {
47
48 DenseMap<TileID, Operation *> tiles;
49 DenseMap<Operation *, SmallVector<BufferOp, 4>> buffers;
50
51 DeviceOp targetOp = AIE::DeviceOp::getForSymbolInModule(module, deviceName);
52 if (!targetOp)
53 return module.emitOpError("expected AIE.device operation at toplevel");
54
55 // Putting the standard header
56 output << hsa_cpp_file_header;
57
58 // Getting the sequence function op which contains the instructions
59 auto sequenceOps = targetOp.getOps<AIEX::RuntimeSequenceOp>();
60 if (sequenceOps.empty()) {
61 // If no sequenceOp then just return
62 return success();
63 } else if (std::distance(sequenceOps.begin(), sequenceOps.end()) > 1) {
64 return module.emitOpError("expected at most one sequence operation");
65 }
66 AIEX::RuntimeSequenceOp sequenceOp = *sequenceOps.begin();
67
68 collectTiles(targetOp, tiles);
69 collectBuffers(targetOp, buffers);
70
71 // Generate dynamic data movement
72 output << "void invoke_data_movement(hsa_queue_t *q, hsa_agent_t *a";
73
74 // Looping over every Memcpy operation so we take the correct number of
75 // buffers
76 int num_ops = 0;
77 for (auto op : sequenceOp.getOps<NpuDmaMemcpyNdOp>()) {
78 // Getting the IDs of the buffers
79 auto memref = op.getMemref();
80 Block &entryBB =
81 op->getParentOfType<AIEX::RuntimeSequenceOp>().getBody().front();
82 int arg_idx = -1;
83 for (int i = 0, e = entryBB.getNumArguments(); i < e; i++) {
84 if (entryBB.getArgument(i) == memref) {
85 arg_idx = i;
86 break;
87 }
88 }
89 num_ops++;
90
91 output << ", void *buf" << arg_idx;
92 }
93
94 output << ") {\n";
95
96 output << "\tuint64_t wr_idx = 0;\n";
97 output << "\tuint64_t packet_id = 0;\n";
98
99 int op_count = 0;
100 for (auto op : sequenceOp.getOps<NpuDmaMemcpyNdOp>()) {
101 auto dev = sequenceOp->getParentOfType<AIE::DeviceOp>();
102 if (!dev) {
103 op.emitOpError("couldn't get DeviceOp");
104 return failure();
105 }
106
107 AIE::ShimDMAAllocationOp infoOp = AIE::ShimDMAAllocationOp::getForSymbol(
108 dev, op.getMetadata().getRootReference());
109 if (!infoOp) {
110 op.emitOpError("couldn't find shim_dma_allocation op");
111 return failure();
112 }
113
114 auto channelDir = infoOp.getChannelDir();
115 uint32_t ChannelId = infoOp.getChannelIndex();
116 bool isMM2S = channelDir == AIE::DMAChannelDir::MM2S;
117 int col = infoOp.getCol();
118 bool isPlio = infoOp.getPlio();
119
120 llvm::SmallVector<int64_t, 4> strides = llvm::map_to_vector(
121 llvm::reverse(op.getMixedStrides()),
122 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
123 ::SmallVector<int64_t, 4> sizes = llvm::map_to_vector(
124 llvm::reverse(op.getMixedSizes()),
125 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
126 ::SmallVector<int64_t, 4> offsets = llvm::map_to_vector(
127 llvm::reverse(op.getMixedOffsets()),
128 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
129
130 // buffer_offset
131 size_t stride = 1;
132 size_t offset = 0;
133 BaseMemRefType my_memref = op.getMemref().getType();
134 auto shape = my_memref.getShape();
135 size_t R = shape.size();
136 size_t el_bit_width = op.getElementTypeBitwidth();
137 assert(el_bit_width % 8 == 0 &&
138 "Expected Memref element bitwidth to be multiple of 8.");
139 size_t S = el_bit_width / 8;
140 for (size_t i = 0; i < R; i++) {
141 offset += offsets[i] * stride * S;
142 stride *= shape[R - i - 1];
143 }
144
145 // Getting the ID of the buffer that we are using
146 auto memref = op.getMemref();
147 Block &entryBB =
148 op->getParentOfType<AIEX::RuntimeSequenceOp>().getBody().front();
149 int arg_idx = -1;
150 for (int i = 0, e = entryBB.getNumArguments(); i < e; i++) {
151 if (entryBB.getArgument(i) == memref) {
152 arg_idx = i;
153 break;
154 }
155 }
156
157 if (strides[0] != 1)
158 return module.emitOpError("nd_memcpy inner-dimension stride != 1 is "
159 "unsupported by HSA target");
160
161 // Writing the packet information to perform the DMA
162 output << "\thsa_agent_dispatch_packet_t pkt" << op_count << " ;\n";
163 output << "\twr_idx = hsa_queue_add_write_index_relaxed(q, 1);\n";
164 output << "\tpacket_id = wr_idx % q->size;\n";
165 output << "\tmlir_aie_packet_nd_memcpy(&pkt" << op_count
166 << ", 0 /* herd_id */, " << col << " /* col */, " << isMM2S
167 << " /* dir */, " << ChannelId
168 << "/* channel */, 4 /* Burst length */, " << (isPlio ? 1 : 2)
169 << " /* Memory space */, "
170 "(uint64_t)buf"
171 << arg_idx << " + " << offset << " /* Address */, " << sizes[0] * 4
172 << " /* 1d_length */, " << (strides[1] ? sizes[1] : 1)
173 << " /* 2d_length */, " << (strides[1] ? strides[1] * 4 : 0)
174 << " /* 2d_stride */, " << (strides[2] ? sizes[2] : 1)
175 << " /* 3d_length */, " << (strides[2] ? strides[2] * 4 : 0)
176 << " /* 3d_stride */ , 1 /* 4d_length */, 0 /* 4d_stride */);\n";
177
178 bool last_op = op_count == (num_ops - 1);
179 // Only ring the doorbell on the last packet
180 if (last_op) {
181 output
182 << "\tmlir_aie_queue_dispatch_and_wait(a, q, packet_id, wr_idx, &pkt"
183 << op_count << ", false);\n\n";
184 } else {
185 output << "\thsa_amd_signal_create_on_agent(1, 0, nullptr, a, 0, &pkt"
186 << op_count << ".completion_signal);\n";
187 output << "\tmlir_aie_write_pkt<hsa_agent_dispatch_packet_t>(q, "
188 "packet_id, &pkt"
189 << op_count << ");\n\n";
190 }
191
192 op_count++;
193 }
194
195 // Waiting to make sure each DMA is complete
196 for (int i = 0; i < op_count; i++) {
197 output << "\twhile (hsa_signal_wait_scacquire(pkt" << i
198 << ".completion_signal,\n";
199 output << "\tHSA_SIGNAL_CONDITION_EQ, 0, 0x80000,\n";
200 output << "\tHSA_WAIT_STATE_ACTIVE) != 0);\n";
201 }
202
203 // Destroying every signal that we created
204 for (int i = 0; i < op_count; i++) {
205 output << "\thsa_signal_destroy(pkt" << i << ".completion_signal);\n";
206 }
207
208 output << "}\n";
209
210 return success();
211}
