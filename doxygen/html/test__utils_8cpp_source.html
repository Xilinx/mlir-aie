81 std::vector<uint32_t> test_utils::load_instr_sequence(std::string instr_path) {
82  std::ifstream instr_file(instr_path);
83  std::string line;
84  std::vector<uint32_t> instr_v;
85  while (std::getline(instr_file, line)) {
86  std::istringstream iss(line);
87  uint32_t a;
88  if (!(iss >> std::hex >> a)) {
89  throw std::runtime_error("Unable to parse instruction file\n");
90  }
91  instr_v.push_back(a);
92  }
93  return instr_v;
94 }

96 std::vector<uint32_t> test_utils::load_instr_binary(std::string instr_path) {
97  std::ifstream instr_file(instr_path);
98  if (!instr_file.is_open()) {
99  throw std::runtime_error("Unable to open instruction file\n");
100  }
101  // read size of file, reserve space in instr_v, then read the file into
102  // instr_v
103  instr_file.seekg(0, instr_file.end);
104  int size = instr_file.tellg();
105  instr_file.seekg(0, instr_file.beg);
106  std::vector<uint32_t> instr_v(size / 4);
107  instr_file.read(reinterpret_cast<char *>(instr_v.data()), size);
108  return instr_v;
109 }

114 void test_utils::init_xrt_load_kernel(xrt::device &device, xrt::kernel &kernel,
115  int verbosity, std::string xclbinFileName,
116  std::string kernelNameInXclbin) {
117  // Get a device handle
118  unsigned int device_index = 0;
119  device = xrt::device(device_index);
120 
121  // Load the xclbin
122  if (verbosity >= 1)
123  std::cout << "Loading xclbin: " << xclbinFileName << "\n";
124  auto xclbin = xrt::xclbin(xclbinFileName);
125 
126  if (verbosity >= 1)
127  std::cout << "Kernel opcode: " << kernelNameInXclbin << "\n";
128 
129  // Get the kernel from the xclbin
130  auto xkernels = xclbin.get_kernels();
131  auto xkernel =
132  *std::find_if(xkernels.begin(), xkernels.end(),
133  [kernelNameInXclbin, verbosity](xrt::xclbin::kernel &k) {
134  auto name = k.get_name();
135  if (verbosity >= 1) {
136  std::cout << "Name: " << name << std::endl;
137  }
138  return name.rfind(kernelNameInXclbin, 0) == 0;
139  });
140  auto kernelName = xkernel.get_name();
141 
142  // Register xclbin
143  if (verbosity >= 1)
144  std::cout << "Registering xclbin: " << xclbinFileName << "\n";
145 
146  device.register_xclbin(xclbin);
147 
148  // Get a hardware context
149  if (verbosity >= 1)
150  std::cout << "Getting hardware context.\n";
151  xrt::hw_context context(device, xclbin.get_uuid());
152 
153  // Get a kernel handle
154  if (verbosity >= 1)
155  std::cout << "Getting handle to kernel:" << kernelName << "\n";
156  kernel = xrt::kernel(context, kernelName);
157 
158  return;
159 }

165 // nearly_equal function adapted from Stack Overflow, License CC BY-SA 4.0
166 // Original author: P-Gn
167 // Source: https://stackoverflow.com/a/32334103
168 bool test_utils::nearly_equal(float a, float b, float epsilon, float abs_th)
169 // those defaults are arbitrary and could be removed
170 {
171  assert(std::numeric_limits<float>::epsilon() <= epsilon);
172  assert(epsilon < 1.f);
173 
174  if (a == b)
175  return true;
176 
177  auto diff = std::abs(a - b);
178  auto norm =
179  std::min((std::abs(a) + std::abs(b)), std::numeric_limits<float>::max());
180  // or even faster: std::min(std::abs(a + b),
181  // std::numeric_limits<float>::max()); keeping this commented out until I
182  // update figures below
183  return diff < std::max(abs_th, epsilon * norm);
184 }
