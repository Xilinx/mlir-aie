68static bool isGemmBTransposedContractionOp(vector::ContractionOp op) {
69 if (op.getKind() != vector::CombiningKind::ADD)
70 return false;
71
72 // Get and check shape of operands
73 auto lhsShape = op.getLhsType().getShape();
74 auto rhsShape = op.getRhsType().getShape();
75 auto accShape = cast<ShapedType>(op.getAccType()).getShape();
76 if (lhsShape.size() < 2 || rhsShape.size() < 2 || accShape.size() < 2)
77 return false;
78
79 // Check that the innermost iterators match gemm-like iterators
80 SmallVector<vector::IteratorType> iterators = op.getIteratorTypesArray();
81 if (iterators.size() < 3)
82 return false;
83 auto innerMostIterators =
84 SmallVector<vector::IteratorType>(iterators.end() - 3, iterators.end());
85 if (vector::IteratorType::parallel != innerMostIterators[0] ||
86 vector::IteratorType::parallel != innerMostIterators[1] ||
87 vector::IteratorType::reduction != innerMostIterators[2])
88 return false;
89
90 // Get indexing maps of iterators for operands
91 SmallVector<AffineMap, 4> indexingMaps(op.getIndexingMapsArray());
92 SmallVector<int64_t> outerMostResults;
93 for (int64_t i = 0; i < indexingMaps[0].getNumResults() - 2; i++)
94 outerMostResults.push_back(i);
95
96 auto innerLhsMap = indexingMaps[0].dropResults(outerMostResults);
97 auto innerRhsMap = indexingMaps[1].dropResults(outerMostResults);
98 auto innerAccMap = indexingMaps[2].dropResults(outerMostResults);
99
100 // Check whether they conform to a "transposed B" gemm
101 auto *ctx = op.getContext();
102 auto mmAidxMap =
103 AffineMap::getPermutationMap(ArrayRef<unsigned>{1, 0, 2}, ctx)
104 .dropResults(0);
105 auto mmBidxMap =
106 AffineMap::getPermutationMap(ArrayRef<unsigned>{0, 1, 2}, ctx)
107 .dropResults(0);
108 auto mmCidxMap =
109 AffineMap::getPermutationMap(ArrayRef<unsigned>{2, 0, 1}, ctx)
110 .dropResults(0);
111 int64_t numOuterMostDims = indexingMaps[0].getNumDims() - 3;
112 return innerLhsMap == mmAidxMap.shiftDims(numOuterMostDims) &&
113 innerRhsMap == mmBidxMap.shiftDims(numOuterMostDims) &&
114 innerAccMap == mmCidxMap.shiftDims(numOuterMostDims);
115}

121// This pattern converts a `vector.transfer_read` with an unaligned access
122// into an aligned `vector.transfer_read` twice as long, followed by a
123// `vector.extract_strided_slice` selecting the subvector matching the
124// original `vector.transfer_read`.
125struct SplitUnalignedTransferReadPattern
126 : public OpConversionPattern<vector::TransferReadOp> {
127 using OpConversionPattern<vector::TransferReadOp>::OpConversionPattern;
128
129 SplitUnalignedTransferReadPattern(MLIRContext *context, int64_t maxVectorSize,
130 int64_t alignment)
131 : OpConversionPattern<vector::TransferReadOp>(context),
132 maxVectorSize(maxVectorSize), vectorAlignment(alignment) {}
133
134 LogicalResult
135 matchAndRewrite(vector::TransferReadOp readOp, OpAdaptor adaptor,
136 ConversionPatternRewriter &rewriter) const override {
137 // Check that it's not a splat transfer read.
138 if (adaptor.getPermutationMap().isConstant())
139 return failure();
140
141 // Check if the transfer is unaligned.
142 auto vType = readOp.getVectorType();
143 int64_t offset =
144 getTransferReadAlignmentOffset(adaptor, vType, vectorAlignment)
145 .value_or(0);
146 if (offset == 0)
147 return failure();
148
149 // Verify that we can load a vector 2x as long as the original
150 auto vLen = vType.getShape().back();
151 auto longVecTy = VectorType::get(2 * vLen, vType.getElementType());
152 auto longVecSize = getElementSizeInBits(vType) * 2 * vLen;
153 if (longVecSize > maxVectorSize)
154 return failure();
155
156 // Calculate the aligned indices for the lower and higher parts.
157 // TODO: Add support for cases where the offset is greater than the
158 // TODO: vector length.
159 auto loc = readOp.getLoc();
160 Value oldInnerMostIdx = adaptor.getIndices().back();
161 auto offsetCorrectionMap =
162 AffineMap::get(1, 0, getAffineDimExpr(0, readOp.getContext()) - offset);
163 Value newInnerMostIdx = affine::AffineApplyOp::create(
164 rewriter, readOp.getLoc(), offsetCorrectionMap,
165 SmallVector<Value, 1>({oldInnerMostIdx}))
166 .getResult();
167 SmallVector<Value, 8> alignedIdx;
168 alignedIdx.append(adaptor.getIndices().begin(), adaptor.getIndices().end());
169 alignedIdx[alignedIdx.size() - 1] = newInnerMostIdx;
170
171 // Create the aligned transfer read for a vector 2x as long that covers the
172 // elements of the unaligned vector.
173 auto newReadOp = vector::TransferReadOp::create(
174 rewriter, loc, longVecTy, adaptor.getBase(), alignedIdx,
175 adaptor.getPadding());
176
177 // Create a `vector.extract_strided_slice` to extract the unaligned vector.
178 rewriter.replaceOpWithNewOp<vector::ExtractStridedSliceOp>(
179 readOp, newReadOp.getResult(), offset, vLen, 1);
180
181 return success();
182 }
183
184 int64_t maxVectorSize;
185 int64_t vectorAlignment;
186};

188// This pattern converts a `vector.transfer_read` with a splat permutation map
189// into a contiguous `vector.transfer_read` followed by a `vector.extract` to
190// obtain the splat value and a `vector.broadcast` to broadcast it into a
191// vector of the right size.
192struct ConvertSplatTransferReadToBroadcastPattern
193 : public OpConversionPattern<vector::TransferReadOp> {
194 using OpConversionPattern<vector::TransferReadOp>::OpConversionPattern;
195
196 ConvertSplatTransferReadToBroadcastPattern(MLIRContext *context)
197 : OpConversionPattern<vector::TransferReadOp>(context) {}
198
199 LogicalResult
200 matchAndRewrite(vector::TransferReadOp readOp, OpAdaptor adaptor,
201 ConversionPatternRewriter &rewriter) const override {
202 AffineMap map = readOp.getPermutationMap();
203 if (!map.isConstant())
204 return failure();
205
206 Value srcMemRef = adaptor.getBase();
207 SmallVector<Value, 8> indices;
208 Value newIdx;
209 int64_t offset = 0;
210 // If it's a zero-rank memory access
211 if (cast<MemRefType>(srcMemRef.getType()).getRank() == 0) {
212 srcMemRef = memref::ExpandShapeOp::create(
213 rewriter, readOp.getLoc(), SmallVector<int64_t, 1>({1}),
214 srcMemRef, SmallVector<ReassociationIndices, 1>({}))
215 .getResult();
216 newIdx = arith::ConstantOp::create(rewriter, readOp.getLoc(),
217 rewriter.getIndexAttr(0L));
218 indices.push_back(newIdx);
219 } else {
220 indices.append(adaptor.getIndices().begin(), adaptor.getIndices().end());
221 newIdx = indices[indices.size() - 1];
222 // If the innermost index comes from an `affine.apply` op, take the base
223 // as the new innermost index for the new `vector.transfer_read`, and the
224 // offset as the index for the `aievec.broadcast` op.
225 if (auto applyOp = newIdx.getDefiningOp<affine::AffineApplyOp>())
226 if (applyOp.getAffineMap().getNumDims() == 1) {
227 newIdx = applyOp.getMapOperands()[0];
228 offset = applyOp.getAffineMap().compose(ArrayRef<int64_t>{0})[0];
229 }
230 }
231 // XXX: We assume we are reading 1D vectors
232 int64_t vlen = readOp.getVector().getType().getShape()[0];
233 if (offset >= vlen) {
234 // If the splat element is beyond the first vector, we calculate the
235 // address of the vector containing the element.
236 int64_t numElemsToSkip = vlen * (offset / vlen);
237 offset = offset % vlen;
238 auto newAddrMap = AffineMap::get(
239 1, 0, getAffineDimExpr(0, readOp.getContext()) + numElemsToSkip);
240 newIdx =
241 affine::AffineApplyOp::create(rewriter, readOp.getLoc(), newAddrMap,
242 SmallVector<Value, 1>({newIdx}))
243 .getResult();
244 }
245 indices[indices.size() - 1] = newIdx;
246 auto newReadOp = vector::TransferReadOp::create(
247 rewriter, readOp.getLoc(), readOp.getVector().getType(), srcMemRef,
248 indices, adaptor.getPadding());
249 auto extractOp = vector::ExtractOp::create(rewriter, readOp.getLoc(),
250 newReadOp.getResult(),
251 ArrayRef<int64_t>{offset});
252 rewriter.replaceOpWithNewOp<vector::BroadcastOp>(
253 readOp, newReadOp.getVector().getType(), extractOp.getResult());
254 return success();
255 }
256};

258// This pattern moves cast operations as close as possible to the source of
259// the data. This helps to simplify dealing with patterns that may vary only
260// by these sorts of casts between data manipulation operations and arithmetic
261// ops.
262// TODO: Generalize this op and instantiate for different types of cast ops.
263struct HoistCastOpToDataSourcePattern : public RewritePattern {
264 HoistCastOpToDataSourcePattern(MLIRContext *context)
265 : RewritePattern(arith::ExtSIOp::getOperationName(), /*benefit=*/1,
266 context) {}
267
268 LogicalResult matchAndRewrite(Operation *op,
269 PatternRewriter &rewriter) const override {
270 arith::ExtSIOp extOp = cast<arith::ExtSIOp>(op);
271 Operation *defOp = extOp.getIn().getDefiningOp();
272 // If it's a data source op, we're done.
273 if (!defOp || isa<vector::TransferReadOp, memref::LoadOp,
274 affine::AffineLoadOp, func::CallOp>(defOp))
275 return failure();
276
277 // At the moment, we only accept ops we know we can swap with cast.
278 if (!isa<vector::BroadcastOp, vector::ExtractOp,
279 vector::ExtractStridedSliceOp>(defOp))
280 return failure();
281
282 Type extOpInTy = extOp.getIn().getType();
283 SmallVector<Value, 4> inputs;
284 for (Value operand : defOp->getOperands()) {
285 Type operandTy = operand.getType();
286 VectorType extOpInVecTy = dyn_cast<VectorType>(extOpInTy);
287 VectorType operandVecTy = dyn_cast<VectorType>(operandTy);
288 if (operandTy == extOpInTy) {
289 Type outTy = extOp.getOut().getType();
290 inputs.push_back(
291 arith::ExtSIOp::create(rewriter, extOp.getLoc(), outTy, operand)
292 .getOut());
293 } else if (extOpInVecTy && extOpInVecTy.getElementType() == operandTy) {
294 // Promote from vector to scalar -> scalar conversion for this operand
295 Type outTy =
296 cast<VectorType>(extOp.getOut().getType()).getElementType();
297 inputs.push_back(
298 arith::ExtSIOp::create(rewriter, extOp.getLoc(), outTy, operand)
299 .getOut());
300 } else if (operandVecTy && operandVecTy.getElementType() == extOpInTy) {
301 // Promote from scalar to vector -> vector conversion for this operand
302 Type outTy =
303 VectorType::get(operandVecTy.getShape(), extOp.getOut().getType());
304 inputs.push_back(
305 arith::ExtSIOp::create(rewriter, extOp.getLoc(), outTy, operand)
306 .getOut());
307 } else if (extOpInVecTy && operandVecTy &&
308 (extOpInVecTy.getElementType() ==
309 operandVecTy.getElementType())) {
310 // Hoist through a vector shape change
311 Type outTy = VectorType::get(
312 operandVecTy.getShape(),
313 cast<VectorType>(extOp.getOut().getType()).getElementType());
314 inputs.push_back(
315 arith::ExtSIOp::create(rewriter, extOp.getLoc(), outTy, operand)
316 .getOut());
317 } else {
318 inputs.push_back(operand);
319 }
320 }
321
322 auto *newOp =
323 rewriter.create(extOp->getLoc(), defOp->getName().getIdentifier(),
324 inputs, {extOp.getOut().getType()}, defOp->getAttrs());
325 rewriter.replaceOp(extOp, newOp->getResult(0));
326 return success();
327 }
328};

330// This pattern swaps a UnaryOpA followed by UnaryOpB. This pattern can be used
331// to improve pattern matching for mixed-type arithmetic ops, by getting sign
332// extension ops closer to the single-type arithmetic operations.
333template <class UnaryOpA, class UnaryOpB>
334struct SwapUnaryOpsPattern : public OpRewritePattern<UnaryOpB> {
335 using OpRewritePattern<UnaryOpB>::OpRewritePattern;
336 // This function takes the chain of operations A->B, and returns the new type
337 // between B and A after the swap.
338 using InferTypeB2AFnTy = std::function<Type(UnaryOpA aOp, UnaryOpB bOp)>;
339 InferTypeB2AFnTy inferTypeB2A = nullptr;
340
341 SwapUnaryOpsPattern(MLIRContext *context, InferTypeB2AFnTy inferType)
342 : OpRewritePattern<UnaryOpB>(context), inferTypeB2A(inferType) {}
343
344 LogicalResult matchAndRewrite(UnaryOpB bOp,
345 PatternRewriter &rewriter) const override {
346 static_assert(
347 UnaryOpA::template hasTrait<OpTrait::OneOperand>(),
348 "SwapUnaryOps can only be instantiated for single-operand ops");
349 static_assert(
350 UnaryOpB::template hasTrait<OpTrait::OneOperand>(),
351 "SwapUnaryOps can only be instantiated for single-operand ops");
352 UnaryOpA aOp = bOp.getOperand().template getDefiningOp<UnaryOpA>();
353 if (!aOp)
354 return rewriter.notifyMatchFailure(bOp, UnaryOpB::getOperationName() +
355 " not preceeded by " +
356 UnaryOpA::getOperationName());
357
358 Type newA2BTy = inferTypeB2A(aOp, bOp);
359
360 auto newA =
361 UnaryOpB::create(rewriter, bOp->getLoc(), SmallVector<Type>({newA2BTy}),
362 aOp->getOperands(), bOp->getAttrs());
363 auto newB = UnaryOpA::create(rewriter, bOp->getLoc(),
364 SmallVector<Type>({bOp.getResult().getType()}),
365 newA->getResults(), aOp->getAttrs());
366 rewriter.replaceOp(bOp, newB.getResult());
367 return success();
368 }
369};

371static SmallVector<Value> collapseInnerMostDimIndices(PatternRewriter &b,
372 Location loc, int numDims,
373 ValueRange indices,
374 ArrayRef<int64_t> shape,
375 AffineMap layout) {
376 // TODO: Don't assume trivial layout
377 assert(layout.isMinorIdentity() &&
378 "dimension collapse in non-identity layout is not implemented");
379 auto newIdxExpr = b.getAffineDimExpr(numDims - 1);
380 int64_t stride = 1;
381 for (int64_t dim = numDims - 2; dim >= 0; dim--) {
382 stride *= shape[shape.size() - (numDims - dim - 1)];
383 newIdxExpr = newIdxExpr + b.getAffineDimExpr(dim) * stride;
384 }
385 auto newIndexMap = AffineMap::get(numDims, 0, newIdxExpr);
386 Value newInnerMostIdxValue =
387 affine::AffineApplyOp::create(b, loc, newIndexMap,
388 indices.take_back(numDims))
389 .getResult();
390 SmallVector<Value> newIdxRange;
391 for (auto idx : indices.drop_back(numDims))
392 newIdxRange.push_back(idx);
393 newIdxRange.push_back(newInnerMostIdxValue);
394 return newIdxRange;
395}

397static Value collapseInnerMostShapeDims(PatternRewriter &b, Location loc,
398 int numDims, Value val) {
399 auto memRefTy = cast<MemRefType>(val.getType());
400 auto shape = memRefTy.getShape();
401 int64_t newInnerMostDim = std::accumulate(shape.end() - numDims, shape.end(),
402 1, std::multiplies<>());
403 SmallVector<int64_t, 4> newShape{shape.begin(), shape.end() - numDims + 1};
404 newShape[shape.size() - numDims] = newInnerMostDim;
405 auto newNumDims = newShape.size();
406 auto *ctx = b.getContext();
407 auto newMemRefTy = MemRefType::get(
408 newShape, memRefTy.getElementType(),
409 AffineMap::getMinorIdentityMap(newNumDims, newNumDims, ctx),
410 memRefTy.getMemorySpace());
411 auto reassocIndices =
412 getReassociationIndicesForCollapse(shape, newShape).value();
413 return memref::CollapseShapeOp::create(b, loc, newMemRefTy, val,
414 reassocIndices)
415 .getResult();
416}

418// This pattern flatten multidimensional `vector.transfer_read` operations
419// replacing them with a `memref.collapse_shape`, a 1D `vector.transfer_read`,
420// and a `vector.shape_cast`.
421struct FlattenMultDimTransferReadPattern
422 : public OpConversionPattern<vector::TransferReadOp> {
423 using OpConversionPattern<vector::TransferReadOp>::OpConversionPattern;
424
425 LogicalResult
426 matchAndRewrite(vector::TransferReadOp readOp, OpAdaptor adaptor,
427 ConversionPatternRewriter &rewriter) const override {
428 // We can only deal with unmasked transfer ops with an identity permutation
429 // map.
430 if (!adaptor.getPermutationMap().isMinorIdentity() || adaptor.getMask())
431 return failure();
432 VectorType vectorTy = readOp.getVector().getType();
433 if (vectorTy.getRank() < 2)
434 return failure();
435 // Work only on bufferized reads
436 MemRefType memRefTy = dyn_cast<MemRefType>(adaptor.getBase().getType());
437 if (!memRefTy)
438 return failure();
439 auto memRefShape = memRefTy.getShape();
440 auto vecShape = vectorTy.getShape();
441
442 auto newVectorTy =
443 VectorType::get({std::accumulate(vecShape.begin(), vecShape.end(), 1,
444 std::multiplies<>())},
445 vectorTy.getElementType());
446 AffineMap layout = memRefTy.getLayout().getAffineMap();
447 auto newIndices =
448 collapseInnerMostDimIndices(rewriter, readOp.getLoc(), vecShape.size(),
449 adaptor.getIndices(), memRefShape, layout);
450 auto newSource = collapseInnerMostShapeDims(
451 rewriter, readOp.getLoc(), vecShape.size(), adaptor.getBase());
452 auto newVector = vector::TransferReadOp::create(
453 rewriter, readOp.getLoc(), newVectorTy, newSource, newIndices,
454 /*padding*/
455 arith::getZeroConstant(rewriter, readOp.getLoc(),
456 newVectorTy.getElementType()));
457
458 auto inBoundsArrayAttrOpt = adaptor.getInBounds();
459 if (inBoundsArrayAttrOpt) {
460 SmallVector<bool> inBounds =
461 llvm::to_vector(inBoundsArrayAttrOpt.getAsValueRange<BoolAttr>());
462 SmallVector<bool> newInBounds({false});
463 newInBounds[0] = std::all_of(inBounds.begin(), inBounds.end(),
464 [](bool v) { return v; });
465 newVector.getProperties().setInBounds(
466 rewriter.getBoolArrayAttr(newInBounds));
467 }
468
469 rewriter.replaceOpWithNewOp<vector::ShapeCastOp>(readOp, vectorTy,
470 newVector);
471
472 return success();
473 }
474};

532// This pattern extracts an implicit transposition of the 2 innermost
533// dimensions of `rhs` in a gemm-like contraction op, making it an explicit
534// `vector.transpose` op.
535// If `rhs` is coming from a widening op (`extf`/`extsi`/`extui`), the
536// transposition will be hoisted above the widening op.
537struct ExtractTransposeFromContractionOp
538 : public OpConversionPattern<vector::ContractionOp> {
539 using OpConversionPattern<vector::ContractionOp>::OpConversionPattern;
540
541 static VectorType getTransposedVectorType(VectorType vecTy) {
542 SmallVector<int64_t> shape{vecTy.getShape()};
543 auto nDim = shape.size();
544 int64_t dimNm1 = shape[nDim - 1];
545 shape[nDim - 1] = shape[nDim - 2];
546 shape[nDim - 2] = dimNm1;
547 auto elemTy = vecTy.getElementType();
548 return VectorType::get(shape, elemTy);
549 }
550
551 LogicalResult
552 matchAndRewrite(vector::ContractionOp contractOp, OpAdaptor adaptor,
553 ConversionPatternRewriter &rewriter) const override {
554 if (!isGemmBTransposedContractionOp(contractOp))
555 return failure();
556
557 Location loc = contractOp.getLoc();
558 auto *ctx = rewriter.getContext();
559
560 Value rhsVal = adaptor.getRhs();
561 VectorType rhsVecTy = contractOp.getRhsType();
562 Type rhsElemTy = rhsVecTy.getElementType();
563
564 bool doExtF = false, doExtSI = false, doExtUI = false;
565 if (auto extfRhsOp = rhsVal.getDefiningOp<arith::ExtFOp>()) {
566 rhsVal = extfRhsOp.getIn();
567 rhsVecTy = cast<VectorType>(rhsVal.getType());
568 doExtF = true;
569 } else if (auto extsiRhsOp = rhsVal.getDefiningOp<arith::ExtSIOp>()) {
570 rhsVal = extsiRhsOp.getIn();
571 rhsVecTy = cast<VectorType>(rhsVal.getType());
572 doExtSI = true;
573 } else if (auto extuiRhsOp = rhsVal.getDefiningOp<arith::ExtUIOp>()) {
574 rhsVal = extuiRhsOp.getIn();
575 rhsVecTy = cast<VectorType>(rhsVal.getType());
576 doExtUI = true;
577 }
578
579 int64_t nDim = rhsVecTy.getShape().size();
580 SmallVector<int64_t> rhsPermutation;
581 for (int64_t i = 0; i < nDim - 2; i++)
582 rhsPermutation.push_back(i);
583 rhsPermutation.push_back(nDim - 1);
584 rhsPermutation.push_back(nDim - 2);
585 auto transpRhsVecTy = getTransposedVectorType(rhsVecTy);
586 rhsVal = vector::TransposeOp::create(rewriter, loc, transpRhsVecTy, rhsVal,
587 rhsPermutation)
588 .getResult();
589
590 if (doExtF)
591 rhsVal =
592 arith::ExtFOp::create(
593 rewriter, loc,
594 VectorType::get(transpRhsVecTy.getShape(), rhsElemTy), rhsVal)
595 .getOut();
596 if (doExtSI)
597 rhsVal =
598 arith::ExtSIOp::create(
599 rewriter, loc,
600 VectorType::get(transpRhsVecTy.getShape(), rhsElemTy), rhsVal)
601 .getOut();
602 if (doExtUI)
603 rhsVal =
604 arith::ExtUIOp::create(
605 rewriter, loc,
606 VectorType::get(transpRhsVecTy.getShape(), rhsElemTy), rhsVal)
607 .getOut();
608
609 SmallVector<AffineMap, 4> oldIdxMaps(contractOp.getIndexingMapsArray());
610
611 nDim = oldIdxMaps[1].getNumDims();
612 SmallVector<int64_t> innerDimPerm;
613 for (int64_t i = 0; i < nDim - 2; i++)
614 innerDimPerm.push_back(i);
615 innerDimPerm.push_back(nDim - 1);
616 innerDimPerm.push_back(nDim - 2);
617 auto transpPermMap = AffineMap::getPermutationMap(innerDimPerm, ctx);
618
619 auto newIdxMaps = rewriter.getAffineMapArrayAttr(
620 {oldIdxMaps[0], oldIdxMaps[1].compose(transpPermMap), oldIdxMaps[2]});
621
622 rewriter.replaceOpWithNewOp<vector::ContractionOp>(
623 contractOp, contractOp.getResult().getType(), adaptor.getLhs(), rhsVal,
624 adaptor.getAcc(), newIdxMaps, contractOp.getIteratorTypes());
625
626 return success();
627 }

644/// Utility function to convert OpFoldResult offsets from a SubView operation
645/// into a vector of Values.
646static SmallVector<Value> opFoldResultsToValues(PatternRewriter &rewriter,
647 Location loc,
648 memref::SubViewOp subViewOp) {
649 OpBuilder::InsertionGuard g(rewriter);
650 rewriter.setInsertionPoint(subViewOp);

665/// Pattern to canonicalize trivial vector.transfer_read operations on subviews.
666///
667/// This pattern eliminates unnecessary memref.subview operations when the
668/// transfer_read accesses the subview with all-zero indices. It transforms:
669///
670/// INPUT:
671/// %subview = memref.subview %memref [offset0, offset1, ...]
672/// %result = vector.transfer_read %subview[0, 0, ...]
673///
674/// OUTPUT:
675/// %result = vector.transfer_read %memref[offset0, offset1, ...]
676///
677/// The pattern only matches when:
678/// - The base of transfer_read is defined by a memref.subview operation
679/// - All indices in the transfer_read are constant zeros
680struct CanonicalizeTrivialReadAccessSubviewOpPattern
681 : public OpRewritePattern<vector::TransferReadOp> {
682 using OpRewritePattern<vector::TransferReadOp>::OpRewritePattern;
683
684 LogicalResult matchAndRewrite(vector::TransferReadOp readOp,
685 PatternRewriter &rewriter) const override {
686 // Check if the base memref comes from a subview operation
687 auto subViewOp = dyn_cast_if_present<memref::SubViewOp>(
688 readOp.getBase().getDefiningOp());
689 if (!subViewOp)
690 return failure();
691
692 // Verify that all access indices are zero
693 if (failed(isAllZeroOffsetAccess(readOp.getIndices())))
694 return failure();
695
696 // Convert subview offsets to explicit index values
697 SmallVector<Value> newIndices =
698 opFoldResultsToValues(rewriter, readOp.getLoc(), subViewOp);
699
700 // Replace with direct access to the original memref using subview offsets
701 rewriter.replaceOpWithNewOp<vector::TransferReadOp>(
702 readOp, readOp.getType(), subViewOp.getSource(), newIndices,
703 readOp.getPadding(), readOp.getInBoundsValues());
704 return success();
705 }
706};

708/// Pattern to canonicalize trivial vector.transfer_write operations on
709/// subviews.
710///
711/// This pattern eliminates unnecessary memref.subview operations when the
712/// transfer_write accesses the subview with all-zero indices. It transforms:
713///
714/// INPUT:
715/// %subview = memref.subview %memref [offset0, offset1, ...]
716/// vector.transfer_write %value, %subview[0, 0, ...]
717///
718/// OUTPUT:
719/// vector.transfer_write %value, %memref[offset0, offset1, ...]
720///
721/// The pattern only matches when:
722/// - The base of transfer_write is defined by a memref.subview operation
723/// - All indices in the transfer_write are constant zeros
724struct CanonicalizeTrivialWriteAccessSubviewOpPattern
725 : public OpRewritePattern<vector::TransferWriteOp> {
726 using OpRewritePattern<vector::TransferWriteOp>::OpRewritePattern;
727
728 LogicalResult matchAndRewrite(vector::TransferWriteOp writeOp,
729 PatternRewriter &rewriter) const override {
730 // Check if the base memref comes from a subview operation
731 auto subViewOp = dyn_cast_if_present<memref::SubViewOp>(
732 writeOp.getBase().getDefiningOp());
733 if (!subViewOp)
734 return failure();
735
736 // Verify that all access indices are zero
737 if (failed(isAllZeroOffsetAccess(writeOp.getIndices())))
738 return failure();
739
740 // Convert subview offsets to explicit index values
741 SmallVector<Value> newIndices =
742 opFoldResultsToValues(rewriter, writeOp.getLoc(), subViewOp);
743
744 // Create new transfer_write with direct access to original memref
745 vector::TransferWriteOp::create(rewriter, writeOp.getLoc(),
746 writeOp.getVector(), subViewOp.getSource(),
747 newIndices, writeOp.getInBoundsValues());
748
749 // Remove the original transfer_write operation
750 rewriter.eraseOp(writeOp);
751 return success();
752 }

759struct ConvertLeadingUnitDimInsertToReshapePattern
760 : public OpRewritePattern<vector::InsertOp> {
761
762 using OpRewritePattern<vector::InsertOp>::OpRewritePattern;
763
764 LogicalResult matchAndRewrite(vector::InsertOp insOp,
765 PatternRewriter &rewriter) const override {
766 auto insSrcTy = dyn_cast<VectorType>(insOp.getValueToStoreType());
767 if (!insSrcTy)
768 return failure();
769
770 auto srcShape = insSrcTy.getShape();
771 auto dstShape = insOp.getDestVectorType().getShape();
772
773 unsigned long numLeadUnitDimDst = 0;
774 while (numLeadUnitDimDst < dstShape.size() &&
775 dstShape[numLeadUnitDimDst] == 1)
776 numLeadUnitDimDst++;
777
778 if (!numLeadUnitDimDst)
779 return failure();
780
781 unsigned long numLeadUnitDimSrc = 0;
782 while (numLeadUnitDimSrc < srcShape.size() &&
783 srcShape[numLeadUnitDimSrc] == 1)
784 numLeadUnitDimSrc++;
785
786 SmallVector<int64_t> nonLeadUnitDimDstShape(
787 dstShape.begin() + numLeadUnitDimDst, dstShape.end());
788 SmallVector<int64_t> nonLeadUnitDimSrcShape(
789 srcShape.begin() + numLeadUnitDimSrc, srcShape.end());
790
791 if (nonLeadUnitDimSrcShape != nonLeadUnitDimDstShape)
792 return failure();
793
794 rewriter.replaceOpWithNewOp<vector::ShapeCastOp>(
795 insOp, insOp.getDestVectorType(), insOp.getValueToStore());
796 return success();
797 }
