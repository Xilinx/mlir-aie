28/// Pattern to convert transaction operations to control operations
29struct BlockWriteToControlPacketPattern
30 : public OpRewritePattern<AIEX::NpuBlockWriteOp> {
31 using OpRewritePattern<AIEX::NpuBlockWriteOp>::OpRewritePattern;
32
33 LogicalResult matchAndRewrite(AIEX::NpuBlockWriteOp op,
34 PatternRewriter &rewriter) const override {
35
36 Value memref = op.getData();
37 int64_t width = cast<MemRefType>(memref.getType()).getElementTypeBitWidth();
38 if (width != 32) {
39 return op.emitWarning("Only 32-bit data type is supported for now");
40 }
41
42 memref::GetGlobalOp getGlobal = memref.getDefiningOp<memref::GetGlobalOp>();
43 if (!getGlobal) {
44 return op.emitError("Only MemRefs from memref.get_global are supported");
45 }
46
47 auto global = dyn_cast_if_present<memref::GlobalOp>(
48 op->getParentOfType<AIE::DeviceOp>().lookupSymbol(getGlobal.getName()));
49 if (!global) {
50 return op.emitError("Global symbol not found");
51 }
52
53 auto initVal = global.getInitialValue();
54 if (!initVal) {
55 return op.emitError("Global symbol has no initial value");
56 }
57
58 auto data = dyn_cast<DenseIntElementsAttr>(*initVal);
59 if (!data) {
60 return op.emitError(
61 "Global symbol initial value is not a dense int array");
62 }
63 std::vector<int32_t> dataVec(data.value_begin<int32_t>(),
64 data.value_end<int32_t>());
65 rewriter.create<AIEX::NpuControlPacketOp>(
66 op->getLoc(), op.getAddressAttr(), nullptr,
67 /*opcode*/ rewriter.getI32IntegerAttr(0),
68 /*stream_id*/ rewriter.getI32IntegerAttr(0),
69 DenseI32ArrayAttr::get(op->getContext(), dataVec));
70 rewriter.eraseOp(op);
71 return success();
72 }
73};

75/// Pattern to split control packets into smaller control packets
76struct ControlPacketSplitPattern
77 : public OpRewritePattern<AIEX::NpuControlPacketOp> {
78 using OpRewritePattern<AIEX::NpuControlPacketOp>::OpRewritePattern;
79
80 ControlPacketSplitPattern(MLIRContext *ctx, uint32_t max_payload_size)
81 : OpRewritePattern(ctx), max_payload_size(max_payload_size) {}
82
83 LogicalResult matchAndRewrite(AIEX::NpuControlPacketOp op,
84 PatternRewriter &rewriter) const override {
85 auto data = op.getData();
86 if (!data)
87 return failure();
88
89 uint32_t numElements = op.getDataAttr().size();
90
91 if (numElements <= max_payload_size) {
92 return failure(); // No splitting needed
93 }
94
95 auto chunks = llvm::divideCeil(numElements, max_payload_size);
96 auto context = op.getContext();
97 auto loc = op.getLoc();
98 for (unsigned i = 0; i < chunks; ++i) {
99 uint32_t startIdx = i * max_payload_size;
100 uint32_t endIdx = std::min(startIdx + max_payload_size, numElements);
101
102 SmallVector<int32_t, 4> chunkData;
103 for (auto it = data->begin() + startIdx; it != data->begin() + endIdx;
104 ++it) {
105 chunkData.push_back(*it);
106 }
107
108 // Increment the address for each chunk
109 auto incrementedAddress = rewriter.getUI32IntegerAttr(
110 op.getAddress() + (i * 4 * sizeof(uint32_t)));
111
112 rewriter.create<AIEX::NpuControlPacketOp>(
113 loc, incrementedAddress, nullptr, op.getOpcodeAttr(),
114 op.getStreamIdAttr(), DenseI32ArrayAttr::get(context, chunkData));
115 }
116
117 rewriter.eraseOp(op);
118 return success();
119 }
120
121private:
122 uint32_t max_payload_size;
123};

125struct AIETxnToControlPacketPass
126 : public AIEX::AIETxnToControlPacketBase<AIETxnToControlPacketPass> {
127 void runOnOperation() override {
128 AIE::DeviceOp device = getOperation();
129
130 ConversionTarget target(getContext());
131 target.addIllegalOp<AIEX::NpuBlockWriteOp>();
132 target.addLegalOp<AIEX::NpuControlPacketOp>();
133
134 RewritePatternSet patterns(&getContext());
135 patterns.add<BlockWriteToControlPacketPattern>(&getContext());
136
137 if (failed(applyPartialConversion(device, target, std::move(patterns)))) {
138 signalPassFailure();
139 }
140 }
141};

143struct AIELegalizeControlPacketPass
144 : public AIEX::AIELegalizeControlPacketBase<AIELegalizeControlPacketPass> {
145 void runOnOperation() override {
146 AIE::DeviceOp device = getOperation();
147
148 ConversionTarget target(getContext());
149 target.addDynamicallyLegalOp<AIEX::NpuControlPacketOp>([](Operation *op) {
150 auto packetOp = cast<AIEX::NpuControlPacketOp>(op);
151 // Check the data size
152 return packetOp.getDataAttr().size() <= 4;
153 });
154
155 RewritePatternSet patterns(&getContext());
156 patterns.add<ControlPacketSplitPattern>(&getContext(), 4);
157
158 if (failed(applyPartialConversion(device, target, std::move(patterns)))) {
159 signalPassFailure();
160 }
161 }
162};
