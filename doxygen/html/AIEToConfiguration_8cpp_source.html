40// A TransactionBinaryOperation encapsulates an aie-rt XAie_TxnCmd struct and
41// any additional metadata needed for custom operations that do not map cleanly
42// onto the core command fields.
43struct TransactionBinaryOperation {
44 struct XAie_TxnCmd cmd = {};
45
46 struct SyncPayload {
47 int32_t column;
48 int32_t row;
49 int32_t direction;
50 int32_t channel;
51 int32_t columnCount;
52 int32_t rowCount;
53 };
54
55 struct LoadPdiPayload {
56 uint32_t id;
57 uint32_t size;
58 uint64_t address;
59 };
60
61 struct AddressPatchPayload {
62 uint32_t action;
63 uint32_t addr;
64 int32_t argIdx;
65 int32_t argPlus;
66 };
67
68 std::optional<SyncPayload> sync;
69 std::optional<LoadPdiPayload> loadPdi;
70 std::optional<AddressPatchPayload> addressPatch;
71
72 TransactionBinaryOperation() = default;
73
74 TransactionBinaryOperation(XAie_TxnOpcode opc, uint32_t mask, uint64_t addr,
75 uint32_t value, const uint8_t *data,
76 uint32_t size) {
77 cmd.Opcode = opc;
78 cmd.Mask = mask;
79 cmd.RegOff = addr;
80 cmd.Value = value;
81 cmd.DataPtr = reinterpret_cast<uint64_t>(data);
82 cmd.Size = size;
83 }
84};
85
86constexpr size_t kTxnHeaderBytes = 16;
87
88struct TxnPreemptHeader {
89 uint8_t opcode;
90 uint8_t level;
91 uint16_t reserved;
92};
93
94struct TxnLoadPdiHeader {
95 uint8_t opcode;
96 uint8_t padding;
97 uint16_t id;
98 uint32_t size;
99 uint64_t address;
100};
101} // namespace

103// Parse a TXN binary blob. On success return the number of columns from the
104// header and a vector of parsed operations. On failure return std::nullopt.
105static std::optional<int>
106parseTransactionBinary(const std::vector<uint8_t> &data,
107 std::vector<TransactionBinaryOperation> &ops) {
108
109 if (data.size() < kTxnHeaderBytes) {
110 llvm::errs() << "Transaction binary is too small for header\n";
111 return std::nullopt;
112 }
113
114 uint32_t major = data[0];
115 uint32_t minor = data[1];
116 uint32_t num_cols = data[4];
117
118 uint32_t num_ops, txn_size;
119 std::memcpy(&num_ops, &data[8], 4);
120 std::memcpy(&txn_size, &data[12], 4);
121
122 LLVM_DEBUG(llvm::dbgs() << "Major: " << major << "\n");
123 LLVM_DEBUG(llvm::dbgs() << "Minor: " << minor << "\n");
124 LLVM_DEBUG(llvm::dbgs() << "DevGen: " << data[2] << "\n");
125 LLVM_DEBUG(llvm::dbgs() << "NumRows: " << data[3] << "\n");
126 LLVM_DEBUG(llvm::dbgs() << "NumCols: " << num_cols << "\n");
127 LLVM_DEBUG(llvm::dbgs() << "NumMemTileRows: " << data[5] << "\n");
128 LLVM_DEBUG(llvm::dbgs() << "NumOps: " << num_ops << "\n");
129 LLVM_DEBUG(llvm::dbgs() << "TxnSize: " << txn_size << " bytes\n");
130
131 size_t i = kTxnHeaderBytes;
132
133 auto requireBytes = [&](size_t offset, size_t length) -> bool {
134 if (offset + length > data.size()) {
135 llvm::errs() << "Transaction binary truncated while parsing opcode\n";
136 return false;
137 }
138 return true;
139 };
140
141 auto read32 = [&](size_t offset) -> uint32_t {
142 uint32_t value;
143 std::memcpy(&value, data.data() + offset, sizeof(uint32_t));
144 return value;
145 };
146
147 // Convert opcode from uint8 to a validated opcode byte
148 auto convertOpcode = [](uint8_t opc) -> std::optional<uint8_t> {
149 switch (opc) {
150 case static_cast<uint8_t>(XAie_TxnOpcode::XAIE_IO_WRITE):
151 case static_cast<uint8_t>(XAie_TxnOpcode::XAIE_IO_BLOCKWRITE):
152 case static_cast<uint8_t>(XAie_TxnOpcode::XAIE_IO_MASKWRITE):
153 case 0x6: // XAie_TxnOpcode::XAIE_IO_PREEMPT
154 case 0x8: // XAie_TxnOpcode::XAIE_IO_LOAD_PDI
155 case static_cast<uint8_t>(XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_TCT):
156 case static_cast<uint8_t>(XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_DDR_PATCH):
157 return opc;
158 default:
159 llvm::errs() << "Unhandled opcode: " << std::to_string(opc) << "\n";
160 return std::nullopt;
161 }
162 };
163
164 // Parse the binary blob. There are two versions supported, 0.1 and 1.0.
165 // For both versions, build a list of TransactionBinaryOperation objects
166 // representing the parsed operations.
167 if (major == 0 && minor == 1) {
168 while (i < data.size()) {
169 auto maybeOpcode = convertOpcode(data[i]);
170 if (!maybeOpcode)
171 return std::nullopt;
172 XAie_TxnOpcode opcode = static_cast<XAie_TxnOpcode>(*maybeOpcode);
173 LLVM_DEBUG(llvm::dbgs() << "opcode: " + std::to_string(opcode) << "\n");
174
175 TransactionBinaryOperation op;
176 op.cmd.Opcode = opcode;
177
178 switch (opcode) {
179 case XAie_TxnOpcode::XAIE_IO_WRITE: {
180 LLVM_DEBUG(llvm::dbgs() << "opcode: WRITE (0x00)\n");
181 if (!requireBytes(i, 24))
182 return std::nullopt;
183 uint32_t addrLo = read32(i + 8);
184 uint32_t addrHi = read32(i + 12);
185 uint32_t value = read32(i + 16);
186 uint32_t opSize = read32(i + 20);
187 if (!requireBytes(i, opSize))
188 return std::nullopt;
189 uint64_t addr = (static_cast<uint64_t>(addrHi) << 32) | addrLo;
190 op.cmd.RegOff = addr;
191 op.cmd.Value = value;
192 op.cmd.Size = 0;
193 i += opSize;
194 break;
195 }
196 case XAie_TxnOpcode::XAIE_IO_BLOCKWRITE: {
197 LLVM_DEBUG(llvm::dbgs() << "opcode: BLOCKWRITE (0x01)\n");
198 if (!requireBytes(i, 16))
199 return std::nullopt;
200 uint32_t addr = read32(i + 8);
201 uint32_t opSize = read32(i + 12);
202 if (opSize < 16 || !requireBytes(i, opSize))
203 return std::nullopt;
204 const uint8_t *payload = data.data() + i + 16;
205 uint32_t payloadBytes = opSize - 16;
206 op.cmd.RegOff = addr;
207 op.cmd.DataPtr = reinterpret_cast<uint64_t>(payload);
208 op.cmd.Size = payloadBytes;
209 i += opSize;
210 break;
211 }
212 case XAie_TxnOpcode::XAIE_IO_MASKWRITE: {
213 LLVM_DEBUG(llvm::dbgs() << "opcode: MASKWRITE (0x03)\n");
214 if (!requireBytes(i, 28))
215 return std::nullopt;
216 uint32_t addrLo = read32(i + 8);
217 uint32_t addrHi = read32(i + 12);
218 uint32_t value = read32(i + 16);
219 uint32_t mask = read32(i + 20);
220 uint32_t opSize = read32(i + 24);
221 if (!requireBytes(i, opSize))
222 return std::nullopt;
223 uint64_t addr = (static_cast<uint64_t>(addrHi) << 32) | addrLo;
224 op.cmd.RegOff = addr;
225 op.cmd.Value = value;
226 op.cmd.Mask = mask;
227 op.cmd.Size = opSize;
228 i += opSize;
229 break;
230 }
231 case XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_TCT: {
232 uint32_t opSize = read32(i + 4);
233 if (opSize < 16 || !requireBytes(i, opSize))
234 return std::nullopt;
235 uint32_t descriptor = read32(i + 8);
236 uint32_t config = read32(i + 12);
237 TransactionBinaryOperation::SyncPayload payload{
238 /*column=*/static_cast<int32_t>((descriptor >> 16) & 0xff),
239 /*row=*/static_cast<int32_t>((descriptor >> 8) & 0xff),
240 /*direction=*/static_cast<int32_t>(descriptor & 0xff),
241 /*channel=*/static_cast<int32_t>((config >> 24) & 0xff),
242 /*columnCount=*/static_cast<int32_t>((config >> 16) & 0xff),
243 /*rowCount=*/static_cast<int32_t>((config >> 8) & 0xff)};
244 op.sync = payload;
245 op.cmd.Size = opSize;
246 i += opSize;
247 break;
248 }
249 case 0x8: { // XAie_TxnOpcode::XAIE_IO_LOAD_PDI
250 LLVM_DEBUG(llvm::dbgs() << "opcode: LOAD_PDI (0x08)\n");
251 constexpr size_t opSize = sizeof(TxnLoadPdiHeader);
252 if (!requireBytes(i, opSize))
253 return std::nullopt;
254 TxnLoadPdiHeader header;
255 std::memcpy(&header, data.data() + i, opSize);
256 TransactionBinaryOperation::LoadPdiPayload payload{
257 header.id, header.size, header.address};
258 op.loadPdi = payload;
259 op.cmd.Size = opSize;
260 i += opSize;
261 break;
262 }
263 case XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_DDR_PATCH: {
264 uint32_t opSize = read32(i + 4);
265 if (opSize < 44 || !requireBytes(i, opSize))
266 return std::nullopt;
267 uint32_t action = read32(i + 20);
268 uint32_t addr = read32(i + 24);
269 int32_t argIdx = static_cast<int32_t>(read32(i + 32));
270 int32_t argPlus = static_cast<int32_t>(read32(i + 40));
271 TransactionBinaryOperation::AddressPatchPayload payload{
272 action, addr, argIdx, argPlus};
273 op.addressPatch = payload;
274 op.cmd.Size = opSize;
275 i += opSize;
276 break;
277 }
278 case 0x6: { // XAie_TxnOpcode::XAIE_IO_PREEMPT
279 LLVM_DEBUG(llvm::dbgs() << "opcode: PREEMPT (0x06)\n");
280 constexpr size_t opSize = sizeof(TxnPreemptHeader);
281 if (!requireBytes(i, opSize))
282 return std::nullopt;
283 auto header =
284 reinterpret_cast<const TxnPreemptHeader *>(data.data() + i);
285 op.cmd.Value = header->level;
286 op.cmd.Size = opSize;
287 i += opSize;
288 break;
289 }
290 default:
291 llvm::errs() << "Unhandled opcode: " << std::to_string(opcode)
292 << " for v0.1 transaction\n";
293 return std::nullopt;
294 }
295
296 ops.push_back(std::move(op));
297 }
298 } else if (major == 1 && minor == 0) {
299 while (i < data.size()) {
300 auto maybeOpcode = convertOpcode(data[i]);
301 if (!maybeOpcode)
302 return std::nullopt;
303 XAie_TxnOpcode opcode = static_cast<XAie_TxnOpcode>(*maybeOpcode);
304 LLVM_DEBUG(llvm::dbgs() << "opcode: " + std::to_string(opcode) << "\n");
305
306 TransactionBinaryOperation op;
307 op.cmd.Opcode = opcode;
308
309 switch (opcode) {
310 case XAie_TxnOpcode::XAIE_IO_WRITE: {
311 LLVM_DEBUG(llvm::dbgs() << "opcode: WRITE (0x00)\n");
312 if (!requireBytes(i, 12))
313 return std::nullopt;
314 uint32_t addr = read32(i + 4);
315 uint32_t value = read32(i + 8);
316 op.cmd.RegOff = addr;
317 op.cmd.Value = value;
318 op.cmd.Size = 0;
319 i += 12;
320 break;
321 }
322 case XAie_TxnOpcode::XAIE_IO_BLOCKWRITE: {
323 LLVM_DEBUG(llvm::dbgs() << "opcode: BLOCKWRITE (0x01)\n");
324 if (!requireBytes(i, 12))
325 return std::nullopt;
326 uint32_t addr = read32(i + 4);
327 uint32_t opSize = read32(i + 8);
328 if (opSize < 12 || !requireBytes(i, opSize))
329 return std::nullopt;
330 const uint8_t *payload = data.data() + i + 12;
331 uint32_t payloadBytes = opSize - 12;
332 op.cmd.RegOff = addr;
333 op.cmd.DataPtr = reinterpret_cast<uint64_t>(payload);
334 op.cmd.Size = payloadBytes;
335 i += opSize;
336 break;
337 }
338 case XAie_TxnOpcode::XAIE_IO_MASKWRITE: {
339 LLVM_DEBUG(llvm::dbgs() << "opcode: MASKWRITE (0x03)\n");
340 if (!requireBytes(i, 16))
341 return std::nullopt;
342 uint32_t addr = read32(i + 4);
343 uint32_t value = read32(i + 8);
344 uint32_t mask = read32(i + 12);
345 op.cmd.RegOff = addr;
346 op.cmd.Value = value;
347 op.cmd.Mask = mask;
348 op.cmd.Size = 0;
349 i += 16;
350 break;
351 }
352 case XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_TCT: {
353 uint32_t opSize = read32(i + 4);
354 if (opSize < 16 || !requireBytes(i, opSize))
355 return std::nullopt;
356 uint32_t descriptor = read32(i + 8);
357 uint32_t config = read32(i + 12);
358 TransactionBinaryOperation::SyncPayload payload{
359 /*column=*/static_cast<int32_t>((descriptor >> 16) & 0xff),
360 /*row=*/static_cast<int32_t>((descriptor >> 8) & 0xff),
361 /*direction=*/static_cast<int32_t>(descriptor & 0xff),
362 /*channel=*/static_cast<int32_t>((config >> 24) & 0xff),
363 /*columnCount=*/static_cast<int32_t>((config >> 16) & 0xff),
364 /*rowCount=*/static_cast<int32_t>((config >> 8) & 0xff)};
365 op.sync = payload;
366 op.cmd.Size = opSize;
367 i += opSize;
368 break;
369 }
370 case 0x8: { // XAie_TxnOpcode::XAIE_IO_LOAD_PDI
371 LLVM_DEBUG(llvm::dbgs() << "opcode: LOAD_PDI (0x08)\n");
372 constexpr size_t opSize = sizeof(TxnLoadPdiHeader);
373 if (!requireBytes(i, opSize))
374 return std::nullopt;
375 TxnLoadPdiHeader header;
376 std::memcpy(&header, data.data() + i, opSize);
377 TransactionBinaryOperation::LoadPdiPayload payload{
378 header.id, header.size, header.address};
379 op.loadPdi = payload;
380 op.cmd.Size = opSize;
381 i += opSize;
382 break;
383 }
384 case XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_DDR_PATCH: {
385 uint32_t opSize = read32(i + 4);
386 if (opSize < 44 || !requireBytes(i, opSize))
387 return std::nullopt;
388 uint32_t action = read32(i + 20);
389 uint32_t addr = read32(i + 24);
390 int32_t argIdx = static_cast<int32_t>(read32(i + 32));
391 int32_t argPlus = static_cast<int32_t>(read32(i + 40));
392 TransactionBinaryOperation::AddressPatchPayload payload{
393 action, addr, argIdx, argPlus};
394 op.addressPatch = payload;
395 op.cmd.Size = opSize;
396 i += opSize;
397 break;
398 }
399 case 0x6: { // XAie_TxnOpcode::XAIE_IO_PREEMPT
400 LLVM_DEBUG(llvm::dbgs() << "opcode: PREEMPT (0x06)\n");
401 constexpr size_t opSize = sizeof(TxnPreemptHeader);
402 if (!requireBytes(i, opSize))
403 return std::nullopt;
404 auto header =
405 reinterpret_cast<const TxnPreemptHeader *>(data.data() + i);
406 op.cmd.Value = header->level;
407 op.cmd.Size = opSize;
408 i += opSize;
409 break;
410 }
411 default:
412 llvm::errs() << "Unhandled opcode: " << std::to_string(opcode)
413 << " for v1.0 transaction\n";
414 return std::nullopt;
415 }
416
417 ops.push_back(std::move(op));
418 }
419 } else {
420 llvm::errs() << "Unsupported TXN binary version: " << major << "." << minor
421 << "\n";
422 return std::nullopt;
423 }
424
425 return num_cols;
426}

428static LogicalResult generateTransactions(AIERTControl &ctl,
429 const StringRef workDirPath,
430 DeviceOp &targetOp, bool aieSim,
431 bool enableElfs, bool enableInit,
432 bool enableCores) {
433 if (enableElfs && !targetOp.getOps<CoreOp>().empty() &&
434 failed(ctl.addAieElfs(targetOp, workDirPath, aieSim)))
435 return failure();
436 if (enableInit && failed(ctl.addInitConfig(targetOp)))
437 return failure();
438 if (enableCores && !targetOp.getOps<CoreOp>().empty() &&
439 failed(ctl.addCoreEnable(targetOp)))
440 return failure();
441 return success();
442}

444// Translate vector of TransactionBinaryOperation to a sequence of transaction
445// ops (npu.write32, npu.maskwrite32, npu.blockwrite).
446static LogicalResult
447emitTransactionOps(OpBuilder &builder,
448 std::vector<TransactionBinaryOperation> &operations,
449 std::vector<memref::GlobalOp> &global_data) {
450
451 auto loc = builder.getUnknownLoc();
452
453 // create the txn ops
454 for (auto [op, payload] : llvm::zip(operations, global_data)) {
455
456 if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_WRITE) {
457 AIEX::NpuWrite32Op::create(builder, loc, op.cmd.RegOff, op.cmd.Value,
458 nullptr, nullptr, nullptr);
459 } else if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_BLOCKWRITE) {
460 auto memref = memref::GetGlobalOp::create(builder, loc, payload.getType(),
461 payload.getName());
462 AIEX::NpuBlockWriteOp::create(
463 builder, loc, builder.getUI32IntegerAttr(op.cmd.RegOff),
464 memref.getResult(), nullptr, nullptr, nullptr);
465 } else if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_MASKWRITE) {
466 AIEX::NpuMaskWrite32Op::create(builder, loc, op.cmd.RegOff, op.cmd.Value,
467 op.cmd.Mask, nullptr, nullptr, nullptr);
468 } else if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_TCT) {
469 if (!op.sync) {
470 llvm::errs() << "Missing sync payload while emitting transaction\n";
471 return failure();
472 }
473 const TransactionBinaryOperation::SyncPayload &sync = *op.sync;
474 AIEX::NpuSyncOp::create(builder, loc,
475 builder.getI32IntegerAttr(sync.column),
476 builder.getI32IntegerAttr(sync.row),
477 builder.getI32IntegerAttr(sync.direction),
478 builder.getI32IntegerAttr(sync.channel),
479 builder.getI32IntegerAttr(sync.columnCount),
480 builder.getI32IntegerAttr(sync.rowCount));
481 } else if (op.cmd.Opcode == 0x8 /* XAie_TxnOpcode::XAIE_IO_LOAD_PDI */) {
482 if (!op.loadPdi) {
483 llvm::errs() << "Missing load_pdi payload while emitting transaction\n";
484 return failure();
485 }
486 const TransactionBinaryOperation::LoadPdiPayload &payloadInfo =
487 *op.loadPdi;
488 auto idAttr =
489 builder.getI32IntegerAttr(static_cast<int32_t>(payloadInfo.id));
490 IntegerAttr sizeAttr =
491 builder.getI32IntegerAttr(static_cast<int32_t>(payloadInfo.size));
492
493 auto ui64Ty =
494 IntegerType::get(builder.getContext(), 64, IntegerType::Unsigned);
495 IntegerAttr addressAttr =
496 IntegerAttr::get(ui64Ty, llvm::APInt(64, payloadInfo.address));
497
498 AIEX::NpuLoadPdiOp::create(builder, loc, nullptr, idAttr, sizeAttr,
499 addressAttr);
500 } else if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_CUSTOM_OP_DDR_PATCH) {
501 if (!op.addressPatch) {
502 llvm::errs()
503 << "Missing address_patch payload while emitting transaction\n";
504 return failure();
505 }
506 const TransactionBinaryOperation::AddressPatchPayload &patch =
507 *op.addressPatch;
508 AIEX::NpuAddressPatchOp::create(builder, loc,
509 builder.getUI32IntegerAttr(patch.addr),
510 builder.getI32IntegerAttr(patch.argIdx),
511 builder.getI32IntegerAttr(patch.argPlus));
512 } else if (op.cmd.Opcode == 0x6 /* XAie_TxnOpcode::XAIE_IO_PREEMPT */) {
513 auto ui8Ty =
514 IntegerType::get(builder.getContext(), 8, IntegerType::Unsigned);
515 auto levelAttr = IntegerAttr::get(ui8Ty, llvm::APInt(8, op.cmd.Value));
516 AIEX::NpuPreemptOp::create(builder, loc, levelAttr);
517 } else {
518 llvm::errs() << "Unhandled txn opcode: " << op.cmd.Opcode << "\n";
519 return failure();
520 }
521 }
522 return success();
523}

525// Translate vector of TransactionBinaryOperation to a sequence of control
526// packet ops.
527static LogicalResult
528emitControlPacketOps(OpBuilder &builder,
529 std::vector<TransactionBinaryOperation> &operations,
530 std::vector<memref::GlobalOp> &global_data) {
531
532 auto loc = builder.getUnknownLoc();
533 auto ctx = builder.getContext();
534
535 // create the control packet ops
536 for (auto [op, payload] : llvm::zip(operations, global_data)) {
537
538 if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_WRITE) {
539 AIEX::NpuControlPacketOp::create(
540 builder, loc, builder.getUI32IntegerAttr(op.cmd.RegOff), nullptr,
541 /*opcode*/ builder.getI32IntegerAttr(0),
542 /*stream_id*/ builder.getI32IntegerAttr(0),
543 DenseI32ArrayAttr::get(ctx, ArrayRef<int32_t>(op.cmd.Value)));
544 } else if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_BLOCKWRITE) {
545 if (!payload.getInitialValue())
546 continue;
547 auto blockWriteData =
548 dyn_cast<DenseIntElementsAttr>(*payload.getInitialValue());
549 if (!blockWriteData) {
550 payload.emitError(
551 "Global symbol initial value is not a dense int array");
552 break;
553 }
554 auto blockWriteDataValues = blockWriteData.getValues<int32_t>();
555 // Split block write data into beats of 4 or less, in int32_t.
556 int currAddr = op.cmd.RegOff;
557 for (size_t i = 0; i < blockWriteDataValues.size(); i += 4) {
558 auto last = std::min(blockWriteDataValues.size(), i + 4);
559 SmallVector<int32_t> splitData =
560 SmallVector<int32_t>(blockWriteDataValues.begin() + i,
561 blockWriteDataValues.begin() + last);
562 AIEX::NpuControlPacketOp::create(
563 builder, loc, builder.getUI32IntegerAttr(currAddr), nullptr,
564 /*opcode*/ builder.getI32IntegerAttr(0),
565 /*stream_id*/ builder.getI32IntegerAttr(0),
566 DenseI32ArrayAttr::get(ctx, ArrayRef<int32_t>(splitData)));
567 currAddr += splitData.size() * sizeof(int32_t);
568 }
569
570 } else if (op.cmd.Opcode == XAie_TxnOpcode::XAIE_IO_MASKWRITE) {
571 AIEX::NpuControlPacketOp::create(
572 builder, loc, builder.getUI32IntegerAttr(op.cmd.RegOff), nullptr,
573 /*opcode*/ builder.getI32IntegerAttr(0),
574 /*stream_id*/ builder.getI32IntegerAttr(0),
575 DenseI32ArrayAttr::get(ctx, ArrayRef<int32_t>(op.cmd.Value)));
576 } else {
577 llvm::errs() << "Unhandled txn opcode: " << op.cmd.Opcode << "\n";
578 return failure();
579 }
580 }
581 return success();
582}

584// Perform bitwise or on consecutive control packets operating on the same
585// address, to resolve the lack of mask write in control packets.
586LogicalResult orConsecutiveWritesOnSameAddr(Block *body) {
587 SmallVector<AIEX::NpuControlPacketOp> ctrlPktOps;
588 body->walk(
589 [&](AIEX::NpuControlPacketOp cpOp) { ctrlPktOps.push_back(cpOp); });
590 if (ctrlPktOps.empty())
591 return success();
592
593 SmallVector<Operation *> erased;
594 int addrBuffer = ctrlPktOps[0].getAddress();
595 AIEX::NpuControlPacketOp ctrlPktBuffer = ctrlPktOps[0];
596 for (size_t i = 1; i < ctrlPktOps.size(); i++) {
597 int currentAddrBuffer = ctrlPktOps[i].getAddress();
598 if (addrBuffer != currentAddrBuffer) {
599 addrBuffer = currentAddrBuffer;
600 ctrlPktBuffer = ctrlPktOps[i];
601 continue;
602 }
603 auto bufferedData = ctrlPktBuffer.getData().value();
604 auto currentData = ctrlPktOps[i].getData().value();
605 SmallVector<int> newData;
606 for (unsigned j = 0; j < std::max(bufferedData.size(), currentData.size());
607 j++) {
608 if (j < std::min(bufferedData.size(), currentData.size())) {
609 newData.push_back(bufferedData[j] | currentData[j]);
610 continue;
611 }
612 newData.push_back(j < bufferedData.size() ? bufferedData[j]
613 : currentData[j]);
614 }
615 ctrlPktBuffer.getProperties().data = DenseI32ArrayAttr::get(
616 ctrlPktBuffer->getContext(), ArrayRef<int>{newData});
617 erased.push_back(ctrlPktOps[i]);
618 }
619
620 for (auto e : erased)
621 e->erase();
622
623 return success();
624}

632static LogicalResult convertTransactionOpsToMLIR(
633 OpBuilder builder, AIE::DeviceOp device, OutputType outputType,
634 std::vector<TransactionBinaryOperation> &operations) {
635
636 auto loc = builder.getUnknownLoc();
637
638 // for each blockwrite in the binary, create a GlobalOp with the data
639 std::vector<memref::GlobalOp> global_data;
640 for (auto &op : operations) {
641 if (op.cmd.Opcode != XAIE_IO_BLOCKWRITE) {
642 global_data.push_back(nullptr);
643 continue;
644 }
645 uint32_t size = op.cmd.Size / 4;
646 const uint32_t *d = reinterpret_cast<const uint32_t *>(op.cmd.DataPtr);
647 std::vector<uint32_t> data32(d, d + size);
648
649 int id = 0;
650 std::string name = "blockwrite_data";
651 while (device.lookupSymbol(name))
652 name = "blockwrite_data_" + std::to_string(id++);
653
654 MemRefType memrefType = MemRefType::get({size}, builder.getI32Type());
655 TensorType tensorType = RankedTensorType::get({size}, builder.getI32Type());
656 auto global = memref::GlobalOp::create(
657 builder, loc, name, builder.getStringAttr("private"), memrefType,
658 DenseElementsAttr::get<uint32_t>(tensorType, data32), true, nullptr);
659 global_data.push_back(global);
660 }
661
662 // search for aiex.configure ops in runtime sequences by walking the device
663 // and collect them in a vector. If there are none, create a new runtime
664 // sequence. Otherwise assume the insertion point is the first aiex.configure
665 // op.
666 SmallVector<AIEX::ConfigureOp> configureOps;
667 device.walk([&](AIEX::ConfigureOp op) { configureOps.push_back(op); });
668
669 if (configureOps.empty()) {
670
671 // create aiex.runtime_sequence
672 int id = 0;
673 std::string seq_name = "configure";
674 while (device.lookupSymbol(seq_name))
675 seq_name = "configure" + std::to_string(id++);
676 StringAttr seq_sym_name = builder.getStringAttr(seq_name);
677 auto seq = AIE::RuntimeSequenceOp::create(builder, loc, seq_sym_name);
678 seq.getBody().push_back(new Block);
679
680 builder.setInsertionPointToStart(&seq.getBody().front());
681 } else {
682 builder.setInsertionPoint(configureOps.front());
683 }
684
685 // create the txn ops
686 if (outputType == OutputType::Transaction) {
687 if (failed(emitTransactionOps(builder, operations, global_data)))
688 return failure();
689 } else if (outputType == OutputType::ControlPacket) {
690 if (failed(emitControlPacketOps(builder, operations, global_data)))
691 return failure();
692 // resolve mask writes; control packet doesn't natively support mask write.
693 if (failed(orConsecutiveWritesOnSameAddr(builder.getBlock())))
694 return failure();
695 } else {
696 llvm_unreachable("bad output type");
697 }
698
699 if (!configureOps.empty()) {
700 // splice the body into the current insertion point
701 builder.getBlock()->getOperations().splice(
702 builder.getInsertionPoint(),
703 configureOps.front().getBody().front().getOperations());
704 configureOps.front().erase();
705 }
706 return success();
707}

709// Convert (disassemble) a transaction binary to MLIR. On success return a new
710// ModuleOp containing a DeviceOp containing a runtime sequence with the
711// transaction binary encoded as a sequence of npu.write32, npu.maskwrite32 and
712// npu.blockwrite operations. On failure return std::nullopt.
713std::optional<mlir::ModuleOp>
714xilinx::AIE::convertTransactionBinaryToMLIR(mlir::MLIRContext *ctx,
715 std::vector<uint8_t> &binary) {
716
717 // parse the binary
718 std::vector<TransactionBinaryOperation> operations;
719 auto c = parseTransactionBinary(binary, operations);
720 if (!c) {
721 llvm::errs() << "Failed to parse binary\n";
722 return std::nullopt;
723 }
724 int columns = *c;
725
726 auto loc = mlir::UnknownLoc::get(ctx);
727
728 // create a new ModuleOp and set the insertion point
729 auto module = ModuleOp::create(loc);
730 OpBuilder builder(module.getBodyRegion());
731 builder.setInsertionPointToStart(module.getBody());
732
733 // create aie.device
734 std::vector<AIEDevice> devices{AIEDevice::npu1_1col, AIEDevice::npu1_2col,
735 AIEDevice::npu1_3col, AIEDevice::npu1};
736 auto device = DeviceOp::create(builder, loc, devices[columns - 1],
737 StringAttr::get(builder.getContext()));
738 device.getRegion().emplaceBlock();
739 DeviceOp::ensureTerminator(device.getBodyRegion(), builder, loc);
740 builder.setInsertionPointToStart(device.getBody());
741
742 // convert the parsed ops to MLIR
743 if (failed(convertTransactionOpsToMLIR(builder, device,
744 OutputType::Transaction, operations)))
745 return std::nullopt;
746
747 return module;
748}

750static LogicalResult convertAIEToConfiguration(AIE::DeviceOp device,
751 StringRef clElfDir,
752 OutputType outputType) {
753
754 const AIETargetModel &targetModel =
755 (const AIETargetModel &)device.getTargetModel();
756
757 if (!targetModel.hasProperty(AIETargetModel::IsNPU))
758 return failure();
759
760 bool aieSim = false;
761 bool xaieDebug = false;
762
763 AIERTControl ctl(targetModel);
764 if (failed(ctl.setIOBackend(aieSim, xaieDebug)))
765 return failure();
766
767 // start collecting transations
768 ctl.startTransaction();
769
770 bool generateElfs = clElfDir.size() > 0;
771 if (failed(generateTransactions(ctl, clElfDir, device, aieSim, generateElfs,
772 true, true)))
773 return failure();
774
775 // Export the transactions to a binary buffer
776 std::vector<uint8_t> txn_data = ctl.exportSerializedTransaction();
777
778 // parse the binary data
779 std::vector<TransactionBinaryOperation> operations;
780 if (!parseTransactionBinary(txn_data, operations)) {
781 llvm::errs() << "Failed to parse binary\n";
782 return failure();
783 }
784
785 OpBuilder builder(device.getBodyRegion());
786
787 // convert the parsed ops to MLIR
788 if (failed(
789 convertTransactionOpsToMLIR(builder, device, outputType, operations)))
790 return failure();
791
792 return success();
793}
