39class LockAnalysis {
40 DenseMap<std::pair<Value, int>, int> locksPerTile;
41
42public:
43 LockAnalysis(DeviceOp &device) {
44 // go over the locks created for each tile and update the index in
45 // locksPerTile
46 device.walk([&](LockOp lockOp) {
47 auto tile = lockOp.getTile();
48 auto lockID = lockOp.getLockIDValue();
49 locksPerTile[{tile, lockID}] = 1;
50 });
51 }
52
53 /// Given a tile, returns next usable lockID for that tile.
54 int getLockID(TileOp &tileOp) {
55 const auto &targetModel = getTargetModel(tileOp);
56 for (unsigned i = 0;
57 i < targetModel.getNumLocks(tileOp.getCol(), tileOp.getRow()); i++)
58 if (int usageCnt = locksPerTile[{tileOp, i}]; usageCnt == 0) {
59 locksPerTile[{tileOp, i}] = 1;
60 return i;
61 }
62 return -1;
63 }
64};

69class DMAChannelAnalysis {
70 DenseMap<std::tuple<Value, DMAChannelDir, int>, int> channelsPerTile;
71
72public:
73 DMAChannelAnalysis(DeviceOp &device) {
74 // go over the channels used for each tile and update channel map
75 for (auto memOp : device.getOps<MemOp>()) {
76 Region &r = memOp.getBody();
77 for (auto &bl : r.getBlocks()) {
78 for (auto op : bl.getOps<DMAStartOp>()) {
79 channelsPerTile[{memOp.getTile(), op.getChannelDir(),
80 op.getChannelIndex()}] = 1;
81 }
82 }
83 }
84 for (auto memOp : device.getOps<MemTileDMAOp>()) {
85 Region &r = memOp.getBody();
86 for (auto &bl : r.getBlocks()) {
87 for (auto op : bl.getOps<DMAStartOp>()) {
88 channelsPerTile[{memOp.getTile(), op.getChannelDir(),
89 op.getChannelIndex()}] = 1;
90 }
91 }
92 }
93 for (auto memOp : device.getOps<ShimDMAOp>()) {
94 Region &r = memOp.getBody();
95 for (auto &bl : r.getBlocks()) {
96 for (auto op : bl.getOps<DMAStartOp>()) {
97 channelsPerTile[{memOp.getTile(), op.getChannelDir(),
98 op.getChannelIndex()}] = 1;
99 }
100 }
101 }
102 }
103
104 /// Given a tile and DMAChannelDir, returns next usable channel index for
105 /// that tile.
106 int getDMAChannelIndex(TileOp tileOp, DMAChannelDir dir) {
107 const auto &targetModel = getTargetModel(tileOp);
108 int maxChannelNum = 0;
109 if (tileOp.isShimTile())
110 maxChannelNum = 2;
111 else {
112 if (dir == DMAChannelDir::MM2S)
113 maxChannelNum = targetModel.getNumSourceSwitchboxConnections(
114 tileOp.getCol(), tileOp.getRow(), WireBundle::DMA);
115 else
116 maxChannelNum = targetModel.getNumDestSwitchboxConnections(
117 tileOp.getCol(), tileOp.getRow(), WireBundle::DMA);
118 }
119 for (int i = 0; i < maxChannelNum; i++)
120 if (int usageCnt = channelsPerTile[{tileOp.getResult(), dir, i}];
121 usageCnt == 0) {
122 channelsPerTile[{tileOp.getResult(), dir, i}] = 1;
123 return i;
124 }
125 return -1;
126 }
127};

132struct AIEObjectFifoStatefulTransformPass
133 : AIEObjectFifoStatefulTransformBase<AIEObjectFifoStatefulTransformPass> {
134 DenseMap<ObjectFifoCreateOp, std::vector<BufferOp>>
135 buffersPerFifo; // maps each objFifo to its corresponding buffer
136 DenseMap<ObjectFifoCreateOp, std::vector<ExternalBufferOp>>
137 externalBuffersPerFifo; // maps each objFifo to its corresponding
138 // external buffers
139 DenseMap<ObjectFifoCreateOp, std::vector<LockOp>>
140 locksPerFifo; // maps each objFifo to its corresponding locks
141 std::vector<std::pair<ObjectFifoCreateOp, std::vector<ObjectFifoCreateOp>>>
142 splitFifos; // maps each objFifo between non-adjacent tiles to its
143 // corresponding consumer objectFifos
144 DenseMap<ObjectFifoLinkOp, ObjectFifoCreateOp>
145 objFifoLinks; // maps each ObjectFifoLinkOp to objFifo whose elements
146 // have been created and should be used
147 std::vector<ObjectFifoCreateOp>
148 splitBecauseLink; // objfifos which have been split because they are
149 // part of a Link, not because they didn't have a shared memory module
150
151 /// Function that returns true if two tiles in the AIE array share a memory
152 /// module. share_direction is equal to:
153 /// * -1 if the shared memory module is that of the first input tile,
154 /// * 1 if it is that of the second input tile,
155 /// * 0 is no memory module is shared.
156 bool isSharedMemory(TileOp a, TileOp b, int *share_direction) {
157 const auto &targetModel = getTargetModel(a.getOperation());
158
159 if ((a.isShimTile() && !b.isShimTile()) ||
160 (!a.isShimTile() && b.isShimTile())) {
161 *share_direction = 0;
162 return false;
163 }
164 if ((targetModel.isMemTile(a.getCol(), a.getRow()) &&
165 !targetModel.isMemTile(b.getCol(), b.getRow())) ||
166 (!targetModel.isMemTile(a.getCol(), a.getRow()) &&
167 targetModel.isMemTile(b.getCol(), b.getRow()))) {
168 *share_direction = 0;
169 return false;
170 }
171 bool rightShared = targetModel.isLegalMemAffinity(
172 a.colIndex(), a.rowIndex(), b.colIndex(), b.rowIndex());
173
174 bool leftShared = targetModel.isLegalMemAffinity(
175 b.colIndex(), b.rowIndex(), a.colIndex(), a.rowIndex());
176
177 if (leftShared)
178 *share_direction = -1;
179 else if (rightShared)
180 *share_direction = 1;
181 else
182 *share_direction = 0;
183
184 return leftShared || rightShared;
185 }
186
187 // Return true if the objectFifo created by createOp requires a DMA to be set
188 // up. This is the case if the tiles are not adjacent (no shared memory), if
189 // the objectFifo broadcasts to multiple tiles, if one of the consumers or
190 // the producer wants to use the multi-dimensional address generation
191 // features of the DMA, if the objectFifo is part of a LinkOp, or if the
192 // via_DMA or repeatCount attributes of the objectFifo are set.
193 bool requiresDMAs(ObjectFifoCreateOp createOp, int &share_direction) {
194 bool hasSharedMemory = false;
195 bool atLeastOneConsumerWantsTransform = false;
196 bool isUsedInLinkOp = false;
197
198 if (createOp.getVia_DMA())
199 return true;
200
201 if (createOp.getRepeatCount().has_value())
202 return true;
203
204 if (createOp.getConsumerTiles().size() == 1 &&
205 createOp.getDimensionsToStream().empty()) {
206
207 // Test for shared memory
208 for (auto consumerTile : createOp.getConsumerTiles()) {
209 if (auto consumerTileOp =
210 dyn_cast<TileOp>(consumerTile.getDefiningOp())) {
211 if (std::count(splitBecauseLink.begin(), splitBecauseLink.end(),
212 createOp))
213 hasSharedMemory =
214 isSharedMemory(createOp.getProducerTileOp(),
215 createOp.getProducerTileOp(), &share_direction);
216 else
217 hasSharedMemory = isSharedMemory(createOp.getProducerTileOp(),
218 consumerTileOp, &share_direction);
219 }
220 }
221 }
222
223 // Only test for use of data layout transformations if we are in the shared
224 // memory case; otherwise, we will return `true` in any case.
225 if (hasSharedMemory) {
226 // Even if just one of the consumers in the list of consumers wants to
227 // perform a memory transform, we need to use DMAs.
228 for (BDDimLayoutArrayAttr dims :
229 createOp.getDimensionsFromStreamPerConsumer())
230 if (!dims.empty()) {
231 atLeastOneConsumerWantsTransform = true;
232 break;
233 }
234 }
235
236 // Check if the objectfifo operation can use shared memory for linking. If
237 // the link operation is a distribute or a join operation, or if the link
238 // has different memref types, DMAs are required even if shared memory is
239 // available and the objectfifo should be split. Otherwise also check if the
240 // via_shared_memory attribute of the objectfifo operation is set and try to
241 // apply it.
242 if (hasSharedMemory) {
243 if (auto linkOp = getOptionalLinkOp(createOp)) {
244 isUsedInLinkOp = true;
245 int share_dir = 0;
246 if (!linkOp->isDistribute() && !linkOp->isJoin()) {
247 auto fifoInType = llvm::cast<AIEObjectFifoType>(
248 linkOp->getInputObjectFifos()[0].getElemType());
249 auto producerType =
250 llvm::cast<MemRefType>(fifoInType.getElementType());
251 auto fifoOutType = llvm::cast<AIEObjectFifoType>(
252 linkOp->getOutputObjectFifos()[0].getElemType());
253 auto consumerType =
254 llvm::cast<MemRefType>(fifoOutType.getElementType());
255 if (consumerType != producerType) {
256 // TODO: Support for different memref types through shared
257 // memory without DMAs
258 splitBecauseLink.push_back(createOp);
259 }
260 if (createOp.getViaSharedMem().has_value()) {
261 checkAndApplyViaSharedMemAttribute(createOp, share_dir);
262 if (share_direction == share_dir)
263 isUsedInLinkOp = false;
264 else
265 splitBecauseLink.push_back(createOp);
266 }
267 } else {
268 splitBecauseLink.push_back(createOp);
269 }
270 }
271 }
272
273 return !hasSharedMemory || atLeastOneConsumerWantsTransform ||
274 isUsedInLinkOp;
275 }

277 // Checks if via_shared_mem attribute of the objectfifo is set and if so
278 // tries to apply it. If the desired shared memory module is available to
279 // both producer and consumer then it will be used, otherwise an error is
280 // emitted.
281 void checkAndApplyViaSharedMemAttribute(ObjectFifoCreateOp createOp,
282 int &share_direction) {
283 if (createOp.getViaSharedMem().has_value()) {
284 int desiredSharedTile = createOp.getViaSharedMem().value();
285 int desiredSharedModule = 1;
286 if (desiredSharedTile == 0)
287 desiredSharedModule = -1;
288 if (share_direction != desiredSharedModule) {
289 bool desiredSharedModuleIsShared = false;
290 int newShareDirection = 0;
291 for (auto consumerTile : createOp.getConsumerTiles()) {
292 if (auto consumerTileOp =
293 dyn_cast<TileOp>(consumerTile.getDefiningOp()))
294 if (share_direction == -1)
295 /// * -1 if the shared memory module is that of the first input
296 /// tile,
297 /// * 1 if it is that of the second input tile
298 desiredSharedModuleIsShared =
299 isSharedMemory(consumerTileOp, createOp.getProducerTileOp(),
300 &newShareDirection);
301 }
302 if (desiredSharedModuleIsShared) {
303 if (share_direction == newShareDirection)
304 share_direction = (share_direction == -1) ? 1 : -1;
305 else
306 createOp->emitOpError(
307 "no access to shared memory module specified by "
308 "`via_shared_mem`");
309 }
310 }
311 }
312 }

341 /// Function used to create objectFifo locks based on target architecture.
342 /// Called by createObjectFifoElements().
343 std::vector<LockOp> createObjectFifoLocks(OpBuilder &builder,
344 LockAnalysis &lockAnalysis,
345 ObjectFifoCreateOp op, int numElem,
346 int joinDistribFactor,
347 TileOp creation_tile,
348 int repeatCount) {
349 std::vector<LockOp> locks;
350 if (op.getDisableSynchronization())
351 return locks;
352 auto dev = op->getParentOfType<DeviceOp>();
353 auto &target = dev.getTargetModel();
354 // if shimTile external buffers are collected from input code
355 // create as many locks as there are external buffers
356 if (creation_tile.isShimTile()) {
357 numElem = 1;
358 if (!externalBuffersPerFifo[op].empty())
359 numElem = externalBuffersPerFifo[op].size();
360 }
361 if (target.getTargetArch() == AIEArch::AIE1) {
362 int of_elem_index = 0; // used to give objectFifo elements a symbolic name
363 for (int i = 0; i < numElem; i++) {
364 // create corresponding aie1 locks
365 int initValue = op.getInitValues().has_value() ? 1 : 0;
366 int lockID = lockAnalysis.getLockID(creation_tile);
367 assert(lockID >= 0 && "No more locks to allocate!");
368 auto lock = builder.create<LockOp>(builder.getUnknownLoc(),
369 creation_tile, lockID, initValue);
370 lock.getOperation()->setAttr(
371 SymbolTable::getSymbolAttrName(),
372 builder.getStringAttr(op.name().str() + "_lock_" +
373 std::to_string(of_elem_index)));
374 locks.push_back(lock);
375 of_elem_index++;
376 }
377 } else {
378 // create corresponding aie2 locks
379 auto initValues = op.getInitValues().has_value()
380 ? op.getInitValues().value().size()
381 : 0;
382 int prodLockID = lockAnalysis.getLockID(creation_tile);
383 assert(prodLockID >= 0 && "No more locks to allocate!");
384 int prodLockValue =
385 (numElem - initValues) * joinDistribFactor * repeatCount;
386 auto prodLock = builder.create<LockOp>(
387 builder.getUnknownLoc(), creation_tile, prodLockID, prodLockValue);
388 prodLock.getOperation()->setAttr(
389 SymbolTable::getSymbolAttrName(),
390 builder.getStringAttr(op.name().str() + "_prod_lock"));
391 locks.push_back(prodLock);
392
393 int consLockID = lockAnalysis.getLockID(creation_tile);
394 assert(consLockID >= 0 && "No more locks to allocate!");
395 int consLockValue = initValues * joinDistribFactor * repeatCount;
396 auto consLock = builder.create<LockOp>(
397 builder.getUnknownLoc(), creation_tile, consLockID, consLockValue);
398 consLock.getOperation()->setAttr(
399 SymbolTable::getSymbolAttrName(),
400 builder.getStringAttr(op.name().str() + "_cons_lock"));
401 locks.push_back(consLock);
402 }
403 return locks;
404 }

406 /// Function used to create objectFifo elements and their locks.
407 /// It maps the input objectFifo to associated buffers and locks.
408 void createObjectFifoElements(OpBuilder &builder, LockAnalysis &lockAnalysis,
409 ObjectFifoCreateOp op, int share_direction) {
410 if (!op.size())
411 return;
412
413 std::vector<BufferOp> buffers;
414 auto fifo = llvm::cast<AIEObjectFifoType>(op.getElemType());
415 auto elemType = llvm::cast<MemRefType>(fifo.getElementType());
416 int numElem = op.size();
417 int of_elem_index = 0; // used to give objectFifo elements a symbolic name
418
419 // if this objectFifo is linked to another, check if the other's elements
420 // have already been created: if none of the output objectfifos of the link
421 // have initValues, then the elements that are created are those of the
422 // objFifo with elements of bigger size
423 bool linked = false;
424 auto linkOp = getOptionalLinkOp(op);
425 if (linkOp) {
426 auto fifoIn = linkOp->getInputObjectFifos()[0];
427 auto fifoOut = linkOp->getOutputObjectFifos()[0];
428 linked = true;
429 if (objFifoLinks.find(*linkOp) != objFifoLinks.end())
430 return; // elements have already been created
431 if (linkOp->isJoin()) {
432 // if join, fifoOut has bigger size
433 if (op.name() != fifoOut.name())
434 return;
435 } else if (linkOp->isDistribute()) {
436 // if distribute, fifoIn has bigger size
437 if (op.name() != fifoIn.name())
438 return;
439 } else {
440 // check if output objectfifo has initValues
441 if (fifoOut.getInitValues().has_value()) {
442 if (fifoOut.name() != op.name())
443 return;
444 } else {
445 // check which objectfifo of the link has bigger size
446 auto fifoInType = llvm::cast<AIEObjectFifoType>(fifoIn.getElemType());
447 auto elemInType = llvm::cast<MemRefType>(fifoInType.getElementType());
448 int inSize = elemInType.getNumElements();
449
450 auto fifoOutType =
451 llvm::cast<AIEObjectFifoType>(fifoOut.getElemType());
452 auto elemOutType =
453 llvm::cast<MemRefType>(fifoOutType.getElementType());
454
455 if (int outSize = elemOutType.getNumElements(); inSize >= outSize) {
456 if (op.name() != fifoIn.name())
457 return;
458 } else {
459 if (fifoOut.name() != op.name())
460 return;
461 }
462 }
463 }
464 }
465
466 TileOp creation_tile;
467 if (share_direction == 0 || share_direction == -1)
468 creation_tile = op.getProducerTileOp();
469 else {
470 auto consumerTileOp =
471 dyn_cast<TileOp>(op.getConsumerTiles()[0].getDefiningOp());
472 creation_tile = consumerTileOp;
473 }
474
475 // Reset opbuilder location to after the last tile declaration
476 Operation *t = nullptr;
477 auto dev = op->getParentOfType<DeviceOp>();
478 for (auto tile_op : dev.getBody()->getOps<TileOp>()) {
479 t = tile_op.getOperation();
480 }
481 builder.setInsertionPointAfter(t);
482 for (int i = 0; i < numElem; i++) {
483 mlir::ElementsAttr initValues = nullptr;
484 if (!creation_tile.isShimTile()) {
485 if (op.getInitValues().has_value()) {
486 initValues =
487 llvm::cast<mlir::ElementsAttr>(op.getInitValues().value()[i]);
488 }
489 auto buff = builder.create<BufferOp>(
490 builder.getUnknownLoc(), elemType, creation_tile,
491 builder.getStringAttr(op.name().str() + "_buff_" +
492 std::to_string(of_elem_index)),
493 /*address*/ nullptr, initValues,
494 /*mem_bank*/ nullptr);
495 buffers.push_back(buff);
496 }
497 of_elem_index++;
498 }
499 int repeatCount = 1;
500 int joinDistribFactor = 1;
501 if (op.getRepeatCount().has_value())
502 repeatCount = op.getRepeatCount().value();
503 if (linked) {
504 if (linkOp->getRepeatCount().has_value())
505 repeatCount = linkOp->getRepeatCount().value();
506 if (linkOp->isDistribute())
507 joinDistribFactor *= linkOp->getFifoOuts().size();
508 else if (linkOp->isJoin())
509 joinDistribFactor *= linkOp->getFifoIns().size();
510 objFifoLinks[*linkOp] = op;
511 }
512 std::vector<LockOp> locks = createObjectFifoLocks(
513 builder, lockAnalysis, op, numElem, joinDistribFactor, creation_tile, repeatCount);
514 buffersPerFifo[op] = buffers;
515 locksPerFifo[op] = locks;
516 }

528 /// Function used to create a Bd block.
529 template <typename MyOp>
530 void createBd(OpBuilder &builder, LockOp acqLock, int acqMode,
531 LockAction acqLockAction, LockOp relLock, int relMode,
532 MyOp buff, int offset, int len, Block *succ,
533 BDDimLayoutArrayAttr dims, BDPadLayoutArrayAttr padDimensions) {
534 if (acqLock)
535 builder.create<UseLockOp>(builder.getUnknownLoc(), acqLock, acqLockAction,
536 acqMode);
537
538 if (!dims.getValue().empty() && padDimensions) {
539 builder.create<DMABDOp>(builder.getUnknownLoc(), buff, offset, len, dims,
540 padDimensions);
541 } else if (!dims.getValue().empty()) {
542 builder.create<DMABDOp>(builder.getUnknownLoc(), buff, offset, len, dims);
543 } else {
544 builder.create<DMABDOp>(builder.getUnknownLoc(), buff, offset, len);
545 }
546 if (acqLock)
547 builder.create<UseLockOp>(builder.getUnknownLoc(), relLock,
548 LockAction::Release, relMode);
549 builder.create<NextBDOp>(builder.getUnknownLoc(), succ);
550 }

552 /// Function used to create a Bd block.
553 /// If lockMode is 0 we create a consumerDMA (i.e. on producer tile) else a
554 /// producerDMA (i.e. on consumer tile).
555 template <typename MyOp>
556 void createBdBlock(OpBuilder &builder, ObjectFifoCreateOp op, int lockMode,
557 int acqNum, int relNum, MyOp buff, int offset, int len,
558 DMAChannelDir channelDir, size_t blockIndex, Block *succ,
559 BDDimLayoutArrayAttr dims,
560 BDPadLayoutArrayAttr padDimensions) {
561 LockOp acqLock;
562 LockOp relLock;
563 int acqMode = 1;
564 int relMode = 1;
565 auto acqLockAction = LockAction::Acquire;
566 if (locksPerFifo[op].size() > 0) {
567 auto dev = op->getParentOfType<DeviceOp>();
568 if (auto &target = dev.getTargetModel();
569 target.getTargetArch() == AIEArch::AIE1) {
570 acqMode = lockMode == 0 ? 1 : 0;
571 relMode = lockMode == 0 ? 0 : 1;
572 acqLock = locksPerFifo[op][blockIndex];
573 relLock = locksPerFifo[op][blockIndex];
574 } else {
575 acqMode = acqNum;
576 relMode = relNum;
577 acqLockAction = LockAction::AcquireGreaterEqual;
578 acqLock = channelDir == DMAChannelDir::S2MM ? locksPerFifo[op][0]
579 : locksPerFifo[op][1];
580 relLock = channelDir == DMAChannelDir::S2MM ? locksPerFifo[op][1]
581 : locksPerFifo[op][0];
582 }
583 }
584 createBd(builder, acqLock, acqMode, acqLockAction, relLock, relMode, buff,
585 offset, len, succ, dims, padDimensions);
586 }

608 /// Function used to create a MemOp region with a DMA channel.
609 /// It uses creatBdBlock(), see there for lockMode input.
610 void createAIETileDMA(DeviceOp &device, OpBuilder &builder,
611 ObjectFifoCreateOp op, DMAChannelDir channelDir,
612 int channelIndex, int lockMode,
613 BDDimLayoutArrayAttr dims) {
614 size_t numBlocks = op.size();
615 if (numBlocks == 0)
616 return;
617
618 int acqNum = 1;
619 int relNum = 1;
620
621 auto fifo = llvm::cast<AIEObjectFifoType>(op.getElemType());
622 auto elemType = llvm::cast<MemRefType>(fifo.getElementType());
623 int len = elemType.getNumElements();
624
625 // check for repeat count
626 int repeatCount = 1;
627 if (op.getRepeatCount().has_value())
628 repeatCount = op.getRepeatCount().value();
629
630 // search for the buffers/locks (based on if this objFifo has a link)
631 ObjectFifoCreateOp target = op;
632 if (std::optional<ObjectFifoLinkOp> linkOp = getOptionalLinkOp(op);
633 linkOp.has_value()) {
634 if (objFifoLinks.find(linkOp.value()) != objFifoLinks.end()) {
635 target = objFifoLinks[linkOp.value()];
636 if (target == op) {
637 if (linkOp->getRepeatCount().has_value()) {
638 acqNum *= linkOp->getRepeatCount().value();
639 relNum *= linkOp->getRepeatCount().value();
640 }
641 }
642 }
643 }
644
645 // search for MemOp
646 Operation *producerMem = nullptr;
647 for (auto memOp : device.getOps<MemOp>()) {
648 if (memOp.getTile() == op.getProducerTile()) {
649 producerMem = memOp.getOperation();
650 break;
651 }
652 }
653
654 // if none exists, create one
655 TileOp objFifoTileOp = target.getProducerTileOp();
656 if (producerMem == nullptr) {
657 OpBuilder::InsertionGuard g(builder);
658 builder.setInsertionPoint(device.getBody()->getTerminator());
659 auto newMemOp =
660 builder.create<MemOp>(builder.getUnknownLoc(), objFifoTileOp);
661 {
662 OpBuilder::InsertionGuard g(builder);
663 builder.setInsertionPointToStart(&newMemOp.getRegion().emplaceBlock());
664 builder.create<EndOp>(builder.getUnknownLoc());
665 }
666 producerMem = newMemOp.getOperation();
667 }
668 Block *endBlock = findEndOpBlock(producerMem->getRegion(0));
669 Block *lastDmaBlock = endBlock->getSinglePredecessor();
670 Block *dmaBlock = builder.createBlock(endBlock);
671 Block *bdBlock = builder.createBlock(endBlock);
672
673 // create DMA channel
674 builder.setInsertionPointToStart(dmaBlock);
675 builder.create<DMAStartOp>(builder.getUnknownLoc(), channelDir,
676 channelIndex, repeatCount - 1, bdBlock,
677 endBlock);
678 if (lastDmaBlock != nullptr)
679 lastDmaBlock->getTerminator()->setSuccessor(dmaBlock, 1);
680
681 // create Bd blocks
682 Block *succ;
683 Block *curr = bdBlock;
684 size_t blockIndex = 0;
685 for (size_t i = 0; i < numBlocks; i++) {
686 if (blockIndex >= buffersPerFifo[target].size())
687 break;
688 if (i == numBlocks - 1)
689 succ = bdBlock;
690 else
691 succ = builder.createBlock(endBlock);
692
693 builder.setInsertionPointToStart(curr);
694 createBdBlock<BufferOp>(builder, target, lockMode, acqNum, relNum,
695 buffersPerFifo[target][blockIndex], /*offset*/ 0,
696 len, channelDir, blockIndex, succ, dims, nullptr);
697 curr = succ;
698 blockIndex++;
699 }

776 /// Function used to create a MemTileDMAOp region with a DMA channel.
777 /// It uses creatBdBlock(), see there for lockMode input.
778 void createMemTileDMA(DeviceOp &device, OpBuilder &builder,
779 ObjectFifoCreateOp op, DMAChannelDir channelDir,
780 int channelIndex, int lockMode,
781 BDDimLayoutArrayAttr dims,
782 BDPadLayoutArrayAttr padDimensions) {
783 size_t numBlocks = op.size();
784 if (numBlocks == 0)
785 return;
786
787 auto fifo = llvm::cast<AIEObjectFifoType>(op.getElemType());
788 auto elemType = llvm::cast<MemRefType>(fifo.getElementType());
789 int lenOut = elemType.getNumElements();
790 int acqNum = 1;
791 int relNum = 1;
792
793 // check for repeat count
794 int repeatCount = 1;
795 if (op.getRepeatCount().has_value())
796 repeatCount = op.getRepeatCount().value();
797
798 // search for the buffers/locks (based on if this objFifo has a link)
799 // identify size difference between input and output memrefs
800 ObjectFifoCreateOp target = op;
801 bool isDistribute = false;
802 bool isJoin = false;
803 int extraOffset = 0;
804 if (auto linkOp = getOptionalLinkOp(op)) {
805 if (objFifoLinks.find(*linkOp) != objFifoLinks.end()) {
806 target = objFifoLinks[*linkOp];
807 auto srcOffsets = linkOp->getSrcOffsets();
808 auto dstOffsets = linkOp->getDstOffsets();
809
810 if (linkOp->getRepeatCount().has_value()) {
811 if (linkOp->getInputObjectFifos()[0] == op) {
812 acqNum *= linkOp->getRepeatCount().value();
813 relNum *= linkOp->getRepeatCount().value();
814 }
815 }
816
817 if (linkOp->isJoin()) {
818 // compute offset and length
819 isJoin = true;
820 if (target == op) {
821 acqNum *= linkOp->getFifoIns().size();
822 relNum *= linkOp->getFifoIns().size();
823 } else {
824 int i = 0;
825 for (auto fifoIn : linkOp->getInputObjectFifos()) {
826 if (fifoIn.name() == op.name())
827 break;
828 i++;
829 }
830 extraOffset = *getConstantIntValue(srcOffsets[i]);
831 lenOut = linkOp->getJoinTransferLengths()[i];
832 }
833 } else if (linkOp->isDistribute()) {
834 // compute offset and length
835 isDistribute = true;
836 if (target == op) {
837 acqNum *= linkOp->getFifoOuts().size();
838 relNum *= linkOp->getFifoOuts().size();
839 } else {
840 int i = 0;
841 for (auto fifoOut : linkOp->getOutputObjectFifos()) {
842 if (fifoOut.name() == op.name())
843 break;
844 i++;
845 }
846 extraOffset = *getConstantIntValue(dstOffsets[i]);
847 lenOut = linkOp->getDistributeTransferLengths()[i];
848 }
849 } else {
850 if (target != op) {
851 auto targetFifo =
852 llvm::cast<AIEObjectFifoType>(target.getElemType());
853 auto targetElemType =
854 llvm::cast<MemRefType>(targetFifo.getElementType());
855 lenOut = targetElemType.getNumElements();
856 }
857 }
858
859 // check if current op is of smaller size in link
860 if (target != op)
861 numBlocks = target.size();
862 }
863 }
864
865 // search for MemTileDMAOp
866 Operation *producerDMA = nullptr;
867 for (auto dmaOp : device.getOps<MemTileDMAOp>()) {
868 if (dmaOp.getTile() == target.getProducerTile()) {
869 producerDMA = dmaOp.getOperation();
870 break;
871 }
872 }
873
874 // if none exists, create one
875 TileOp objFifoTileOp = target.getProducerTileOp();
876 if (producerDMA == nullptr) {
877 OpBuilder::InsertionGuard g(builder);
878 builder.setInsertionPoint(device.getBody()->getTerminator());
879 auto newDMAOp =
880 builder.create<MemTileDMAOp>(builder.getUnknownLoc(), objFifoTileOp);
881 {
882 OpBuilder::InsertionGuard g(builder);
883 builder.setInsertionPointToStart(&newDMAOp.getRegion().emplaceBlock());
884 builder.create<EndOp>(builder.getUnknownLoc());
885 }
886 producerDMA = newDMAOp.getOperation();
887 }
888
889 Block *endBlock = findEndOpBlock(producerDMA->getRegion(0));
890 Block *lastDmaBlock = endBlock->getSinglePredecessor();
891 Block *dmaBlock = builder.createBlock(endBlock);
892 Block *bdBlock = builder.createBlock(endBlock);
893
894 // create DMA channel
895 builder.setInsertionPointToStart(dmaBlock);
896 builder.create<DMAStartOp>(builder.getUnknownLoc(), channelDir,
897 channelIndex, repeatCount - 1, bdBlock,
898 endBlock);
899 if (lastDmaBlock != nullptr)
900 lastDmaBlock->getTerminator()->setSuccessor(dmaBlock, 1);
901
902 // create Bd blocks
903 Block *succ;
904 Block *curr = bdBlock;
905 size_t blockIndex = 0;
906 for (size_t i = 0; i < numBlocks; i++) {
907 if (blockIndex >= buffersPerFifo[target].size())
908 break;
909 if (i == numBlocks - 1)
910 succ = bdBlock;
911 else
912 succ = builder.createBlock(endBlock);
913
914 builder.setInsertionPointToStart(curr);
915 int offset = 0;
916 if (isDistribute || isJoin)
917 offset = extraOffset;
918 createBdBlock<BufferOp>(builder, target, lockMode, acqNum, relNum,
919 buffersPerFifo[target][blockIndex], offset,
920 lenOut, channelDir, blockIndex, succ, dims,
921 padDimensions);
922 curr = succ;
923 blockIndex++;
924 }
925 }

936 // Function that unrolls for-loops that contain objectFifo operations.
937 LogicalResult unrollForLoops(DeviceOp &device, OpBuilder &builder,
938 std::set<TileOp> objectFifoTiles) {
939 for (auto coreOp : device.getOps<CoreOp>()) {
940 if (objectFifoTiles.count(coreOp.getTileOp()) > 0) {
941 std::vector<scf::ForOp> unrolledLoops;
942 std::map<Operation *, bool> foundMap;
943 std::map<Operation *, int64_t> remainderMap;
944 std::map<Operation *, int64_t> tripCountMap;
945 WalkResult res = coreOp.walk([&](scf::ForOp forLoop) {
946 // look for operations on objectFifos
947 // when multiple fifos in same loop, must use the smallest
948 // common multiplier as the unroll factor
949 foundMap[forLoop.getOperation()] = false;
950 std::set<int> objFifoSizes;
951 Block *body = forLoop.getBody();
952 remainderMap[forLoop.getOperation()] = 0;
953 for (auto acqOp : body->getOps<ObjectFifoAcquireOp>()) {
954 if (acqOp.getOperation()->getParentOp() == forLoop) {
955 foundMap[forLoop.getOperation()] = true;
956 ObjectFifoCreateOp op = acqOp.getObjectFifo();
957 objFifoSizes.insert(op.size());
958 }
959 }
960 // If the loop doesn't have acquire and release locks
961 // Push it to the unrolledLoops to avoid unrolling
962 if (!foundMap[forLoop.getOperation()]) {
963 unrolledLoops.push_back(forLoop);
964 return WalkResult::advance();
965 }
966 // Walk in the loop region to unroll the loop and its remainder
967 Region *region = forLoop->getParentRegion();
968 scf::ForOp prevLoop;
969 prevLoop = forLoop;
970 tripCountMap[prevLoop.getOperation()] = 0;
971 while (remainderMap[prevLoop.getOperation()] > 1 ||
972 foundMap[prevLoop.getOperation()]) {
973 region->walk([&](scf::ForOp remLoop) {
974 bool skipLoop = false;
975 int64_t tripCount = 0;
976 if (remLoop.getSingleLowerBound() &&
977 remLoop.getSingleUpperBound() && remLoop.getSingleStep()) {
978 tripCount = constantTripCount(*(remLoop.getSingleLowerBound()),
979 *(remLoop.getSingleUpperBound()),
980 *(remLoop.getSingleStep()))
981 .value_or(0);
982 }
983 int unrollFactor =
984 computeLCM(objFifoSizes); // also counts original loop body
985 // Loop ids are not unique.
986 // Sometimes, immediately after unrolling, the unrolled loop
987 // and the one next to it (can be the remainder loop or an
988 // independent loop) will have the same ID. This makes it
989 // difficult to identify which loop needs to be unrolled.
990 // Once it restarts walking from start, it ends up allocating
991 // new ID to each loop.
992 if (remainderMap[prevLoop.getOperation()] > 1 &&
993 foundMap[remLoop.getOperation()] == false &&
994 prevLoop != remLoop) {
995 skipLoop = true;
996 }
997 if (std::count(unrolledLoops.begin(), unrolledLoops.end(),
998 remLoop) == 0 &&
999 !skipLoop) {
1000 tripCountMap[remLoop.getOperation()] = tripCount;
1001 // if loop iterations < unrollFactor, unroll the loop fully
1002 if (tripCountMap[remLoop.getOperation()] < unrollFactor)
1003 unrollFactor = tripCountMap[remLoop.getOperation()];
1004 // If unrollFactor = 0,divide by zero
1005 if (unrollFactor == 0) {
1006 remLoop.emitOpError()
1007 << "could not be unrolled with unrollFactor = 0, check "
1008 "loop boundaries."
1009 << "\n";
1010 return WalkResult::interrupt();
1011 }
1012 remainderMap[remLoop.getOperation()] =
1013 tripCountMap[remLoop.getOperation()] % unrollFactor;
1014 auto step = remLoop.getStep()
1015 .getDefiningOp<arith::ConstantOp>()
1016 .getValue();
1017 int64_t step_value = llvm::dyn_cast<IntegerAttr>(step).getInt();
1018
1019 if (step_value < unrollFactor ||
1020 foundMap[remLoop.getOperation()]) {
1021 // Process the for loop
1022 if (failed(mlir::loopUnrollByFactor(remLoop, unrollFactor))) {
1023 remLoop.emitOpError()
1024 << "could not be unrolled with unrollFactor: "
1025 << unrollFactor << "\n";
1026 return WalkResult::interrupt();
1027 }
1028 unrolledLoops.push_back(remLoop);
1029 foundMap[remLoop.getOperation()] = false;
1030 } else {
1031 remainderMap[remLoop.getOperation()] = 0;
1032 foundMap[remLoop.getOperation()] = false;
1033 }
1034 } else {
1035 remainderMap[remLoop.getOperation()] = 0;
1036 foundMap[remLoop.getOperation()] = false;
1037 }
1038 prevLoop = remLoop;
1039 return WalkResult::advance();
1040 });
1041 }
1042 return WalkResult::advance();
1043 });
1044 if (res.wasInterrupted())
1045 return failure();
1046 }
1047 }
1048 return success();
1049 }

1051 // Function that generates the IR to update runtime state of objectfifo
1052 // accesses. Called by dynamicGlobalObjectFifos().
1053 void updateGlobalNextIndex(OpBuilder &builder, ObjectFifoReleaseOp relOp,
1054 BufferOp globalNextIndex, arith::ConstantOp index,
1055 arith::ConstantOp size) {
1056 builder.setInsertionPointAfter(relOp);
1057 Value oldCounter = builder.create<memref::LoadOp>(
1058 builder.getUnknownLoc(), globalNextIndex,
1059 ValueRange(ArrayRef({index.getResult()})));
1060 Value val = builder.create<arith::ConstantOp>(
1061 oldCounter.getLoc(), builder.getI32IntegerAttr(relOp.getSize()));
1062 Value sum = builder.create<arith::AddIOp>(val.getLoc(), oldCounter, val);
1063 Value isGreaterEqual = builder.create<arith::CmpIOp>(
1064 sum.getLoc(), arith::CmpIPredicate::sge, sum, size);
1065 Value newCounter = builder.create<arith::SelectOp>(
1066 sum.getLoc(), isGreaterEqual,
1067 builder.create<arith::SubIOp>(sum.getLoc(), sum, size), sum);
1068 builder.create<memref::StoreOp>(size.getLoc(), newCounter, globalNextIndex,
1069 ValueRange(ArrayRef({index.getResult()})));
1070 }

1072 // Function that generates the IR for objectfifo accesses to be handled at
1073 // runtime.
1074 LogicalResult dynamicGlobalObjectFifos(DeviceOp &device, OpBuilder &builder,
1075 std::set<TileOp> objectFifoTiles) {
1076 for (auto coreOp : device.getOps<CoreOp>()) {
1077 if (objectFifoTiles.count(coreOp.getTileOp()) <= 0)
1078 continue;
1079 if (objectFifoTiles.count(coreOp.getTileOp()) > 0) {
1080 // For each core: count the number of objectFifos and create
1081 // a global buffer just before the core to track index of
1082 // next object to access.
1083 // !! NOTE !! objectFifos with same producer / consumer tile
1084 // need two counters (accessed based on the ObjectFifoPort)
1085 std::map<std::pair<ObjectFifoCreateOp, ObjectFifoPort>, int> fifoSizes;
1086 // Also, keep a map of the ConstantOps for the indices per OF
1087 // and a map with the ConstantOps for the sizes per OF.
1088 std::map<std::pair<ObjectFifoCreateOp, ObjectFifoPort>,
1089 arith::ConstantOp>
1090 globalIndices;
1091 std::map<std::pair<ObjectFifoCreateOp, ObjectFifoPort>,
1092 arith::ConstantOp>
1093 constantSizes;
1094
1095 int index = 0;
1096 builder.setInsertionPointToStart(&(coreOp.getBody().front()));
1097 Value initVal = builder.create<arith::ConstantOp>(
1098 builder.getUnknownLoc(), builder.getI32IntegerAttr(0));
1099 coreOp.walk([&](ObjectFifoAcquireOp acqOp) {
1100 ObjectFifoCreateOp op = acqOp.getObjectFifo();
1101 ObjectFifoPort port = acqOp.getPort();
1102 if (fifoSizes.find({op, port}) == fifoSizes.end()) {
1103 fifoSizes[{op, port}] = op.size();
1104 auto indexOp = builder.create<arith::ConstantOp>(
1105 initVal.getLoc(), builder.getIndexAttr(index));
1106 globalIndices[{op, port}] = indexOp;
1107 index++;
1108 auto size = builder.create<arith::ConstantOp>(
1109 indexOp.getLoc(), builder.getI32IntegerAttr(op.size()));
1110 constantSizes[{op, port}] = size;
1111 }
1112 });
1113 builder.setInsertionPoint(coreOp);
1114 auto memrefTy =
1115 MemRefType::get(SmallVector<int64_t>{(int64_t)fifoSizes.size()},
1116 builder.getI32Type());
1117 auto globalNextIndex = builder.create<BufferOp>(
1118 builder.getUnknownLoc(), memrefTy, coreOp.getTile(),
1119 /*sym_name*/ nullptr, /*address*/ nullptr,
1120 /*initial_value*/ nullptr, /*mem_bank*/ nullptr);
1121
1122 // Initialize all counters in the global buffers to 0.
1123 for (auto i : constantSizes) {
1124 builder.setInsertionPointAfter(i.second);
1125 builder.create<memref::StoreOp>(
1126 builder.getUnknownLoc(), initVal, globalNextIndex,
1127 ValueRange(ArrayRef({globalIndices[i.first].getResult()})));
1128 }
1129
1130 // Walk the code:
1131 // - after each ObjectFifoReleaseOp:
1132 // - globalNextIndex: add #rel modulo objfifo depth
1133 // - before each ObjectFifoAcquireOp:
1134 // - globalNextIndex: load index and use it to index_switch (one
1135 // IndexSwithOp per AccessOp)
1136 WalkResult res = coreOp.walk([&](Operation *op) {
1137 if (auto relOp = dyn_cast<ObjectFifoReleaseOp>(op)) {
1138 ObjectFifoCreateOp createOp = relOp.getObjectFifo();
1139 ObjectFifoPort port = relOp.getPort();
1140 updateGlobalNextIndex(builder, relOp, globalNextIndex,
1141 globalIndices[{createOp, port}],
1142 constantSizes[{createOp, port}]);
1143 }
1144 if (auto acqOp = dyn_cast<ObjectFifoAcquireOp>(op)) {
1145 std::vector<ObjectFifoSubviewAccessOp> accessOps;
1146 for (auto u : acqOp->getUsers())
1147 if (auto accessOp = dyn_cast<ObjectFifoSubviewAccessOp>(u))
1148 accessOps.push_back(accessOp);
1149
1150 for (auto accessOp : accessOps) {
1151 ObjectFifoCreateOp createOp = acqOp.getObjectFifo();
1152 ObjectFifoPort port = acqOp.getPort();
1153
1154 // Single switch case
1155 if (fifoSizes[{createOp, port}] == 1)
1156 return WalkResult::advance();
1157
1158 // Create a switch for each subview access
1159 builder.setInsertionPointAfter(accessOp);
1160 auto switchIndexAsInteger = builder.create<memref::LoadOp>(
1161 builder.getUnknownLoc(), globalNextIndex,
1162 ValueRange(
1163 ArrayRef({globalIndices[{createOp, port}].getResult()})));
1164 auto switchIndex = builder.create<arith::IndexCastOp>(
1165 builder.getUnknownLoc(), builder.getIndexType(),
1166 switchIndexAsInteger);
1167 unsigned caseRegionCounts = fifoSizes[{createOp, port}];
1168 SmallVector<int64_t, 4> caseValues;
1169 for (int i = 0; i < fifoSizes[{createOp, port}]; ++i) {
1170 caseValues.push_back(i);
1171 }
1172 auto cases =
1173 DenseI64ArrayAttr::get(builder.getContext(), caseValues);
1174 auto switchOp = builder.create<scf::IndexSwitchOp>(
1175 switchIndex.getLoc(),
1176 TypeRange({buffersPerFifo[createOp][0].getType()}),
1177 switchIndex, cases, caseRegionCounts);
1178 // Create default case of IndexSwitchOp
1179 builder.createBlock(&switchOp.getDefaultRegion());
1180 auto bufferIndex = (accessOp.getIndex()) % createOp.size();
1181 builder.setInsertionPointToStart(&(switchOp.getDefaultBlock()));
1182 builder.create<scf::YieldOp>(
1183 builder.getUnknownLoc(),
1184 buffersPerFifo[createOp][bufferIndex].getResult());
1185 for (int i = 0; i < fifoSizes[{createOp, port}]; ++i) {
1186 // Create other cases of IndexSwitchOp
1187 builder.createBlock(&switchOp.getCaseRegions()[i]);
1188 builder.setInsertionPoint(&switchOp.getCaseBlock(i),
1189 switchOp.getCaseBlock(i).begin());
1190 int bufferToBeAccesed =
1191 (accessOp.getIndex() + i) % fifoSizes[{createOp, port}];
1192 builder.create<scf::YieldOp>(
1193 switchOp.getCaseRegions()[i].getLoc(),
1194 buffersPerFifo[createOp][bufferToBeAccesed].getResult());
1195 }
1196
1197 // Replace all uses of accessed objectfifo buffers with
1198 // results of switchOps
1199 accessOp.getOutput().replaceAllUsesWith(switchOp.getResult(0));
1200 }
1201 }
1202 return WalkResult::advance();
1203 });
1204 if (res.wasInterrupted())
1205 return failure();
1206 }
1207 }
1208 return success();
1209 }

1211 /// Function used to create a UseLockOp based on input parameters.
1212 /// acc is an accumulator map that tracks the indices of the next locks to
1213 /// acquire (or release). Uses op to find index of acc for next lockID.
1214 /// Updates acc.
1215 void createUseLocks(OpBuilder &builder, ObjectFifoCreateOp op,
1216 ObjectFifoPort port,
1217 DenseMap<std::pair<ObjectFifoCreateOp, int>, int> &acc,
1218 int numLocks, LockAction lockAction) {
1219 ObjectFifoCreateOp target = op;
1220 auto portNum = port == ObjectFifoPort::Produce ? 0 : 1;
1221 if (auto linkOp = getOptionalLinkOp(op))
1222 if (objFifoLinks.find(*linkOp) != objFifoLinks.end())
1223 target = objFifoLinks[*linkOp];
1224
1225 auto dev = op->getParentOfType<DeviceOp>();
1226 if (!dev.getTargetModel().hasProperty(AIETargetModel::UsesSemaphoreLocks)) {
1227
1228 if (locksPerFifo[target].size() == 0) {
1229 for (int i = 0; i < numLocks; i++) {
1230 int lockID = acc[{op, portNum}];
1231 acc[{op, portNum}] =
1232 (lockID + 1) % op.size(); // update to next objFifo elem
1233 }
1234 return;
1235 }
1236
1237 int lockMode = 0;
1238 if ((port == ObjectFifoPort::Produce &&
1239 lockAction == LockAction::Release) ||
1240 (port == ObjectFifoPort::Consume &&
1241 lockAction == LockAction::Acquire))
1242 lockMode = 1;
1243 for (int i = 0; i < numLocks; i++) {
1244 int lockID = acc[{op, portNum}];
1245 builder.create<UseLockOp>(builder.getUnknownLoc(),
1246 locksPerFifo[target][lockID], lockAction,
1247 lockMode);
1248 acc[{op, portNum}] =
1249 (lockID + 1) % op.size(); // update to next objFifo elem
1250 }
1251 } else {
1252 if (numLocks == 0)
1253 return;
1254
1255 if (locksPerFifo[target].size() == 0) {
1256 acc[{op, portNum}] = (acc[{op, portNum}] + numLocks) %
1257 op.size(); // update to next objFifo elem
1258 return;
1259 }
1260
1261 // search for the correct lock based on the port of the acq/rel
1262 // operation e.g. acq as consumer is the read lock (second)
1263 LockOp lock;
1264 if (lockAction == LockAction::AcquireGreaterEqual) {
1265 if (port == ObjectFifoPort::Produce)
1266 lock = locksPerFifo[target][0];
1267 else
1268 lock = locksPerFifo[target][1];
1269 } else {
1270 if (port == ObjectFifoPort::Produce)
1271 lock = locksPerFifo[target][1];
1272 else
1273 lock = locksPerFifo[target][0];
1274 }
1275 builder.create<UseLockOp>(builder.getUnknownLoc(), lock, lockAction,
1276 numLocks);
1277 acc[{op, portNum}] = (acc[{op, portNum}] + numLocks) %
1278 op.size(); // update to next objFifo elem
1279 }
1280 }

1330 /// Function used to find the size of an objectFifo after split based on
1331 /// the maximum number of elements (of the original objectFifo) acquired
1332 /// by a process running on given tile. If no CoreOp exists for this tile
1333 /// return 0.
1334 int findObjectFifoSize(DeviceOp &device, Value tile,
1335 ObjectFifoCreateOp objFifo) {
1336 if (objFifo.size() == 0)
1337 return 0;
1338
1339 // if memTile, size is equal to objFifo size
1340 if (tile.getDefiningOp<TileOp>().isMemTile())
1341 return objFifo.size();
1342
1343 // if shimTile, size is equal to number of external buffers
1344 if (tile.getDefiningOp<TileOp>().isShimTile())
1345 for (auto regOp : device.getOps<ObjectFifoRegisterExternalBuffersOp>()) {
1346 if (regOp.getTile() == tile)
1347 return regOp.getExternalBuffers().size();
1348 }
1349
1350 int maxAcquire = 0;
1351 for (auto coreOp : device.getOps<CoreOp>())
1352 if (coreOp.getTile() == tile)
1353 coreOp.walk([&](ObjectFifoAcquireOp acqOp) {
1354 if (auto createOp = acqOp.getObjectFifo(); createOp == objFifo)
1355 if (acqOp.acqNumber() > maxAcquire)
1356 maxAcquire = acqOp.acqNumber();
1357 });
1358
1359 if (maxAcquire > 0) {
1360 if (maxAcquire == 1 && objFifo.size() == 1)
1361 return 1;
1362 return maxAcquire + 1;
1363 // +1 because objectFifo size is always 1 bigger than maxAcquire to allow
1364 // for prefetching: simplest case scenario is at least a ping-pong buffer
1365 }
1366
1367 return objFifo.size();
1368 }

1404 void runOnOperation() override {
1405 DeviceOp device = getOperation();
1406 LockAnalysis lockAnalysis(device);
1407 DMAChannelAnalysis dmaAnalysis(device);
1408 OpBuilder builder = OpBuilder::atBlockTerminator(device.getBody());
1409 auto ctx = device->getContext();
1410 auto producerWireType = WireBundle::DMA;
1411 auto consumerWireType = WireBundle::DMA;
1412 std::set<TileOp>
1413 objectFifoTiles; // track cores to check for loops during unrolling
1414
1415 verifyObjectFifoLinks(device);
1416
1417 //===------------------------------------------------------------------===//
1418 // Split objectFifos into a consumer end and producer end if needed
1419 //===------------------------------------------------------------------===//
1420 // We are going to create additional createObjectFifoOps, so get a copy of
1421 // all "original" ones before the loop to avoid looping over newly created
1422 // ones.
1423 std::vector<ObjectFifoCreateOp> createFifoOps;
1424 auto range = device.getOps<ObjectFifoCreateOp>();
1425 createFifoOps.insert(createFifoOps.end(), range.begin(), range.end());
1426 for (auto createOp : createFifoOps) {
1427 std::vector<ObjectFifoCreateOp> splitConsumerFifos;
1428 int consumerIndex = 0;
1429 int consumerDepth = createOp.size();
1430 ArrayRef<BDDimLayoutArrayAttr> consumerDims =
1431 createOp.getDimensionsFromStreamPerConsumer();
1432
1433 // Only FIFOs using DMA are split into two ends;
1434 // skip in shared memory case
1435 if (int share_direction = 0; !requiresDMAs(createOp, share_direction)) {
1436 continue;
1437 }
1438
1439 for (auto consumerTile : createOp.getConsumerTiles()) {
1440 auto consumerTileOp = dyn_cast<TileOp>(consumerTile.getDefiningOp());
1441
1442 if (isa<ArrayAttr>(createOp.getElemNumber())) {
1443 // +1 to account for 1st depth (producer)
1444 consumerDepth = createOp.size(consumerIndex + 1);
1445 } else {
1446 consumerDepth = findObjectFifoSize(device, consumerTileOp, createOp);
1447 }
1448
1449 builder.setInsertionPointAfter(createOp);
1450 auto datatype = llvm::cast<AIEObjectFifoType>(createOp.getElemType());
1451 auto consumerObjFifoSize =
1452 builder.getIntegerAttr(builder.getI32Type(), consumerDepth);
1453 // rename and replace split objectFifo
1454 std::string consumerFifoName;
1455 if (createOp.getConsumerTiles().size() > 1) {
1456 consumerFifoName = createOp.name().str() + "_" +
1457 std::to_string(consumerIndex) + "_cons";
1458 } else {
1459 consumerFifoName = createOp.name().str() + "_cons";
1460 }
1461 BDDimLayoutArrayAttr emptyDims =
1462 BDDimLayoutArrayAttr::get(builder.getContext(), {});
1463 BDDimLayoutArrayAttr singletonFromStreamDims =
1464 BDDimLayoutArrayAttr::get(
1465 builder.getContext(),
1466 ArrayRef<BDDimLayoutAttr>{consumerDims[consumerIndex]});
1467 BDDimLayoutArrayArrayAttr fromStreamDims =
1468 BDDimLayoutArrayArrayAttr::get(builder.getContext(),
1469 singletonFromStreamDims);
1470
1471 ObjectFifoCreateOp consumerFifo = createObjectFifo(
1472 builder, datatype, consumerFifoName, consumerTile, consumerTile,
1473 consumerObjFifoSize, emptyDims, fromStreamDims);
1474 if (createOp.getDisableSynchronization())
1475 consumerFifo.setDisableSynchronization(true);
1476 replaceSplitFifo(createOp, consumerFifo, consumerTileOp);
1477
1478 // identify external buffers that were registered to the consumer fifo
1479 if (consumerTile.getDefiningOp<TileOp>().isShimTile())
1480 detectExternalBuffers(device, createOp, consumerFifo, consumerTile);
1481
1482 // record that this objectFifo was split; it will require DMA config
1483 splitConsumerFifos.push_back(consumerFifo);
1484
1485 // update the linkOp if the split objFifo was originally its start point
1486 if (auto linkOp = getOptionalLinkOp(createOp))
1487 for (ObjectFifoCreateOp fifoIn : linkOp->getInputObjectFifos())
1488 if (fifoIn.name() == createOp.name() &&
1489 consumerTile == *linkOp->getOptionalSharedTile())
1490 if (failed(SymbolTable::replaceAllSymbolUses(
1491 createOp, consumerFifo.name(), linkOp->getOperation())))
1492 llvm::report_fatal_error("unable to update all symbol uses");
1493
1494 consumerIndex++;
1495 }
1496
1497 if (!splitConsumerFifos.empty()) {
1498 splitFifos.emplace_back(createOp, splitConsumerFifos);
1499 }
1500 }
1501
1502 //===------------------------------------------------------------------===//
1503 // - Create objectFifo buffers and locks.
1504 // - Populate a list of tiles containing objectFifos for later processing of
1505 // the acquires/releases (uses of the FIFO).
1506 // - Global release counter tracker to keep track of the objectFifo state
1507 //===------------------------------------------------------------------===//
1508
1509 for (auto createOp : device.getOps<ObjectFifoCreateOp>()) {
1510 int share_direction = 0;
1511 bool shared = !requiresDMAs(createOp, share_direction);
1512
1513 // add all tiles that contain an objectFifo to objectFifoTiles for later
1514 // loop unrolling pass
1515 objectFifoTiles.insert(createOp.getProducerTileOp());
1516 for (auto consumerTile : createOp.getConsumerTiles()) {
1517 auto consumerTileOp = dyn_cast<TileOp>(consumerTile.getDefiningOp());
1518 objectFifoTiles.insert(consumerTileOp);
1519 }
1520
1521 // identify external buffers that were registered to
1522 // the producer objectFifo
1523 if (createOp.getProducerTileOp().isShimTile())
1524 detectExternalBuffers(device, createOp, createOp,
1525 createOp.getProducerTile());
1526
1527 // if split, the necessary size for producer fifo might change
1528 if (shared) {
1529 checkAndApplyViaSharedMemAttribute(createOp, share_direction);
1530 createObjectFifoElements(builder, lockAnalysis, createOp,
1531 share_direction);
1532 } else {
1533 if (createOp.getViaSharedMem().has_value())
1534 createOp->emitOpError(
1535 "no access to shared memory module specified by "
1536 "`via_shared_mem`");
1537
1538 if (isa<ArrayAttr>(createOp.getElemNumber()))
1539 createOp.setElemNumberAttr(
1540 builder.getI32IntegerAttr(createOp.size()));
1541 else {
1542 if (!createOp.getInitValues().has_value()) {
1543 int prodMaxAcquire = findObjectFifoSize(
1544 device, createOp.getProducerTileOp(), createOp);
1545 createOp.setElemNumberAttr(
1546 builder.getI32IntegerAttr(prodMaxAcquire));
1547 }
1548 }
1549 createObjectFifoElements(builder, lockAnalysis, createOp,
1550 share_direction);
1551 }
1552 }
1553
1554 //===------------------------------------------------------------------===//
1555 // Create flows and tile DMAs
1556 //===------------------------------------------------------------------===//
1557 // Only the objectFifos we split above require DMA communication; the others
1558 // rely on shared memory and share the same buffers.
1559 for (auto &[producer, consumers] : splitFifos) {
1560 // create producer tile DMA
1561 int producerChanIndex = dmaAnalysis.getDMAChannelIndex(
1562 producer.getProducerTileOp(), DMAChannelDir::MM2S);
1563 if (producerChanIndex == -1)
1564 producer.getProducerTileOp().emitOpError(
1565 "number of output DMA channel exceeded!");
1566 DMAChannel producerChan = {DMAChannelDir::MM2S, producerChanIndex};
1567 createDMA(device, builder, producer, producerChan.direction,
1568 producerChan.channel, 0, producer.getDimensionsToStreamAttr(),
1569 producer.getPadDimensionsAttr());
1570 // generate objectFifo allocation info
1571 builder.setInsertionPoint(device.getBody()->getTerminator());
1572
1573 if (producer.getProducerTileOp().isShimTile())
1574 createObjectFifoAllocationInfo(
1575 builder, ctx, SymbolRefAttr::get(ctx, producer.getName()),
1576 producer.getProducerTileOp().colIndex(), producerChan.direction,
1577 producerChan.channel, producer.getPlio());
1578
1579 for (auto consumer : consumers) {
1580
1581 // create consumer tile DMA
1582 int consumerChanIndex = dmaAnalysis.getDMAChannelIndex(
1583 consumer.getProducerTileOp(), DMAChannelDir::S2MM);
1584 if (consumerChanIndex == -1)
1585 consumer.getProducerTileOp().emitOpError(
1586 "number of input DMA channel exceeded!");
1587 DMAChannel consumerChan = {DMAChannelDir::S2MM, consumerChanIndex};
1588 BDDimLayoutArrayAttr consumerDims =
1589 consumer.getDimensionsFromStreamPerConsumer()[0];
1590 createDMA(device, builder, consumer, consumerChan.direction,
1591 consumerChan.channel, 1, consumerDims, nullptr);
1592 // generate objectFifo allocation info
1593 builder.setInsertionPoint(device.getBody()->getTerminator());
1594
1595 // If we have PLIO then figure out the direction and make that a PLIO
1596 if (producer.getPlio()) {
1597 producerWireType = producer.getProducerTileOp().isShimTile()
1598 ? WireBundle::PLIO
1599 : WireBundle::DMA;
1600 consumerWireType = consumer.getProducerTileOp().isShimTile()
1601 ? WireBundle::PLIO
1602 : WireBundle::DMA;
1603 } else {
1604 producerWireType = WireBundle::DMA;
1605 consumerWireType = WireBundle::DMA;
1606 }
1607
1608 if (consumer.getProducerTileOp().isShimTile())
1609 createObjectFifoAllocationInfo(
1610 builder, ctx, SymbolRefAttr::get(ctx, producer.getName()),
1611 consumer.getProducerTileOp().colIndex(), consumerChan.direction,
1612 consumerChan.channel, producer.getPlio());
1613
1614 // create flow
1615 builder.setInsertionPointAfter(producer);
1616 builder.create<FlowOp>(builder.getUnknownLoc(),
1617 producer.getProducerTile(), producerWireType,
1618 producerChan.channel, consumer.getProducerTile(),
1619 consumerWireType, consumerChan.channel);
1620 }
1621 }
1622
1623 //===------------------------------------------------------------------===//
1624 // Statically unroll for loops or use dynamic objectFifos
1625 //===------------------------------------------------------------------===//
1626 if (clDynamicObjectFifos) {
1627 if (failed(dynamicGlobalObjectFifos(device, builder, objectFifoTiles)))
1628 signalPassFailure();
1629 } else {
1630 std::set<TileOp> dynamicTiles;
1631 std::set<TileOp> unrollTiles;
1632 for (auto c : device.getOps<CoreOp>()) {
1633 TileOp t = c.getTileOp();
1634 if (objectFifoTiles.count(t) > 0) {
1635 if (c.getDynamicObjfifoLowering().has_value()) {
1636 if (c.getDynamicObjfifoLowering().value())
1637 dynamicTiles.insert(t);
1638 else
1639 unrollTiles.insert(t);
1640 } else {
1641 unrollTiles.insert(t);
1642 }
1643 }
1644 }
1645 if (failed(dynamicGlobalObjectFifos(device, builder, dynamicTiles)))
1646 signalPassFailure();
1647 if (failed(unrollForLoops(device, builder, unrollTiles)))
1648 signalPassFailure();
1649 }
1650
1651 //===------------------------------------------------------------------===//
1652 // Replace ops
1653 //===------------------------------------------------------------------===//
1654 for (auto coreOp : device.getOps<CoreOp>()) {
1655 DenseMap<ObjectFifoAcquireOp, std::vector<BufferOp *>>
1656 subviews; // maps each "subview" to its buffer references (subviews
1657 // are created by AcquireOps)
1658 DenseMap<std::pair<ObjectFifoCreateOp, int>, std::vector<int>>
1659 acquiresPerFifo; // maps each objFifo to indices of buffers acquired
1660 // in latest subview of that objFifo (useful to
1661 // cascade acquired elements to next AcquireOp)
1662 DenseMap<std::pair<ObjectFifoCreateOp, int>,
1663 std::vector<ObjectFifoReleaseOp>>
1664 releaseOps; // useful to check which ReleaseOp has taken place before
1665 // an AcquireOp per objFifo
1666 DenseMap<std::pair<ObjectFifoCreateOp, int>, int>
1667 acqPerFifo; // maps each objFifo to its next index to acquire within
1668 // this CoreOp
1669 DenseMap<std::pair<ObjectFifoCreateOp, int>, int>
1670 relPerFifo; // maps each objFifo to its next index to release within
1671 // this CoreOp
1672
1673 //===----------------------------------------------------------------===//
1674 // Replace objectFifo.release ops
1675 //===----------------------------------------------------------------===//
1676 coreOp.walk([&](ObjectFifoReleaseOp releaseOp) {
1677 builder.setInsertionPointAfter(releaseOp);
1678 ObjectFifoCreateOp op = releaseOp.getObjectFifo();
1679 auto port = releaseOp.getPort();
1680 auto portNum = port == ObjectFifoPort::Produce ? 0 : 1;
1681 auto core = releaseOp->getParentOfType<CoreOp>();
1682
1683 if (auto linkOp = getOptionalLinkOp(op)) {
1684 if (core.getTile() == *linkOp->getOptionalSharedTile()) {
1685 releaseOp->emitOpError("currently cannot access objectFifo used in "
1686 "ObjectFifoLinkOp");
1687 return;
1688 }
1689 }
1690
1691 // update index of next element to release for this objectFifo
1692 updateAndReturnIndex(relPerFifo, {op, portNum});
1693
1694 // release locks
1695 int numLocks = releaseOp.relNumber();
1696 // account for repetition
1697 if (op.getRepeatCount().has_value())
1698 numLocks *= op.getRepeatCount().value();
1699 createUseLocks(builder, op, port, relPerFifo, numLocks,
1700 LockAction::Release);
1701
1702 // register release op
1703 if (releaseOps.find({op, portNum}) != releaseOps.end()) {
1704 releaseOps[{op, portNum}].push_back(releaseOp);
1705 } else {
1706 std::vector release = {releaseOp};
1707 releaseOps[{op, portNum}] = release;
1708 }
1709 });
1710
1711 //===----------------------------------------------------------------===//
1712 // Replace objectFifo.acquire ops
1713 //===----------------------------------------------------------------===//
1714 coreOp.walk([&](ObjectFifoAcquireOp acquireOp) {
1715 ObjectFifoCreateOp op = acquireOp.getObjectFifo();
1716 builder.setInsertionPointAfter(acquireOp);
1717 auto port = acquireOp.getPort();
1718 auto portNum = port == ObjectFifoPort::Produce ? 0 : 1;
1719 auto core = acquireOp->getParentOfType<CoreOp>();
1720
1721 auto linkOp = getOptionalLinkOp(op);
1722 if (linkOp) {
1723 if (core.getTile() == *linkOp->getOptionalSharedTile()) {
1724 acquireOp->emitOpError("currently cannot access objectFifo used in "
1725 "ObjectFifoLinkOp");
1726 return;
1727 }
1728 }
1729
1730 // index of next element to acquire for this objectFifo
1731 int start = updateAndReturnIndex(
1732 acqPerFifo, {op, portNum}); // useful for keeping track of which
1733 // indices are acquired
1734
1735 // check how many elements have been released in between this AcquireOp
1736 // and the previous one
1737 // !!! operations may not be in the same block !!!
1738 int numRel = 0;
1739 for (std::vector<ObjectFifoReleaseOp>::iterator relOp =
1740 releaseOps[{op, portNum}].begin();
1741 relOp != releaseOps[{op, portNum}].end();) {
1742 bool erased = false;
1743 Operation *acqBlockDefOp = acquireOp.getOperation();
1744 do {
1745 Operation *relBlockDefOp = (*relOp).getOperation();
1746 do {
1747 if (acqBlockDefOp->getBlock() == relBlockDefOp->getBlock()) {
1748 if (relBlockDefOp->isBeforeInBlock(acqBlockDefOp)) {
1749 numRel += (*relOp).relNumber();
1750 relOp = releaseOps[{op, portNum}].erase(relOp);
1751 // to ensure that we do not account
1752 // the ReleaseOps again later,
1753 // after the subview is created
1754 erased = true;
1755 }
1756 }
1757 } while ((relBlockDefOp = relBlockDefOp->getParentOp()) &&
1758 !isa<DeviceOp>(relBlockDefOp) && !erased);
1759 } while ((acqBlockDefOp = acqBlockDefOp->getParentOp()) &&
1760 !isa<DeviceOp>(acqBlockDefOp) && !erased);
1761 if (!erased)
1762 ++relOp;
1763 }
1764
1765 // track indices of elements to acquire
1766 std::vector<int> acquiredIndices;
1767 if (!acquiresPerFifo[{op, portNum}].empty()) {
1768 // take into account what has already been acquired by previous
1769 // AcquireOp in program order
1770 acquiredIndices = acquiresPerFifo[{op, portNum}];
1771 // take into account what has been released in-between
1772 if (static_cast<size_t>(numRel) > acquiredIndices.size()) {
1773 acquireOp->emitOpError("cannot release more elements than are "
1774 "already acquired");
1775 return;
1776 }
1777 for (int i = 0; i < numRel; i++)
1778 acquiredIndices.erase(acquiredIndices.begin());
1779 }
1780
1781 // acquire locks
1782 int numLocks = acquireOp.acqNumber();
1783 int alreadyAcq = acquiredIndices.size();
1784 int numCreate;
1785 if (numLocks > alreadyAcq)
1786 numCreate = numLocks - alreadyAcq;
1787 else
1788 numCreate = 0;
1789
1790 // account for repetition
1791 if (op.getRepeatCount().has_value())
1792 numCreate *= op.getRepeatCount().value();
1793
1794 auto dev = op->getParentOfType<DeviceOp>();
1795 if (auto &targetArch = dev.getTargetModel();
1796 targetArch.getTargetArch() == AIEArch::AIE1)
1797 createUseLocks(builder, op, port, acqPerFifo, numCreate,
1798 LockAction::Acquire);
1799 else
1800 createUseLocks(builder, op, port, acqPerFifo, numCreate,
1801 LockAction::AcquireGreaterEqual);
1802
1803 // if objFifo was linked with others, find which objFifos
1804 // elements to use
1805 ObjectFifoCreateOp target = op;
1806 if (linkOp)
1807 if (objFifoLinks.find(*linkOp) != objFifoLinks.end())
1808 target = objFifoLinks[*linkOp];
1809
1810 // create subview: buffers that were already acquired + new acquires
1811 for (int i = 0; i < numCreate; i++) {
1812 acquiredIndices.push_back(start);
1813 start = (start + 1) % op.size();
1814 }
1815 std::vector<BufferOp *> subviewRefs;
1816 subviewRefs.reserve(acquiredIndices.size());
1817 for (auto index : acquiredIndices)
1818 subviewRefs.push_back(&buffersPerFifo[target][index]);
1819
1820 subviews[acquireOp] = subviewRefs;
1821 acquiresPerFifo[{op, portNum}] = acquiredIndices;
1822 });
1823
1824 //===----------------------------------------------------------------===//
1825 // Replace subview.access ops
1826 //===----------------------------------------------------------------===//
1827 coreOp.walk([&](ObjectFifoSubviewAccessOp accessOp) {
1828 auto acqOp = accessOp.getSubview().getDefiningOp<ObjectFifoAcquireOp>();
1829 if (ObjectFifoCreateOp op = acqOp.getObjectFifo()) {
1830 if (auto linkOp = getOptionalLinkOp(op); linkOp.has_value()) {
1831 if (!linkOp->isDistribute() && !linkOp->isJoin()) {
1832 for (auto consumerTile : op.getConsumerTiles()) {
1833 if (auto consumerTileOp =
1834 dyn_cast<TileOp>(consumerTile.getDefiningOp())) {
1835 int share_dir_value = 0;
1836 bool sharing = isSharedMemory(
1837 op.getProducerTileOp(), consumerTileOp, &share_dir_value);
1838 if (!sharing)
1839 accessOp->emitOpError(
1840 "currently cannot access objectFifo used in "
1841 "ObjectFifoLinkOp if the tiles don't share memory");
1842 }
1843 }
1844 } else
1845 accessOp->emitOpError(
1846 "currently cannot access objectFifo used in "
1847 "ObjectFifoLinkOp if it is a distribute or join link");
1848 }
1849 }
1850 accessOp.getOutput().replaceAllUsesWith(
1851 subviews[acqOp][accessOp.getIndex()]->getBuffer());
1852 });
