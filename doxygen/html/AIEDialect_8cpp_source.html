77  if (!(op.getMixedOffsets().size() == 1 && maxRanks[0] == 1) && // NOLINT
78  op.getMixedOffsets().size() != op.getMixedSizes().size())

83  if (failed(verifyListOfOperandsOrIntegers(
84  op, "offset", maxRanks[0], op.getStaticOffsets(), op.getOffsets())))
85  return failure();
86  if (failed(verifyListOfOperandsOrIntegers(
87  op, "size", maxRanks[1], op.getStaticSizes(), op.getSizes())))
88  return failure();
89  if (failed(verifyListOfOperandsOrIntegers(
90  op, "stride", maxRanks[2], op.getStaticStrides(), op.getStrides())))

93  if (offset < 0 && !ShapedType::isDynamic(offset))

95  << offset;
96  for (int64_t size : op.getStaticSizes())
97  if (size < 0 && !ShapedType::isDynamic(size))
98  return op->emitError("expected sizes to be non-negative, but got ")

161 struct UsesAreAccessible {
162  static LogicalResult verifyTrait(Operation *op) {
163  auto thisElement = cast<TileElement>(op);
164  auto thisID = thisElement.getTileID();
165  auto users = op->getResult(0).getUsers();
166  const auto &targetModel = getTargetModel(op);
167  for (auto *user : users) {
168  // AIE.useLock may be used in a device to set the lock's default value
169  // Allow in a toplevel module for backward compatibility
170  if (llvm::isa_and_nonnull<DeviceOp, ModuleOp>(user->getParentOp())) {
171  continue;
172  }
173  // If any parent prescribes that accessibility checks be skipped,
174  // skip the check for that user.
175  if (user->getParentWithTrait<SkipAccessibilityCheckTrait>()) {
176  continue;
177  }
178  TileElement element = llvm::dyn_cast<TileElement>(user);
179  if (!element) {
180  element = getParentTileElement(user);
181  }
182  if (!element) {
183  // This should probably be caught elsewhere as well.
184  return op->emitOpError("is accessed outside of a tile")
185  .attachNote(user->getLoc())
186  << "user";
187  }
188  auto tileID = element.getTileID();
189  if (!targetModel.isLegalMemAffinity(tileID.col, tileID.row, thisID.col,
190  thisID.row)) {
191  return (op->emitOpError("in Column ")
192  << thisID.col << " and Row " << thisID.row
193  << " is accessed from an unreachable tile in Column "
194  << tileID.col << " and Row " << tileID.row)
195  .attachNote(user->getLoc())
196  << "user";
197  }
198  }
199  return success();

301 /// Parse an instance of a type registered to the AIE dialect.
302 /// Parse an AIE type in the following forms:
303 /// AIE-type
304 /// ::= `objectfifo` `<` type `>`
305 /// ::= `objectfifosubview` `<` type `>`
306 static OptionalParseResult aieTypeParser(DialectAsmParser &parser,
307  StringRef name, Type &result) {
308  if (name == "objectfifo") {
309  MemRefType elementType;
310  SMLoc typeLoc = parser.getCurrentLocation();
311  if (parser.parseLess() || parser.parseType(elementType) ||
312  parser.parseGreater())
313  return failure();
314 
315  // Check that the type is a MemRef type.
316  if (!llvm::isa<MemRefType>(elementType)) {
317  parser.emitError(typeLoc, "element type for an objectFifo must be "
318  "a MemRefType, got: ")
319  << elementType;
320  return failure();
321  }
322 
323  return result = AIEObjectFifoType::get(elementType), success();
324  }
325 
326  if (name == "objectfifosubview") {
327  if (parser.parseLess())
328  return failure();
329 
330  // Parse the element type of the struct.
331  MemRefType elementType;
332  // Parse the current element type.
333  SMLoc typeLoc = parser.getCurrentLocation();
334  if (parser.parseType(elementType))
335  return failure();
336 
337  // Check that the type is a MemRefType.
338  if (!llvm::isa<MemRefType>(elementType)) {
339  parser.emitError(typeLoc, "element type for a subview must be "
340  "a MemRefType, got: ")
341  << elementType;
342  return failure();
343  }
344 
345  // Parse: `>`
346  if (parser.parseGreater())
347  return failure();
348 
349  return result = AIEObjectFifoSubviewType::get(elementType), success();
350  }
351 
352  return {};
353 }

454 // Check that the given DMA-like op (e.g. MemOp, ShimDMAOp)
455 // has valid DMA channels.
456 template <typename ConcreteType>
457 LogicalResult HasValidDMAChannels<ConcreteType>::verifyTrait(Operation *op) {
458  DenseSet<DMAChannel> inputChannels;
459  DenseSet<DMAChannel> outputChannels;
460  auto element = cast<ConcreteType>(op);
461  Region &body = element.getBody();
462  if (body.empty())
463  return op->emitOpError("should have non-empty body");
464  for (auto &bodyOp : body.getOps()) {
465  // check for duplicate DMA channels within the same ShimDMAOp
466  if (auto dmaStart = dyn_cast<DMAStartOp>(bodyOp)) {
467  DMAChannel dmaChan = {dmaStart.getChannelDir(),
468  dmaStart.getChannelIndex()};
469  // check if number of input and output channels is more than available
470  // hardware
471  if (dmaChan.direction == DMAChannelDir::S2MM)
472  inputChannels.insert(dmaChan);
473  else
474  outputChannels.insert(dmaChan);
475  }
476  }
477 
478  if (inputChannels.size() >
479  element.getTileOp().getNumSourceConnections(WireBundle::DMA))
480  return op->emitOpError(
481  "uses more input channels than available on this tile");
482 
483  if (outputChannels.size() >
484  element.getTileOp().getNumDestConnections(WireBundle::DMA))
485  return op->emitOpError(
486  "uses more output channels than available on this tile");
487  return success();
488 }

562  OpAsmParser &parser, SmallVectorImpl<OpAsmParser::UnresolvedOperand> &tiles,
563  BDDimLayoutArrayArrayAttr &dimensions) {
564  // parseCommaSeparatedList doesn't handle the missing case for "none",
565  // so we handle it custom here.
566  std::vector<BDDimLayoutArrayAttr> tileDims = {};
567 
568  auto parseOneOperand = [&]() -> ParseResult {
569  if (parser.parseOperand(tiles.emplace_back(), true)) {
570  return failure();
571  }
572  // By default, create empty dimensions array for each consumer; this way,
573  // we can be certain to have as many entries in the dimensions array as

576  BDDimLayoutArrayAttr::get(parser.getContext(), {});
577 
578  if (succeeded(parser.parseOptionalKeyword("dimensionsFromStream"))) {

587  };

590  parseOneOperand, " in operand list"))
591  return failure();

674 LogicalResult ObjectFifoLinkOp::verify() {
675  if (isJoin() && isDistribute())
676  return emitError("ObjectFifoLinkOp does not support 'join' and "
677  "'distribute' at the same time");
678 
679  auto sharedTile = getOptionalSharedTile();
680  if (!sharedTile)
681  return emitError("ObjectFifoLinkOp must have a link point, i.e., a "
682  "shared tile between objectFifos");
683 
684  TileOp tile = cast<TileOp>(sharedTile.value().getDefiningOp());
685  if (!tile.isMemTile()) {
686  if (isJoin() || isDistribute())
687  return emitError("ObjectFifoLinkOp join and distribute are "
688  "unavailable on compute or shim tiles");
689  }
690 
691  if (isJoin()) {
692  if (getFifoIns().size() != getSrcOffsets().size())
693  return emitOpError("number of provided src offsets must be equal "
694  "to the number of input objectFifos");
695 
696  if (!getDstOffsets().empty())
697  return emitOpError("dst offsets should be empty for join");
698 
699  } else if (isDistribute()) {
700  if (getFifoOuts().size() != getDstOffsets().size())
701  return emitOpError("number of provided dst offsets must be equal "
702  "to the number of output objectFifos");
703 
704  if (!getSrcOffsets().empty())
705  return emitOpError("src offsets should be empty for distribute");
706 
707  ObjectFifoCreateOp fifoIn = getInputObjectFifos()[0];
708  if (!fifoIn.getDimensionsToStream().empty()) {
709  return emitOpError("currently does not support objectFifos with "
710  "dimensionsToStream.");
711  }
712  for (auto dims : fifoIn.getDimensionsFromStreamPerConsumer()) {
713  if (!dims.empty())
714  return emitOpError("currently does not support objectFifos with "
715  "dimensionsFromStreamPerConsumer.");
716  }
717 
718  for (auto fifoOut : getOutputObjectFifos()) {
719  for (auto dims : fifoOut.getDimensionsFromStreamPerConsumer()) {
720  if (!dims.empty())
721  return emitOpError("currently does not support objectFifos with "
722  "dimensionsFromStreamPerConsumer.");
723  }
724  }
725 
726  std::vector<int> repeat_counts;
727  for (auto fifoOut : getOutputObjectFifos()) {
728  if (fifoOut.getRepeatCount().has_value())
729  repeat_counts.push_back(fifoOut.getRepeatCount().value());
730  else
731  repeat_counts.push_back(0);
732  }
733  for (auto repeat : repeat_counts)
734  if (repeat_counts[0] != repeat)
735  return emitError("repeat counts of output object FIFOs must be equal");
736 
737  } else {
738  if (!getSrcOffsets().empty() && !getDstOffsets().empty())
739  return emitOpError("all offsets should be empty if there is no "
740  "join or distribute");
741  }
742 
743  return success();
744 }

1514  "MemTileDMAOp has zero region!");

1517  .failed())
1518  return failure();
1519 
1520  for (auto &bodyOp : getBody().getOps()) {
1521  if (auto allocOp = dyn_cast<memref::AllocOp>(bodyOp)) {
1522  if (!allocOp->getAttr("id"))
1523  return allocOp.emitOpError()
1524  << "allocOp in MemTileDMAOp region should have an id attribute";
1525  }
1526  if (auto startOp = dyn_cast<DMAStartOp>(bodyOp)) {
1527  if (startOp.getChannelIndex() > 3) {
1528  // Channels 4 and 5 in a memtile are restricted to only access local
1529  // buffers and locks.
1530

1532  // Set of blocks found to be reachable within a given region.

1536  reachable.insert(firstBD);
1537  worklist.push_back(firstBD);
1538  while (!worklist.empty()) {
1539  Block *block = worklist.pop_back_val();

1541  continue;

1543  for (auto *i : successors) {
1544  if (!reachable.contains(i)) {
1545  reachable.insert(i);
1546  worklist.push_back(i);
1547  }
1548  }
1549  }
1550  for (Block *b : reachable) {
1551  for (DMABDOp bd : b->getOps<DMABDOp>()) {
1552  if (auto bufferOp = bd.getBufferOp();
1553  bufferOp.getTileOp().colIndex() != colIndex() ||
1554  bufferOp.getTileOp().rowIndex() != rowIndex()) {
1555  InFlightDiagnostic err =
1556  bd.emitOpError()
1557  << "is reachable from DMA channel "
1558  << startOp.getChannelIndex()
1559  << " and attempts to access a non-local buffer\n";
1560  err.attachNote(startOp->getLoc()) << "channel";
1561  err.attachNote(bufferOp->getLoc()) << "buffer";
1562  return err;
1563  }
1564  }

1566  if (auto lockOp = useLock.getLockOp();
1567  lockOp.getTileOp().colIndex() != colIndex() ||
1568  lockOp.getTileOp().rowIndex() != rowIndex()) {
1569  InFlightDiagnostic err =
1570  useLock.emitOpError()
1571  << "is reachable from DMA channel "
1572  << startOp.getChannelIndex()
1573  << " and attempts to access a non-local lock\n";

1575  err.attachNote(lockOp->getLoc()) << "lock";
1576  return err;
1577  }
1578  }
1579  }
1580  }
1581  }
1582  }
1583

1785 LogicalResult DMABDOp::verify() {
1786  // Skip verification of the BDOp outside of mem operations.
1787  // BDOps may appear elsewhere and subsequent lowerings will place them in the
1788  // correct mem ops.
1789  Operation *p = (*this)->getParentOp();
1790  if (!llvm::isa<MemOp, MemTileDMAOp, ShimDMAOp, DMAOp>(*p)) {
1791  return success();
1792  }
1793 
1794  if (!isa<BufferOp, ExternalBufferOp>(getBuffer().getDefiningOp()))
1795  return emitOpError(
1796  "BDs only support BufferOp or ExternalBufferOp operands.");
1797 
1798  if (getLenInBytes() % 4)
1799  return emitOpError("transfer length must be multiple of 4 (i.e., represent "
1800  "4 byte aligned address)");
1801 
1802  TileID parentTileId = getParentTileElement(getOperation()).getTileID();
1803 
1804  if (getOperation()->getParentOfType<MemOp>() &&
1805  (getBufferOp().getTileOp().colIndex() != parentTileId.col ||
1806  getBufferOp().getTileOp().rowIndex() != parentTileId.row))
1807  return emitOpError(
1808  "Core tile DMAs can only access a buffer in the same tile.");
1809 
1810  const AIETargetModel &targetModel = getTargetModel(getOperation());
1811 
1812  uint32_t maxBds = targetModel.getNumBDs(parentTileId.col, parentTileId.row);
1813  if (std::optional<int32_t> bdId = getBdId();
1814  bdId.has_value() && static_cast<uint32_t>(*bdId) >= maxBds)
1815  return emitOpError("bdId attribute exceeds max: ") << maxBds - 1;
1816  if (std::optional<int32_t> nextBdId = getNextBdId();
1817  nextBdId.has_value() && static_cast<uint32_t>(*nextBdId) >= maxBds)
1818  return emitOpError("nextBdId attribute exceeds max: ") << maxBds - 1;
1819  if (auto dims = getDimensions(); dims.has_value()) {
1820  size_t maxNDims = 3;
1821  if (getOperation()->getParentOfType<MemTileDMAOp>())
1822  maxNDims = 4;
1823  if (dims->size() > maxNDims)
1824  return emitOpError() << "Cannot give more than "
1825  << std::to_string(maxNDims)
1826  << " dimensions for step sizes and wraps in this "
1827  " tile (got "
1828  << std::to_string(dims->size()) << " dimensions).";
1829 
1830  MemRefType buffer = getBuffer().getType();
1831  int64_t maxIdx = 0;
1832  for (BDDimLayoutAttr dim : *dims) {
1833  maxIdx += dim.getStride() * (dim.getSize() - 1);
1834  if (0 == dim.getStride())
1835  return emitOpError()
1836  << "Invalid step size; must be a positive integer.";
1837  if (dim.getStride() > buffer.getNumElements())
1838  return emitOpError()
1839  << "Step size " << std::to_string(dim.getStride()) << " "
1840  << "exceeds memref size "
1841  << std::to_string(buffer.getNumElements());
1842  if (dim.getSize() >= (1UL << 9) + 1)
1843  return emitOpError() << "Size may not exceed 1023.";
1844  if (dim.getStride() >= (1UL << 19))
1845  return emitOpError() << "Stride may not exceed " << (1 << 20);
1846  }
1847 
1848  if (buffer.getNumElements() <= maxIdx)
1849  return emitOpError() << "Specified stride(s) and size(s) result in out "
1850  "of bounds access in buffer, for index "
1851  << std::to_string(maxIdx) << " in memref of length "
1852  << std::to_string(buffer.getNumElements()) << ".";
1853 
1854  // Since streams read 32b words, there's no way to read eg 16b with stride
1855  // of 2 (ie lower halfs of each 32b). So force it to be 1 (and then in
1856  // CDODirect/XAIEV2 scale the size by 4/getBufferElementTypeWidthInBytes).
1857  if (getBufferElementTypeWidthInBytes() < 4 && dims->back().getStride() != 1)
1858  return emitOpError(
1859  "For <32b width datatypes, inner-most dim stride must be 1");
1860  }
1861  if (auto paddims = getPadDimensions(); paddims.has_value()) {
1862  auto dims = getDimensions();
1863  if (!dims.has_value())
1864  return emitOpError() << "Padding requires n-d data layouts expressed as"
1865  << " wrap(s) and stride(s).";
1866  if (!targetModel.isMemTile(parentTileId.col, parentTileId.row))
1867  return emitOpError() << "Padding is only supported by memtile dma bds.";
1868  if (dims->size() != paddims->size())
1869  return emitOpError() << "Mismatch number of dimensions between padding(s)"
1870  << " and wrap(s) and stride(s).";
1871  int actuallen = 1;

1931  reachable.insert(firstBD);
1932  worklist.push_back(firstBD);
1933  while (!worklist.empty()) {
1934  Block *block = worklist.pop_back_val();
1935  if (block->empty())
1936  continue;
1937  auto successors = block->getTerminator()->getSuccessors();
1938  for (auto *i : successors) {
1939  if (!reachable.contains(i)) {
1940  reachable.insert(i);
1941  worklist.push_back(i);
1942  }
1943  }
1944  }

1947  // folding not applicable.
1948  if (isa<EndOp>((reachable.back())->getTerminator()))
1949  return failure();

1952  auto areEquivalentBDs = [](Block *b1, Block *b2) {
1953  auto b1OpRange = b1->without_terminator();

1958  [](std::tuple<Operation &, Operation &> pair) {
1959  return OperationEquivalence::isEquivalentTo(
1960  &std::get<0>(pair), &std::get<1>(pair),
1961  OperationEquivalence::IgnoreLocations);
1962  }))
1963  return false;
1964  return true;
1965  };
1966

1968  SmallVector<Block *> uniquePattern;
1969  auto patternIt = reachable.begin();
1970  while (patternIt != reachable.end() &&
1971  llvm::none_of(uniquePattern, [patternIt, areEquivalentBDs](Block *b1) {
1972  return areEquivalentBDs(*patternIt, b1);
1973  })) {
1974  uniquePattern.push_back(*patternIt);
1975  patternIt++;
1976  }
1977 
1978  unsigned idx = 0;
1979  while (patternIt != reachable.end()) {
1980  // BD repetition found. Check if repeating pattern.
1981  if (!areEquivalentBDs(*patternIt, uniquePattern[idx]))
1982  return failure();

1984  idx = (++idx) % uniquePattern.size();
1985  }
1986 
1987  // Repeating BD chains detected. Erasing repetitions.
1988  auto lastBDTerm = dyn_cast<NextBDOp>(reachable.back()->getTerminator());
1989  auto lastUniqueBDTerm =
1990  dyn_cast<NextBDOp>(uniquePattern.back()->getTerminator());
1991  lastUniqueBDTerm.setSuccessor(lastBDTerm.getSuccessor());
1992
