155 myargs.trace_size, myargs.trace_file);

160 std::chrono::duration_cast<std::chrono::microseconds>(stop - start)
161 .count();
162
163 npu_time_total += npu_time;
164 npu_time_min = (npu_time < npu_time_min) ? npu_time : npu_time_min;
165 npu_time_max = (npu_time > npu_time_max) ? npu_time : npu_time_max;
166 }

193 << std::endl;
