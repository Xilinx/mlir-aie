28struct DMAStartTaskOpPattern : OpConversionPattern<DMAStartTaskOp> {
29 using OpConversionPattern::OpConversionPattern;
30
31 LogicalResult
32 matchAndRewrite(DMAStartTaskOp op, OpAdaptor adaptor,
33 ConversionPatternRewriter &rewriter) const override {
34 DMAConfigureTaskOp task_op = op.getTaskOp();
35 if (!task_op) {
36 // Cannot rewrite this; probably points to a DMAStartTaskForOp,
37 // which we will lower once it has been rewritten into a DMAStartTaskOp.
38 return failure();
39 }
40 AIE::TileOp tile = task_op.getTileOp();
41 std::optional<uint32_t> first_bd_id = task_op.getFirstBdId();
42 if (!first_bd_id) {
43 auto err = op.emitOpError(
44 "First buffer descriptor in chain has not been assigned an ID");
45 err.attachNote() << "Run the `aie-assign-runtime-buffer-descriptor-ids` "
46 "pass first or manually assign an ID.";
47 return failure();
48 }
49 rewriter.replaceOpWithNewOp<NpuPushQueueOp>(
50 op, tile.getCol(), tile.getRow(), task_op.getDirection(),
51 task_op.getChannel(), task_op.getIssueToken(), task_op.getRepeatCount(),
52 *first_bd_id);
53 return success();
54 }
55};

59
60 LogicalResult
61 matchAndRewrite(DMAAwaitTaskOp op, OpAdaptor adaptor,
62 ConversionPatternRewriter &rewriter) const override {
63 DMAConfigureTaskOp task_op = op.getTaskOp();
64 if (!task_op) {
65 return failure();
66 }

77 task_op.getChannel(), 1, 1);
78 return success();
79 }

84 bool shouldSkipBlock(Block &block) {
85 // Allow blocks in the input IR that contain nothing but a next_bd operation
86 // as the entry block. We will skip these blocks and not lower them to
87 // anything.
88 auto it = block.without_terminator();
89 return block.isEntryBlock() && it.begin() == it.end();
90 }

92 LogicalResult verifyBdInBlock(Block &block) {
93 auto bd_ops = block.getOps<AIE::DMABDOp>();
94 // Exactly one BD op per block
95 int n_bd_ops = std::distance(bd_ops.begin(), bd_ops.end());
96 if (n_bd_ops < 1) {
97 auto error = block.getTerminator()->emitError(
98 "Block ending in this terminator does not contain a required "
99 "aie.dma_bd operation.");
100 error.attachNote(block.getParentOp()->getLoc())
101 << "Error encountered while lowering this BD configuration.";
102 return failure();
103 } else if (n_bd_ops > 1) {
104 auto error = block.getTerminator()->emitOpError(
105 "This block contains multiple aie.dma_bd operations. Exactly one is "
106 "required.");
107 auto it = bd_ops.begin();
108 ++it;
109 for (; it != bd_ops.end(); ++it) {
110 error.attachNote((*it)->getLoc()) << "Extra aie.dma_bd operation here.";
111 }
112 return failure();
113 }
114 AIE::DMABDOp bd_op = *bd_ops.begin();
115 if (!bd_op.getBdId().has_value()) {
116 auto error = bd_op.emitOpError(
117 "Cannot lower buffer descriptor without assigned ID.");
118 error.attachNote()
119 << "Run the `--aie-assign-runtime-sequence-bd-ids` pass first or "
120 "manually assign an ID to this buffer descriptor.";
121 error.attachNote(block.getParentOp()->getLoc())
122 << "Error encountered while lowering this BD configuration.";
123 return failure();
124 }
125 return success();
126 }
127
128 LogicalResult verifyOptionalLocksInBlock(Block &block) {
129 auto lock_ops = block.getOps<AIE::UseLockOp>();
130 int n_lock_ops = std::distance(lock_ops.begin(), lock_ops.end());
131 // Allow exactly 0 or 2 lock ops (acquire and release)
132 if (n_lock_ops != 0 && n_lock_ops != 2) {
133 AIE::UseLockOp lock_op = *lock_ops.begin();
134 lock_op.emitOpError(
135 "BD blocks must have either 0 or 2 lock operations (acquire and "
136 "release). Found ")
137 << n_lock_ops << " lock operations.";
138 return failure();
139 }
140 return success();
141 }
142
143 LogicalResult verifyNoUnsupportedOpsInBlock(Block &block) {
144 WalkResult unsupported_ops = block.walk([&](Operation *inner_op) {
145 return llvm::TypeSwitch<Operation *, WalkResult>(inner_op)
146 .Case<AIE::DMABDOp>(
147 [&](AIE::DMABDOp bd_op) { return WalkResult::advance(); })
148 .Case<AIE::UseLockOp>(
149 [&](AIE::UseLockOp lock_op) { return WalkResult::advance(); })
150 .Case<AIE::NextBDOp>(
151 [&](AIE::NextBDOp lock_op) { return WalkResult::advance(); })
152 .Case<AIE::EndOp>(
153 [&](AIE::EndOp lock_op) { return WalkResult::advance(); })
154 .Default([&](Operation *inner_op) {
155 auto error = block.getParentOp()->emitOpError(
156 "Unsupported operation within BD block.");
157 error.attachNote(inner_op->getLoc())
158 << "No lowering to NPU instructions available for this "
159 "operation.";
160 return WalkResult::interrupt();
161 });
162 });
163 if (unsupported_ops.wasInterrupted()) {
164 return failure();
165 }
166 return success();
167 }

169 AIE::DMABDOp getBdForBlock(Block &block) {
170 auto bd_ops = block.getOps<AIE::DMABDOp>();
171 AIE::DMABDOp bd_op = *bd_ops.begin(); // Dereference first (and only, after
172 // previous checks) bd op iterator
173 return bd_op;
174 }
175
176 // Returns pair of (acquire_lock_op, release_lock_op) if present
177 std::optional<std::pair<AIE::UseLockOp, AIE::UseLockOp>>
178 getOptionalLockOpsForBlock(Block &block) {
179 auto lock_ops = block.getOps<AIE::UseLockOp>();
180 int n_lock_ops = std::distance(lock_ops.begin(), lock_ops.end());
181 if (n_lock_ops != 2) {
182 return std::nullopt;
183 }
184
185 AIE::UseLockOp acquire_op = nullptr;
186 AIE::UseLockOp release_op = nullptr;
187
188 for (auto lock_op : lock_ops) {
189 if (lock_op.acquire() || lock_op.acquireGE()) {
190 acquire_op = lock_op;
191 } else if (lock_op.release()) {
192 release_op = lock_op;
193 }
194 }
195
196 if (acquire_op && release_op) {
197 return std::make_pair(acquire_op, release_op);
198 }
199 return std::nullopt;
200 }

202 LogicalResult setAddressForSingleBD(OpBuilder &builder, AIE::DMABDOp &bd_op,
203 AIE::TileOp &tile) {
204 uint32_t bd_id = bd_op.getBdId().value();
205 const AIE::AIETargetModel &target_model = AIE::getTargetModel(bd_op);
206 auto buf = bd_op.getBuffer();
207 auto col = tile.getCol();
208 auto row = tile.getRow();
209 uint64_t register_addr = target_model.getDmaBdAddress(col, row, bd_id) +
210 target_model.getDmaBdAddressOffset(col, row);
211
212 // A buffer descriptor can refer to a statically allocated aie.buffer, or to
213 // a DDR buffer which will be passed as a runtime argument (block
214 // argument). Try to find the root block argument, either directly or
215 // through subviews/casts.
216 mlir::BlockArgument buf_arg = nullptr;
217 int64_t offset = 0;
218
219 if (auto directArg = llvm::dyn_cast<mlir::BlockArgument>(buf)) {
220 buf_arg = directArg;
221 offset = 0;
222 } else if (auto traceResult = traceSubviewToBlockArgument(buf)) {
223 buf_arg = traceResult->rootArg;
224 offset = traceResult->offsetInBytes;
225 }
226
227 if (buf_arg) {
228 if (!target_model.isShimNOCTile(tile.getCol(), tile.getRow())) {
229 return bd_op->emitOpError("DDR memory (runtime input arguments) can "
230 "only be referred to on shim tiles.");
231 }
232
233 unsigned arg_idx = buf_arg.getArgNumber();
234 offset += bd_op.getOffsetInBytes();
235 NpuAddressPatchOp::create(builder, bd_op.getLoc(),
236 /*addr*/ register_addr,
237 /*arg_idx*/ arg_idx,
238 /*arg_plus*/ offset);
239 } else if (AIE::BufferOp buffer =
240 llvm::dyn_cast<AIE::BufferOp>(buf.getDefiningOp())) {
241 uint64_t buf_addr;
242 if (!buffer.getAddress().has_value()) {
243 return bd_op->emitOpError(
244 "Cannot lower buffer without associated address. Run pass "
245 "--aie-assign-buffer-addresses first or manually assign an "
246 "address.");
247 }
248 buf_addr = *buffer.getAddress();
249 buf_addr += bd_op.getOffsetInBytes();
250 if (target_model.isCoreTile(col, row)) {
251 NpuMaskWrite32Op::create(builder, bd_op.getLoc(), register_addr,
252 (buf_addr / 4) << 14, 0x0fffc000, nullptr,
253 nullptr, nullptr);
254 } else if (target_model.isMemTile(col, row)) {
255 // On AIE2p (NPU2), memtile DMAs use an offset-based address
256 // space where the base depends on the relative position of the
257 // buffer's tile (west=0, internal=getMemTileSize, east=2x).
258 // On AIE2 (NPU1), memtile DMAs address local memory directly
259 // starting at 0. Only add the offset for AIE2p.
260 if (target_model.getTargetArch() == AIE::AIEArch::AIE2p) {
261 auto addrOffset = target_model.getMemLocalBaseAddress(
262 col, row, buffer.getTileOp().getCol(),
263 buffer.getTileOp().getRow());
264 if (addrOffset)
265 buf_addr += addrOffset.value();
266 }
267 NpuMaskWrite32Op::create(builder, bd_op.getLoc(), register_addr,
268 buf_addr / 4, 0x0007FFFF, nullptr, nullptr,
269 nullptr);
270 } else {
271 NpuWrite32Op::create(builder, bd_op.getLoc(), register_addr, buf_addr,
272 nullptr, nullptr, nullptr);
273 }
274 } else {
275 return bd_op->emitOpError(
276 "Buffer argument must be a constant aie.buffer, a runtime sequence "
277 "input argument, or a (chain of) subview(s) or cast(s) of a block "
278 "argument with constant offsets and strides equal to one.");
279 }
280 return success();
281 }

283 LogicalResult
284 rewriteSingleBD(OpBuilder &builder, Block &block, AIE::TileOp &tile,
285 AIE::DMAChannelDir channelDir,
286 std::optional<xilinx::AIE::PacketInfoAttr> packet) {
287 AIE::DMABDOp bd_op = getBdForBlock(block);
288 const auto &target_model = AIE::getTargetModel(bd_op);
289 auto buffer_type = llvm::cast<BaseMemRefType>(bd_op.getBuffer().getType());
290 uint32_t addr_granularity = target_model.getAddressGenGranularity();
291
292 uint32_t bd_id = bd_op.getBdId().value();
293 int64_t offset = bd_op.getOffsetInBytes();
294 uint64_t len = bd_op.getLenInBytes();
295 uint64_t len_addr_granularity = len * 8 / addr_granularity;
296
297 if (offset * 8 % addr_granularity != 0) {
298 return bd_op->emitOpError("Offset must be aligned to ")
299 << (addr_granularity / 8) << " byte boundary.";
300 }
301
302 if (len < addr_granularity / 8) {
303 return bd_op->emitOpError("Transfer size of ")
304 << len << " bytes falls below minimum hardware transfer unit of "
305 << (addr_granularity / 8) << " bytes.";
306 }
307 // Process strides/wraps
308 std::optional<llvm::ArrayRef<AIE::BDDimLayoutAttr>> dims =
309 bd_op.getDimensions();
310 llvm::SmallVector<int64_t, 4> sizes = llvm::SmallVector<int64_t, 4>(4, 0);
311 llvm::SmallVector<int64_t, 4> strides = llvm::SmallVector<int64_t, 4>(4, 0);
312
313 // Padding
314 std::optional<llvm::ArrayRef<AIE::BDPadLayoutAttr>> padDims =
315 bd_op.getPadDimensions();
316 llvm::SmallVector<int64_t, 4> padBefore =
317 llvm::SmallVector<int64_t, 4>(4, 0);
318 llvm::SmallVector<int64_t, 4> padAfter =
319 llvm::SmallVector<int64_t, 4>(4, 0);
320 std::fill(padBefore.begin(), padBefore.end(), 0);
321 std::fill(padAfter.begin(), padAfter.end(), 0);
322
323 auto enable_packet = 0;
324 auto out_of_order_id = 0;
325 auto packet_id = 0;
326 auto packet_type = 0;
327 auto d0size = 0;
328 auto d0stride = 0;
329 auto d1size = 0;
330 auto d1stride = 0;
331 auto d2size = 0;
332 auto d2stride = 0;
333 auto iteration_size = 0;
334 auto iteration_stride = 0;
335
336 if (dims && dims->size() > 0) {
337 llvm::SmallVector<int64_t, 4> input_sizes =
338 llvm::SmallVector<int64_t, 4>(4, 1);
339 llvm::SmallVector<int64_t, 4> input_strides =
340 llvm::SmallVector<int64_t, 4>(4, 0);
341 if (dims->size() > 4) {
342 return bd_op->emitOpError("At most four data layout transformation "
343 "dimensions may be provided.");
344 }
345
346 for (size_t i = 0; i < dims->size(); i++) {
347 // Pass down dimensions in reverse order; in the MLIR, this allows
348 // us to specify step sizes/wraps in the same order as we would
349 // access a multi-dim C array, with the highest dimension first.
350 int j = dims->size() - i - 1;
351 input_sizes[i] = (*dims)[j].getSize();
352 input_strides[i] = (*dims)[j].getStride();
353 }
354
355 // Do not check input_sizes[3] because a repeat can still be considered a
356 // linear transfer
357 bool isLinearTransfer = (input_sizes[0] >= 1) && (input_sizes[1] == 1) &&
358 (input_sizes[2] == 1);
359
360 if (dims->size() > 2) {
361 d2size = (target_model.isMemTile(tile.getCol(), tile.getRow()))
362 ? (*dims)[2].getSize()
363 : 0;
364 }
365 if (padDims.has_value()) {
366 if (!target_model.isMemTile(tile.getCol(), tile.getRow()))
367 return bd_op->emitOpError()
368 << "Padding is only supported by memtile dma bds.";
369 if (padDims->size() > dims->size())
370 return bd_op->emitOpError()
371 << "Mismatch number of dimensions between padding(s)"
372 << " and wrap(s) and stride(s).";
373 if (channelDir == AIE::DMAChannelDir::MM2S) {
374 for (size_t i = 0; i < padDims->size(); i++) {
375 int j = padDims->size() - i - 1;
376 padBefore[i] = (*padDims)[j].getConstPadBefore();
377 padAfter[i] = (*padDims)[j].getConstPadAfter();
378 }
379 for (size_t i = padDims->size(); i < dims->size(); i++) {
380 padBefore[i] = 0;
381 padAfter[i] = 0;
382 }
383 } else
384 return bd_op->emitOpError()
385 << "supports padding only for MM2S direction on MemTiles.";
386 }
387 getHardwareStridesWraps(target_model, bd_op, buffer_type, input_sizes,
388 input_strides, sizes, strides);
389
390 if (failed(verifyStridesWraps(bd_op, buffer_type, tile.getCol(),
391 tile.getRow(), input_sizes, input_strides,
392 sizes, strides, isLinearTransfer))) {
393 return failure();
394 }
395
396 iteration_size = sizes[3];
397 iteration_stride = strides[3];
398
399 if (!isLinearTransfer) {
400 // d0_size, d0_stride
401 d0size = sizes[0];
402 d0stride = strides[0];
403
404 // d1_size, d1_stride
405 d1size = sizes[1];
406 d1stride = strides[1];
407
408 // d2_stride
409 d2stride = strides[2];
410 // d2_size set elsewhere
411 }
412 if (input_sizes[3] > 1 && input_strides[3] == 0) {
413 // We allow users to encode the repeat_count as a dimension 3 stride
414 // of 0. This must lower to a iteration wrap of 0, so no stride is
415 // ever added. We then repeat the BD using the repeat_count in
416 // NpuPushQueueOp.
417 iteration_size = 0;
418 iteration_stride = 0;
419 }
420
421 // Ensure the total transfer length and the length expressed in the lowest
422 // three dimensions of strides/wraps agree. (Fourth dimension is
423 // iteration/repeat count and repeats the whole BD, so should not be
424 // incorporated in length of a single BD invocation.)
425 uint64_t len_dims_addr_granularity = 1;
426 for (size_t i = 0; i < 3; i++) {
427 len_dims_addr_granularity *= sizes[i];
428 }
429 if (len_dims_addr_granularity != len_addr_granularity) {
430 auto err =
431 bd_op->emitOpError(
432 "Buffer descriptor length does not match length of transfer "
433 "expressed by lowest three dimensions of data layout "
434 "transformation strides/wraps. ")
435 << "BD length is " << (len_addr_granularity * addr_granularity / 8)
436 << " bytes. "
437 << "Lowest three dimensions of data layout transformation would "
438 "result in transfer of "
439 << (len_dims_addr_granularity * addr_granularity / 8) << " bytes. ";
440 err.attachNote() << "Do not include the highest dimension size in "
441 "transfer length, as this is the BD repeat count.";
442 return failure();
443 }
444 } else {
445 if (padDims && target_model.isMemTile(tile.getCol(), tile.getRow()) &&
446 channelDir == AIE::DMAChannelDir::MM2S) {
447 return bd_op->emitOpError()
448 << "Padding requires n-d data layouts expressed as "
449 << "wrap(s) and stride(s).";
450 } else if (padDims) {
451 return bd_op->emitOpError() << "Padding is supported only on MemTiles.";
452 }
453 }
454 // find next BD ID, if any
455 uint32_t use_next_bd = 0;
456 uint32_t next_bd_id = 0;
457 if (bd_op.getNextBdId().has_value()) {
458 next_bd_id = bd_op.getNextBdId().value();
459 use_next_bd = 1;
460 }
461
462 // enable_packet
463 // auto info = bd_op.getPacket() ? bd_op.getPacket() : packet;
464 auto info = bd_op.getPacket().value_or(packet.value_or(nullptr));
465 if (info) {
466 enable_packet = 1;
467 packet_type = info.getPktType();
468 packet_id = info.getPktId();
469 }
470
471 // Extract lock information if present
472 int32_t lock_rel_val = 0;
473 int32_t lock_rel_id = 0;
474 int32_t lock_acq_enable = 0;
475 int32_t lock_acq_val = 0;
476 int32_t lock_acq_id = 0;
477
478 auto lock_ops = getOptionalLockOpsForBlock(block);
479 if (lock_ops) {
480 auto [acquire_op, release_op] = *lock_ops;
481
482 // Get lock IDs from the lock operations
483 AIE::LockOp acq_lock = acquire_op.getLockOp();
484 AIE::LockOp rel_lock = release_op.getLockOp();
485
486 if (acq_lock.getLockID().has_value()) {
487 lock_acq_id = acq_lock.getLockID().value();
488 lock_acq_val = acquire_op.getLockValue();
489 // For AcquireGreaterEqual, negate the value to signal the hardware
490 // to use >= comparison instead of == comparison.
491 if (acquire_op.acquireGE())
492 lock_acq_val = -lock_acq_val;
493 lock_acq_enable = 1;
494 }
495
496 if (rel_lock.getLockID().has_value()) {
497 lock_rel_id = rel_lock.getLockID().value();
498 lock_rel_val = release_op.getLockValue();
499 }
500
501 // For memtile, add lock offset using getLockLocalBaseIndex.
502 // This matches AIERT.cpp implementation.
503 if (target_model.isMemTile(tile.getCol(), tile.getRow())) {
504 auto lockOffset = target_model.getLockLocalBaseIndex(
505 tile.getCol(), tile.getRow(), acq_lock.colIndex(),
506 acq_lock.rowIndex());
507 if (lockOffset && acq_lock.getLockID().has_value())
508 lock_acq_id += lockOffset.value();
509 if (lockOffset && rel_lock.getLockID().has_value())
510 lock_rel_id += lockOffset.value();
511 }
512 }

514 NpuWriteBdOp::create(
515 builder, bd_op.getLoc(), tile.getCol(), bd_id, len_addr_granularity,
516 offset,
517 /*enable_packet=*/enable_packet,

541 LogicalResult hoistNextBdOpsIntoAttrs(DMAConfigureTaskOp op) {
542 Region &body = op.getBody();
543 for (auto it = body.begin(); it != body.end(); ++it) {
544 Block &block = *it;
545 if (shouldSkipBlock(block)) {
546 continue;
547 }
548 AIE::DMABDOp bd_op = getBdForBlock(block);
549 if (AIE::NextBDOp next_bd_op =
550 llvm::dyn_cast<AIE::NextBDOp>(block.getTerminator())) {
551 if (bd_op.getNextBdId().has_value()) {
552 auto error =
553 bd_op.emitOpError("Cannot specify both next_bd_id attribute and "
554 "aie.next_bd operation.");
555 error.attachNote(next_bd_op.getLoc())
556 << "Potentially conflicting next buffer descriptor ID specified "
557 "here.";
558 return failure();
559 }
560 Block &next_bd_block = *next_bd_op.getDest();
561 AIE::DMABDOp next_dma_bd_op = getBdForBlock(next_bd_block);
562 assert(next_dma_bd_op.getBdId()
563 .has_value()); // Next BD should have assigned ID, and this
564 // should have been checked by earlier
565 // verifyBdInBlock() call
566 bd_op.setNextBdId(next_dma_bd_op.getBdId().value());
567 OpBuilder builder(next_bd_op);
568 AIE::EndOp::create(builder, next_bd_op.getLoc());
569 next_bd_op.erase();
570 }
571 }
572 return success();
573 }

575 LogicalResult rewriteSingleDMAConfigureTaskOp(DMAConfigureTaskOp op) {
576 OpBuilder builder(op);
577 AIE::TileOp tile = op.getTileOp();
578
579 if (!op.use_empty()) {
580 auto err = op.emitOpError("Cannot lower while op still has uses.");
581 mlir::Operation::use_range uses = op.getOperation()->getUses();
582 for (auto it = uses.begin(); it != uses.end(); ++it) {
583 err.attachNote(it->getOwner()->getLoc()) << "Used here.";
584 }
585 return failure();
586 }
587
588 Region &body = op.getBody();
589
590 // Verify each BD block first; subsequent functions rely on them being
591 // well-formed
592 for (auto it = body.begin(); it != body.end(); ++it) {
593 if (shouldSkipBlock(*it)) {
594 continue;
595 }
596 if (failed(verifyNoUnsupportedOpsInBlock(*it))) {
597 return failure();
598 }
599 if (failed(verifyBdInBlock(*it))) {
600 return failure();
601 }
602 if (failed(verifyOptionalLocksInBlock(*it))) {
603 return failure();
604 }
605 }
606
607 // Hoist next_bd operations into next_bd_id attribute of the dma_bd
608 if (failed(hoistNextBdOpsIntoAttrs(op))) {
609 return failure();
610 }
611
612 auto channelDir = op.getDirection();
613 auto packet = op.getPacket();
614
615 // Lower all BDs
616 for (auto it = body.begin(); it != body.end(); ++it) {
617 Block &block = *it;
618 if (shouldSkipBlock(block)) {
619 continue;
620 }
621 if (failed(rewriteSingleBD(builder, block, tile, channelDir, packet))) {
622 return failure();
623 }
624 }
625
626 op.erase();
627
628 return success();
629 }

631 LogicalResult rewriteDMAConfigureTaskOp(AIE::DeviceOp device) {
632 WalkResult result = device.walk([&](DMAConfigureTaskOp op) {
633 if (failed(rewriteSingleDMAConfigureTaskOp(op))) {
634 return WalkResult::interrupt();
635 }
636 return WalkResult::advance();
637 });
638 if (result.wasInterrupted()) {
639 return failure();
640 }
641 return success();
642 }

644 void runOnOperation() override {
645 AIE::DeviceOp device = getOperation();
646
647 // Convert DMAStartBD and DMAAwaitBD ops
648 ConversionTarget target(getContext());
649 target.addLegalDialect<AIEXDialect>();
650 target.addIllegalOp<DMAStartTaskOp>();
651 target.addIllegalOp<DMAAwaitTaskOp>();
652 RewritePatternSet patterns(&getContext());
653 patterns.insert<DMAStartTaskOpPattern>(&getContext());
654 patterns.insert<DMAAwaitTaskOpPattern>(&getContext());
655 if (failed(applyPartialConversion(device, target, std::move(patterns)))) {
656 signalPassFailure();
657 }
658
659 // Lower the configuration for the BDs
660 if (failed(rewriteDMAConfigureTaskOp(device))) {
661 signalPassFailure();
662 }
663 }
