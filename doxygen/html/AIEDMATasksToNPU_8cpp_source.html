26struct DMAStartTaskOpPattern : OpConversionPattern<DMAStartTaskOp> {
27 using OpConversionPattern::OpConversionPattern;
28
29 LogicalResult
30 matchAndRewrite(DMAStartTaskOp op, OpAdaptor adaptor,
31 ConversionPatternRewriter &rewriter) const override {
32 DMAConfigureTaskOp task_op = op.getTaskOp();
33 if (!task_op) {
34 // Cannot rewrite this; probably points to a DMAStartTaskForOp,
35 // which we will lower once it has been rewritten into a DMAStartTaskOp.
36 return failure();
37 }
38 AIE::TileOp tile = task_op.getTileOp();
39 std::optional<uint32_t> first_bd_id = task_op.getFirstBdId();
40 if (!first_bd_id) {
41 auto err = op.emitOpError(
42 "First buffer descriptor in chain has not been assigned an ID");
43 err.attachNote() << "Run the `aie-assign-runtime-buffer-descriptor-ids` "
44 "pass first or manually assign an ID.";
45 return failure();
46 }
47 rewriter.replaceOpWithNewOp<NpuPushQueueOp>(
48 op, tile.getCol(), tile.getRow(), task_op.getDirection(),
49 task_op.getChannel(), task_op.getIssueToken(), task_op.getRepeatCount(),
50 *first_bd_id);
51 return success();
52 }
53};

56 using OpConversionPattern::OpConversionPattern;
57
58 LogicalResult
59 matchAndRewrite(DMAAwaitTaskOp op, OpAdaptor adaptor,
60 ConversionPatternRewriter &rewriter) const override {
61 DMAConfigureTaskOp task_op = op.getTaskOp();
62 if (!task_op) {
63 return failure();
64 }
65 if (!task_op.getIssueToken()) {
66 auto err = op.emitOpError(
67 "Cannot wait on a BD that is not configured to issue a token.");
68 err.attachNote(task_op.getLoc())
69 << "Consider adding attribute `issue_token=true` here.";
70 return err;
71 }
72 AIE::TileOp tile = task_op.getTileOp();
73 rewriter.replaceOpWithNewOp<NpuSyncOp>(op, tile.getCol(), tile.getRow(),
74 (uint32_t)task_op.getDirection(),
75 task_op.getChannel(), 1, 1);
76 return success();
77 }

82 bool shouldSkipBlock(Block &block) {
83 // Allow blocks in the input IR that contain nothing but a next_bd operation
84 // as the entry block. We will skip these blocks and not lower them to
85 // anything.
86 auto it = block.without_terminator();
87 return block.isEntryBlock() && it.begin() == it.end();
88 }
89
90 LogicalResult verifyBdInBlock(Block &block) {
91 auto bd_ops = block.getOps<AIE::DMABDOp>();
92 // Exactly one BD op per block
93 int n_bd_ops = std::distance(bd_ops.begin(), bd_ops.end());
94 if (n_bd_ops < 1) {
95 auto error = block.getTerminator()->emitError(
96 "Block ending in this terminator does not contain a required "
97 "aie.dma_bd operation.");
98 error.attachNote(block.getParentOp()->getLoc())
99 << "Error encountered while lowering this BD configuration.";
100 return failure();
101 } else if (n_bd_ops > 1) {
102 auto error = block.getTerminator()->emitOpError(
103 "This block contains multiple aie.dma_bd operations. Exactly one is "
104 "required.");
105 auto it = bd_ops.begin();
106 ++it;
107 for (; it != bd_ops.end(); ++it) {
108 error.attachNote((*it)->getLoc()) << "Extra aie.dma_bd operation here.";
109 }
110 return failure();
111 }
112 AIE::DMABDOp bd_op = *bd_ops.begin();
113 if (!bd_op.getBdId().has_value()) {
114 auto error = bd_op.emitOpError(
115 "Cannot lower buffer descriptor without assigned ID.");
116 error.attachNote()
117 << "Run the `--aie-assign-runtime-sequence-bd-ids` pass first or "
118 "manually assign an ID to this buffer descriptor.";
119 error.attachNote(block.getParentOp()->getLoc())
120 << "Error encountered while lowering this BD configuration.";
121 return failure();
122 }
123 return success();
124 }
125
126 LogicalResult verifyOptionalLocksInBlock(Block &block) {
127 auto lock_ops = block.getOps<AIE::UseLockOp>();
128 // Exactly 0 or 2 lock ops
129 int n_lock_ops = std::distance(lock_ops.begin(), lock_ops.end());
130 if (n_lock_ops > 0) {
131 // TODO: Not yet implemented
132 AIE::UseLockOp lock_op = *lock_ops.begin();
133 lock_op.emitOpError("Lowering for lock operations in NPU runtime "
134 "configuration is not yet implemented.");
135 return failure();
136 }
137 return success();
138 }
139
140 LogicalResult verifyNoUnsupportedOpsInBlock(Block &block) {
141 WalkResult unsupported_ops = block.walk([&](Operation *inner_op) {
142 return llvm::TypeSwitch<Operation *, WalkResult>(inner_op)
143 .Case<AIE::DMABDOp>(
144 [&](AIE::DMABDOp bd_op) { return WalkResult::advance(); })
145 .Case<AIE::UseLockOp>(
146 [&](AIE::UseLockOp lock_op) { return WalkResult::advance(); })
147 .Case<AIE::NextBDOp>(
148 [&](AIE::NextBDOp lock_op) { return WalkResult::advance(); })
149 .Case<AIE::EndOp>(
150 [&](AIE::EndOp lock_op) { return WalkResult::advance(); })
151 .Default([&](Operation *inner_op) {
152 auto error = block.getParentOp()->emitOpError(
153 "Unsupported operation within BD block.");
154 error.attachNote(inner_op->getLoc())
155 << "No lowering to NPU instructions available for this "
156 "operation.";
157 return WalkResult::interrupt();
158 });
159 });
160 if (unsupported_ops.wasInterrupted()) {
161 return failure();
162 }
163 return success();
164 }

179 LogicalResult setAddressForSingleBD(OpBuilder &builder, AIE::DMABDOp &bd_op,
180 AIE::TileOp &tile) {
181 uint32_t bd_id = bd_op.getBdId().value();
182 const AIE::AIETargetModel &target_model = AIE::getTargetModel(bd_op);
183 auto buf = bd_op.getBuffer();
184 uint64_t register_addr =
185 target_model.getDmaBdAddress(tile.getCol(), tile.getRow(), bd_id) +
186 target_model.getDmaBdAddressOffset(tile.getCol(), tile.getRow());
187 if (mlir::BlockArgument buf_arg =
188 llvm::dyn_cast<mlir::BlockArgument>(buf)) {
189 if (!target_model.isShimNOCTile(tile.getCol(), tile.getRow())) {
190 return bd_op->emitOpError("DDR memory (runtime input arguments) can "
191 "only be referred to on shim tiles.");
192 }
193 unsigned arg_idx = buf_arg.getArgNumber();
194 int64_t offset = bd_op.getOffsetInBytes();
195 builder.create<NpuAddressPatchOp>(bd_op.getLoc(),
196 /*addr*/ register_addr,
197 /*arg_idx*/ arg_idx,
198 /*arg_plus*/ offset);
199 } else if (AIE::BufferOp buffer =
200 llvm::dyn_cast<AIE::BufferOp>(buf.getDefiningOp())) {
201 uint64_t buf_addr;
202 if (!buffer.getAddress().has_value()) {
203 return bd_op->emitOpError(
204 "Cannot lower buffer without associated address. Run pass "
205 "--aie-assign-buffer-addresses first or manually assign an "
206 "address.");
207 }
208 buf_addr = *buffer.getAddress();
209 builder.create<NpuWrite32Op>(bd_op.getLoc(), register_addr, buf_addr,
210 nullptr, nullptr, nullptr);
211 } else {
212 return bd_op->emitOpError("Buffer argument must be either a constant "
213 "aie.buffer or a runtime "
214 "sequence input argument.");
215 }
216 return success();
217 }

219 LogicalResult rewriteSingleBD(OpBuilder &builder, Block &block,
220 AIE::TileOp &tile,
221 AIE::DMAChannelDir channelDir) {
222 AIE::DMABDOp bd_op = getBdForBlock(block);
223 const auto &target_model = AIE::getTargetModel(bd_op);
224 MemRefType buffer_type = bd_op.getBuffer().getType();
225 uint32_t addr_granularity = target_model.getAddressGenGranularity();
226
227 uint32_t bd_id = bd_op.getBdId().value();
228 int64_t offset = bd_op.getOffsetInBytes();
229 uint64_t len = bd_op.getLenInBytes();
230 uint64_t len_addr_granularity = len * 8 / addr_granularity;
231
232 if (offset * 8 % addr_granularity != 0) {
233 return bd_op->emitOpError("Offset must be aligned to ")
234 << (addr_granularity / 8) << " byte boundary.";
235 }
236
237 if (len < addr_granularity / 8) {
238 return bd_op->emitOpError("Transfer size of ")
239 << len << " bytes falls below minimum hardware transfer unit of "
240 << (addr_granularity / 8) << " bytes.";
241 }
242 // Process strides/wraps
243 std::optional<llvm::ArrayRef<AIE::BDDimLayoutAttr>> dims =
244 bd_op.getDimensions();
245 llvm::SmallVector<int64_t, 4> sizes = llvm::SmallVector<int64_t, 4>(4, 0);
246 llvm::SmallVector<int64_t, 4> strides = llvm::SmallVector<int64_t, 4>(4, 0);
247
248 // Padding
249 std::optional<llvm::ArrayRef<AIE::BDPadLayoutAttr>> padDims =
250 bd_op.getPadDimensions();
251 llvm::SmallVector<int64_t, 4> padBefore =
252 llvm::SmallVector<int64_t, 4>(4, 0);
253 llvm::SmallVector<int64_t, 4> padAfter =
254 llvm::SmallVector<int64_t, 4>(4, 0);
255 std::fill(padBefore.begin(), padBefore.end(), 0);
256 std::fill(padAfter.begin(), padAfter.end(), 0);
257
258 auto d0size = 0;
259 auto d0stride = 0;
260 auto d1size = 0;
261 auto d1stride = 0;
262 auto d2size = 0;
263 auto d2stride = 0;
264 auto iteration_size = 0;
265 auto iteration_stride = 0;
266
267 if (dims && dims->size() > 0) {
268 llvm::SmallVector<int64_t, 4> input_sizes =
269 llvm::SmallVector<int64_t, 4>(4, 1);
270 llvm::SmallVector<int64_t, 4> input_strides =
271 llvm::SmallVector<int64_t, 4>(4, 0);
272 if (dims->size() > 4) {
273 return bd_op->emitOpError("At most four data layout transformation "
274 "dimensions may be provided.");
275 }
276
277 for (size_t i = 0; i < dims->size(); i++) {
278 // Pass down dimensions in reverse order; in the MLIR, this allows
279 // us to specify step sizes/wraps in the same order as we would
280 // access a multi-dim C array, with the highest dimension first.
281 int j = dims->size() - i - 1;
282 input_sizes[i] = (*dims)[j].getSize();
283 input_strides[i] = (*dims)[j].getStride();
284 }
285
286 // Do not check input_sizes[3] because a repeat can still be considered a
287 // linear transfer
288 bool isLinearTransfer = (input_sizes[0] >= 1) && (input_sizes[1] == 1) &&
289 (input_sizes[2] == 1);
290
291 if (dims->size() > 2) {
292 d2size = (target_model.isMemTile(tile.getCol(), tile.getRow()))
293 ? (*dims)[2].getSize()
294 : 0;
295 }
296 if (padDims.has_value()) {
297 if (!target_model.isMemTile(tile.getCol(), tile.getRow()))
298 return bd_op->emitOpError()
299 << "Padding is only supported by memtile dma bds.";
300 if (padDims->size() > dims->size())
301 return bd_op->emitOpError()
302 << "Mismatch number of dimensions between padding(s)"
303 << " and wrap(s) and stride(s).";
304 if (channelDir == AIE::DMAChannelDir::MM2S) {
305 for (size_t i = 0; i < padDims->size(); i++) {
306 int j = padDims->size() - i - 1;
307 padBefore[i] = (*padDims)[j].getConstPadBefore();
308 padAfter[i] = (*padDims)[j].getConstPadAfter();
309 }
310 for (size_t i = padDims->size(); i < dims->size(); i++) {
311 padBefore[i] = 0;
312 padAfter[i] = 0;
313 }
314 } else
315 return bd_op->emitOpError()
316 << "supports padding only for MM2S direction on MemTiles.";
317 }
318 getHardwareStridesWraps(target_model, buffer_type, input_sizes,
319 input_strides, sizes, strides);
320
321 if (failed(verifyStridesWraps(bd_op, buffer_type, tile.getCol(),
322 tile.getRow(), input_sizes, input_strides,
323 sizes, strides, isLinearTransfer))) {
324 return failure();
325 }
326
327 iteration_size = sizes[3];
328 iteration_stride = strides[3];
329
330 if (!isLinearTransfer) {
331 // d0_size, d0_stride
332 d0size = sizes[0];
333 d0stride = strides[0];
334
335 // d1_size, d1_stride
336 d1size = sizes[1];
337 d1stride = strides[1];
338
339 // d2_stride
340 d2stride = strides[2];
341 // d2_size set elsewhere
342 }
343 if (input_sizes[3] > 1 && input_strides[3] == 0) {
344 // We allow users to encode the repeat_count as a dimension 3 stride
345 // of 0. This must lower to a iteration wrap of 0, so no stride is
346 // ever added. We then repeat the BD using the repeat_count in
347 // NpuPushQueueOp.
348 iteration_size = 0;
349 iteration_stride = 0;
350 }
351
352 // Ensure the total transfer length and the length expressed in the lowest
353 // three dimensions of strides/wraps agree. (Fourth dimension is
354 // iteration/repeat count and repeats the whole BD, so should not be
355 // incorporated in length of a single BD invocation.)
356 uint64_t len_dims_addr_granularity = 1;
357 for (size_t i = 0; i < 3; i++) {
358 len_dims_addr_granularity *= sizes[i];
359 }
360 if (len_dims_addr_granularity != len_addr_granularity) {
361 auto err =
362 bd_op->emitOpError(
363 "Buffer descriptor length does not match length of transfer "
364 "expressed by lowest three dimensions of data layout "
365 "transformation strides/wraps. ")
366 << "BD length is " << (len_addr_granularity * addr_granularity / 8)
367 << " bytes. "
368 << "Lowest three dimensions of data layout transformation would "
369 "result in transfer of "
370 << (len_dims_addr_granularity * addr_granularity / 8) << " bytes. ";
371 err.attachNote() << "Do not include the highest dimension size in "
372 "transfer length, as this is the BD repeat count.";
373 return failure();
374 }
375 } else {
376 if (padDims && target_model.isMemTile(tile.getCol(), tile.getRow()) &&
377 channelDir == AIE::DMAChannelDir::MM2S) {
378 return bd_op->emitOpError()
379 << "Padding requires n-d data layouts expressed as "
380 << "wrap(s) and stride(s).";
381 } else if (padDims) {
382 return bd_op->emitOpError() << "Padding is supported only on MemTiles.";
383 }
384 }
385 // find next BD ID, if any
386 uint32_t use_next_bd = 0;
387 uint32_t next_bd_id = 0;
388 if (bd_op.getNextBdId().has_value()) {
389 next_bd_id = bd_op.getNextBdId().value();
390 use_next_bd = 1;
391 }
392
393 builder.create<NpuWriteBdOp>(
394 bd_op.getLoc(), tile.getCol(), bd_id, len_addr_granularity, offset, 0,
395 0, 0, 0,
396 /* TODO: Strides/Wraps */
397 /*d0_size=*/d0size, /*d0_stride=*/d0stride,
398 /*d1_size=*/d1size, /*d1_stride=*/d1stride,
399 /*d2_size=*/d2size, /*d2_stride=*/d2stride,
400 /*iteration_current=*/0, /*iteration_size=*/iteration_size,
401 /*iteration_stride=*/iteration_stride,
402 /* TODO: Next BD */
403 /*next_bd=*/next_bd_id,
404 /*row=*/tile.getRow(),
405 /*use_next_bd=*/use_next_bd,
406 /*valid_bd=*/1,
407 /* TODO: Locks */
408 /*lock_rel_val=*/0, /*lock_rel_id=*/0, /*lock_acq_enable=*/0,
409 /*lock_acq_val=*/0, /*lock_ackq_id=*/0, /*d0_zero_before=*/padBefore[0],
410 /*d1_zero_before=*/padBefore[1], /*d2_zero_before=*/padBefore[2],
411 /*d0_zero_after=*/padAfter[0], /*d1_zero_after=*/padAfter[1],
412 /*d2_zero_after=*/padAfter[2],
413 /*burst_length=*/bd_op.getBurstLength());
414 return setAddressForSingleBD(builder, bd_op, tile);
415 }

417 LogicalResult hoistNextBdOpsIntoAttrs(DMAConfigureTaskOp op) {
418 Region &body = op.getBody();
419 for (auto it = body.begin(); it != body.end(); ++it) {
420 Block &block = *it;
421 if (shouldSkipBlock(block)) {
422 continue;
423 }
424 AIE::DMABDOp bd_op = getBdForBlock(block);
425 if (AIE::NextBDOp next_bd_op =
426 llvm::dyn_cast<AIE::NextBDOp>(block.getTerminator())) {
427 if (bd_op.getNextBdId().has_value()) {
428 auto error =
429 bd_op.emitOpError("Cannot specify both next_bd_id attribute and "
430 "aie.next_bd operation.");
431 error.attachNote(next_bd_op.getLoc())
432 << "Potentially conflicting next buffer descriptor ID specified "
433 "here.";
434 return failure();
435 }
436 Block &next_bd_block = *next_bd_op.getDest();
437 AIE::DMABDOp next_dma_bd_op = getBdForBlock(next_bd_block);
438 assert(next_dma_bd_op.getBdId()
439 .has_value()); // Next BD should have assigned ID, and this
440 // should have been checked by earlier
441 // verifyBdInBlock() call
442 bd_op.setNextBdId(next_dma_bd_op.getBdId().value());
443 OpBuilder builder(next_bd_op);
444 builder.create<AIE::EndOp>(next_bd_op.getLoc());
445 next_bd_op.erase();
446 }
447 }
448 return success();
449 }

451 LogicalResult rewriteSingleDMAConfigureTaskOp(DMAConfigureTaskOp op) {
452 OpBuilder builder(op);
453 AIE::TileOp tile = op.getTileOp();
454
455 if (!op.use_empty()) {
456 auto err = op.emitOpError("Cannot lower while op still has uses.");
457 mlir::Operation::use_range uses = op.getOperation()->getUses();
458 for (auto it = uses.begin(); it != uses.end(); ++it) {
459 err.attachNote(it->getOwner()->getLoc()) << "Used here.";
460 }
461 return failure();
462 }
463
464 Region &body = op.getBody();
465
466 // Verify each BD block first; subsequent functions rely on them being
467 // well-formed
468 for (auto it = body.begin(); it != body.end(); ++it) {
469 if (shouldSkipBlock(*it)) {
470 continue;
471 }
472 if (failed(verifyNoUnsupportedOpsInBlock(*it))) {
473 return failure();
474 }
475 if (failed(verifyBdInBlock(*it))) {
476 return failure();
477 }
478 if (failed(verifyOptionalLocksInBlock(*it))) {
479 return failure();
480 }
481 }
482
483 // Hoist next_bd operations into next_bd_id attribute of the dma_bd
484 if (failed(hoistNextBdOpsIntoAttrs(op))) {
485 return failure();
486 }
487
488 auto channelDir = op.getDirection();
489
490 // Lower all BDs
491 for (auto it = body.begin(); it != body.end(); ++it) {
492 Block &block = *it;
493 if (shouldSkipBlock(block)) {
494 continue;
495 }
496 if (failed(rewriteSingleBD(builder, block, tile, channelDir))) {
497 return failure();
498 }
499 }
500
501 op.erase();
502
503 return success();
504 }

506 LogicalResult rewriteDMAConfigureTaskOp(AIE::DeviceOp device) {
507 WalkResult result = device.walk([&](DMAConfigureTaskOp op) {
508 if (failed(rewriteSingleDMAConfigureTaskOp(op))) {
509 return WalkResult::interrupt();
510 }
511 return WalkResult::advance();
512 });
513 if (result.wasInterrupted()) {
514 return failure();
515 }
516 return success();
517 }

519 void runOnOperation() override {
520 AIE::DeviceOp device = getOperation();
521
522 // Convert DMAStartBD and DMAAwaitBD ops
523 ConversionTarget target(getContext());
524 target.addLegalDialect<AIEXDialect>();
525 target.addIllegalOp<DMAStartTaskOp>();
526 target.addIllegalOp<DMAAwaitTaskOp>();
527 RewritePatternSet patterns(&getContext());
528 patterns.insert<DMAStartTaskOpPattern>(&getContext());
529 patterns.insert<DMAAwaitTaskOpPattern>(&getContext());
530 if (failed(applyPartialConversion(device, target, std::move(patterns)))) {
531 signalPassFailure();
532 }
533
534 // Lower the configuration for the BDs
535 if (failed(rewriteDMAConfigureTaskOp(device))) {
536 signalPassFailure();
537 }
538 }
