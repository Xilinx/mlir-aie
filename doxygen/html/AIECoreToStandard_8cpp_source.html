38static StringRef getArchIntrinsicString(AIEArch arch) {
39 switch (arch) {
40 case AIEArch::AIE1:
41 return "aie";
42 case AIEArch::AIE2:
43 return "aie2";
44 case AIEArch::AIE2p:
45 return "aie2p";
46 }
47 llvm::report_fatal_error("unsupported arch");
48}

54static auto getAIE1Intrinsics(OpBuilder &builder) {
55 Type int32Type = IntegerType::get(builder.getContext(), 32);
56 Type int128Type = IntegerType::get(builder.getContext(), 128);
57 Type int384Type = IntegerType::get(builder.getContext(), 384);
58 Type floatType = Float32Type::get(builder.getContext());
59
60 // Note that not all of these are valid for a particular design, or needed.
61 // For right now, we will just accept the noise.
62 IntrinsicDecls functions = {
63 {"debug_i32", {int32Type}, {}},
64 {"llvm.aie.event0", {}, {}},
65 {"llvm.aie.event1", {}, {}},
66 {"llvm.aie.put.ms",
67 {int32Type, int32Type},
68 {}}, //(%channel, %value) -> ()
69 {"llvm.aie.put.wms",
70 {int32Type, int128Type},
71 {}}, //(%channel, %value) -> ()
72 {"llvm.aie.put.fms",
73 {int32Type, floatType},
74 {}}, //(%channel, %value) -> ()
75 {"llvm.aie.get.ss", {int32Type}, {int32Type}}, //(%channel, %value) -> ()
76 {"llvm.aie.get.wss",
77 {int32Type},
78 {int128Type}}, //(%channel, %value) -> ()
79 {"llvm.aie.get.fss", {int32Type}, {floatType}}, //(%channel, %value) -> ()
80 {"llvm.aie.put.mcd", {int384Type}, {}},
81 {"llvm.aie.get.scd", {}, {int384Type}},
82 {"llvm.aie.lock.acquire.reg",
83 {int32Type, int32Type},
84 {}}, //(%lock_id, %lock_val) -> ()
85 {"llvm.aie.lock.release.reg",
86 {int32Type, int32Type},
87 {}}, //(%lock_id, %lock_val) -> ()
88 };
89 return functions;
90}

144static void declareAIEIntrinsics(AIEArch arch, OpBuilder &builder) {
145 auto registerIntrinsics = [&builder](IntrinsicDecls functions) {
146 for (auto &i : functions) {
147 auto [name, argTypes, retTypes] = i;
148 builder
149 .create<func::FuncOp>(
150 builder.getUnknownLoc(), name,
151 FunctionType::get(builder.getContext(), argTypes, retTypes))
152 .setPrivate();
153 }
154 };
155 switch (arch) {
156 case AIEArch::AIE1:
157 registerIntrinsics(getAIE1Intrinsics(builder));
158 return;
159 case AIEArch::AIE2:
160 registerIntrinsics(getAIE2Intrinsics(builder));
161 return;
162 case AIEArch::AIE2p:
163 registerIntrinsics(getAIE2pIntrinsics(builder));
164 return;
165 }
166 llvm::report_fatal_error("unsupported arch");
167}

218 LogicalResult
219 matchAndRewrite(PutStreamOp op, OpAdaptor adaptor,

244 args.push_back(op.getChannel());
245 args.push_back(op.getStreamValue());
246 } else {
247 args.push_back(op.getStreamValue());
248 args.push_back(rewriter.create<arith::ConstantOp>(
249 op.getLoc(), IntegerType::get(rewriter.getContext(), 32),
250 rewriter.getI32IntegerAttr(0))); // tlast
251 }
252 rewriter.create<func::CallOp>(rewriter.getUnknownLoc(), putMSFunc, args);
253 rewriter.eraseOp(op);

301struct AIEPutCascadeToStdLowering : OpConversionPattern<PutCascadeOp> {
302 using OpConversionPattern::OpConversionPattern;
303 ModuleOp &module;
304
305 AIEPutCascadeToStdLowering(MLIRContext *context, ModuleOp &m,
306 PatternBenefit benefit = 1)
307 : OpConversionPattern(context, benefit), module(m) {}
308
309 LogicalResult
310 matchAndRewrite(PutCascadeOp op, OpAdaptor adaptor,
311 ConversionPatternRewriter &rewriter) const override {
312 auto device = op->getParentOfType<DeviceOp>();
313 const auto &targetModel = device.getTargetModel();
314 std::string funcName;
315 if (targetModel.getTargetArch() == AIEArch::AIE1)
316 funcName = "llvm.aie.put.mcd";
317 else if (targetModel.getTargetArch() == AIEArch::AIE2)
318 funcName = "llvm.aie2.mcd.write.vec";
319 else
320 funcName = "llvm.aie2p.mcd.write.vec";
321 auto putMCDFunc = module.lookupSymbol<func::FuncOp>(funcName);
322 if (!putMCDFunc)
323 return op.emitOpError("Could not find the intrinsic function ")
324 << funcName;
325 SmallVector<Value, 2> args;
326 Value cascadeValue = op.getCascadeValue();
327
328 // Check if we need a bitcast for the input value
329 Type expectedInputType = putMCDFunc.getFunctionType().getInput(0);
330 Type actualInputType = cascadeValue.getType();
331
332 if (expectedInputType != actualInputType) {
333 // Create a bitcast operation to convert from actual input type to
334 // expected type
335 cascadeValue = rewriter.create<vector::BitCastOp>(
336 op.getLoc(), expectedInputType, cascadeValue);
337 }
338
339 args.push_back(cascadeValue);
340 if (isa<AIE2TargetModel>(targetModel))
341 args.push_back(rewriter.create<arith::ConstantOp>(
342 op.getLoc(), IntegerType::get(rewriter.getContext(), 32),
343 rewriter.getI32IntegerAttr(1))); // enable
344
345 rewriter.create<func::CallOp>(rewriter.getUnknownLoc(), putMCDFunc, args);
346 rewriter.eraseOp(op);
347 return success();
348 }

401struct AIEUseLockToStdLowering : OpConversionPattern<UseLockOp> {
402 using OpConversionPattern::OpConversionPattern;
403 ModuleOp &module;
404
405 AIEUseLockToStdLowering(MLIRContext *context, ModuleOp &m,
406 PatternBenefit benefit = 1)
407 : OpConversionPattern(context, benefit), module(m) {}
408 LogicalResult
409 matchAndRewrite(UseLockOp useLock, OpAdaptor adaptor,
410 ConversionPatternRewriter &rewriter) const override {
411 if (!isa<DeviceOp>(useLock->getParentOp())) {
412 auto device = useLock->getParentOfType<DeviceOp>();
413 if (!device) {
414 return module.emitOpError("Device Not found!");
415 }
416 const auto &targetModel = device.getTargetModel();
417
418 // Generate the intrinsic name
419 std::string funcName;
420 if (targetModel.getTargetArch() == AIEArch::AIE1)
421 funcName = "llvm.aie.lock.";
422 else if (targetModel.getTargetArch() == AIEArch::AIE2)
423 funcName = "llvm.aie2.";
424 else
425 funcName = "llvm.aie2p.";
426 if (useLock.acquire() || useLock.acquireGE())
427 funcName += "acquire";
428 else if (useLock.release())
429 funcName += "release";
430 if (targetModel.getTargetArch() == AIEArch::AIE1)
431 funcName += ".reg";
432
433 auto useLockFunc = module.lookupSymbol<func::FuncOp>(funcName);
434 if (!useLockFunc)
435 return useLock.emitOpError("Could not find the intrinsic function!");
436
437 SmallVector<Value, 2> args;
438 auto lockValue = useLock.getLockValue();
439
440 // AIE2 acquire greater equal is encoded as a negative value.
441 if (useLock.acquireGE()) {
442 lockValue = -lockValue;
443 }
444 args.push_back(rewriter.create<arith::IndexCastOp>(
445 useLock.getLoc(), IntegerType::get(rewriter.getContext(), 32),
446 useLock.getLock()));
447 args.push_back(rewriter.create<arith::ConstantOp>(
448 useLock.getLoc(), IntegerType::get(rewriter.getContext(), 32),
449 rewriter.getI32IntegerAttr(lockValue)));
450
451 rewriter.create<func::CallOp>(rewriter.getUnknownLoc(), useLockFunc,
452 args);
453 }
454 rewriter.eraseOp(useLock);
455 return success();
456 }

459struct AIEBufferToStandard : OpConversionPattern<BufferOp> {
460 using OpConversionPattern::OpConversionPattern;
461 ModuleOp &module;
462 int tileCol = 0;
463 int tileRow = 0;
464 AIEBufferToStandard(MLIRContext *context, ModuleOp &m,
465 PatternBenefit benefit = 1, int tileCol = -1,
466 int tileRow = -1)
467 : OpConversionPattern(context, benefit), module(m), tileCol(tileCol),
468 tileRow(tileRow) {}
469 LogicalResult
470 matchAndRewrite(BufferOp buffer, OpAdaptor adaptor,
471 ConversionPatternRewriter &rewriter) const override {
472 rewriter.setInsertionPointToStart(module.getBody());
473 auto t = llvm::cast<MemRefType>(buffer.getType());
474 int col = llvm::cast<TileOp>(buffer.getTile().getDefiningOp()).getCol();
475 int row = llvm::cast<TileOp>(buffer.getTile().getDefiningOp()).getRow();
476 auto symName = buffer.name().getValue();
477 mlir::ElementsAttr initValue = buffer.getInitialValueAttr();
478 // Don't emit initialization for cores that don't "own" the buffer (to
479 // prevent duplication in the data section of the elf/object file)
480 if ((tileRow != row && tileRow != -1) || (tileCol != col && tileCol != -1))
481 initValue = nullptr;
482 rewriter.create<memref::GlobalOp>(
483 rewriter.getUnknownLoc(), symName, rewriter.getStringAttr("public"),
484 buffer.getType(), initValue, /*constant*/ false,
485 /*alignment*/ nullptr);
486
487 for (auto &use : make_early_inc_range(buffer.getResult().getUses())) {
488 Operation *user = use.getOwner();
489 rewriter.setInsertionPoint(user);
490 auto allocated = rewriter.create<memref::GetGlobalOp>(
491 rewriter.getUnknownLoc(), t, symName);
492 // Assume that buffers are aligned so they can be vectorized.
493 rewriter.create<memref::AssumeAlignmentOp>(rewriter.getUnknownLoc(),
494 allocated, 32);
495
496 use.set(allocated.getResult());
497 }
498
499 rewriter.eraseOp(buffer);
500 return success();
501 }

504struct AIECoreToStandardFunc : OpConversionPattern<CoreOp> {
505 using OpConversionPattern::OpConversionPattern;
506 ModuleOp &module;
507 IRMapping &mapper;
508 DenseMap<Operation *, SmallVector<BufferOp, 4>> &tileToBuffers;
509 int tileCol = 0;
510 int tileRow = 0;
511
512 AIECoreToStandardFunc(
513 MLIRContext *context, ModuleOp &m, IRMapping &mapper,
514 DenseMap<Operation *, SmallVector<BufferOp, 4>> &tileToBuffers,
515 PatternBenefit benefit = 1, int tileCol = 1, int tileRow = 1)
516 : OpConversionPattern(context, benefit), module(m), mapper(mapper),
517 tileToBuffers(tileToBuffers), tileCol(tileCol), tileRow(tileRow) {}
518
519 LogicalResult
520 matchAndRewrite(CoreOp op, OpAdaptor adaptor,
521 ConversionPatternRewriter &rewriter) const override {
522
523 int col = op.colIndex();
524 int row = op.rowIndex();
525
526 // Only pull code for the indicated function
527 if ((tileRow != row && tileRow != -1) ||
528 (tileCol != col && tileCol != -1)) {
529 rewriter.eraseOp(op);
530 return success();
531 }
532
533 // The parent should be an AIE.device op.
534 rewriter.setInsertionPointAfter(op->getParentOp());
535
536 std::string coreName("core_" + std::to_string(col) + "_" +
537 std::to_string(row));
538 auto coreFunc = rewriter.create<func::FuncOp>(
539 rewriter.getUnknownLoc(), coreName,
540 FunctionType::get(rewriter.getContext(), {}, {}));
541
542 rewriter.cloneRegionBefore(op.getBody(), coreFunc.getBody(),
543 coreFunc.getBody().begin(), mapper);
544
545 // Rewrite the AIE.end() op
546 coreFunc.getBody().walk([&](Operation *childOp) {
547 rewriter.setInsertionPointAfter(childOp);
548
549 if (isa<EndOp>(childOp)) {
550 rewriter.create<func::ReturnOp>(rewriter.getUnknownLoc(),
551 ValueRange({}));
552 rewriter.eraseOp(childOp);
553 }
554 });
555
556 rewriter.eraseOp(op);
557 return success();
558 }
559};

572// Lower AIE.event to llvm.aie.event intrinsic
573struct AIEEventOpToStdLowering : OpConversionPattern<EventOp> {
574 using OpConversionPattern::OpConversionPattern;
575 ModuleOp &module;
576
577 AIEEventOpToStdLowering(MLIRContext *context, ModuleOp &m,
578 PatternBenefit benefit = 1)
579 : OpConversionPattern(context, benefit), module(m) {}
580
581 LogicalResult
582 matchAndRewrite(EventOp op, OpAdaptor adaptor,
583 ConversionPatternRewriter &rewriter) const override {
584 auto device = op->getParentOfType<DeviceOp>();
585 std::string funcName;
586 SmallVector<Value, 1> args;
587 switch (device.getTargetModel().getTargetArch()) {
588 case AIEArch::AIE1:
589 funcName = "llvm.aie.event" + std::to_string(op.getVal());
590 break;
591 case AIEArch::AIE2:
592 funcName = "llvm.aie2.event";
593 args.push_back(rewriter.create<arith::ConstantOp>(
594 op.getLoc(), rewriter.getI32Type(),
595 rewriter.getI32IntegerAttr(op.getVal())));
596 break;
597 case AIEArch::AIE2p:
598 funcName = "llvm.aie2p.event";
599 args.push_back(rewriter.create<arith::ConstantOp>(
600 op.getLoc(), rewriter.getI32Type(),
601 rewriter.getI32IntegerAttr(op.getVal())));
602 break;
603 default:
604 return op->emitOpError("Unsupported AIEArch for EventOp lowering");
605 }
606 auto eventFunc = module.lookupSymbol<func::FuncOp>(funcName);
607 if (!eventFunc)
608 return op.emitOpError("Could not find the intrinsic function ")
609 << funcName;
610 rewriter.create<func::CallOp>(rewriter.getUnknownLoc(), eventFunc, args);
611 rewriter.eraseOp(op);
612 return success();
613 }

616struct AIECoreToStandardPass : AIECoreToStandardBase<AIECoreToStandardPass> {
617 void runOnOperation() override {
618
619 ModuleOp m = getOperation();
620 OpBuilder builder = OpBuilder::atBlockEnd(m.getBody());
621
622 DeviceOp deviceOp = DeviceOp::getForSymbolInModuleOrError(m, deviceName);
623 if (!deviceOp) {
624 return signalPassFailure();
625 }
626 const auto &targetModel = deviceOp.getTargetModel();
627
628 // Copy data layout attribute from DeviceOp to ModuleOp if present
629 if (auto dlAttr = deviceOp->getAttr(DLTIDialect::kDataLayoutAttrName)) {
630 m->setAttr(DLTIDialect::kDataLayoutAttrName, dlAttr);
631 }
632
