25LogicalResult DynamicTileAnalysis::runAnalysis(DeviceOp &device) {
26 LLVM_DEBUG(llvm::dbgs() << "\t---Begin DynamicTileAnalysis Constructor---\n");
27 // find the maxCol and maxRow
28 maxCol = device.getTargetModel().columns();
29 maxRow = device.getTargetModel().rows();
30
31 pathfinder->initialize(maxCol, maxRow, device.getTargetModel());
32
33 // For each flow (circuit + packet) in the device, add it to pathfinder. Each
34 // source can map to multiple different destinations (fanout). Control packet
35 // flows to be routed (as prioritized routings). Then followed by normal
36 // packet flows.
37 for (PacketFlowOp pktFlowOp : device.getOps<PacketFlowOp>()) {
38 Region &r = pktFlowOp.getPorts();
39 Block &b = r.front();
40 Port srcPort, dstPort;
41 TileOp srcTile, dstTile;
42 TileID srcCoords, dstCoords;
43 for (Operation &Op : b.getOperations()) {
44 if (auto pktSource = dyn_cast<PacketSourceOp>(Op)) {
45 srcTile = dyn_cast<TileOp>(pktSource.getTile().getDefiningOp());
46 srcPort = pktSource.port();
47 srcCoords = {srcTile.colIndex(), srcTile.rowIndex()};
48 } else if (auto pktDest = dyn_cast<PacketDestOp>(Op)) {
49 dstTile = dyn_cast<TileOp>(pktDest.getTile().getDefiningOp());
50 dstPort = pktDest.port();
51 dstCoords = {dstTile.colIndex(), dstTile.rowIndex()};
52 LLVM_DEBUG(llvm::dbgs()
53 << "\tAdding Packet Flow: (" << srcCoords.col << ", "
54 << srcCoords.row << ")"
55 << stringifyWireBundle(srcPort.bundle) << srcPort.channel
56 << " -> (" << dstCoords.col << ", " << dstCoords.row << ")"
57 << stringifyWireBundle(dstPort.bundle) << dstPort.channel
58 << "\n");
59 // todo: support many-to-one & many-to-many?
60 bool priorityFlow =
61 pktFlowOp.getPriorityRoute()
62 ? *pktFlowOp.getPriorityRoute()
63 : false; // Flows such as control packet flows are routed in
64 // priority, to ensure routing consistency.
65 pathfinder->addFlow(srcCoords, srcPort, dstCoords, dstPort,
66 /*isPktFlow*/ true, priorityFlow);
67 }
68 }
69 }
70
71 // Sort ctrlPktFlows into a deterministic order; concat ctrlPktFlows to flows
72 pathfinder->sortFlows(device.getTargetModel().columns(),
73 device.getTargetModel().rows());
74
75 // Add circuit flows.
76 for (FlowOp flowOp : device.getOps<FlowOp>()) {
77 TileOp srcTile = cast<TileOp>(flowOp.getSource().getDefiningOp());
78 TileOp dstTile = cast<TileOp>(flowOp.getDest().getDefiningOp());
79 TileID srcCoords = {srcTile.colIndex(), srcTile.rowIndex()};
80 TileID dstCoords = {dstTile.colIndex(), dstTile.rowIndex()};
81 Port srcPort = {flowOp.getSourceBundle(), flowOp.getSourceChannel()};
82 Port dstPort = {flowOp.getDestBundle(), flowOp.getDestChannel()};
83 LLVM_DEBUG(llvm::dbgs()
84 << "\tAdding Flow: (" << srcCoords.col << ", " << srcCoords.row
85 << ")" << stringifyWireBundle(srcPort.bundle) << srcPort.channel
86 << " -> (" << dstCoords.col << ", " << dstCoords.row << ")"
87 << stringifyWireBundle(dstPort.bundle) << dstPort.channel
88 << "\n");
89 pathfinder->addFlow(srcCoords, srcPort, dstCoords, dstPort,
90 /*isPktFlow*/ false, /*isPriorityFlow*/ false);
91 }
92
93 // add existing connections so Pathfinder knows which resources are
94 // available search all existing SwitchBoxOps for exising connections
95 for (SwitchboxOp switchboxOp : device.getOps<SwitchboxOp>()) {
96 if (!pathfinder->addFixedConnection(switchboxOp))
97 return switchboxOp.emitOpError() << "Unable to add fixed connections";
98 }
99
100 // all flows are now populated, call the congestion-aware pathfinder
101 // algorithm
102 // check whether the pathfinder algorithm creates a legal routing
103 if (auto maybeFlowSolutions = pathfinder->findPaths(maxIterations))
104 flowSolutions = maybeFlowSolutions.value();
105 else
106 return device.emitError("Unable to find a legal routing");
107
108 // initialize all flows as unprocessed to prep for rewrite
109 for (const auto &[PathEndPoint, switchSetting] : flowSolutions) {
110 processedFlows[PathEndPoint] = false;
111 }
112
113 // fill in coords to TileOps, SwitchboxOps, and ShimMuxOps
114 for (auto tileOp : device.getOps<TileOp>()) {
115 int col, row;
116 col = tileOp.colIndex();
117 row = tileOp.rowIndex();
118 assert(coordToTile.count({col, row}) == 0);
119 coordToTile[{col, row}] = tileOp;
120 }
121 for (auto switchboxOp : device.getOps<SwitchboxOp>()) {
122 int col = switchboxOp.colIndex();
123 int row = switchboxOp.rowIndex();
124 assert(coordToSwitchbox.count({col, row}) == 0);
125 coordToSwitchbox[{col, row}] = switchboxOp;
126 }
127 for (auto shimmuxOp : device.getOps<ShimMuxOp>()) {
128 int col = shimmuxOp.colIndex();
129 int row = shimmuxOp.rowIndex();
130 assert(coordToShimMux.count({col, row}) == 0);
131 coordToShimMux[{col, row}] = shimmuxOp;
132 }
133
134 LLVM_DEBUG(llvm::dbgs() << "\t---End DynamicTileAnalysis Constructor---\n");
135 return success();
136}

138TileOp DynamicTileAnalysis::getTile(OpBuilder &builder, int col, int row) {
139 if (coordToTile.count({col, row})) {
140 return coordToTile[{col, row}];
141 }
142 auto tileOp = TileOp::create(builder, builder.getUnknownLoc(), col, row);
143 coordToTile[{col, row}] = tileOp;
144 return tileOp;
145}

147TileOp DynamicTileAnalysis::getTile(OpBuilder &builder, const TileID &tileId) {
148 return getTile(builder, tileId.col, tileId.row);
149}

151SwitchboxOp DynamicTileAnalysis::getSwitchbox(OpBuilder &builder, int col,
152 int row) {
153 assert(col >= 0);
154 assert(row >= 0);
155 if (coordToSwitchbox.count({col, row})) {
156 return coordToSwitchbox[{col, row}];
157 }
158 auto switchboxOp = SwitchboxOp::create(builder, builder.getUnknownLoc(),
159 getTile(builder, col, row));
160 SwitchboxOp::ensureTerminator(switchboxOp.getConnections(), builder,
161 builder.getUnknownLoc());
162 coordToSwitchbox[{col, row}] = switchboxOp;
163 return switchboxOp;
164}
165
166ShimMuxOp DynamicTileAnalysis::getShimMux(OpBuilder &builder, int col) {
167 assert(col >= 0);
168 int row = 0;
169 if (coordToShimMux.count({col, row})) {
170 return coordToShimMux[{col, row}];
171 }
172 assert(getTile(builder, col, row).isShimNOCorPLTile());
173 auto switchboxOp = ShimMuxOp::create(builder, builder.getUnknownLoc(),
174 getTile(builder, col, row));
175 SwitchboxOp::ensureTerminator(switchboxOp.getConnections(), builder,
176 builder.getUnknownLoc());
177 coordToShimMux[{col, row}] = switchboxOp;
178 return switchboxOp;
179}

181void Pathfinder::initialize(int maxCol, int maxRow,
182 const AIETargetModel &targetModel) {
183
184 std::map<WireBundle, int> maxChannels;
185 auto intraconnect = [&](int col, int row) {
186 TileID coords = {col, row};
187 SwitchboxConnect sb = {coords};
188
189 for (int i = 0, e = getMaxEnumValForWireBundle() + 1; i < e; ++i) {
190 WireBundle bundle = symbolizeWireBundle(i).value();
191 // get all ports into current switchbox
192 int channels =
193 targetModel.getNumSourceSwitchboxConnections(col, row, bundle);
194 if (channels == 0 && targetModel.isShimNOCorPLTile(col, row)) {
195 // wordaround for shimMux
196 channels = targetModel.getNumSourceShimMuxConnections(col, row, bundle);
197 }
198 for (int channel = 0; channel < channels; channel++) {
199 sb.srcPorts.push_back(Port{bundle, channel});
200 }
201 // get all ports out of current switchbox
202 channels = targetModel.getNumDestSwitchboxConnections(col, row, bundle);
203 if (channels == 0 && targetModel.isShimNOCorPLTile(col, row)) {
204 // wordaround for shimMux
205 channels = targetModel.getNumDestShimMuxConnections(col, row, bundle);
206 }
207 for (int channel = 0; channel < channels; channel++) {
208 sb.dstPorts.push_back(Port{bundle, channel});
209 }
210 maxChannels[bundle] = channels;
211 }
212 // initialize matrices
213 sb.resize();
214 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
215 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
216 auto &pIn = sb.srcPorts[i];
217 auto &pOut = sb.dstPorts[j];
218 if (targetModel.isLegalTileConnection(col, row, pIn.bundle, pIn.channel,
219 pOut.bundle, pOut.channel))
220 sb.connectivity[i][j] = Connectivity::AVAILABLE;
221 else {
222 sb.connectivity[i][j] = Connectivity::INVALID;
223 if (targetModel.isShimNOCorPLTile(col, row)) {
224 // wordaround for shimMux
225 auto isBundleInList = [](WireBundle bundle,
226 std::vector<WireBundle> bundles) {
227 return std::find(bundles.begin(), bundles.end(), bundle) !=
228 bundles.end();
229 };
230 const std::vector<WireBundle> bundles = {
231 WireBundle::DMA, WireBundle::NOC, WireBundle::PLIO};
232 if (isBundleInList(pIn.bundle, bundles) ||
233 isBundleInList(pOut.bundle, bundles))
234 sb.connectivity[i][j] = Connectivity::AVAILABLE;
235 }
236 }
237 }
238 }
239 graph[std::make_pair(coords, coords)] = sb;
240 };
241
242 auto interconnect = [&](int col, int row, int targetCol, int targetRow,
243 WireBundle srcBundle, WireBundle dstBundle) {
244 SwitchboxConnect sb = {{col, row}, {targetCol, targetRow}};
245 for (int channel = 0; channel < maxChannels[srcBundle]; channel++) {
246 sb.srcPorts.push_back(Port{srcBundle, channel});
247 sb.dstPorts.push_back(Port{dstBundle, channel});
248 }
249 sb.resize();
250 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
251 sb.connectivity[i][i] = Connectivity::AVAILABLE;
252 }
253 graph[std::make_pair(TileID{col, row}, TileID{targetCol, targetRow})] = sb;
254 };
255
256 for (int row = 0; row <= maxRow; row++) {
257 for (int col = 0; col <= maxCol; col++) {
258 maxChannels.clear();
259 // connections within the same switchbox
260 intraconnect(col, row);
261
262 // connections between switchboxes
263 if (row > 0) {
264 // from south to north
265 interconnect(col, row, col, row - 1, WireBundle::South,
266 WireBundle::North);
267 }
268 if (row < maxRow) {
269 // from north to south
270 interconnect(col, row, col, row + 1, WireBundle::North,
271 WireBundle::South);
272 }
273 if (col > 0) {
274 // from east to west
275 interconnect(col, row, col - 1, row, WireBundle::West,
276 WireBundle::East);
277 }
278 if (col < maxCol) {
279 // from west to east
280 interconnect(col, row, col + 1, row, WireBundle::East,
281 WireBundle::West);
282 }
283 }
284 }
285}

287// Add a flow from src to dst can have an arbitrary number of dst locations
288// due to fanout.
289void Pathfinder::addFlow(TileID srcCoords, Port srcPort, TileID dstCoords,
290 Port dstPort, bool isPacketFlow, bool isPriorityFlow) {
291 // check if a flow with this source already exists
292 for (auto &[_, prioritized, src, dsts] : flows) {
293 if (src.coords == srcCoords && src.port == srcPort) {
294 if (isPriorityFlow) {
295 prioritized = true;
296 dsts.emplace(dsts.begin(), PathEndPoint{dstCoords, dstPort});
297 } else
298 dsts.emplace_back(PathEndPoint{dstCoords, dstPort});
299 return;
300 }
301 }
302
303 // Assign a group ID for packet flows
304 // any overlapping in source/destination will lead to the same group ID
305 // channel sharing will happen within the same group ID
306 // for circuit flows, group ID is always -1, and no channel sharing
307 int packetGroupId = -1;
308 if (isPacketFlow) {
309 bool found = false;
310 for (auto &[existingId, _, src, dsts] : flows) {
311 if (src.coords == srcCoords && src.port == srcPort) {
312 packetGroupId = existingId;
313 found = true;
314 break;
315 }
316 for (auto &dst : dsts) {
317 if (dst.coords == dstCoords && dst.port == dstPort) {
318 packetGroupId = existingId;
319 found = true;
320 break;
321 }
322 }
323 packetGroupId = std::max(packetGroupId, existingId);
324 }
325 if (!found) {
326 packetGroupId++;
327 }
328 }
329 // If no existing flow was found with this source, create a new flow.
330 flows.push_back(
331 Flow{packetGroupId, isPriorityFlow, PathEndPoint{srcCoords, srcPort},
332 std::vector<PathEndPoint>{PathEndPoint{dstCoords, dstPort}}});
333}

335// Sort flows to (1) get deterministic routing, and (2) perform routings on
336// prioritized flows before others, for routing consistency on those flows.
337void Pathfinder::sortFlows(const int maxCol, const int maxRow) {
338 std::vector<Flow> priorityFlows;
339 std::vector<Flow> normalFlows;
340 for (auto f : flows) {
341 if (f.isPriorityFlow)
342 priorityFlows.push_back(f);
343 else
344 normalFlows.push_back(f);
345 }
346 std::sort(priorityFlows.begin(), priorityFlows.end(),
347 [](const auto &lhs, const auto &rhs) {
348 // Compare tuple of properties in priority order:
349 // (col, row, bundle, channel)
350 auto lhsKey =
351 std::make_tuple(lhs.src.coords.col, lhs.src.coords.row,
352 getWireBundleAsInt(lhs.src.port.bundle),
353 lhs.src.port.channel);
354 auto rhsKey =
355 std::make_tuple(rhs.src.coords.col, rhs.src.coords.row,
356 getWireBundleAsInt(rhs.src.port.bundle),
357 rhs.src.port.channel);
358 return lhsKey < rhsKey;
359 });
360 flows = priorityFlows;
361 flows.insert(flows.end(), normalFlows.begin(), normalFlows.end());
362}

364// Keep track of connections already used in the AIE; Pathfinder algorithm
365// will avoid using these.
366bool Pathfinder::addFixedConnection(SwitchboxOp switchboxOp) {
367 int col = switchboxOp.colIndex();
368 int row = switchboxOp.rowIndex();
369 TileID coords = {col, row};
370 auto &sb = graph[std::make_pair(coords, coords)];
371 for (ConnectOp connectOp : switchboxOp.getOps<ConnectOp>()) {
372 bool found = false;
373 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
374 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
375 if (sb.srcPorts[i] == connectOp.sourcePort() &&
376 sb.dstPorts[j] == connectOp.destPort() &&
377 sb.connectivity[i][j] == Connectivity::AVAILABLE) {
378 sb.connectivity[i][j] = Connectivity::INVALID;
379 found = true;
380 }
381 }
382 }
383 if (!found) {
384 // could not add such a fixed connection
385 return false;
386 }
387 }
388 return true;
389}

391static constexpr double INF = std::numeric_limits<double>::max();
392
393std::map<PathEndPoint, PathEndPoint>
394Pathfinder::dijkstraShortestPaths(PathEndPoint src) {
395 // Use std::map instead of DenseMap because DenseMap doesn't let you
396 // overwrite tombstones.
397 std::map<PathEndPoint, double> distance;
398 std::map<PathEndPoint, PathEndPoint> preds;
399 std::map<PathEndPoint, uint64_t> indexInHeap;
400 enum Color { WHITE, GRAY, BLACK };
401 std::map<PathEndPoint, Color> colors;
402 typedef d_ary_heap_indirect<
403 /*Value=*/PathEndPoint, /*Arity=*/4,
404 /*IndexInHeapPropertyMap=*/std::map<PathEndPoint, uint64_t>,
405 /*DistanceMap=*/std::map<PathEndPoint, double> &,
406 /*Compare=*/std::less<>>
407 MutableQueue;
408 MutableQueue Q(distance, indexInHeap);
409
410 distance[src] = 0.0;
411 Q.push(src);
412 while (!Q.empty()) {
413 src = Q.top();
414 Q.pop();
415
416 // get all channels src connects to
417 if (channels.count(src) == 0) {
418 auto &sb = graph[std::make_pair(src.coords, src.coords)];
419 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
420 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
421 if (sb.srcPorts[i] == src.port &&
422 sb.connectivity[i][j] == Connectivity::AVAILABLE) {
423 // connections within the same switchbox
424 channels[src].push_back(PathEndPoint{src.coords, sb.dstPorts[j]});
425 }
426 }
427 }
428 // connections to neighboring switchboxes
429 std::vector<std::pair<TileID, Port>> neighbors = {
430 {{src.coords.col, src.coords.row - 1},
431 {WireBundle::North, src.port.channel}},
432 {{src.coords.col - 1, src.coords.row},
433 {WireBundle::East, src.port.channel}},
434 {{src.coords.col, src.coords.row + 1},
435 {WireBundle::South, src.port.channel}},
436 {{src.coords.col + 1, src.coords.row},
437 {WireBundle::West, src.port.channel}}};
438
439 for (const auto &[neighborCoords, neighborPort] : neighbors) {
440 if (graph.count(std::make_pair(src.coords, neighborCoords)) > 0 &&
441 src.port.bundle == getConnectingBundle(neighborPort.bundle)) {
442 auto &sb = graph[std::make_pair(src.coords, neighborCoords)];
443 if (std::find(sb.dstPorts.begin(), sb.dstPorts.end(), neighborPort) !=
444 sb.dstPorts.end())
445 channels[src].push_back({neighborCoords, neighborPort});
446 }
447 }
448 std::sort(channels[src].begin(), channels[src].end());
449 }
450
451 for (auto &dest : channels[src]) {
452 if (distance.count(dest) == 0)
453 distance[dest] = INF;
454 auto &sb = graph[std::make_pair(src.coords, dest.coords)];
455 size_t i = std::distance(
456 sb.srcPorts.begin(),
457 std::find(sb.srcPorts.begin(), sb.srcPorts.end(), src.port));
458 size_t j = std::distance(
459 sb.dstPorts.begin(),
460 std::find(sb.dstPorts.begin(), sb.dstPorts.end(), dest.port));
461 assert(i < sb.srcPorts.size());
462 assert(j < sb.dstPorts.size());
463 bool relax = distance[src] + sb.demand[i][j] < distance[dest];
464 if (colors.count(dest) == 0) {
465 // was WHITE
466 if (relax) {
467 distance[dest] = distance[src] + sb.demand[i][j];
468 preds[dest] = src;
469 colors[dest] = GRAY;
470 }
471 Q.push(dest);
472 } else if (colors[dest] == GRAY && relax) {
473 distance[dest] = distance[src] + sb.demand[i][j];
474 preds[dest] = src;
475 }
476 }
477 colors[src] = BLACK;
478 }
479
480 return preds;
481}
482
483// Perform congestion-aware routing for all flows which have been added.
484// Use Dijkstra's shortest path to find routes, and use "demand" as the
485// weights. If the routing finds too much congestion, update the demand
486// weights and repeat the process until a valid solution is found. Returns a
487// map specifying switchbox settings for all flows. If no legal routing can be
488// found after maxIterations, returns empty vector.
489std::optional<std::map<PathEndPoint, SwitchSettings>>
490Pathfinder::findPaths(const int maxIterations) {
491 LLVM_DEBUG(llvm::dbgs() << "\t---Begin Pathfinder::findPaths---\n");
492 std::map<PathEndPoint, SwitchSettings> routingSolution;
493 // initialize all Channel histories to 0
494 for (auto &[_, sb] : graph) {
495 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
496 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
497 sb.usedCapacity[i][j] = 0;
498 sb.overCapacity[i][j] = 0;
499 sb.isPriority[i][j] = false;
500 }
501 }
502 }
503
504 // group flows based on packetGroupId
505 llvm::MapVector<int, std::vector<Flow>> groupedFlows;
506 for (auto &f : flows) {
507 if (groupedFlows.count(f.packetGroupId) == 0) {
508 groupedFlows[f.packetGroupId] = std::vector<Flow>();
509 }
510 groupedFlows[f.packetGroupId].push_back(f);
511 }
512
513 int iterationCount = -1;
514 int illegalEdges = 0;
515#ifndef NDEBUG
516 int totalPathLength = 0;
517#endif
518 do {
519 // if reach maxIterations, throw an error since no routing can be found
520 if (++iterationCount >= maxIterations) {
521 LLVM_DEBUG(llvm::dbgs()
522 << "\t\tPathfinder: maxIterations has been exceeded ("
523 << maxIterations
524 << " iterations)...unable to find routing for flows.\n");
525 return std::nullopt;
526 }
527
528 LLVM_DEBUG(llvm::dbgs() << "\t\t---Begin findPaths iteration #"
529 << iterationCount << "---\n");
530 // update demand at the beginning of each iteration
531 for (auto &[_, sb] : graph) {
532 sb.updateDemand();
533 }
534
535 // "rip up" all routes
536 illegalEdges = 0;
537#ifndef NDEBUG
538 totalPathLength = 0;
539#endif
540 routingSolution.clear();
541 for (auto &[_, sb] : graph) {
542 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
543 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
544 sb.usedCapacity[i][j] = 0;
545 sb.packetFlowCount[i][j] = 0;
546 sb.packetGroupId[i][j] = -1;
547 }
548 }
549 }
550
551 // for each flow, find the shortest path from source to destination
552 // update used_capacity for the path between them
553
554 for (const auto &[_, flows] : groupedFlows) {
555 for (const auto &[packetGroupId, isPriority, src, dsts] : flows) {
556 // Use dijkstra to find path given current demand from the start
557 // switchbox; find the shortest paths to each other switchbox. Output is
558 // in the predecessor map, which must then be processed to get
559 // individual switchbox settings
560 std::set<PathEndPoint> processed;
561 std::map<PathEndPoint, PathEndPoint> preds = dijkstraShortestPaths(src);
562
563 // trace the path of the flow backwards via predecessors
564 // increment used_capacity for the associated channels
565 SwitchSettings switchSettings;
566 processed.insert(src);
567 for (auto endPoint : dsts) {
568 if (endPoint == src) {
569 // route to self
570 switchSettings[src.coords].srcs.push_back(src.port);
571 switchSettings[src.coords].dsts.push_back(src.port);
572 }
573 auto curr = endPoint;
574 // trace backwards until a vertex already processed is reached
575 while (!processed.count(curr)) {
576 auto &sb = graph[std::make_pair(preds[curr].coords, curr.coords)];
577 size_t i =
578 std::distance(sb.srcPorts.begin(),
579 std::find(sb.srcPorts.begin(), sb.srcPorts.end(),
580 preds[curr].port));
581 size_t j = std::distance(
582 sb.dstPorts.begin(),
583 std::find(sb.dstPorts.begin(), sb.dstPorts.end(), curr.port));
584 assert(i < sb.srcPorts.size());
585 assert(j < sb.dstPorts.size());
586 sb.isPriority[i][j] = isPriority;
587 if (packetGroupId >= 0 &&
588 (sb.packetGroupId[i][j] == -1 ||
589 sb.packetGroupId[i][j] == packetGroupId)) {
590 for (size_t k = 0; k < sb.srcPorts.size(); k++) {
591 for (size_t l = 0; l < sb.dstPorts.size(); l++) {
592 if (k == i || l == j) {
593 sb.packetGroupId[k][l] = packetGroupId;
594 }
595 }
596 }
597 sb.packetFlowCount[i][j]++;
598 // maximum packet stream sharing per channel
599 if (sb.packetFlowCount[i][j] >= MAX_PACKET_STREAM_CAPACITY) {
600 sb.packetFlowCount[i][j] = 0;
601 sb.usedCapacity[i][j]++;
602 }
603 } else {
604 sb.usedCapacity[i][j]++;
605 }
606 // if at capacity, bump demand to discourage using this Channel
607 // this means the order matters!
608 sb.bumpDemand(i, j);
609 if (preds[curr].coords == curr.coords) {
610 switchSettings[preds[curr].coords].srcs.push_back(
611 preds[curr].port);
612 switchSettings[curr.coords].dsts.push_back(curr.port);
613 }
614 processed.insert(curr);
615 curr = preds[curr];
616 }
617 }
618 // add this flow to the proposed solution
619 routingSolution[src] = switchSettings;
620 }
621 for (auto &[_, sb] : graph) {
622 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
623 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
624 // fix used capacity for packet flows
625 if (sb.packetFlowCount[i][j] > 0) {
626 sb.packetFlowCount[i][j] = 0;
627 sb.usedCapacity[i][j]++;
628 }
629 sb.bumpDemand(i, j);
630 }
631 }
632 }
633 }
634
635 for (auto &[_, sb] : graph) {
636 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
637 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
638 // check that every channel does not exceed max capacity
639 if (sb.usedCapacity[i][j] > MAX_CIRCUIT_STREAM_CAPACITY) {
640 sb.overCapacity[i][j]++;
641 illegalEdges++;
642 LLVM_DEBUG(
643 llvm::dbgs()
644 << "\t\t\tToo much capacity on (" << sb.srcCoords.col << ","
645 << sb.srcCoords.row << ") " << sb.srcPorts[i].bundle
646 << sb.srcPorts[i].channel << " -> (" << sb.dstCoords.col << ","
647 << sb.dstCoords.row << ") " << sb.dstPorts[j].bundle
648 << sb.dstPorts[j].channel << ", used_capacity = "
649 << sb.usedCapacity[i][j] << ", demand = " << sb.demand[i][j]
650 << ", over_capacity_count = " << sb.overCapacity[i][j] << "\n");
651 }
652#ifndef NDEBUG
653 // calculate total path length (across switchboxes)
654 if (sb.srcCoords != sb.dstCoords) {
655 totalPathLength += sb.usedCapacity[i][j];
656 }
657#endif
658 }
659 }
660 }
661
662#ifndef NDEBUG
663 for (const auto &[PathEndPoint, switchSetting] : routingSolution) {
664 LLVM_DEBUG(llvm::dbgs()
665 << "\t\t\tFlow starting at (" << PathEndPoint.coords.col << ","
666 << PathEndPoint.coords.row << "):\t");
667 LLVM_DEBUG(llvm::dbgs() << switchSetting);
668 }
669 LLVM_DEBUG(llvm::dbgs()
670 << "\t\t---End findPaths iteration #" << iterationCount
671 << " , illegal edges count = " << illegalEdges
672 << ", total path length = " << totalPathLength << "---\n");
673#endif
674 } while (illegalEdges >
675 0); // continue iterations until a legal routing is found
676
677 LLVM_DEBUG(llvm::dbgs() << "\t---End Pathfinder::findPaths---\n");
678 return routingSolution;
679}
