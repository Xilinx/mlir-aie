25LogicalResult DynamicTileAnalysis::runAnalysis(DeviceOp &device) {
26 LLVM_DEBUG(llvm::dbgs() << "\t---Begin DynamicTileAnalysis Constructor---\n");
27 // find the maxCol and maxRow
28 maxCol = 0;
29 maxRow = 0;
30 for (TileOp tileOp : device.getOps<TileOp>()) {
31 maxCol = std::max(maxCol, tileOp.colIndex());
32 maxRow = std::max(maxRow, tileOp.rowIndex());
33 }
34
35 pathfinder->initialize(maxCol, maxRow, device.getTargetModel());
36
37 // For each flow (circuit + packet) in the device, add it to pathfinder. Each
38 // source can map to multiple different destinations (fanout). Control packet
39 // flows to be routed (as prioritized routings). Then followed by normal
40 // packet flows.
41 for (PacketFlowOp pktFlowOp : device.getOps<PacketFlowOp>()) {
42 Region &r = pktFlowOp.getPorts();
43 Block &b = r.front();
44 Port srcPort, dstPort;
45 TileOp srcTile, dstTile;
46 TileID srcCoords, dstCoords;
47 for (Operation &Op : b.getOperations()) {
48 if (auto pktSource = dyn_cast<PacketSourceOp>(Op)) {
49 srcTile = dyn_cast<TileOp>(pktSource.getTile().getDefiningOp());
50 srcPort = pktSource.port();
51 srcCoords = {srcTile.colIndex(), srcTile.rowIndex()};
52 } else if (auto pktDest = dyn_cast<PacketDestOp>(Op)) {
53 dstTile = dyn_cast<TileOp>(pktDest.getTile().getDefiningOp());
54 dstPort = pktDest.port();
55 dstCoords = {dstTile.colIndex(), dstTile.rowIndex()};
56 LLVM_DEBUG(llvm::dbgs()
57 << "\tAdding Packet Flow: (" << srcCoords.col << ", "
58 << srcCoords.row << ")"
59 << stringifyWireBundle(srcPort.bundle) << srcPort.channel
60 << " -> (" << dstCoords.col << ", " << dstCoords.row << ")"
61 << stringifyWireBundle(dstPort.bundle) << dstPort.channel
62 << "\n");
63 // todo: support many-to-one & many-to-many?
64 bool priorityFlow =
65 pktFlowOp.getPriorityRoute()
66 ? *pktFlowOp.getPriorityRoute()
67 : false; // Flows such as control packet flows are routed in
68 // priority, to ensure routing consistency.
69 pathfinder->addFlow(srcCoords, srcPort, dstCoords, dstPort,
70 /*isPktFlow*/ true, priorityFlow);
71 }
72 }
73 }
74
75 // Sort ctrlPktFlows into a deterministic order; concat ctrlPktFlows to flows
76 pathfinder->sortFlows(device.getTargetModel().columns(),
77 device.getTargetModel().rows());
78
79 // Add circuit flows.
80 for (FlowOp flowOp : device.getOps<FlowOp>()) {
81 TileOp srcTile = cast<TileOp>(flowOp.getSource().getDefiningOp());
82 TileOp dstTile = cast<TileOp>(flowOp.getDest().getDefiningOp());
83 TileID srcCoords = {srcTile.colIndex(), srcTile.rowIndex()};
84 TileID dstCoords = {dstTile.colIndex(), dstTile.rowIndex()};
85 Port srcPort = {flowOp.getSourceBundle(), flowOp.getSourceChannel()};
86 Port dstPort = {flowOp.getDestBundle(), flowOp.getDestChannel()};
87 LLVM_DEBUG(llvm::dbgs()
88 << "\tAdding Flow: (" << srcCoords.col << ", " << srcCoords.row
89 << ")" << stringifyWireBundle(srcPort.bundle) << srcPort.channel
90 << " -> (" << dstCoords.col << ", " << dstCoords.row << ")"
91 << stringifyWireBundle(dstPort.bundle) << dstPort.channel
92 << "\n");
93 pathfinder->addFlow(srcCoords, srcPort, dstCoords, dstPort,
94 /*isPktFlow*/ false, /*isPriorityFlow*/ false);
95 }
96
97 // add existing connections so Pathfinder knows which resources are
98 // available search all existing SwitchBoxOps for exising connections
99 for (SwitchboxOp switchboxOp : device.getOps<SwitchboxOp>()) {
100 if (!pathfinder->addFixedConnection(switchboxOp))
101 return switchboxOp.emitOpError() << "Unable to add fixed connections";
102 }
103
104 // all flows are now populated, call the congestion-aware pathfinder
105 // algorithm
106 // check whether the pathfinder algorithm creates a legal routing
107 if (auto maybeFlowSolutions = pathfinder->findPaths(maxIterations))
108 flowSolutions = maybeFlowSolutions.value();
109 else
110 return device.emitError("Unable to find a legal routing");
111
112 // initialize all flows as unprocessed to prep for rewrite
113 for (const auto &[PathEndPoint, switchSetting] : flowSolutions) {
114 processedFlows[PathEndPoint] = false;
115 }
116
117 // fill in coords to TileOps, SwitchboxOps, and ShimMuxOps
118 for (auto tileOp : device.getOps<TileOp>()) {
119 int col, row;
120 col = tileOp.colIndex();
121 row = tileOp.rowIndex();
122 maxCol = std::max(maxCol, col);
123 maxRow = std::max(maxRow, row);
124 assert(coordToTile.count({col, row}) == 0);
125 coordToTile[{col, row}] = tileOp;
126 }
127 for (auto switchboxOp : device.getOps<SwitchboxOp>()) {
128 int col = switchboxOp.colIndex();
129 int row = switchboxOp.rowIndex();
130 assert(coordToSwitchbox.count({col, row}) == 0);
131 coordToSwitchbox[{col, row}] = switchboxOp;
132 }
133 for (auto shimmuxOp : device.getOps<ShimMuxOp>()) {
134 int col = shimmuxOp.colIndex();
135 int row = shimmuxOp.rowIndex();
136 assert(coordToShimMux.count({col, row}) == 0);
137 coordToShimMux[{col, row}] = shimmuxOp;
138 }
139
140 LLVM_DEBUG(llvm::dbgs() << "\t---End DynamicTileAnalysis Constructor---\n");
141 return success();
142}

144TileOp DynamicTileAnalysis::getTile(OpBuilder &builder, int col, int row) {
145 if (coordToTile.count({col, row})) {
146 return coordToTile[{col, row}];
147 }
148 auto tileOp = builder.create<TileOp>(builder.getUnknownLoc(), col, row);
149 coordToTile[{col, row}] = tileOp;
150 maxCol = std::max(maxCol, col);
151 maxRow = std::max(maxRow, row);
152 return tileOp;
153}
154
155SwitchboxOp DynamicTileAnalysis::getSwitchbox(OpBuilder &builder, int col,
156 int row) {
157 assert(col >= 0);
158 assert(row >= 0);
159 if (coordToSwitchbox.count({col, row})) {
160 return coordToSwitchbox[{col, row}];
161 }
162 auto switchboxOp = builder.create<SwitchboxOp>(builder.getUnknownLoc(),
163 getTile(builder, col, row));
164 SwitchboxOp::ensureTerminator(switchboxOp.getConnections(), builder,
165 builder.getUnknownLoc());
166 coordToSwitchbox[{col, row}] = switchboxOp;
167 maxCol = std::max(maxCol, col);
168 maxRow = std::max(maxRow, row);
169 return switchboxOp;
170}

172ShimMuxOp DynamicTileAnalysis::getShimMux(OpBuilder &builder, int col) {
173 assert(col >= 0);
174 int row = 0;
175 if (coordToShimMux.count({col, row})) {
176 return coordToShimMux[{col, row}];
177 }
178 assert(getTile(builder, col, row).isShimNOCorPLTile());
179 auto switchboxOp = builder.create<ShimMuxOp>(builder.getUnknownLoc(),
180 getTile(builder, col, row));
181 SwitchboxOp::ensureTerminator(switchboxOp.getConnections(), builder,
182 builder.getUnknownLoc());
183 coordToShimMux[{col, row}] = switchboxOp;
184 maxCol = std::max(maxCol, col);
185 maxRow = std::max(maxRow, row);
186 return switchboxOp;
187}

189void Pathfinder::initialize(int maxCol, int maxRow,
190 const AIETargetModel &targetModel) {
191
192 std::map<WireBundle, int> maxChannels;
193 auto intraconnect = [&](int col, int row) {
194 TileID coords = {col, row};
195 SwitchboxConnect sb = {coords};
196
197 for (int i = 0, e = getMaxEnumValForWireBundle() + 1; i < e; ++i) {
198 WireBundle bundle = symbolizeWireBundle(i).value();
199 // get all ports into current switchbox
200 int channels =
201 targetModel.getNumSourceSwitchboxConnections(col, row, bundle);
202 if (channels == 0 && targetModel.isShimNOCorPLTile(col, row)) {
203 // wordaround for shimMux
204 channels = targetModel.getNumSourceShimMuxConnections(col, row, bundle);
205 }
206 for (int channel = 0; channel < channels; channel++) {
207 sb.srcPorts.push_back(Port{bundle, channel});
208 }
209 // get all ports out of current switchbox
210 channels = targetModel.getNumDestSwitchboxConnections(col, row, bundle);
211 if (channels == 0 && targetModel.isShimNOCorPLTile(col, row)) {
212 // wordaround for shimMux
213 channels = targetModel.getNumDestShimMuxConnections(col, row, bundle);
214 }
215 for (int channel = 0; channel < channels; channel++) {
216 sb.dstPorts.push_back(Port{bundle, channel});
217 }
218 maxChannels[bundle] = channels;
219 }
220 // initialize matrices
221 sb.resize();
222 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
223 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
224 auto &pIn = sb.srcPorts[i];
225 auto &pOut = sb.dstPorts[j];
226 if (targetModel.isLegalTileConnection(col, row, pIn.bundle, pIn.channel,
227 pOut.bundle, pOut.channel))
228 sb.connectivity[i][j] = Connectivity::AVAILABLE;
229 else {
230 sb.connectivity[i][j] = Connectivity::INVALID;
231 if (targetModel.isShimNOCorPLTile(col, row)) {
232 // wordaround for shimMux
233 auto isBundleInList = [](WireBundle bundle,
234 std::vector<WireBundle> bundles) {
235 return std::find(bundles.begin(), bundles.end(), bundle) !=
236 bundles.end();
237 };
238 const std::vector<WireBundle> bundles = {
239 WireBundle::DMA, WireBundle::NOC, WireBundle::PLIO};
240 if (isBundleInList(pIn.bundle, bundles) ||
241 isBundleInList(pOut.bundle, bundles))
242 sb.connectivity[i][j] = Connectivity::AVAILABLE;
243 }
244 }
245 }
246 }
247 graph[std::make_pair(coords, coords)] = sb;
248 };
249
250 auto interconnect = [&](int col, int row, int targetCol, int targetRow,
251 WireBundle srcBundle, WireBundle dstBundle) {
252 SwitchboxConnect sb = {{col, row}, {targetCol, targetRow}};
253 for (int channel = 0; channel < maxChannels[srcBundle]; channel++) {
254 sb.srcPorts.push_back(Port{srcBundle, channel});
255 sb.dstPorts.push_back(Port{dstBundle, channel});
256 }
257 sb.resize();
258 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
259 sb.connectivity[i][i] = Connectivity::AVAILABLE;
260 }
261 graph[std::make_pair(TileID{col, row}, TileID{targetCol, targetRow})] = sb;
262 };
263
264 for (int row = 0; row <= maxRow; row++) {
265 for (int col = 0; col <= maxCol; col++) {
266 maxChannels.clear();
267 // connections within the same switchbox
268 intraconnect(col, row);
269
270 // connections between switchboxes
271 if (row > 0) {
272 // from south to north
273 interconnect(col, row, col, row - 1, WireBundle::South,
274 WireBundle::North);
275 }
276 if (row < maxRow) {
277 // from north to south
278 interconnect(col, row, col, row + 1, WireBundle::North,
279 WireBundle::South);
280 }
281 if (col > 0) {
282 // from east to west
283 interconnect(col, row, col - 1, row, WireBundle::West,
284 WireBundle::East);
285 }
286 if (col < maxCol) {
287 // from west to east
288 interconnect(col, row, col + 1, row, WireBundle::East,
289 WireBundle::West);
290 }
291 }
292 }
293}

295// Add a flow from src to dst can have an arbitrary number of dst locations
296// due to fanout.
297void Pathfinder::addFlow(TileID srcCoords, Port srcPort, TileID dstCoords,
298 Port dstPort, bool isPacketFlow, bool isPriorityFlow) {
299 // check if a flow with this source already exists
300 for (auto &[_, prioritized, src, dsts] : flows) {
301 if (src.coords == srcCoords && src.port == srcPort) {
302 if (isPriorityFlow) {
303 prioritized = true;
304 dsts.emplace(dsts.begin(), PathEndPoint{dstCoords, dstPort});
305 } else
306 dsts.emplace_back(PathEndPoint{dstCoords, dstPort});
307 return;
308 }
309 }
310
311 // Assign a group ID for packet flows
312 // any overlapping in source/destination will lead to the same group ID
313 // channel sharing will happen within the same group ID
314 // for circuit flows, group ID is always -1, and no channel sharing
315 int packetGroupId = -1;
316 if (isPacketFlow) {
317 bool found = false;
318 for (auto &[existingId, _, src, dsts] : flows) {
319 if (src.coords == srcCoords && src.port == srcPort) {
320 packetGroupId = existingId;
321 found = true;
322 break;
323 }
324 for (auto &dst : dsts) {
325 if (dst.coords == dstCoords && dst.port == dstPort) {
326 packetGroupId = existingId;
327 found = true;
328 break;
329 }
330 }
331 packetGroupId = std::max(packetGroupId, existingId);
332 }
333 if (!found) {
334 packetGroupId++;
335 }
336 }
337 // If no existing flow was found with this source, create a new flow.
338 flows.push_back(
339 Flow{packetGroupId, isPriorityFlow, PathEndPoint{srcCoords, srcPort},
340 std::vector<PathEndPoint>{PathEndPoint{dstCoords, dstPort}}});
341}

343// Sort flows to (1) get deterministic routing, and (2) perform routings on
344// prioritized flows before others, for routing consistency on those flows.
345void Pathfinder::sortFlows(const int maxCol, const int maxRow) {
346 std::vector<Flow> priorityFlows;
347 std::vector<Flow> normalFlows;
348 for (auto f : flows) {
349 if (f.isPriorityFlow)
350 priorityFlows.push_back(f);
351 else
352 normalFlows.push_back(f);
353 }
354 // Get unique int identifier from a vector if pairs of int properties and
355 // their maximums.
356 auto getUniqueIdFromVecOfProperties =
357 [](std::vector<std::pair<int, int>> propertiesAndLimits) {
358 int uniqueId = 0;
359 int multiplier = 1;
360 for (auto pair : propertiesAndLimits) {
361 uniqueId += pair.first * multiplier;
362 multiplier *= pair.second;
363 }
364 return uniqueId;
365 };
366 std::sort(priorityFlows.begin(), priorityFlows.end(),
367 [maxCol, maxRow, getUniqueIdFromVecOfProperties](const auto &lhs,
368 const auto &rhs) {
369 // List of properties used in sorting: src col, src row, src
370 // wirebundle and src channel.
371 std::vector<std::pair<int, int>> lhsProperties = {
372 {lhs.src.coords.col, maxCol},
373 {lhs.src.coords.row, maxRow},
374 {getWireBundleAsInt(lhs.src.port.bundle),
375 AIE::getMaxEnumValForWireBundle()},
376 {lhs.src.port.channel, /*don't care*/ 0}};
377 int lhsUniqueID = getUniqueIdFromVecOfProperties(lhsProperties);
378 std::vector<std::pair<int, int>> rhsProperties = {
379 {rhs.src.coords.col, maxCol},
380 {rhs.src.coords.row, maxRow},
381 {getWireBundleAsInt(rhs.src.port.bundle),
382 AIE::getMaxEnumValForWireBundle()},
383 {rhs.src.port.channel, /*don't care*/ 0}};
384 int rhsUniqueID = getUniqueIdFromVecOfProperties(rhsProperties);
385 return lhsUniqueID < rhsUniqueID;
386 });
387 flows = priorityFlows;
388 flows.insert(flows.end(), normalFlows.begin(), normalFlows.end());
389}

391// Keep track of connections already used in the AIE; Pathfinder algorithm
392// will avoid using these.
393bool Pathfinder::addFixedConnection(SwitchboxOp switchboxOp) {
394 int col = switchboxOp.colIndex();
395 int row = switchboxOp.rowIndex();
396 TileID coords = {col, row};
397 auto &sb = graph[std::make_pair(coords, coords)];
398 for (ConnectOp connectOp : switchboxOp.getOps<ConnectOp>()) {
399 bool found = false;
400 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
401 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
402 if (sb.srcPorts[i] == connectOp.sourcePort() &&
403 sb.dstPorts[j] == connectOp.destPort() &&
404 sb.connectivity[i][j] == Connectivity::AVAILABLE) {
405 sb.connectivity[i][j] = Connectivity::INVALID;
406 found = true;
407 }
408 }
409 }
410 if (!found) {
411 // could not add such a fixed connection
412 return false;
413 }
414 }
415 return true;
416}

418static constexpr double INF = std::numeric_limits<double>::max();
419
420std::map<PathEndPoint, PathEndPoint>
421Pathfinder::dijkstraShortestPaths(PathEndPoint src) {
422 // Use std::map instead of DenseMap because DenseMap doesn't let you
423 // overwrite tombstones.
424 std::map<PathEndPoint, double> distance;
425 std::map<PathEndPoint, PathEndPoint> preds;
426 std::map<PathEndPoint, uint64_t> indexInHeap;
427 enum Color { WHITE, GRAY, BLACK };
428 std::map<PathEndPoint, Color> colors;
429 typedef d_ary_heap_indirect<
430 /*Value=*/PathEndPoint, /*Arity=*/4,
431 /*IndexInHeapPropertyMap=*/std::map<PathEndPoint, uint64_t>,
432 /*DistanceMap=*/std::map<PathEndPoint, double> &,
433 /*Compare=*/std::less<>>
434 MutableQueue;
435 MutableQueue Q(distance, indexInHeap);
436
437 distance[src] = 0.0;
438 Q.push(src);
439 while (!Q.empty()) {
440 src = Q.top();
441 Q.pop();
442
443 // get all channels src connects to
444 if (channels.count(src) == 0) {
445 auto &sb = graph[std::make_pair(src.coords, src.coords)];
446 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
447 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
448 if (sb.srcPorts[i] == src.port &&
449 sb.connectivity[i][j] == Connectivity::AVAILABLE) {
450 // connections within the same switchbox
451 channels[src].push_back(PathEndPoint{src.coords, sb.dstPorts[j]});
452 }
453 }
454 }
455 // connections to neighboring switchboxes
456 std::vector<std::pair<TileID, Port>> neighbors = {
457 {{src.coords.col, src.coords.row - 1},
458 {WireBundle::North, src.port.channel}},
459 {{src.coords.col - 1, src.coords.row},
460 {WireBundle::East, src.port.channel}},
461 {{src.coords.col, src.coords.row + 1},
462 {WireBundle::South, src.port.channel}},
463 {{src.coords.col + 1, src.coords.row},
464 {WireBundle::West, src.port.channel}}};
465
466 for (const auto &[neighborCoords, neighborPort] : neighbors) {
467 if (graph.count(std::make_pair(src.coords, neighborCoords)) > 0 &&
468 src.port.bundle == getConnectingBundle(neighborPort.bundle)) {
469 auto &sb = graph[std::make_pair(src.coords, neighborCoords)];
470 if (std::find(sb.dstPorts.begin(), sb.dstPorts.end(), neighborPort) !=
471 sb.dstPorts.end())
472 channels[src].push_back({neighborCoords, neighborPort});
473 }
474 }
475 std::sort(channels[src].begin(), channels[src].end());
476 }
477
478 for (auto &dest : channels[src]) {
479 if (distance.count(dest) == 0)
480 distance[dest] = INF;
481 auto &sb = graph[std::make_pair(src.coords, dest.coords)];
482 size_t i = std::distance(
483 sb.srcPorts.begin(),
484 std::find(sb.srcPorts.begin(), sb.srcPorts.end(), src.port));
485 size_t j = std::distance(
486 sb.dstPorts.begin(),
487 std::find(sb.dstPorts.begin(), sb.dstPorts.end(), dest.port));
488 assert(i < sb.srcPorts.size());
489 assert(j < sb.dstPorts.size());
490 bool relax = distance[src] + sb.demand[i][j] < distance[dest];
491 if (colors.count(dest) == 0) {
492 // was WHITE
493 if (relax) {
494 distance[dest] = distance[src] + sb.demand[i][j];
495 preds[dest] = src;
496 colors[dest] = GRAY;
497 }
498 Q.push(dest);
499 } else if (colors[dest] == GRAY && relax) {
500 distance[dest] = distance[src] + sb.demand[i][j];
501 preds[dest] = src;
502 }
503 }
504 colors[src] = BLACK;
505 }
506
507 return preds;
508}

510// Perform congestion-aware routing for all flows which have been added.
511// Use Dijkstra's shortest path to find routes, and use "demand" as the
512// weights. If the routing finds too much congestion, update the demand
513// weights and repeat the process until a valid solution is found. Returns a
514// map specifying switchbox settings for all flows. If no legal routing can be
515// found after maxIterations, returns empty vector.
516std::optional<std::map<PathEndPoint, SwitchSettings>>
517Pathfinder::findPaths(const int maxIterations) {
518 LLVM_DEBUG(llvm::dbgs() << "\t---Begin Pathfinder::findPaths---\n");
519 std::map<PathEndPoint, SwitchSettings> routingSolution;
520 // initialize all Channel histories to 0
521 for (auto &[_, sb] : graph) {
522 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
523 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
524 sb.usedCapacity[i][j] = 0;
525 sb.overCapacity[i][j] = 0;
526 sb.isPriority[i][j] = false;
527 }
528 }
529 }
530
531 // group flows based on packetGroupId
532 llvm::MapVector<int, std::vector<Flow>> groupedFlows;
533 for (auto &f : flows) {
534 if (groupedFlows.count(f.packetGroupId) == 0) {
535 groupedFlows[f.packetGroupId] = std::vector<Flow>();
536 }
537 groupedFlows[f.packetGroupId].push_back(f);
538 }
539
540 int iterationCount = -1;
541 int illegalEdges = 0;
542#ifndef NDEBUG
543 int totalPathLength = 0;
544#endif
545 do {
546 // if reach maxIterations, throw an error since no routing can be found
547 if (++iterationCount >= maxIterations) {
548 LLVM_DEBUG(llvm::dbgs()
549 << "\t\tPathfinder: maxIterations has been exceeded ("
550 << maxIterations
551 << " iterations)...unable to find routing for flows.\n");
552 return std::nullopt;
553 }
554
555 LLVM_DEBUG(llvm::dbgs() << "\t\t---Begin findPaths iteration #"
556 << iterationCount << "---\n");
557 // update demand at the beginning of each iteration
558 for (auto &[_, sb] : graph) {
559 sb.updateDemand();
560 }
561
562 // "rip up" all routes
563 illegalEdges = 0;
564#ifndef NDEBUG
565 totalPathLength = 0;
566#endif
567 routingSolution.clear();
568 for (auto &[_, sb] : graph) {
569 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
570 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
571 sb.usedCapacity[i][j] = 0;
572 sb.packetFlowCount[i][j] = 0;
573 sb.packetGroupId[i][j] = -1;
574 }
575 }
576 }
577
578 // for each flow, find the shortest path from source to destination
579 // update used_capacity for the path between them
580
581 for (const auto &[_, flows] : groupedFlows) {
582 for (const auto &[packetGroupId, isPriority, src, dsts] : flows) {
583 // Use dijkstra to find path given current demand from the start
584 // switchbox; find the shortest paths to each other switchbox. Output is
585 // in the predecessor map, which must then be processed to get
586 // individual switchbox settings
587 std::set<PathEndPoint> processed;
588 std::map<PathEndPoint, PathEndPoint> preds = dijkstraShortestPaths(src);
589
590 // trace the path of the flow backwards via predecessors
591 // increment used_capacity for the associated channels
592 SwitchSettings switchSettings;
593 processed.insert(src);
594 for (auto endPoint : dsts) {
595 if (endPoint == src) {
596 // route to self
597 switchSettings[src.coords].srcs.push_back(src.port);
598 switchSettings[src.coords].dsts.push_back(src.port);
599 }
600 auto curr = endPoint;
601 // trace backwards until a vertex already processed is reached
602 while (!processed.count(curr)) {
603 auto &sb = graph[std::make_pair(preds[curr].coords, curr.coords)];
604 size_t i =
605 std::distance(sb.srcPorts.begin(),
606 std::find(sb.srcPorts.begin(), sb.srcPorts.end(),
607 preds[curr].port));
608 size_t j = std::distance(
609 sb.dstPorts.begin(),
610 std::find(sb.dstPorts.begin(), sb.dstPorts.end(), curr.port));
611 assert(i < sb.srcPorts.size());
612 assert(j < sb.dstPorts.size());
613 sb.isPriority[i][j] = isPriority;
614 if (packetGroupId >= 0 &&
615 (sb.packetGroupId[i][j] == -1 ||
616 sb.packetGroupId[i][j] == packetGroupId)) {
617 for (size_t k = 0; k < sb.srcPorts.size(); k++) {
618 for (size_t l = 0; l < sb.dstPorts.size(); l++) {
619 if (k == i || l == j) {
620 sb.packetGroupId[k][l] = packetGroupId;
621 }
622 }
623 }
624 sb.packetFlowCount[i][j]++;
625 // maximum packet stream sharing per channel
626 if (sb.packetFlowCount[i][j] >= MAX_PACKET_STREAM_CAPACITY) {
627 sb.packetFlowCount[i][j] = 0;
628 sb.usedCapacity[i][j]++;
629 }
630 } else {
631 sb.usedCapacity[i][j]++;
632 }
633 // if at capacity, bump demand to discourage using this Channel
634 // this means the order matters!
635 sb.bumpDemand(i, j);
636 if (preds[curr].coords == curr.coords) {
637 switchSettings[preds[curr].coords].srcs.push_back(
638 preds[curr].port);
639 switchSettings[curr.coords].dsts.push_back(curr.port);
640 }
641 processed.insert(curr);
642 curr = preds[curr];
643 }
644 }
645 // add this flow to the proposed solution
646 routingSolution[src] = switchSettings;
647 }
648 for (auto &[_, sb] : graph) {
649 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
650 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
651 // fix used capacity for packet flows
652 if (sb.packetFlowCount[i][j] > 0) {
653 sb.packetFlowCount[i][j] = 0;
654 sb.usedCapacity[i][j]++;
655 }
656 sb.bumpDemand(i, j);
657 }
658 }
659 }
660 }
661
662 for (auto &[_, sb] : graph) {
663 for (size_t i = 0; i < sb.srcPorts.size(); i++) {
664 for (size_t j = 0; j < sb.dstPorts.size(); j++) {
665 // check that every channel does not exceed max capacity
666 if (sb.usedCapacity[i][j] > MAX_CIRCUIT_STREAM_CAPACITY) {
667 sb.overCapacity[i][j]++;
668 illegalEdges++;
669 LLVM_DEBUG(
670 llvm::dbgs()
671 << "\t\t\tToo much capacity on (" << sb.srcCoords.col << ","
672 << sb.srcCoords.row << ") " << sb.srcPorts[i].bundle
673 << sb.srcPorts[i].channel << " -> (" << sb.dstCoords.col << ","
674 << sb.dstCoords.row << ") " << sb.dstPorts[j].bundle
675 << sb.dstPorts[j].channel << ", used_capacity = "
676 << sb.usedCapacity[i][j] << ", demand = " << sb.demand[i][j]
677 << ", over_capacity_count = " << sb.overCapacity[i][j] << "\n");
678 }
679#ifndef NDEBUG
680 // calculate total path length (across switchboxes)
681 if (sb.srcCoords != sb.dstCoords) {
682 totalPathLength += sb.usedCapacity[i][j];
683 }
684#endif
685 }
686 }
687 }
688
689#ifndef NDEBUG
690 for (const auto &[PathEndPoint, switchSetting] : routingSolution) {
691 LLVM_DEBUG(llvm::dbgs()
692 << "\t\t\tFlow starting at (" << PathEndPoint.coords.col << ","
693 << PathEndPoint.coords.row << "):\t");
694 LLVM_DEBUG(llvm::dbgs() << switchSetting);
695 }
696 LLVM_DEBUG(llvm::dbgs()
697 << "\t\t---End findPaths iteration #" << iterationCount
698 << " , illegal edges count = " << illegalEdges
699 << ", total path length = " << totalPathLength << "---\n");
700#endif
701 } while (illegalEdges >
702 0); // continue iterations until a legal routing is found
703
704 LLVM_DEBUG(llvm::dbgs() << "\t---End Pathfinder::findPaths---\n");
705 return routingSolution;
706}
