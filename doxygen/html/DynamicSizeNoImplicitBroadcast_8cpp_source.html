7// This file contains rewrites to the arith dialect to enable the support of
8// dynamic sized tensor/memref for the auto-vectorization to CPP flow.
9// MLIR-AIE auto-vectorization to CPP flow currently doesn't support to
10// implicitly broadcast a dynamic dimension of size `1`. Hence, we assume that
11// dynamic dimensions are not with size '1' that can be interpreted to various
12// broadcasting scenarios. The effectiveness of this rewrite pattern is guarded
13// by the attribute `tosa.no_implicit_broadcast_of_dynamic_sizes`.
14//===----------------------------------------------------------------------===//

32//============================================================================//
33//=========================== Rewrite Patterns ===============================//
34//============================================================================//
35
36// This pattern replaces a arith::CmpIOp with a arith::ConstantOp `false` only
37// when the CmpIOp compares the equality of a dynamic dimension's runtime size
38// to a constant 1, and is guarded by the attribute
39// `tosa.no_implicit_broadcast_of_dynamic_sizes`.
40struct DynamicSizeNoImplicitBroadcastPattern : RewritePattern {
41 DynamicSizeNoImplicitBroadcastPattern(MLIRContext *context)
42 : RewritePattern(arith::CmpIOp::getOperationName(), /*benefit=*/1,
43 context) {}
44
45 LogicalResult matchAndRewrite(Operation *op,
46 PatternRewriter &rewriter) const override {
47 if (!isAssumingNoImplicitBroadcastOfDynamicSizes(rewriter))
48 return failure();
49
50 arith::CmpIOp cmpiOp = cast<arith::CmpIOp>(op);
51
52 if (cmpiOp.getPredicate() != arith::CmpIPredicate::eq)
53 return failure();
54
55 auto lhsOp = cmpiOp.getLhs().getDefiningOp();
56 auto rhsOp = cmpiOp.getRhs().getDefiningOp();
57 if (!((isa<memref::DimOp>(lhsOp) || isa<tensor::DimOp>(lhsOp)) &&
58 isa<arith::ConstantOp>(rhsOp)) &&
59 !((isa<memref::DimOp>(rhsOp) || isa<tensor::DimOp>(rhsOp)) &&
60 isa<arith::ConstantOp>(lhsOp)))
61 return failure();
62
63 // Make sure rhsOp is ConstantOp and lhsOp is DimOp
64 if (isa<memref::DimOp>(rhsOp) || isa<tensor::DimOp>(rhsOp))
65 std::swap(lhsOp, rhsOp);
66
67 // If ConstantOp is 1 for Integer/Index, replace cmpiOp as constant 0
68 auto constantOp = cast<arith::ConstantOp>(rhsOp);
69 if (cast<IntegerAttr>(constantOp.getValue()).getValue().getZExtValue() != 1)
70 return failure();
71
72 // Check the DimOp's input is a dynamic dim from the given index
73 auto constIndexOp = lhsOp->getOperand(1).getDefiningOp<arith::ConstantOp>();
74 if (!constIndexOp)
75 return failure();
76
77 auto index =
78 cast<IntegerAttr>(constIndexOp.getValue()).getValue().getZExtValue();
79 auto inputDimType = dyn_cast<ShapedType>(lhsOp->getOperand(0).getType());
80 if (!inputDimType || !inputDimType.isDynamicDim(index))
81 return failure();
82
83 rewriter.replaceOpWithNewOp<arith::ConstantOp>(
84 cmpiOp, rewriter.getIntegerAttr(rewriter.getI1Type(), 0));
85
86 return success();
87 }
88};

94struct DynamicSizeNoImplicitBroadcastPass
95 : PassWrapper<DynamicSizeNoImplicitBroadcastPass, OperationPass<>> {
96
97 StringRef getArgument() const final {
98 return "test-dynamic-size-no-implicit-broadcast";
99 }
100
101 StringRef getDescription() const final {
102 return "Test rewriting arith operations when assuming no implict "
103 "broadcast of dynamic sizes";
104 }
105
106 void runOnOperation() override {
107 auto op = getOperation();
108 MLIRContext *context = &getContext();
109 RewritePatternSet patterns(context);
110
111 patterns.add<DynamicSizeNoImplicitBroadcastPattern>(patterns.getContext());
112
113 (void)applyPatternsGreedily(op, std::move(patterns));
114 }
115};
