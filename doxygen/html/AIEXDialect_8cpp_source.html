55/* Return the correct values to write to the hardware registers to configure
56 strides and wraps given the input user-facing strides and wraps.
57
58 In the IR, we express strides in units of element data type, but the hardware
59 requires it in units of address granularity. Address granularity currently is
60 4 bytes for all hardware.
61
62
63 User-facing strides/wraps relate to hardware as follows:
64
65 - By default, stride 0 and size 1 is assumed if unspecified.
66 - If only N strides/wraps are defined, those define the lowest N dimensions.
67
68 inputStride[3] == iteration_stride / elemSizeFac + 1
69 inputWrap[3] == iteration_size + 1
70 Highest-dimension stride/wrap is iteration count / iteration stride.
71 inputStride[2] == d2_stride / elemSizeFac + 1
72 Note: d2_size is not specified in hardware as it is
73 implicit from the total buffer transfer length
74 inputStride[1] == d1_stride / elemSizeFac + 1
75 inputSize[1] == d1_size
76 inputStride[0] == d0_stride / elemSizeFac + 1
77 inputSize[0] == d0_size / elemSizeFac
78
79 where elemSizeFac == bufferElementSize / addressGranularity
80 where bufferElementSize == size in bytes of elements in buffer,
81 e.g. 4 for int32
82 where addressGranularity == transfer granularity in hardware, which is
83 4 bytes for all current hardware
84
85 Note: strides are expressed offset by one from user input strides, because the
86 hardware does not support a 0 stride (repeat).
87 */
88void AIEX::getHardwareStridesWraps(const AIE::AIETargetModel &targetModel,
89 mlir::Operation *op,
90 mlir::BaseMemRefType referencedBufType,
91 llvm::SmallVector<int64_t, 4> inputSizes,
92 llvm::SmallVector<int64_t, 4> inputStrides,
93 llvm::SmallVector<int64_t, 4> &sizes,
94 llvm::SmallVector<int64_t, 4> &strides) {
95 assert(inputSizes.size() == inputStrides.size());
96 assert(sizes.size() == 4);
97 assert(strides.size() == 4);
98
99 DataLayout dataLayout = DataLayout::closest(op);
100 auto elemWidth =
101 dataLayout.getTypeSizeInBits(referencedBufType.getElementType());
102 auto addressGranularity = targetModel.getAddressGenGranularity();
103
104 // Output strides and sizes are default-initialized to 0
105 std::fill(sizes.begin(), sizes.end(), 0);
106 std::fill(strides.begin(), strides.end(), 0);
107
108 if (inputSizes[0] == 0) {
109 // Illegal input, this won't transfer anything at all.
110 // Leave it to the verification functions to complain to the user.
111 return;
112 }
113
114 // d0_size, d0_stride
115 sizes[0] = inputSizes[0] * elemWidth / addressGranularity;
116 if (inputStrides[0] * elemWidth < addressGranularity ||
117 (elemWidth > addressGranularity)) {
118 // First check:
119 // While the hardware cannot transfer less than addressGranularity bits at
120 // a time, the user may expresses a contiguous transfer of multiple
121 // elements with a stride smaller than addressGranularity. We can thus set
122 // the stride to 1 (encoded in hardware as 0) here to allow such transfers.
123 // The verification function should ensure that
124 // inputStrides[0] * elemWidth < addressGranularity
125 // iff. inputSize[0] * elemWidth > addressGranularity.
126 // Second check:
127 // If the element width is larger than addressGranularity, we need to make
128 // sure that all bytes are properly copied and therefore the stride must be
129 // set to 1 (encoded in hardware as 0).
130 // The verification function should ensure that
131 // inputStrides[0] * elemWidth % addressGranularity == 0
132 // && inputStrides[0] == 1 if elemWidth > addressGranularity
133 // This makes it impossible to have a stride greater than 1 for
134 // elemWidths bigger than addressGranularity, even if they are a multiple of
135 // it. Such operations should make use of an additional dimension instead.
136 strides[0] = 0;
137 } else {
138 strides[0] = inputStrides[0] * elemWidth / addressGranularity - 1;
139 }
140
141 // d1_size, d1_stride
142 sizes[1] = inputSizes[1];
143 if (inputSizes[1] > 1) {
144 // Stride only matters if we have more than one iteration.
145 strides[1] = inputStrides[1] * elemWidth / addressGranularity - 1;
146 }
147
148 // d2_size, d2_stride
149 sizes[2] = inputSizes[2];
150 if (inputSizes[2] > 1) {
151 // Stride only matters if we have more than one iteration.
152 strides[2] = inputStrides[2] * elemWidth / addressGranularity - 1;
153 }
154
155 // iteration_size, iteration_stride
156 if (inputSizes[3] > 1) {
157 // Stride only matters if we have more than one iteration.
158 sizes[3] = inputSizes[3] - 1;
159 // Note that the iteration_stride must be positive, just like the other
160 // dimensions. However, one can encode a zero-stride "repeat" of the same
161 // transfer by setting a positive repeat_count on the pushToQueue instr,
162 // and setting the size here to 1. This causes the BD to "wrap" at every
163 // single iteration, effectively never adding the specified stride, in turn
164 // equalling a repeat without stride.
165 if (inputStrides[3] > 0) {
166 strides[3] = inputStrides[3] * elemWidth / addressGranularity - 1;
167 }
168 }
169}

171mlir::LogicalResult
172AIEX::verifyStridesWraps(mlir::Operation *forOp,
173 mlir::BaseMemRefType referencedBufType, int tileCol,
174 int tileRow, llvm::SmallVector<int64_t, 4> inputSizes,
175 llvm::SmallVector<int64_t, 4> inputStrides,
176 llvm::SmallVector<int64_t, 4> hardwareSizes,
177 llvm::SmallVector<int64_t, 4> hardwareStrides,
178 bool skipTransformationChecks) {
179 const auto &targetModel = AIE::getTargetModel(forOp);
180 auto addressGranularity = targetModel.getAddressGenGranularity();
181 DataLayout dataLayout = DataLayout::closest(forOp);
182 auto elemWidth =
183 dataLayout.getTypeSizeInBits(referencedBufType.getElementType());
184
185 uint32_t wrap_bits = 0;
186 uint32_t step_bits = 0;
187 uint32_t iter_bits = 6;
188 if (targetModel.isShimNOCTile(tileCol, tileRow)) {
189 step_bits = 20; // XAIEMLGBL_NOC_MODULE_DMA_BD0_3_D0_STEPSIZE_WIDTH
190 wrap_bits = 10; // XAIEMLGBL_NOC_MODULE_DMA_BD0_3_D0_WRAP_WIDTH
191 } else if (targetModel.isMemTile(tileCol, tileRow)) {
192 step_bits = 17; // XAIEMLGBL_MEM_TILE_MODULE_DMA_BD0_2_D0_STEPSIZE_WIDTH
193 wrap_bits = 10; // XAIEMLGBL_MEM_TILE_MODULE_DMA_BD0_2_D0_WRAP_WIDTH
194 } else if (targetModel.isCoreTile(tileCol, tileRow)) {
195 step_bits = 13; // XAIEMLGBL_MEMORY_MODULE_DMA_BD0_2_D0_STEPSIZE_WIDTH
196 wrap_bits = 8; // XAIEMLGBL_MEMORY_MODULE_DMA_BD0_3_D0_WRAP_WIDTH
197 } else {
198 return forOp->emitOpError(
199 "Unsupported tile type at (" + std::to_string(tileCol) + ", " +
200 std::to_string(tileRow) + ") Must be ShimNOC, Mem or Core.");
201 }
202
203 for (int i = 0; i < 4; i++) {
204 if (inputSizes[i] <= 0) {
205 return forOp->emitOpError("Size ") << i << " must be a positive integer.";
206 }
207 }
208
209 if (inputSizes[0] * elemWidth % addressGranularity != 0) {
210 std::stringstream msg;
211 msg << "Transfer sizes must be multiples of " << (addressGranularity / 8)
212 << " bytes. " << inputSizes[0] << " elements at " << (elemWidth / 8)
213 << " bytes each equal " << (inputSizes[0] * elemWidth / 8)
214 << " bytes, which is not divisible by " << (addressGranularity / 8)
215 << ". ";
216 return forOp->emitOpError(msg.str());
217 }
218
219 for (int i = 0; i < 3; i++) {
220 if (inputSizes[i] > 1 && inputStrides[i] < 1) {
221 // If inputSize[i] == 1, anything is allowable in the stride, since that
222 // stride will never be applied. For any larger size, we must verify that
223 // the stride is positive.
224 return forOp->emitOpError("Stride ")
225 << i << " must be a positive integer.";
226 }
227 }
228 // A value of zero is allowable for the fourth-dimension stride
229 // (this indicates an interation stride for the repeat of 0)
230 if (inputSizes[3] > 1 && inputStrides[3] < 0) {
231 return forOp->emitOpError("Stride 3 must be a non-negative integer.");
232 }
233
234 for (int i = 0; i < 4; i++) {
235 // strides[0] == 1 is ok iff the transfer size is a multiple of
236 // addressGranularity, which is checked below
237 if (i == 0 && inputStrides[i] == 1)
238 continue;
239 if (inputStrides[i] * elemWidth % addressGranularity != 0) {
240 std::stringstream msg;
241 msg << "Stride " << i << " is " << inputStrides[i] << " elements * "
242 << (elemWidth / 8) << " bytes = " << (inputStrides[i] * elemWidth / 8)
243 << " bytes, which is not divisible by " << (addressGranularity / 8)
244 << ". ";
245 return forOp->emitOpError(msg.str());
246 }
247 }
248
249 if (!skipTransformationChecks && hardwareSizes[0] > (1 << wrap_bits) - 1)
250 return forOp->emitOpError(
251 "Size 0 exceeds the [0:" + std::to_string((1 << wrap_bits) - 1) +
252 "] range.");
253 if (hardwareSizes[1] > (1 << wrap_bits) - 1)
254 return forOp->emitOpError(
255 "Size 1 exceeds the [0:" + std::to_string((1 << wrap_bits) - 1) +
256 "] range.");
257 if (hardwareSizes[3] > (1 << iter_bits))
258 return forOp->emitOpError(
259 "Size 3 exceeds the [1:" + std::to_string(1 << iter_bits) + "] range.");
260 if (hardwareStrides[0] > (1 << step_bits))
261 return forOp->emitOpError("Stride 0 exceeds the [1:" +
262 std::to_string(1 << step_bits) + "] range.");
263 if (hardwareStrides[1] > (1 << step_bits))
264 return forOp->emitOpError("Stride 1 exceeds the [1:" +
265 std::to_string(1 << step_bits) + "] range.");
266 if (hardwareStrides[2] > (1 << step_bits))
267 return forOp->emitOpError("Stride 2 exceeds the [1:" +
268 std::to_string(1 << step_bits) + "] range.");
269 // strides[3] exceeding the range is ok iff the sizes[3] is one, which is
270 // checked below
271 if (hardwareStrides[3] > (1 << step_bits) && hardwareSizes[3] > 0)
272 return forOp->emitOpError("Stride 3 exceeds the [1:" +
273 std::to_string(1 << step_bits) + "] range.");
274
275 return success();
276}

324/* Calculates the offset value to be written to the
325 */
326int64_t AIEX::NpuDmaMemcpyNdOp::getOffsetInBytes() {
327 llvm::SmallVector<int64_t, 4> offsets =
328 llvm::map_to_vector(llvm::reverse(getMixedOffsets()), [](OpFoldResult s) {
329 return getConstantIntValue(s).value();
330 });
331 llvm::SmallVector<int64_t, 4> strides =
332 llvm::map_to_vector(llvm::reverse(getMixedStrides()), [](OpFoldResult s) {
333 return getConstantIntValue(s).value();
334 });
335 size_t offset = 0;
336 size_t R = offsets.size();
337 size_t el_bit_width = getElementTypeBitwidth();
338 assert(el_bit_width % 8 == 0 &&
339 "Expected Memref element bitwidth to be multiple of 8.");
340 size_t S = el_bit_width / 8;
341 for (size_t i = 0; i < R; i++)
342 offset += offsets[i] * strides[i] * S;
343 return offset;
344}

346// dma_memcpy_nd transfers of the form [*, 1, 1, len][*, 0, 0, 1] do not
347// specify any data layout transformation, but simply express a contiguous
348// transfer of `len`. We exclude checks to 4th dimension, because repeat count
349// is still possible without a data layout transformation.
350bool AIEX::NpuDmaMemcpyNdOp::isLinearTransferWithoutTransformation() {
351 llvm::SmallVector<int64_t, 4> inputSizes =
352 llvm::map_to_vector(llvm::reverse(getMixedSizes()), [](OpFoldResult s) {
353 return getConstantIntValue(s).value();
354 });
355 llvm::SmallVector<int64_t, 4> inputStrides =
356 llvm::map_to_vector(llvm::reverse(getMixedStrides()), [](OpFoldResult s) {
357 return getConstantIntValue(s).value();
358 });
359 return (inputSizes[1] == 1 && inputSizes[2] == 1 && inputStrides[0] == 1 &&
360 inputStrides[1] == 0 && inputStrides[2] == 0);
361}

363// Helper method to check if a requested burst length is supported by the target
364// model. Returns an error message if the burst length is not supported or an
365// empty option otherwise.
366static std::optional<std::string>
367checkBurstLength(const xilinx::AIE::AIETargetModel &targetModel,
368 uint32_t requestedBurstLength) {
369 if (requestedBurstLength != 0) {
370 auto bel = targetModel.getShimBurstEncodingsAndLengths();
371 auto pair = std::find_if(bel.begin(), bel.end(),
372 [=](const std::pair<uint32_t, uint32_t> &p) {
373 return p.second == requestedBurstLength;
374 });
375
376 if (pair == bel.end()) {
377 std::string errorMessage =
378 "Requested burst length is not supported by the target. "
379 "Supported burst lengths:";
380
381 errorMessage =
382 std::accumulate(bel.begin(), bel.end(), errorMessage,
383 [](const std::string &a, auto b) {
384 return a + " " + std::to_string(b.second);
385 });
386
387 return errorMessage;
388 }
389 }
390
391 return std::nullopt;
392}

394LogicalResult AIEX::NpuDmaMemcpyNdOp::verify() {
395 BaseMemRefType buffer = getMemref().getType();
396 const auto &targetModel = AIE::getTargetModel(*this);
397 auto addressGranularity = targetModel.getAddressGenGranularity();
398
399 if (getElementTypeBitwidth() > addressGranularity) {
400 return emitOpError("Maximum element bit width allowed is ")
401 << addressGranularity << "bits. ";
402 }
403 if (buffer.hasStaticShape() &&
404 (buffer.getNumElements() * getElementTypeBitwidth()) <
405 addressGranularity) {
406 return emitOpError("Minimum data transfer size required is ")
407 << addressGranularity << "bits. ";
408 }
409 if (!llvm::all_of(getMixedStrides(), [](OpFoldResult s) {
410 return getConstantIntValue(s).has_value();
411 }))
412 return emitOpError("Only constant strides currently supported.");
413 if (!llvm::all_of(getMixedSizes(), [](OpFoldResult s) {
414 return getConstantIntValue(s).has_value();
415 }))
416 return emitOpError("Only constant sizes currently supported.");
417 if (!llvm::all_of(getMixedOffsets(), [](OpFoldResult s) {
418 return getConstantIntValue(s).has_value();
419 }))
420 return emitOpError("Only constant offsets currently supported.");
421
422 llvm::SmallVector<int64_t, 4> inputSizes =
423 llvm::map_to_vector(llvm::reverse(getMixedSizes()), [](OpFoldResult s) {
424 return getConstantIntValue(s).value();
425 });
426 llvm::SmallVector<int64_t, 4> inputStrides =
427 llvm::map_to_vector(llvm::reverse(getMixedStrides()), [](OpFoldResult s) {
428 return getConstantIntValue(s).value();
429 });
430 llvm::SmallVector<int64_t, 4> hardwareSizes(4);
431 llvm::SmallVector<int64_t, 4> hardwareStrides(4);
432 getHardwareStridesWraps(targetModel, getOperation(), buffer, inputSizes,
433 inputStrides, hardwareSizes, hardwareStrides);
434 int64_t offset = getOffsetInBytes();
435
436 auto errorMessage = checkBurstLength(targetModel, getBurstLength());
437 if (errorMessage.has_value()) {
438 return emitOpError(errorMessage.value());
439 }
440
441 // The experimental HSA target uses this op on AIE1, skip all the AIE2
442 // specific checks
443 if (targetModel.getTargetArch() == AIE::AIEArch::AIE1)
444 return success();
445
446 if (offset % 4 != 0) {
447 return emitOpError("Offset must be 4-byte-aligned.");
448 }
449
450 // dma_memcpy_nd transfers of the form [1, 1, 1, len][0, 0, 0, 1] do not
451 // specify any data layout transformation, but simply express a contiguous
452 // transfer of `len`. For backwards compatibility, we allow this to proceed
453 // even if it exceeds the maximum stride/wrap size of any one dimension,
454 // and simply do not lower any data layout transformations, since there is
455 // no other way to express this at the dma_memcpy_nd interface otherwise.
456 AIE::DeviceOp dev = getOperation()->getParentOfType<AIE::DeviceOp>();
457 if (auto allocOp = AIE::ShimDMAAllocationOp::getForSymbol(
458 dev, getMetadata().getRootReference())) {
459 AIE::TileOp tile = allocOp.getTileOp();
460 if (!tile) {
461 return emitOpError("shim DMA allocation must reference a valid TileOp");
462 }
463 int col = tile.getCol();
464 int row = tile.getRow();
465 bool skipTransformationChecks = isLinearTransferWithoutTransformation();
466 if (failed(verifyStridesWraps(*this, buffer, col, row, inputSizes,
467 inputStrides, hardwareSizes, hardwareStrides,
468 skipTransformationChecks))) {
469 return failure();
470 }
471 }
472
473 // packet header
474 if (auto packetInfo = getPacket()) {
475 if (packetInfo->getPktType() > 7)
476 return emitOpError("Packet type field can only hold 3 bits.");
477 if (packetInfo->getPktId() > 31)
478 return emitOpError("Packet ID field can only hold 5 bits.");
479 }
480
481 return success();
482}

511//===----------------------------------------------------------------------===//
512// NpuWriteBdOp
513//===----------------------------------------------------------------------===//
514
515LogicalResult AIEX::NpuWriteBdOp::verify() {
516 const auto &targetModel = AIE::getTargetModel(*this);
517 auto numBds = targetModel.getNumBDs(getColumn(), getRow());
518 bool isLinearTransfer =
519 (getD0Size() >= 1) && (getD1Size() == 1) && (getIterationSize() == 0);
520 if (getBdId() > numBds)
521 return emitOpError("BD ID exceeds the maximum ID.");
522 if (getPacketId() > 31)
523 return emitOpError("Packet ID exceeds the maximum supported by 5 bits.");
524 if (getPacketType() > 7)
525 return emitOpError("Packet Type exceeds the maximum supported by 3 bits.");
526 if (!isLinearTransfer && getD0Size() > 0x3FF)
527 return emitOpError("D0 Size exceeds the [0:1023] range.");
528 if (getD0Stride() > 0xFFFFF)
529 return emitOpError("D0 Stride exceeds the [0:1M-1] range.");
530 if (getD1Size() > 0x3FF)
531 return emitOpError("D1 Size exceeds the [0:1023] range.");
532 if (getD1Stride() > 0xFFFFF)
533 return emitOpError("D1 Stride exceeds the [0:1M-1] range.");
534 if (getD2Stride() > 0xFFFFF)
535 return emitOpError("D2 Stride exceeds the [0:1M-1] range.");
536 if (getIterationSize() > 0x3F)
537 return emitOpError("Iteration Size exceeds the [0:63] range.");
538 if (getIterationStride() > 0xFFFFF)
539 return emitOpError("Iteration Stride exceeds the [0:1M-1] range.");
540 if (targetModel.isShimNOCTile(getColumn(), getRow()) && getD2Size() != 0)
541 return emitOpError("ShimTile only supports 3 dimensions of sizes.");
542 if (targetModel.isShimNOCTile(getColumn(), getRow()) &&
543 (getD0ZeroBefore() != 0 || getD0ZeroAfter() != 0 ||
544 getD1ZeroBefore() != 0 || getD1ZeroAfter() != 0 ||
545 getD2ZeroBefore() != 0 || getD2ZeroAfter() != 0))
546 return emitOpError("ShimTile doesn't support zero padding.");
547 if (!targetModel.isShimNOCTile(getColumn(), getRow()) &&
548 getBurstLength() != 0)
549 return emitOpError("Only ShimTiles support burst length.");
550 auto errorMessage = checkBurstLength(targetModel, getBurstLength());
551 if (errorMessage.has_value()) {
552 return emitOpError(errorMessage.value());
553 }
554
555 return success();
556}

562template <typename T>
563static std::optional<uint32_t> getAbsoluteAddress(T *op) {
564 AIE::DeviceOp device =
565 op->getOperation()->template getParentOfType<AIE::DeviceOp>();
566 if (!device) {
567 op->emitError("Must be inside a device.");
568 return std::nullopt;
569 }
570 const AIE::AIETargetModel &tm = device.getTargetModel();
571
572 uint32_t address = 0;
573
574 // If blockwrite references a buffer, the given address is understood to be
575 // relative to the buffer's start address.
576 if (op->getBuffer()) {
577 AIE::BufferOp buffer = device.lookupSymbol<AIE::BufferOp>(*op->getBuffer());
578 if (!buffer) {
579 op->emitError() << "buffer '" << *op->getBuffer()
580 << "' not found in device";
581 return std::nullopt;
582 }
583
584 if (!buffer.getAddress()) {
585 mlir::InFlightDiagnostic err =
586 op->emitError("referenced buffer must have address assigned");
587 err.attachNote(buffer.getLoc()) << "This buffer must have an address.";
588 return std::nullopt;
589 }
590
591 uint32_t col = buffer.getTileOp().getCol();
592 uint32_t row = buffer.getTileOp().getRow();
593 address = static_cast<uint32_t>(*buffer.getAddress()) +
594 op->getAddress() * sizeof(uint32_t);
595 address = ((col & 0xff) << tm.getColumnShift()) |
596 ((row & 0xff) << tm.getRowShift()) | (address & 0xfffff);
597 } else { // otherwise, the given address is absolute
598 address = op->getAddress();
599 std::optional<uint32_t> col = op->getColumn();
600 std::optional<uint32_t> row = op->getRow();
601 if (col && row) {
602 // If col and row are set, only the lower 20 bits of the address are
603 // used, and col and row dictate the upper bits (ignored)
604 address = ((*col & 0xff) << tm.getColumnShift()) |
605 ((*row & 0xff) << tm.getRowShift()) | (address & 0xfffff);
606 }
607 }
608
609 return address;
610}

676 Region &body = getBody();

678 return std::nullopt;
679 }

681 if (bd_ops.empty() && body.front().getNumSuccessors() == 1) {

684 // within the constraints of MLIR syntax.

686 bd_ops = chain_entry.getOps<AIE::DMABDOp>();

688 if (bd_ops.empty()) {
689 return std::nullopt;
690 }
691 AIE::DMABDOp bd = *bd_ops.begin();

693 return std::nullopt;

698LogicalResult
699AIEX::DMAConfigureTaskOp::canonicalize(AIEX::DMAConfigureTaskOp op,
700 PatternRewriter &rewriter) {
701 // Remove blocks that contain nothing but a terminator
702 Region &body = op.getBody();
703 bool did_rewrite = false;
704 for (auto it = body.begin(); it != body.end(); ++it) {
705 Block &block = *it;
706 if (block.empty()) {
707 continue;
708 }
709 auto ops_it = block.without_terminator();
710 if (std::distance(ops_it.begin(), ops_it.end()) == 0) {
711 rewriter.eraseOp(block.getTerminator());
712 did_rewrite = true;
713 }
714 }
715 if (did_rewrite) {
716 return success();
717 }
718 return failure();
719}

721LogicalResult AIEX::DMAConfigureTaskOp::verify() {
722 Region &body = getBody();
723 for (auto it = body.begin(); it != body.end(); ++it) {
724 Block &block = *it;
725 if (block.empty()) {
726 continue;
727 }
728 if (block.hasNoPredecessors() && !block.isEntryBlock()) {
729 auto error = block.getTerminator()->emitError(
730 "Block ending in this terminator does not form a chain with "
731 "entry block.");
732 return failure();
733 }
734
735 const AIE::AIETargetModel &targetModel =
736 AIE::getTargetModel(getOperation());
737
738 // This is a layering violation on the DMABDOps, but they are never verified
739 // otherwise Because DMAConfigureTaskOps are not yet merged into the AIE
740 // dialect. The normal DMABDOp verify operation will skip over any BD inside
741 // a DMAConfigureTaskOp
742 LogicalResult result = success();
743 block.walk([&](AIE::DMABDOp bd) {
744 if (bd.getBurstLength() != 0 &&
745 !targetModel.isShimNOCTile(getTileID().col, getTileID().row)) {
746 bd.emitOpError("Burst length is only supported in Shim NOC tiles that "
747 "are connected to the memory-mapped NOC.");
748 result = failure();
749 }
750 });
751 if (failed(result)) {
752 return result;
753 }
754 }
755 return success();
756}

810
811LogicalResult AIEX::SetLockOp::verify() {
812 const auto &targetModel = AIE::getTargetModel(*this);
813
814 if (targetModel.getTargetArch() == AIE::AIEArch::AIE1)
815 return emitOpError("SetLockOp is not supported on AIE1.");
816
817 if (getValue() > targetModel.getMaxLockValue())
818 return emitOpError("Lock value exceeds the maximum value of " +
819 std::to_string(targetModel.getMaxLockValue()));
820
821 auto lockOp = getLockOp();
822 auto lockIDOpt = getLockOp().getLockID();
823 // Note that the lockID may not be assigned initially, so lets wait until it
824 // is to verify the lockID dependent conditions
825 if (!lockIDOpt) {
826 return success();
827 }
828
829 auto col = lockOp.colIndex();
830 auto row = lockOp.rowIndex();
831 uint32_t lockID = lockOp.getLockIDValue();
832
833 if (lockID >= targetModel.getNumLocks(col, row)) {
834 return emitOpError("Lock ID out of range for given tile. Max ID: " +
835 std::to_string(targetModel.getNumLocks(col, row) - 1));
836 }
837
838 if (!targetModel.getLocalLockAddress(lockID, lockOp.getTileID())) {
839 return emitOpError("Invalid lock ID and tile combination when trying to "
840 "retrieve the local lock address.");
841 }
842
843 return success();
