46 /* Return the correct values to write to the hardware registers to configure
47  strides and wraps given the input user-facing strides and wraps.
48 
49  In the IR, we express strides in units of element data type, but the hardware
50  requires it in units of address granularity. Address granularity currently is
51  4 bytes for all hardware.
52 
53 
54  User-facing strides/wraps relate to hardware as follows:
55 
56  - By default, stride 0 and size 1 is assumed if unspecified.
57  - If only N strides/wraps are defined, those define the lowest N dimensions.
58 
59  inputStride[3] == iteration_stride / elemSizeFac + 1
60  inputWrap[3] == iteration_size + 1
61  Highest-dimension stride/wrap is iteration count / iteration stride.
62  inputStride[2] == d2_stride / elemSizeFac + 1
63  Note: d2_size is not specified in hardware as it is
64  implicit from the total buffer transfer length
65  inputStride[1] == d1_stride / elemSizeFac + 1
66  inputSize[1] == d1_size
67  inputStride[0] == d0_stride / elemSizeFac + 1
68  inputSize[0] == d0_size / elemSizeFac
69 
70  where elemSizeFac == bufferElementSize / addressGranularity
71  where bufferElementSize == size in bytes of elements in buffer,
72  e.g. 4 for int32
73  where addressGranularity == transfer granularity in hardware, which is
74  4 bytes for all current hardware
75 
76  Note: strides are expressed offset by one from user input strides, because the
77  hardware does not support a 0 stride (repeat).
78  */
79 void AIEX::getHardwareStridesWraps(const AIE::AIETargetModel &targetModel,
80  mlir::MemRefType referencedBufType,
81  llvm::SmallVector<int64_t, 4> inputSizes,
82  llvm::SmallVector<int64_t, 4> inputStrides,
83  llvm::SmallVector<int64_t, 4> &sizes,
84  llvm::SmallVector<int64_t, 4> &strides) {
85  assert(inputSizes.size() == inputStrides.size());
86  assert(sizes.size() == 4);
87  assert(strides.size() == 4);
88 
89  auto elemWidth = referencedBufType.getElementTypeBitWidth();
90  auto addressGranularity = targetModel.getAddressGenGranularity();
91 
92  // Output strides and sizes are default-initialized to 0
93  std::fill(sizes.begin(), sizes.end(), 0);
94  std::fill(strides.begin(), strides.end(), 0);
95 
96  if (inputSizes[0] == 0) {
97  // Illegal input, this won't transfer anything at all.
98  // Leave it to the verification functions to complain to the user.
99  return;
100  }
101 
102  // d0_size, d0_stride
103  sizes[0] = inputSizes[0] * elemWidth / addressGranularity;
104  if (inputStrides[0] * elemWidth < addressGranularity) {
105  // While the hardware cannot transfer less than addressGranularity bits at
106  // a time, the user may expresses a contiguous transfer of multiple
107  // elements with a stride smaller than addressGranularity. We can thus set
108  // the stride to 1 (encoded in hardware as 0) here to allow such transfers.
109  // The verification function should ensure that
110  // inputStrides[0] * elemWidth < addressGranularity
111  // iff. inputSize[0] * elemWidth > addressGranularity.
112  strides[0] = 0;
113  } else {
114  strides[0] = inputStrides[0] * elemWidth / addressGranularity - 1;
115  }
116 
117  // d1_size, d1_stride
118  sizes[1] = inputSizes[1];
119  if (inputSizes[1] > 1) {
120  // Stride only matters if we have more than one iteration.
121  strides[1] = inputStrides[1] * elemWidth / addressGranularity - 1;
122  }
123 
124  // d2_size, d2_stride
125  sizes[2] = inputSizes[2];
126  if (inputSizes[2] > 1) {
127  // Stride only matters if we have more than one iteration.
128  strides[2] = inputStrides[2] * elemWidth / addressGranularity - 1;
129  }
130 
131  // iteration_size, iteration_stride
132  if (inputSizes[3] > 1) {
133  // Stride only matters if we have more than one iteration.
134  sizes[3] = inputSizes[3] - 1;
135  // Note that the iteration_stride must be positive, just like the other
136  // dimensions. However, one can encode a zero-stride "repeat" of the same
137  // transfer by setting a positive repeat_count on the pushToQueue instr,
138  // and setting the size here to 1. This causes the BD to "wrap" at every
139  // single iteration, effectively never adding the specified stride, in turn
140  // equalling a repeat without stride.
141  if (inputStrides[3] > 0) {
142  strides[3] = inputStrides[3] * elemWidth / addressGranularity - 1;
143  }
144  }
145 }

147 mlir::LogicalResult
148 AIEX::verifyStridesWraps(mlir::Operation *forOp,
149  mlir::MemRefType referencedBufType, int tileCol,
150  int tileRow, llvm::SmallVector<int64_t, 4> inputSizes,
151  llvm::SmallVector<int64_t, 4> inputStrides,
152  llvm::SmallVector<int64_t, 4> hardwareSizes,
153  llvm::SmallVector<int64_t, 4> hardwareStrides,
154  bool skipTransformationChecks) {
155  const auto &targetModel = AIE::getTargetModel(forOp);
156  auto addressGranularity = targetModel.getAddressGenGranularity();
157  auto elemWidth = referencedBufType.getElementTypeBitWidth();
158 
159  uint32_t wrap_bits = 0;
160  uint32_t step_bits = 0;
161  uint32_t iter_bits = 6;
162  if (targetModel.isShimNOCTile(tileCol, tileRow)) {
163  step_bits = 20; // XAIEMLGBL_NOC_MODULE_DMA_BD0_3_D0_STEPSIZE_WIDTH
164  wrap_bits = 10; // XAIEMLGBL_NOC_MODULE_DMA_BD0_3_D0_WRAP_WIDTH
165  } else if (targetModel.isMemTile(tileCol, tileRow)) {
166  step_bits = 17; // XAIEMLGBL_MEM_TILE_MODULE_DMA_BD0_2_D0_STEPSIZE_WIDTH
167  wrap_bits = 10; // XAIEMLGBL_MEM_TILE_MODULE_DMA_BD0_2_D0_WRAP_WIDTH
168  } else if (targetModel.isCoreTile(tileCol, tileRow)) {
169  step_bits = 13; // XAIEMLGBL_MEMORY_MODULE_DMA_BD0_2_D0_STEPSIZE_WIDTH
170  wrap_bits = 8; // XAIEMLGBL_MEMORY_MODULE_DMA_BD0_3_D0_WRAP_WIDTH
171  } else {
172  return forOp->emitOpError(
173  "Unsupported tile type at (" + std::to_string(tileCol) + ", " +
174  std::to_string(tileRow) + ") Must be ShimNOC, Mem or Core.");
175  }
176 
177  for (int i = 0; i < 4; i++) {
178  if (inputSizes[i] <= 0) {
179  return forOp->emitOpError("Size ") << i << " must be a positive integer.";
180  }
181  }
182 
183  if (inputSizes[0] * elemWidth % addressGranularity != 0) {
184  std::stringstream msg;
185  msg << "Transfer sizes must be multiples of " << (addressGranularity / 8)
186  << " bytes. " << inputSizes[0] << " elements at " << (elemWidth / 8)
187  << " bytes each equal " << (inputSizes[0] * elemWidth / 8)
188  << " bytes, which is not divisible by " << (addressGranularity / 8)
189  << ". ";
190  return forOp->emitOpError(msg.str());
191  }
192 
193  for (int i = 0; i < 3; i++) {
194  if (inputSizes[i] > 1 && inputStrides[i] < 1) {
195  // If inputSize[i] == 1, anything is allowable in the stride, since that
196  // stride will never be applied. For any larger size, we must verify that
197  // the stride is positive.
198  return forOp->emitOpError("Stride ")
199  << i << " must be a positive integer.";
200  }
201  }
202  // A value of zero is allowable for the fourth-dimension stride
203  // (this indicates an interation stride for the repeat of 0)
204  if (inputSizes[3] > 1 && inputStrides[3] < 0) {
205  return forOp->emitOpError("Stride 3 must be a non-negative integer.");
206  }
207 
208  for (int i = 0; i < 4; i++) {
209  // strides[0] == 1 is ok iff the transfer size is a multiple of
210  // addressGranularity, which is checked below
211  if (i == 0 && inputStrides[i] == 1)
212  continue;
213  if (inputStrides[i] * elemWidth % addressGranularity != 0) {
214  std::stringstream msg;
215  msg << "Stride " << i << " is " << inputStrides[i] << " elements * "
216  << (elemWidth / 8) << " bytes = " << (inputStrides[i] * elemWidth / 8)
217  << " bytes, which is not divisible by " << (addressGranularity / 8)
218  << ". ";
219  return forOp->emitOpError(msg.str());
220  }
221  }
222 
223  if (!skipTransformationChecks && hardwareSizes[0] > (1 << wrap_bits) - 1)
224  return forOp->emitOpError(
225  "Size 0 exceeds the [0:" + std::to_string((1 << wrap_bits) - 1) +
226  "] range.");
227  if (hardwareSizes[1] > (1 << wrap_bits) - 1)
228  return forOp->emitOpError(
229  "Size 1 exceeds the [0:" + std::to_string((1 << wrap_bits) - 1) +
230  "] range.");
231  if (hardwareSizes[3] > (1 << iter_bits))
232  return forOp->emitOpError(
233  "Size 3 exceeds the [1:" + std::to_string(1 << iter_bits) + "] range.");
234  if (hardwareStrides[0] > (1 << step_bits))
235  return forOp->emitOpError("Stride 0 exceeds the [1:" +
236  std::to_string(1 << step_bits) + "] range.");
237  if (hardwareStrides[1] > (1 << step_bits))
238  return forOp->emitOpError("Stride 1 exceeds the [1:" +
239  std::to_string(1 << step_bits) + "] range.");
240  if (hardwareStrides[2] > (1 << step_bits))
241  return forOp->emitOpError("Stride 2 exceeds the [1:" +
242  std::to_string(1 << step_bits) + "] range.");
243  // strides[3] exceeding the range is ok iff the sizes[3] is one, which is
244  // checked below
245  if (hardwareStrides[3] > (1 << step_bits) && hardwareSizes[3] > 0)
246  return forOp->emitOpError("Stride 3 exceeds the [1:" +
247  std::to_string(1 << step_bits) + "] range.");
248 
249  return success();
250 }

298 /* Calculates the offset value to be written to the
299  */
300 int64_t AIEX::NpuDmaMemcpyNdOp::getOffsetInBytes() {
301  llvm::SmallVector<int64_t, 4> offsets =
302  llvm::map_to_vector(llvm::reverse(getMixedOffsets()), [](OpFoldResult s) {
303  return getConstantIntValue(s).value();
304  });
305  llvm::SmallVector<int64_t, 4> strides =
306  llvm::map_to_vector(llvm::reverse(getMixedStrides()), [](OpFoldResult s) {
307  return getConstantIntValue(s).value();
308  });
309  size_t offset = 0;
310  MemRefType my_memref = getMemref().getType();
311  size_t R = offsets.size();
312  size_t el_bit_width = my_memref.getElementTypeBitWidth();
313  assert(el_bit_width % 8 == 0 &&
314  "Expected Memref element bitwidth to be multiple of 8.");
315  size_t S = el_bit_width / 8;
316  for (size_t i = 0; i < R; i++)
317  offset += offsets[i] * strides[i] * S;
318  return offset;
319 }

321 // dma_memcpy_nd transfers of the form [*, 1, 1, len][*, 0, 0, 1] do not
322 // specify any data layout transformation, but simply express a contiguous
323 // transfer of `len`. We exclude checks to 4th dimension, because repeat count
324 // is still possible without a data layout transformation.
325 bool AIEX::NpuDmaMemcpyNdOp::isLinearTransferWithoutTransformation() {
326  llvm::SmallVector<int64_t, 4> inputSizes =
327  llvm::map_to_vector(llvm::reverse(getMixedSizes()), [](OpFoldResult s) {
328  return getConstantIntValue(s).value();
329  });
330  llvm::SmallVector<int64_t, 4> inputStrides =
331  llvm::map_to_vector(llvm::reverse(getMixedStrides()), [](OpFoldResult s) {
332  return getConstantIntValue(s).value();
333  });
334  return (inputSizes[1] == 1 && inputSizes[2] == 1 && inputStrides[0] == 1 &&
335  inputStrides[1] == 0 && inputStrides[2] == 0);
336 }

338 LogicalResult AIEX::NpuDmaMemcpyNdOp::verify() {
339  MemRefType buffer = getMemref().getType();
340  const auto &targetModel = AIE::getTargetModel(*this);
341  auto addressGranularity = targetModel.getAddressGenGranularity();
342 
343  if (buffer.getElementTypeBitWidth() > addressGranularity) {
344  return emitOpError("Maximum element bit width allowed is ")
345  << addressGranularity << "bits. ";
346  } else if (buffer.hasStaticShape() &&
347  (buffer.getNumElements() * buffer.getElementTypeBitWidth()) <
348  addressGranularity) {
349  return emitOpError("Minimum data transfer size required is ")
350  << addressGranularity << "bits. ";
351  }
352  if (!llvm::all_of(getMixedStrides(), [](OpFoldResult s) {
353  return getConstantIntValue(s).has_value();
354  }))
355  return emitOpError("Only constant strides currently supported.");
356  if (!llvm::all_of(getMixedSizes(), [](OpFoldResult s) {
357  return getConstantIntValue(s).has_value();
358  }))
359  return emitOpError("Only constant sizes currently supported.");
360  if (!llvm::all_of(getMixedOffsets(), [](OpFoldResult s) {
361  return getConstantIntValue(s).has_value();
362  }))
363  return emitOpError("Only constant offsets currently supported.");
364 
365  llvm::SmallVector<int64_t, 4> inputSizes =
366  llvm::map_to_vector(llvm::reverse(getMixedSizes()), [](OpFoldResult s) {
367  return getConstantIntValue(s).value();
368  });
369  llvm::SmallVector<int64_t, 4> inputStrides =
370  llvm::map_to_vector(llvm::reverse(getMixedStrides()), [](OpFoldResult s) {
371  return getConstantIntValue(s).value();
372  });
373  llvm::SmallVector<int64_t, 4> hardwareSizes(4);
374  llvm::SmallVector<int64_t, 4> hardwareStrides(4);
375  getHardwareStridesWraps(targetModel, buffer, inputSizes, inputStrides,
376  hardwareSizes, hardwareStrides);
377  int64_t offset = getOffsetInBytes();
378 
379  // The experimental HSA target uses this op on AIE1, skip all the AIE2
380  // specific checks
381  if (targetModel.getTargetArch() == AIE::AIEArch::AIE1)
382  return success();
383 
384  if (offset % 4 != 0) {
385  return emitOpError("Offset must be 4-byte-aligned.");
386  }
387 
388  // dma_memcpy_nd transfers of the form [1, 1, 1, len][0, 0, 0, 1] do not
389  // specify any data layout transformation, but simply express a contiguous
390  // transfer of `len`. For backwards compatibility, we allow this to proceed
391  // even if it exceeds the maximum stride/wrap size of any one dimension,
392  // and simply do not lower any data layout transformations, since there is
393  // no other way to express this at the dma_memcpy_nd interface otherwise.
394  bool skipTransformationChecks = isLinearTransferWithoutTransformation();
395  if (failed(verifyStridesWraps(*this, buffer, getX(), getY(), inputSizes,
396  inputStrides, hardwareSizes, hardwareStrides,
397  skipTransformationChecks))) {
398  return failure();
399  }
400 
401  // packet header
402  if (auto packetInfo = getPacket()) {
403  if (packetInfo->getPktType() > 7)
404  return emitOpError("Packet type field can only hold 3 bits.");
405  if (packetInfo->getPktId() > 31)
406  return emitOpError("Packet ID field can only hold 5 bits.");
407  }
408 
409  return success();
410 }

503  ParseResult bodyParseResult = parser.parseRegion(*body, entryArgs, false);
504  if (bodyParseResult) {
505  return bodyParseResult;
506  }
507 
508  return success();

548 std::optional<uint32_t> AIEX::DMAConfigureTaskOp::getFirstBdId() {
549  Region &body = getBody();
550  if (body.empty()) {
551  return std::nullopt;
552  }
553  auto bd_ops = body.front().getOps<AIE::DMABDOp>();
554  if (bd_ops.empty() && body.front().getNumSuccessors() == 1) {
555  // Allow the first block to be empty and point to the entry point of the
556  // chain. This allows for specifying cyclying BD chains (infinite loops)
557  // within the constraints of MLIR syntax.
558  Block &chain_entry = *body.front().getSuccessor(0);
559  bd_ops = chain_entry.getOps<AIE::DMABDOp>();
