29// allocates channels between switchboxes ( but does not assign them)
30// instantiates shim-muxes AND allocates channels ( no need to rip these up in )
31struct ConvertFlowsToInterconnect : OpConversionPattern<FlowOp> {
32 using OpConversionPattern::OpConversionPattern;
33 DeviceOp &device;
34 DynamicTileAnalysis &analyzer;
35 ConvertFlowsToInterconnect(MLIRContext *context, DeviceOp &d,
36 DynamicTileAnalysis &a, PatternBenefit benefit = 1)
37 : OpConversionPattern(context, benefit), device(d), analyzer(a) {}
38
39 void addConnection(ConversionPatternRewriter &rewriter,
40 // could be a shim-mux or a switchbox.
41 Interconnect op, FlowOp flowOp, WireBundle inBundle,
42 int inIndex, WireBundle outBundle, int outIndex) const {
43
44 Region &r = op.getConnections();
45 Block &b = r.front();
46 auto point = rewriter.saveInsertionPoint();
47 rewriter.setInsertionPoint(b.getTerminator());
48
49 rewriter.create<ConnectOp>(rewriter.getUnknownLoc(), inBundle, inIndex,
50 outBundle, outIndex);
51
52 rewriter.restoreInsertionPoint(point);
53
54 LLVM_DEBUG(llvm::dbgs()
55 << "\t\taddConnection() (" << op.colIndex() << ","
56 << op.rowIndex() << ") " << stringifyWireBundle(inBundle)
57 << inIndex << " -> " << stringifyWireBundle(outBundle)
58 << outIndex << "\n");
59 }
60
61 mlir::LogicalResult
62 matchAndRewrite(FlowOp flowOp, OpAdaptor adaptor,
63 ConversionPatternRewriter &rewriter) const override {
64 Operation *Op = flowOp.getOperation();
65 DeviceOp d = flowOp->getParentOfType<DeviceOp>();
66 if (!d) {
67 flowOp->emitOpError("This operation must be contained within a device");
68 return failure();
69 }
70 rewriter.setInsertionPoint(d.getBody()->getTerminator());
71
72 auto srcTile = cast<TileOp>(flowOp.getSource().getDefiningOp());
73 TileID srcCoords = {srcTile.colIndex(), srcTile.rowIndex()};
74 auto srcBundle = flowOp.getSourceBundle();
75 auto srcChannel = flowOp.getSourceChannel();
76 Port srcPort = {srcBundle, srcChannel};
77
78#ifndef NDEBUG
79 auto dstTile = cast<TileOp>(flowOp.getDest().getDefiningOp());
80 TileID dstCoords = {dstTile.colIndex(), dstTile.rowIndex()};
81 auto dstBundle = flowOp.getDestBundle();
82 auto dstChannel = flowOp.getDestChannel();
83 LLVM_DEBUG(llvm::dbgs()
84 << "\n\t---Begin rewrite() for flowOp: (" << srcCoords.col
85 << ", " << srcCoords.row << ")" << stringifyWireBundle(srcBundle)
86 << srcChannel << " -> (" << dstCoords.col << ", "
87 << dstCoords.row << ")" << stringifyWireBundle(dstBundle)
88 << dstChannel << "\n\t");
89#endif
90
91 // if the flow (aka "net") for this FlowOp hasn't been processed yet,
92 // add all switchbox connections to implement the flow
93 TileID srcSbId = {srcCoords.col, srcCoords.row};
94 PathEndPoint srcPoint = {srcSbId, srcPort};
95 if (analyzer.processedFlows[srcPoint]) {
96 LLVM_DEBUG(llvm::dbgs() << "Flow already processed!\n");
97 rewriter.eraseOp(Op);
98 return failure();
99 }
100 // std::map<TileID, SwitchSetting>
101 SwitchSettings settings = analyzer.flowSolutions[srcPoint];
102 // add connections for all the Switchboxes in SwitchSettings
103 for (const auto &[tileId, setting] : settings) {
104 int col = tileId.col;
105 int row = tileId.row;
106 SwitchboxOp swOp = analyzer.getSwitchbox(rewriter, col, row);
107 int shimCh = srcChannel;
108 bool isShim = analyzer.getTile(rewriter, tileId).isShimNOCorPLTile();
109
110 // TODO: must reserve N3, N7, S2, S3 for DMA connections
111 if (isShim && tileId == srcSbId) {
112
113 // shim DMAs at start of flows
114 if (srcBundle == WireBundle::DMA)
115 // must be either DMA0 -> N3 or DMA1 -> N7
116 shimCh = srcChannel == 0 ? 3 : 7;
117 else if (srcBundle == WireBundle::NOC)
118 // must be NOC0/NOC1 -> N2/N3 or NOC2/NOC3 -> N6/N7
119 shimCh = srcChannel >= 2 ? srcChannel + 4 : srcChannel + 2;
120 else if (srcBundle == WireBundle::PLIO)
121 shimCh = srcChannel;
122
123 ShimMuxOp shimMuxOp = analyzer.getShimMux(rewriter, col);
124 addConnection(rewriter, cast<Interconnect>(shimMuxOp.getOperation()),
125 flowOp, srcBundle, srcChannel, WireBundle::North, shimCh);
126 }
127 assert(setting.srcs.size() == setting.dsts.size());
128 for (size_t i = 0; i < setting.srcs.size(); i++) {
129 Port src = setting.srcs[i];
130 Port dest = setting.dsts[i];
131
132 // handle special shim connectivity
133 if (isShim && tileId == srcSbId) {
134 addConnection(rewriter, cast<Interconnect>(swOp.getOperation()),
135 flowOp, WireBundle::South, shimCh, dest.bundle,
136 dest.channel);
137 } else if (isShim && (dest.bundle == WireBundle::DMA ||
138 dest.bundle == WireBundle::PLIO ||
139 dest.bundle == WireBundle::NOC)) {
140
141 // shim DMAs at end of flows
142 if (dest.bundle == WireBundle::DMA)
143 // must be either N2 -> DMA0 or N3 -> DMA1
144 shimCh = dest.channel == 0 ? 2 : 3;
145 else if (dest.bundle == WireBundle::NOC)
146 // must be either N2/3/4/5 -> NOC0/1/2/3
147 shimCh = dest.channel + 2;
148 else if (dest.bundle == WireBundle::PLIO)
149 shimCh = dest.channel;
150
151 ShimMuxOp shimMuxOp = analyzer.getShimMux(rewriter, col);
152 addConnection(rewriter, cast<Interconnect>(shimMuxOp.getOperation()),
153 flowOp, WireBundle::North, shimCh, dest.bundle,
154 dest.channel);
155 addConnection(rewriter, cast<Interconnect>(swOp.getOperation()),
156 flowOp, src.bundle, src.channel, WireBundle::South,
157 shimCh);
158 } else {
159 // otherwise, regular switchbox connection
160 addConnection(rewriter, cast<Interconnect>(swOp.getOperation()),
161 flowOp, src.bundle, src.channel, dest.bundle,
162 dest.channel);
163 }
164 }
165
166 LLVM_DEBUG(llvm::dbgs() << tileId << ": " << setting << " | "
167 << "\n");
168 }
169
170 LLVM_DEBUG(llvm::dbgs()
171 << "\n\t\tFinished adding ConnectOps to implement flowOp.\n");
172
173 analyzer.processedFlows[srcPoint] = true;
174 rewriter.eraseOp(Op);
175 return success();
176 }
177};

183void AIEPathfinderPass::runOnFlow(DeviceOp d, DynamicTileAnalysis &analyzer) {
184 // Apply rewrite rule to switchboxes to add assignments to every 'connect'
185 // operation inside
186 ConversionTarget target(getContext());
187 target.addLegalOp<TileOp>();
188 target.addLegalOp<ConnectOp>();
189 target.addLegalOp<SwitchboxOp>();
190 target.addLegalOp<ShimMuxOp>();
191 target.addLegalOp<EndOp>();
192
193 RewritePatternSet patterns(&getContext());
194 patterns.insert<ConvertFlowsToInterconnect>(d.getContext(), d, analyzer);
195 if (failed(applyPartialConversion(d, target, std::move(patterns))))
196 return signalPassFailure();
197}

217bool AIEPathfinderPass::findPathToDest(SwitchSettings settings, TileID currTile,
218 WireBundle currDestBundle,
219 int currDestChannel, TileID finalTile,
220 WireBundle finalDestBundle,
221 int finalDestChannel) {
222
223 if ((currTile == finalTile) && (currDestBundle == finalDestBundle) &&
224 (currDestChannel == finalDestChannel)) {
225 return true;
226 }
227
228 WireBundle neighbourSourceBundle;
229 TileID neighbourTile;
230 if (currDestBundle == WireBundle::East) {
231 neighbourSourceBundle = WireBundle::West;
232 neighbourTile = {currTile.col + 1, currTile.row};
233 } else if (currDestBundle == WireBundle::West) {
234 neighbourSourceBundle = WireBundle::East;
235 neighbourTile = {currTile.col - 1, currTile.row};
236 } else if (currDestBundle == WireBundle::North) {
237 neighbourSourceBundle = WireBundle::South;
238 neighbourTile = {currTile.col, currTile.row + 1};
239 } else if (currDestBundle == WireBundle::South) {
240 neighbourSourceBundle = WireBundle::North;
241 neighbourTile = {currTile.col, currTile.row - 1};
242 } else {
243 return false;
244 }
245
246 int neighbourSourceChannel = currDestChannel;
247 for (const auto &[sbNode, setting] : settings) {
248 TileID tile = {sbNode.col, sbNode.row};
249 if (tile == neighbourTile) {
250 assert(setting.srcs.size() == setting.dsts.size());
251 for (size_t i = 0; i < setting.srcs.size(); i++) {
252 Port src = setting.srcs[i];
253 Port dest = setting.dsts[i];
254 if ((src.bundle == neighbourSourceBundle) &&
255 (src.channel == neighbourSourceChannel)) {
256 if (findPathToDest(settings, neighbourTile, dest.bundle, dest.channel,
257 finalTile, finalDestBundle, finalDestChannel)) {
258 return true;
259 }
260 }
261 }
262 }
263 }
264
265 return false;
266}

268void AIEPathfinderPass::runOnPacketFlow(DeviceOp device, OpBuilder &builder,
269 DynamicTileAnalysis &analyzer) {
270
271 ConversionTarget target(getContext());
272
273 std::map<TileID, mlir::Operation *> tiles;
274
275 // Map from a port and flowID to
276 std::map<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>> packetFlows;
277 std::map<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>> ctrlPacketFlows;
278 SmallVector<std::pair<PhysPort, int>, 4> slavePorts;
279 DenseMap<std::pair<PhysPort, int>, int> slaveAMSels;
280 // Flag to keep packet header at packet flow destination
281 DenseMap<PhysPort, BoolAttr> keepPktHeaderAttr;
282 // Map from tileID and master ports to flags labelling control packet flows
283 DenseMap<std::pair<PhysPort, int>, bool> ctrlPktFlows;
284
285 for (auto tileOp : device.getOps<TileOp>()) {
286 int col = tileOp.colIndex();
287 int row = tileOp.rowIndex();
288 tiles[{col, row}] = tileOp;
289 }
290
291 // The logical model of all the switchboxes.
292 std::map<TileID, SmallVector<std::pair<Connect, int>, 8>> switchboxes;
293 for (PacketFlowOp pktFlowOp : device.getOps<PacketFlowOp>()) {
294 Region &r = pktFlowOp.getPorts();
295 Block &b = r.front();
296 int flowID = pktFlowOp.IDInt();
297 Port srcPort, destPort;
298 TileOp srcTile, destTile;
299 TileID srcCoords, destCoords;
300
301 for (Operation &Op : b.getOperations()) {
302 if (auto pktSource = dyn_cast<PacketSourceOp>(Op)) {
303 srcTile = dyn_cast<TileOp>(pktSource.getTile().getDefiningOp());
304 srcPort = pktSource.port();
305 srcCoords = {srcTile.colIndex(), srcTile.rowIndex()};
306 } else if (auto pktDest = dyn_cast<PacketDestOp>(Op)) {
307 destTile = dyn_cast<TileOp>(pktDest.getTile().getDefiningOp());
308 destPort = pktDest.port();
309 destCoords = {destTile.colIndex(), destTile.rowIndex()};
310 // Assign "keep_pkt_header flag"
311 auto keep = pktFlowOp.getKeepPktHeader();
312 keepPktHeaderAttr[{destTile.getTileID(), destPort}] =
313 keep ? BoolAttr::get(Op.getContext(), *keep) : nullptr;
314
315 TileID srcSB = {srcCoords.col, srcCoords.row};
316 if (PathEndPoint srcPoint = {srcSB, srcPort};
317 !analyzer.processedFlows[srcPoint]) {
318 SwitchSettings settings = analyzer.flowSolutions[srcPoint];
319 // add connections for all the Switchboxes in SwitchSettings
320 for (const auto &[curr, setting] : settings) {
321 assert(setting.srcs.size() == setting.dsts.size());
322 TileID currTile = {curr.col, curr.row};
323 for (size_t i = 0; i < setting.srcs.size(); i++) {
324 Port src = setting.srcs[i];
325 Port dest = setting.dsts[i];
326 // reject false broadcast
327 if (!findPathToDest(settings, currTile, dest.bundle, dest.channel,
328 destCoords, destPort.bundle,
329 destPort.channel))
330 continue;
331 Connect connect = {{src.bundle, src.channel},
332 {dest.bundle, dest.channel}};
333 if (std::find(switchboxes[currTile].begin(),
334 switchboxes[currTile].end(),
335 std::pair{connect, flowID}) ==
336 switchboxes[currTile].end())
337 switchboxes[currTile].push_back({connect, flowID});
338 // Assign "control packet flows" flag per switchbox, based on
339 // packet flow op attribute
340 auto ctrlPkt = pktFlowOp.getPriorityRoute();
341 ctrlPktFlows[{{currTile, dest}, flowID}] =
342 ctrlPkt ? *ctrlPkt : false;
343 }
344 }
345 }
346 }
347 }
348 }
349
350 LLVM_DEBUG(llvm::dbgs() << "Check switchboxes\n");
351
352 for (const auto &[tileId, connects] : switchboxes) {
353 LLVM_DEBUG(llvm::dbgs() << "***switchbox*** " << tileId.col << " "
354 << tileId.row << '\n');
355 for (const auto &[conn, flowID] : connects) {
356 Port sourcePort = conn.src;
357 Port destPort = conn.dst;
358 auto sourceFlow =
359 std::make_pair(std::make_pair(tileId, sourcePort), flowID);
360 if (ctrlPktFlows[{{tileId, destPort}, flowID}])
361 ctrlPacketFlows[sourceFlow].push_back({tileId, destPort});
362 else
363 packetFlows[sourceFlow].push_back({tileId, destPort});
364 slavePorts.push_back(sourceFlow);
365 LLVM_DEBUG(llvm::dbgs() << "flowID " << flowID << ':'
366 << stringifyWireBundle(sourcePort.bundle) << " "
367 << sourcePort.channel << " -> "
368 << stringifyWireBundle(destPort.bundle) << " "
369 << destPort.channel << "\n");
370 }
371 }
372
373 // amsel()
374 // masterset()
375 // packetrules()
376 // rule()
377
378 // Compute arbiter assignments. Each arbiter has four msels.
379 // Therefore, the number of "logical" arbiters is 6 x 4 = 24
380 // A master port can only be associated with one arbiter
381
382 // Constants for arbiter configuration
383 constexpr int INVALID_AMSEL_VALUE = -1;
384 constexpr int INVALID_ARBITER_VALUE = -1;
385
386 // A map from Tile and master selectValue to the ports targetted by that
387 // master select.
388 std::map<std::pair<TileID, int>, SmallVector<Port, 4>> masterAMSels;
389
390 // Track which arbiter each port is assigned to (to prevent conflicts)
391 std::map<PhysPort, int> portToArbiter;
392
393 // Count of currently used logical arbiters for each tile.
394 DenseMap<Operation *, int> amselValues;
395 int numMselsPerArbiter = 4;
396 int numArbiters = 6;
397
398 // Get arbiter id from amsel
399 auto getArbiterIDFromAmsel = [numArbiters](int amsel) {
400 return amsel % numArbiters;
401 };
402 // Get amsel from arbiter id and msel
403 auto getAmselFromArbiterIDAndMsel = [numArbiters](int arbiter, int msel) {
404 return arbiter + msel * numArbiters;
405 };
406 // Get a new unique amsel from masterAMSels on tile op. Prioritize on
407 // incrementing arbiter id, before incrementing msel
408 auto getNewUniqueAmsel =
409 [&](const std::map<std::pair<TileID, int>, SmallVector<Port, 4>>
410 &masterAMSels,
411 TileOp tileOp, bool isCtrlPkt) {
412 if (isCtrlPkt) { // Higher AMsel first
413 for (int i = numMselsPerArbiter - 1; i >= 0; i--)
414 for (int a = numArbiters - 1; a >= 0; a--)
415 if (!masterAMSels.count(
416 {tileOp.getTileID(), getAmselFromArbiterIDAndMsel(a, i)}))
417 return getAmselFromArbiterIDAndMsel(a, i);
418 } else { // Lower AMsel first
419 for (int i = 0; i < numMselsPerArbiter; i++)
420 for (int a = 0; a < numArbiters; a++)
421 if (!masterAMSels.count(
422 {tileOp.getTileID(), getAmselFromArbiterIDAndMsel(a, i)}))
423 return getAmselFromArbiterIDAndMsel(a, i);
424 }
425 tileOp->emitOpError(
426 "tile op has used up all arbiter-msel combinations");
427 return INVALID_AMSEL_VALUE;
428 };
429 // Get a new unique amsel from masterAMSels on tile op with given arbiter id
430 auto getNewUniqueAmselPerArbiterID =
431 [&](const std::map<std::pair<TileID, int>, SmallVector<Port, 4>>
432 &masterAMSels,
433 TileOp tileOp, int arbiter) {
434 for (int i = 0; i < numMselsPerArbiter; i++)
435 if (!masterAMSels.count({tileOp.getTileID(),
436 getAmselFromArbiterIDAndMsel(arbiter, i)}))
437 return getAmselFromArbiterIDAndMsel(arbiter, i);
438 tileOp->emitOpError("tile op arbiter ")
439 << std::to_string(arbiter) << " has used up all its msels";
440 return INVALID_AMSEL_VALUE;
441 };
442
443 // Order the packet flows in order to get determinsitic amsel allocation;
444 // allocate amsels for control packet flows before others to ensure
445 // consistency in control packet flow overlay.
446 auto getSortedPacketFlows =
447 [&](std::map<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>> pktFlows,
448 std::map<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>>
449 ctrlPktFlows) {
450 std::vector<
451 std::pair<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>>>
452 sortedpktFlows(pktFlows.begin(), pktFlows.end());
453 std::vector<
454 std::pair<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>>>
455 sortedctrlpktFlows(ctrlPktFlows.begin(), ctrlPktFlows.end());
456 sortedctrlpktFlows.insert(sortedctrlpktFlows.end(),
457 sortedpktFlows.begin(), sortedpktFlows.end());
458 return sortedctrlpktFlows;
459 };
460
461 std::vector<std::pair<std::pair<PhysPort, int>, SmallVector<PhysPort, 4>>>
462 sortedPacketFlows = getSortedPacketFlows(packetFlows, ctrlPacketFlows);
463
464 packetFlows.insert(ctrlPacketFlows.begin(), ctrlPacketFlows.end());
465
466 // Helper function to assign ports to a given amsel value
467 // Updates masterAMSels and portToArbiter, skipping ports already on different
468 // arbiters
469 auto assignPortsToAmsel = [&](TileID tileId, int amselValue,
470 const SmallVector<PhysPort, 4> &destinations) {
471 int targetArbiter = getArbiterIDFromAmsel(amselValue);
472 for (auto dest : destinations) {
473 Port port = dest.second;
474 PhysPort physPort = {tileId, port};
475
476 // Skip this port if it's already assigned to a different arbiter
477 if (portToArbiter.count(physPort) &&
478 portToArbiter[physPort] != targetArbiter) {
479 LLVM_DEBUG(llvm::dbgs()
480 << "Skipping port " << stringifyWireBundle(port.bundle)
481 << ":" << port.channel << " - already on arbiter "
482 << portToArbiter[physPort] << ", target is " << targetArbiter
483 << "\n");
484 continue;
485 }
486
487 masterAMSels[{tileId, amselValue}].push_back(port);
488 portToArbiter[physPort] = targetArbiter;
489 }
490 };
491
492 // Helper function to find existing arbiter assignment for ports in a flow
493 // Returns -1 if no existing assignment found
494 auto findExistingArbiter =
495 [&](TileID tileId, const SmallVector<PhysPort, 4> &destinations) -> int {
496 for (auto dest : destinations) {
497 Port port = dest.second;
498 PhysPort physPort = {tileId, port};
499 if (portToArbiter.count(physPort)) {
500 return portToArbiter[physPort];
501 }
502 }
503 return INVALID_ARBITER_VALUE;
504 };
505
506 // Check all multi-cast flows (same source, same ID). They should be
507 // assigned the same arbiter and msel so that the flow can reach all the
508 // destination ports at the same time For destination ports that appear in
509 // different (multicast) flows, it should have a different <arbiterID, msel>
510 // value pair for each flow
511 for (const auto &packetFlow : sortedPacketFlows) {
512 // The Source Tile of the flow
513 TileID tileId = packetFlow.first.first.first;
514 TileOp tileOp = analyzer.getTile(builder, tileId);
515 if (amselValues.count(tileOp) == 0)
516 amselValues[tileOp] = 0;
517
518 // arb0: 6*0, 6*1, 6*2, 6*3
519 // arb1: 6*0+1, 6*1+1, 6*2+1, 6*3+1
520 // arb2: 6*0+2, 6*1+2, 6*2+2, 6*3+2
521 // arb3: 6*0+3, 6*1+3, 6*2+3, 6*3+3
522 // arb4: 6*0+4, 6*1+4, 6*2+4, 6*3+4
523 // arb5: 6*0+5, 6*1+5, 6*2+5, 6*3+5
524
525 int amselValue = amselValues[tileOp];
526 assert(amselValue < numArbiters && "Could not allocate new arbiter!");
527
528 // Find existing arbiter and amsel assignments for this flow
529 // Strategy: Look for existing amsel entries that match the flow's
530 // destinations
531 // - Complete match: reuse existing amsel
532 // - Partial match: create new amsel on same arbiter
533 // - No match: create new amsel (on existing arbiter if ports already
534 // assigned)
535 bool hasMatchingAmselEntry = false;
536 int partialMatchArbiterID = INVALID_ARBITER_VALUE;
537
538 // Check if any ports in this flow already have arbiter assignments
539 int existingArbiter = findExistingArbiter(tileId, packetFlow.second);
540
541 // Search for matching amsel entries, prioritizing those that match the
542 // existing arbiter
543 for (const auto &amselEntry : masterAMSels) {
544 if (amselEntry.first.first != tileId)
545 continue;
546 amselValue = amselEntry.first.second;
547 int thisArbiter = getArbiterIDFromAmsel(amselValue);
548
549 // If we have an existing arbiter assignment, only consider amsels from
550 // that arbiter
551 if (existingArbiter != INVALID_ARBITER_VALUE &&
552 existingArbiter != thisArbiter) {
553 continue;
554 }
555
556 // Check if destinations match (completely or partially)
557 const SmallVector<Port, 4> &existingPorts =
558 masterAMSels[{tileId, amselValue}];
559 bool hasOverlap = false;
560 bool hasNonOverlap = false;
561
562 for (auto dest : packetFlow.second) {
563 Port port = dest.second;
564 if (std::find(existingPorts.begin(), existingPorts.end(), port) ==
565 existingPorts.end())
566 hasNonOverlap = true;
567 else
568 hasOverlap = true;
569 }
570
571 if (hasOverlap) {
572 hasMatchingAmselEntry = true;
573 // Partial match if some ports don't overlap or sizes differ
574 if (hasNonOverlap || existingPorts.size() != packetFlow.second.size())
575 partialMatchArbiterID = thisArbiter;
576 break;
577 }
578 }
579
580 if (!hasMatchingAmselEntry) {
581 // This packet flow switchbox's output ports completely mismatches with
582 // any existing amsel. Creating a new amsel.
583
584 // Determine target arbiter: use existing if available, otherwise allocate
585 // based on priority
586 int targetArbiter = existingArbiter;
587
588 if (targetArbiter == INVALID_ARBITER_VALUE) {
589 // No existing assignment, choose based on control packet priority
590 bool isCtrlPkt =
591 llvm::any_of(packetFlow.second, [&](PhysPort destPhysPort) {
592 Port port = destPhysPort.second;
593 return ctrlPktFlows[{{tileId, port}, packetFlow.first.second}];
594 });
595
596 amselValue = getNewUniqueAmsel(masterAMSels, tileOp, isCtrlPkt);
597 } else {
598 // Use existing arbiter to maintain consistency
599 amselValue =
600 getNewUniqueAmselPerArbiterID(masterAMSels, tileOp, targetArbiter);
601 if (amselValue == INVALID_AMSEL_VALUE) {
602 // No more msels available on this arbiter - this is a routing
603 // conflict
604 tileOp->emitOpError("cannot assign flow: arbiter ")
605 << targetArbiter
606 << " has no free msels, but flow requires this arbiter due to "
607 "existing port assignments";
608 return;
609 }
610 }
611
612 // Update masterAMSels with new amsel, skipping ports already assigned to
613 // different arbiters
614 assignPortsToAmsel(tileId, amselValue, packetFlow.second);
615 } else if (partialMatchArbiterID != INVALID_ARBITER_VALUE) {
616 // This packet flow switchbox's output ports partially overlaps with
617 // some existing amsel. Create a NEW amsel with the SAME arbiter for this
618 // flow. The comment states: "destination ports that appear in different
619 // (multicast) flows should have a different <arbiterID, msel> value pair
620 // for each flow" but use the same arbiter to maintain the constraint.
621
622 // Use the arbiter we found (which should match existingArbiter if set)
623 int targetArbiter = partialMatchArbiterID;
624 if (existingArbiter != INVALID_ARBITER_VALUE &&
625 existingArbiter != targetArbiter) {
626 // Conflict detected - should have been caught earlier, but add safety
627 // check
628 tileOp->emitOpError(
629 "internal error: arbiter conflict in partial match");
630 return;
631 }
632
633 amselValue =
634 getNewUniqueAmselPerArbiterID(masterAMSels, tileOp, targetArbiter);
635
636 // Update masterAMSels with new amsel, skipping ports already assigned to
637 // different arbiters
638 assignPortsToAmsel(tileId, amselValue, packetFlow.second);
639 } else {
640 // Complete match - reuse the existing amsel
641 // Track arbiter assignments for all ports in this flow
642 int arbiter = getArbiterIDFromAmsel(amselValue);
643 for (auto dest : packetFlow.second) {
644 Port port = dest.second;
645 PhysPort physPort = {tileId, port};
646 // Update tracking even for reused amsels
647 if (!portToArbiter.count(physPort)) {
648 portToArbiter[physPort] = arbiter;
649 }
650 }
651 }
652
653 slaveAMSels[packetFlow.first] = amselValue;
654 amselValues[tileOp] = getArbiterIDFromAmsel(amselValue);
655 }
656
657 // Compute the master set IDs
658 // A map from a switchbox output port to its associated amsel values
659 std::map<PhysPort, SmallVector<int, 4>> mastersets;
660 for (const auto &[physPort, ports] : masterAMSels) {
661 TileID tileId = physPort.first;
662 int amselValue = physPort.second;
663 for (auto port : ports) {
664 PhysPort physPort = {tileId, port};
665 mastersets[physPort].push_back(amselValue);
666 }
667 }
668
669 // Validate that each port only has amsels from a single arbiter
670 for (const auto &[physPort, amselList] : mastersets) {
671 int assignedArbiter = INVALID_ARBITER_VALUE;
672 for (auto amsel : amselList) {
673 int thisArbiter = getArbiterIDFromAmsel(amsel);
674 if (assignedArbiter != INVALID_ARBITER_VALUE &&
675 assignedArbiter != thisArbiter) {
676 TileID tileId = physPort.first;
677 Port port = physPort.second;
678 TileOp tileOp = analyzer.getTile(builder, tileId);
679 tileOp->emitOpError("port ")
680 << stringifyWireBundle(port.bundle) << ":" << port.channel
681 << " assigned to multiple arbiters: " << assignedArbiter << " and "
682 << thisArbiter << " (amsels: " << amselList[0];
683 for (size_t i = 1; i < amselList.size(); i++) {
684 llvm::errs() << ", " << amselList[i];
685 }
686 llvm::errs() << ")\n";
687 return signalPassFailure();
688 }
689 assignedArbiter = thisArbiter;
690 }
691 }
692
693 LLVM_DEBUG(llvm::dbgs() << "CHECK mastersets\n");
694#ifndef NDEBUG
695 for (const auto &[physPort, values] : mastersets) {
696 TileID tileId = physPort.first;
697 WireBundle bundle = physPort.second.bundle;
698 int channel = physPort.second.channel;
699 LLVM_DEBUG(llvm::dbgs()
700 << "master " << tileId << " " << stringifyWireBundle(bundle)
701 << " : " << channel << '\n');
702 for (auto value : values)
703 LLVM_DEBUG(llvm::dbgs() << "amsel: " << value << '\n');
704 }
705#endif
706
707 // Compute mask values
708 // Merging as many stream flows as possible
709 // The flows must originate from the same source port and have different IDs
710 // Two flows can be merged if they share the same destinations
711 SmallVector<SmallVector<std::pair<PhysPort, int>, 4>, 4> slaveGroups;
712 SmallVector<std::pair<PhysPort, int>, 4> workList(slavePorts);
713 while (!workList.empty()) {
714 auto slave1 = workList.pop_back_val();
715 Port slavePort1 = slave1.first.second;
716
717 bool foundgroup = false;
718 for (auto &group : slaveGroups) {
719 auto slave2 = group.front();
720 if (Port slavePort2 = slave2.first.second; slavePort1 != slavePort2)
721 continue;
722
723 bool matched = true;
724 auto dests1 = packetFlows[slave1];
725 auto dests2 = packetFlows[slave2];
726 if (dests1.size() != dests2.size())
727 continue;
728
729 for (auto dest1 : dests1) {
730 if (std::find(dests2.begin(), dests2.end(), dest1) == dests2.end()) {
731 matched = false;
732 break;
733 }
734 }
735
736 if (matched) {
737 group.push_back(slave1);
738 foundgroup = true;
739 break;
740 }
741 }
742
743 if (!foundgroup) {
744 SmallVector<std::pair<PhysPort, int>, 4> group({slave1});
745 slaveGroups.push_back(group);
746 }
747 }
748
749 std::map<std::pair<PhysPort, int>, int> slaveMasks;
750 for (const auto &group : slaveGroups) {
751 // Iterate over all the ID values in a group
752 // If bit n-th (n <= 5) of an ID value differs from bit n-th of another ID
753 // value, the bit position should be "don't care", and we will set the
754 // mask bit of that position to 0
755 int mask[5] = {-1, -1, -1, -1, -1};
756 for (auto port : group) {
757 int ID = port.second;
758 for (int i = 0; i < 5; i++) {
759 if (mask[i] == -1)
760 mask[i] = ID >> i & 0x1;
761 else if (mask[i] != (ID >> i & 0x1))
762 mask[i] = 2; // found bit difference --> mark as "don't care"
763 }
764 }
765
766 int maskValue = 0;
767 for (int i = 4; i >= 0; i--) {
768 if (mask[i] == 2) // don't care
769 mask[i] = 0;
770 else
771 mask[i] = 1;
772 maskValue = (maskValue << 1) + mask[i];
773 }
774 for (auto port : group)
775 slaveMasks[port] = maskValue;
776 }
777
778#ifndef NDEBUG
779 LLVM_DEBUG(llvm::dbgs() << "CHECK Slave Masks\n");
780 for (auto map : slaveMasks) {
781 PhysPort port = map.first.first;
782 TileOp tile = analyzer.getTile(builder, port.first);
783 WireBundle bundle = port.second.bundle;
784 int channel = port.second.channel;
785 int ID = map.first.second;
786 int mask = map.second;
787
788 LLVM_DEBUG(llvm::dbgs()
789 << "Port " << tile << " " << stringifyWireBundle(bundle) << " "
790 << channel << '\n');
791 LLVM_DEBUG(llvm::dbgs() << "Mask "
792 << "0x" << llvm::Twine::utohexstr(mask) << '\n');
793 LLVM_DEBUG(llvm::dbgs() << "ID "
794 << "0x" << llvm::Twine::utohexstr(ID) << '\n');
795 for (int i = 0; i < 31; i++) {
796 if ((i & mask) == (ID & mask))
797 LLVM_DEBUG(llvm::dbgs() << "matches flow ID "
798 << "0x" << llvm::Twine::utohexstr(i) << '\n');
799 }
800 }
801#endif
802
803 // Realize the routes in MLIR
804
805 // Update tiles map if any new tile op declaration is needed for constructing
806 // the flow.
807 for (const auto &swMap : mastersets) {
808 TileID tileId = swMap.first.first;
809 TileOp tileOp = analyzer.getTile(builder, tileId);
810 if (std::none_of(tiles.begin(), tiles.end(),
811 [&tileOp](const std::pair<const xilinx::AIE::TileID,
812 Operation *> &tileMapEntry) {
813 return tileMapEntry.second == tileOp.getOperation();
814 })) {
815 tiles[{tileOp.colIndex(), tileOp.rowIndex()}] = tileOp;
816 }
817 }
818
819 for (auto map : tiles) {
820 Operation *tileOp = map.second;
821 TileOp tile = cast<TileOp>(map.second);
822 TileID tileId = tile.getTileID();
823
824 // Create a switchbox for the routes and insert inside it.
825 builder.setInsertionPointAfter(tileOp);
826 SwitchboxOp swbox =
827 analyzer.getSwitchbox(builder, tile.colIndex(), tile.rowIndex());
828 SwitchboxOp::ensureTerminator(swbox.getConnections(), builder,
829 builder.getUnknownLoc());
830 Block &b = swbox.getConnections().front();
831 builder.setInsertionPoint(b.getTerminator());
832
833 std::vector<bool> amselOpNeededVector(numMselsPerArbiter * numArbiters);
834 for (const auto &map : mastersets) {
835 if (tileId != map.first.first)
836 continue;
837
838 for (auto value : map.second) {
839 amselOpNeededVector[value] = true;
840 }
841 }
842 // Create all the amsel Ops
843 std::map<int, AMSelOp> amselOps;
844 for (int i = 0; i < numMselsPerArbiter; i++) {
845 for (int a = 0; a < numArbiters; a++) {
846 auto amselValue = getAmselFromArbiterIDAndMsel(a, i);
847 if (amselOpNeededVector[amselValue]) {
848 int arbiterID = a;
849 int msel = i;
850 auto amsel =
851 builder.create<AMSelOp>(builder.getUnknownLoc(), arbiterID, msel);
852 amselOps[amselValue] = amsel;
853 }
854 }
855 }
856 // Create all the master set Ops
857 // First collect the master sets for this tile.
858 SmallVector<Port, 4> tileMasters;
859 for (const auto &map : mastersets) {
860 if (tileId != map.first.first)
861 continue;
862 tileMasters.push_back(map.first.second);
863 }
864 // Sort them so we get a reasonable order
865 std::sort(tileMasters.begin(), tileMasters.end());
866 for (auto tileMaster : tileMasters) {
867 WireBundle bundle = tileMaster.bundle;
868 int channel = tileMaster.channel;
869 SmallVector<int, 4> msels = mastersets[{tileId, tileMaster}];
870 SmallVector<Value, 4> amsels;
871 for (auto msel : msels) {
872 assert(amselOps.count(msel) == 1);
873 amsels.push_back(amselOps[msel]);
874 }
875
876 builder.create<MasterSetOp>(
877 builder.getUnknownLoc(), builder.getIndexType(), bundle, channel,
878 amsels, keepPktHeaderAttr[{tileId, tileMaster}]);
879 }
880
881 // Generate the packet rules
882 DenseMap<Port, PacketRulesOp> slaveRules;
883 for (auto group : slaveGroups) {
884 builder.setInsertionPoint(b.getTerminator());
885
886 auto port = group.front().first;
887 if (tileId != port.first)
888 continue;
889
890 WireBundle bundle = port.second.bundle;
891 int channel = port.second.channel;
892 auto slave = port.second;
893
894 int mask = slaveMasks[group.front()];
895 int ID = group.front().second & mask;
896
897 // Verify that we actually map all the ID's correctly.
898#ifndef NDEBUG
899 for (auto slave : group)
900 assert((slave.second & mask) == ID);
901#endif
902 Value amsel = amselOps[slaveAMSels[group.front()]];
903
904 PacketRulesOp packetrules;
905 if (slaveRules.count(slave) == 0) {
906 packetrules = builder.create<PacketRulesOp>(builder.getUnknownLoc(),
907 bundle, channel);
908 PacketRulesOp::ensureTerminator(packetrules.getRules(), builder,
909 builder.getUnknownLoc());
910 slaveRules[slave] = packetrules;
911 } else
912 packetrules = slaveRules[slave];
913
914 Block &rules = packetrules.getRules().front();
915
916 // Verify ID mapping against all other rules of the same slave.
917 for (auto rule : rules.getOps<PacketRuleOp>()) {
918 auto verifyMask = rule.maskInt();
919 auto verifyValue = rule.valueInt();
920 if ((group.front().second & verifyMask) == verifyValue) {
921 rule->emitOpError("can lead to false packet id match for id ")
922 << ID << ", which is not supposed to pass through this port.";
923 rule->emitRemark("Please consider changing all uses of packet id ")
924 << ID << " to avoid deadlock.";
925 }
926 }
927
928 builder.setInsertionPoint(rules.getTerminator());
929 builder.create<PacketRuleOp>(builder.getUnknownLoc(), mask, ID, amsel);
930 }
931 }
932
933 // Add support for shimDMA
934 // From shimDMA to BLI: 1) shimDMA 0 --> North 3
935 // 2) shimDMA 1 --> North 7
936 // From BLI to shimDMA: 1) North 2 --> shimDMA 0
937 // 2) North 3 --> shimDMA 1
938
939 for (auto switchbox : make_early_inc_range(device.getOps<SwitchboxOp>())) {
940 auto retVal = switchbox->getOperand(0);
941 auto tileOp = retVal.getDefiningOp<TileOp>();
942
943 // Check if it is a shim Tile
944 if (!tileOp.isShimNOCTile())
945 continue;
946
947 // Check if the switchbox is empty
948 if (&switchbox.getBody()->front() == switchbox.getBody()->getTerminator())
949 continue;
950
951 Region &r = switchbox.getConnections();
952 Block &b = r.front();
953
954 // Find if the corresponding shimmux exsists or not
955 int shimExist = 0;
956 ShimMuxOp shimOp;
957 for (auto shimmux : device.getOps<ShimMuxOp>()) {
958 if (shimmux.getTile() == tileOp) {
959 shimExist = 1;
960 shimOp = shimmux;
961 break;
962 }
963 }
964
965 for (Operation &Op : b.getOperations()) {
966 if (auto pktrules = dyn_cast<PacketRulesOp>(Op)) {
967
968 // check if there is MM2S DMA in the switchbox of the 0th row
969 if (pktrules.getSourceBundle() == WireBundle::DMA) {
970
971 // If there is, then it should be put into the corresponding shimmux
972 // If shimmux not defined then create shimmux
973 if (!shimExist) {
974 builder.setInsertionPointAfter(tileOp);
975 shimOp = analyzer.getShimMux(builder, tileOp.colIndex());
976 shimExist = 1;
977 }
978
979 Region &r0 = shimOp.getConnections();
980 Block &b0 = r0.front();
981 builder.setInsertionPointToStart(&b0);
982
983 pktrules.setSourceBundle(WireBundle::South);
984 if (pktrules.getSourceChannel() == 0) {
985 pktrules.setSourceChannel(3);
986 builder.create<ConnectOp>(builder.getUnknownLoc(), WireBundle::DMA,
987 0, WireBundle::North, 3);
988 }
989 if (pktrules.getSourceChannel() == 1) {
990 pktrules.setSourceChannel(7);
991 builder.create<ConnectOp>(builder.getUnknownLoc(), WireBundle::DMA,
992 1, WireBundle::North, 7);
993 }
994 }
995 }
996
997 if (auto mtset = dyn_cast<MasterSetOp>(Op)) {
998
999 // check if there is S2MM DMA in the switchbox of the 0th row
1000 if (mtset.getDestBundle() == WireBundle::DMA) {
1001
1002 // If there is, then it should be put into the corresponding shimmux
1003 // If shimmux not defined then create shimmux
1004 if (!shimExist) {
1005 builder.setInsertionPointAfter(tileOp);
1006 shimOp = analyzer.getShimMux(builder, tileOp.colIndex());
1007 shimExist = 1;
1008 }
1009
1010 Region &r0 = shimOp.getConnections();
1011 Block &b0 = r0.front();
1012 builder.setInsertionPointToStart(&b0);
1013
1014 mtset.setDestBundle(WireBundle::South);
1015 if (mtset.getDestChannel() == 0) {
1016 mtset.setDestChannel(2);
1017 builder.create<ConnectOp>(builder.getUnknownLoc(),
1018 WireBundle::North, 2, WireBundle::DMA, 0);
1019 }
1020 if (mtset.getDestChannel() == 1) {
1021 mtset.setDestChannel(3);
1022 builder.create<ConnectOp>(builder.getUnknownLoc(),
1023 WireBundle::North, 3, WireBundle::DMA, 1);
1024 }
1025 }
1026 }
1027 }
1028 }
1029
1030 RewritePatternSet patterns(&getContext());
1031
1032 if (failed(applyPartialConversion(device, target, std::move(patterns))))
1033 signalPassFailure();
1034}

1036void AIEPathfinderPass::runOnOperation() {
1037
1038 // create analysis pass with routing graph for entire device
1039 LLVM_DEBUG(llvm::dbgs() << "---Begin AIEPathfinderPass---\n");
1040
1041 DeviceOp d = getOperation();
1042 DynamicTileAnalysis &analyzer = getAnalysis<DynamicTileAnalysis>();
1043 if (failed(analyzer.runAnalysis(d)))
1044 return signalPassFailure();
1045 OpBuilder builder = OpBuilder::atBlockTerminator(d.getBody());
1046
1047 if (clRouteCircuit)
1048 runOnFlow(d, analyzer);
1049 if (clRoutePacket)
1050 runOnPacketFlow(d, builder, analyzer);
1051
1052 // Populate wires between switchboxes and tiles.
1053 builder.setInsertionPoint(d.getBody()->getTerminator());
1054 for (int col = 0; col <= analyzer.getMaxCol(); col++) {
1055 for (int row = 0; row <= analyzer.getMaxRow(); row++) {
1056 TileOp tile;
1057 if (analyzer.coordToTile.count({col, row}))
1058 tile = analyzer.coordToTile[{col, row}];
1059 else
1060 continue;
1061 SwitchboxOp sw;
1062 if (analyzer.coordToSwitchbox.count({col, row}))
1063 sw = analyzer.coordToSwitchbox[{col, row}];
1064 else
1065 continue;
1066 if (col > 0) {
1067 // connections east-west between stream switches
1068 if (analyzer.coordToSwitchbox.count({col - 1, row})) {
1069 auto westsw = analyzer.coordToSwitchbox[{col - 1, row}];
1070 builder.create<WireOp>(builder.getUnknownLoc(), westsw,
1071 WireBundle::East, sw, WireBundle::West);
1072 }
1073 }
1074 if (row > 0) {
1075 // connections between abstract 'core' of tile
1076 builder.create<WireOp>(builder.getUnknownLoc(), tile, WireBundle::Core,
1077 sw, WireBundle::Core);
1078 // connections between abstract 'dma' of tile
1079 builder.create<WireOp>(builder.getUnknownLoc(), tile, WireBundle::DMA,
1080 sw, WireBundle::DMA);
1081 // connections north-south inside array ( including connection to shim
1082 // row)
1083 if (analyzer.coordToSwitchbox.count({col, row - 1})) {
1084 auto southsw = analyzer.coordToSwitchbox[{col, row - 1}];
1085 builder.create<WireOp>(builder.getUnknownLoc(), southsw,
1086 WireBundle::North, sw, WireBundle::South);
1087 }
1088 } else if (row == 0) {
1089 if (tile.isShimNOCTile()) {
1090 if (analyzer.coordToShimMux.count({col, 0})) {
1091 auto shimsw = analyzer.coordToShimMux[{col, 0}];
1092 builder.create<WireOp>(
1093 builder.getUnknownLoc(), shimsw,
1094 WireBundle::North, // Changed to connect into the north
1095 sw, WireBundle::South);
1096 // PLIO is attached to shim mux
1097 if (analyzer.coordToPLIO.count(col)) {
1098 auto plio = analyzer.coordToPLIO[col];
1099 builder.create<WireOp>(builder.getUnknownLoc(), plio,
1100 WireBundle::North, shimsw,
1101 WireBundle::South);
1102 }
1103
1104 // abstract 'DMA' connection on tile is attached to shim mux ( in
1105 // row 0 )
1106 builder.create<WireOp>(builder.getUnknownLoc(), tile,
1107 WireBundle::DMA, shimsw, WireBundle::DMA);
1108 }
1109 } else if (tile.isShimPLTile()) {
1110 // PLIO is attached directly to switch
1111 if (analyzer.coordToPLIO.count(col)) {
1112 auto plio = analyzer.coordToPLIO[col];
1113 builder.create<WireOp>(builder.getUnknownLoc(), plio,
1114 WireBundle::North, sw, WireBundle::South);
1115 }
1116 }
1117 }
1118 }
1119 }
1120}
