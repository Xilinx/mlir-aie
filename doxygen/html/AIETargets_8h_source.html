28// One entry in the sidecar location map produced by AIETranslateNpuToBinary
29// (and AIETranslateControlPacketsToUI32Vec). Each entry describes one
30// transaction operation in the .bin: the byte range it occupies, its opcode,
31// the absolute device address it touches (when applicable), the originating
32// aiex.npu.* op's MLIR Location, and an optional regdb register name resolved
33// from (address, tile-module). The vector is written alongside the .bin as a
34// JSON sidecar (see emitNpuLocmapJSON) so consumers can correlate transaction
35// words back to the source op's MLIR Location.
36struct TxnLocEntry {
37 uint32_t byteOffset = 0;
38 uint32_t byteSize = 0;
39 std::string opcodeName; // "WRITE32", "BLOCKWRITE", etc.
40 std::string sourceOpName; // "aiex.npu.write32", etc.
41 std::optional<uint64_t> address; // absolute device address when applicable
42 std::string registerName; // regdb-resolved register name; empty if unknown
43 std::string
44 registerModule; // regdb module: "core", "memory", "memory_tile", "shim"
45 std::optional<mlir::Location> loc;
46};
