31struct AIETraceToConfigPass
32 : xilinx::AIE::impl::AIETraceToConfigBase<AIETraceToConfigPass> {
33 void runOnOperation() override {
34 DeviceOp device = getOperation();
35 OpBuilder builder(device);
36 const auto &targetModel = device.getTargetModel();
37
38 // Collect all trace operations
39 SmallVector<TraceOp> traces;
40 device.walk([&](TraceOp trace) { traces.push_back(trace); });
41
42 for (auto trace : traces) {
43 // Create config symbol name
44 std::string configName = (trace.getSymName().str() + "_config");
45 auto tile = cast<TileOp>(trace.getTile().getDefiningOp());
46 TileID tileID = {tile.getCol(), tile.getRow()};
47
48 // Find packet type (if any)
49 TracePacketType packetType = TracePacketType::Core; // default
50 for (auto &op : trace.getBody().getOps()) {
51 if (auto packetOp = dyn_cast<TracePacketOp>(op)) {
52 packetType = packetOp.getType();
53 break;
54 }
55 }
56
57 // Insert trace.config after trace declaration
58 builder.setInsertionPointAfter(trace);
59 auto configOp = builder.create<TraceConfigOp>(
60 trace.getLoc(), trace.getTile(), builder.getStringAttr(configName),
61 TracePacketTypeAttr::get(builder.getContext(), packetType));
62
63 // Build register writes inside config body
64 Block *configBody = new Block();
65 configOp.getBody().push_back(configBody);
66 OpBuilder configBuilder = OpBuilder::atBlockEnd(configBody);
67
68 bool isMem = (packetType == TracePacketType::Mem);
69
70 // Process combo/edge events FIRST (before other trace config)
71 // This ensures COMBO_EVENT_*/EDGE_DETECTION_EVENT_* are configured
72 // before they can be referenced in trace.event operations
73
74 // 0a. Emit combo event configurations
75 for (auto &op : trace.getBody().getOps()) {
76 if (auto comboOp = dyn_cast<TraceComboEventOp>(op)) {
77 uint32_t slot = comboOp.getSlot();
78
79 // Get input events - use getEventName() helper
80 std::string eventAName = comboOp.getEventA().getEventName();
81 std::string eventBName = comboOp.getEventB().getEventName();
82 ComboLogic logic = comboOp.getLogic();
83
84 // If enum, use enum value directly; otherwise lookup by name
85 std::optional<uint32_t> eventANum, eventBNum;
86
87 if (auto enumValA = comboOp.getEventA().getEnumValue()) {
88 eventANum = static_cast<uint32_t>(*enumValA);
89 } else {
90 eventANum = targetModel.lookupEvent(eventAName, tileID, isMem);
91 }
92
93 if (auto enumValB = comboOp.getEventB().getEnumValue()) {
94 eventBNum = static_cast<uint32_t>(*enumValB);
95 } else {
96 eventBNum = targetModel.lookupEvent(eventBName, tileID, isMem);
97 }
98
99 if (!eventANum) {
100 comboOp.emitError("unknown trace event '") << eventAName << "'";
101 return signalPassFailure();
102 }
103 if (!eventBNum) {
104 comboOp.emitError("unknown trace event '") << eventBName << "'";
105 return signalPassFailure();
106 }
107
108 // Map slot to input event fields
109 StringRef eventAField, eventBField, controlField;
110 if (slot == 0) {
111 eventAField = "eventA";
112 eventBField = "eventB";
113 controlField = "combo0";
114 } else if (slot == 1) {
115 eventAField = "eventC";
116 eventBField = "eventD";
117 controlField = "combo1";
118 } else if (slot == 2) {
119 // Combo2 is hierarchical - reuses eventA/B fields but represents
120 // combo0/combo1
121 eventAField = "eventA";
122 eventBField = "eventB";
123 controlField = "combo2";
124 }
125
126 // Emit Combo_event_inputs register fields
127 configBuilder.create<TraceRegOp>(
128 comboOp.getLoc(), builder.getStringAttr("Combo_event_inputs"),
129 builder.getStringAttr(eventAField), comboOp.getEventA(),
130 /*mask=*/nullptr,
131 builder.getStringAttr("combo" + std::to_string(slot) +
132 " eventA"));
133
134 configBuilder.create<TraceRegOp>(
135 comboOp.getLoc(), builder.getStringAttr("Combo_event_inputs"),
136 builder.getStringAttr(eventBField), comboOp.getEventB(),
137 /*mask=*/nullptr,
138 builder.getStringAttr("combo" + std::to_string(slot) +
139 " eventB"));
140
141 // Emit Combo_event_control register field
142 configBuilder.create<TraceRegOp>(
143 comboOp.getLoc(), builder.getStringAttr("Combo_event_control"),
144 builder.getStringAttr(controlField),
145 builder.getI32IntegerAttr(static_cast<uint32_t>(logic)),
146 /*mask=*/nullptr,
147 builder.getStringAttr("combo" + std::to_string(slot) + " logic"));
148 }
149 }
150
151 // 0b. Emit edge detection configurations
152 for (auto &op : trace.getBody().getOps()) {
153 if (auto edgeOp = dyn_cast<TraceEdgeEventOp>(op)) {
154 uint32_t slot = edgeOp.getSlot();
155 std::string eventName = edgeOp.getEvent().getEventName();
156 EdgeTrigger trigger = edgeOp.getTrigger();
157
158 // If enum, use enum value directly; otherwise lookup by name
159 std::optional<uint32_t> eventNum;
160
161 if (auto enumVal = edgeOp.getEvent().getEnumValue()) {
162 eventNum = static_cast<uint32_t>(*enumVal);
163 } else {
164 eventNum = targetModel.lookupEvent(eventName, tileID, isMem);
165 }
166
167 if (!eventNum) {
168 edgeOp.emitError("unknown trace event '") << eventName << "'";
169 return signalPassFailure();
170 }
171
172 // Map slot to field names
173 StringRef eventField =
174 (slot == 0) ? "Edge_Detection_Event_0" : "Edge_Detection_Event_1";
175 StringRef risingField = (slot == 0)
176 ? "Edge_Detection_0_Trigger_Rising"
177 : "Edge_Detection_1_Trigger_Rising";
178 StringRef fallingField = (slot == 0)
179 ? "Edge_Detection_0_Trigger_Falling"
180 : "Edge_Detection_1_Trigger_Falling";
181
182 // Source event
183 configBuilder.create<TraceRegOp>(
184 edgeOp.getLoc(),
185 builder.getStringAttr("Edge_Detection_event_control"),
186 builder.getStringAttr(eventField), edgeOp.getEvent(),
187 /*mask=*/nullptr,
188 builder.getStringAttr("edge" + std::to_string(slot) + " source"));
189
190 // Trigger mode
191 bool rising =
192 (trigger == EdgeTrigger::RISING || trigger == EdgeTrigger::BOTH);
193 bool falling =
194 (trigger == EdgeTrigger::FALLING || trigger == EdgeTrigger::BOTH);
195
196 configBuilder.create<TraceRegOp>(
197 edgeOp.getLoc(),
198 builder.getStringAttr("Edge_Detection_event_control"),
199 builder.getStringAttr(risingField),
200 builder.getI32IntegerAttr(rising ? 1 : 0),
201 /*mask=*/nullptr,
202 builder.getStringAttr("edge" + std::to_string(slot) + " rising"));
203
204 configBuilder.create<TraceRegOp>(
205 edgeOp.getLoc(),
206 builder.getStringAttr("Edge_Detection_event_control"),
207 builder.getStringAttr(fallingField),
208 builder.getI32IntegerAttr(falling ? 1 : 0),
209 /*mask=*/nullptr,
210 builder.getStringAttr("edge" + std::to_string(slot) +
211 " falling"));
212 }
213 }
214
215 // 1. Emit Trace_Control0 fields
216 // Check for start/stop events
217 for (auto &op : trace.getBody().getOps()) {
218 if (auto startOp = dyn_cast<TraceStartEventOp>(op)) {
219 uint32_t startEvent = 0;
220 if (startOp.getBroadcast()) {
221 startEvent = *startOp.getBroadcast();
222 } else if (auto eventAttr = startOp.getEvent()) {
223 // Use getEventName() helper and check for enum
224 std::string eventName = eventAttr->getEventName();
225
226 std::optional<uint32_t> eventNum;
227 if (auto enumVal = eventAttr->getEnumValue()) {
228 eventNum = static_cast<uint32_t>(*enumVal);
229 } else {
230 eventNum = targetModel.lookupEvent(eventName, tileID, isMem);
231 }
232
233 if (eventNum) {
234 startEvent = *eventNum;
235 } else {
236 startOp.emitError("unknown trace event '") << eventName << "'";
237 return signalPassFailure();
238 }
239 }
240
241 configBuilder.create<TraceRegOp>(
242 trace.getLoc(), builder.getStringAttr("Trace_Control0"),
243 builder.getStringAttr("Trace_Start_Event"),
244 builder.getI32IntegerAttr(startEvent),
245 /*mask=*/nullptr, builder.getStringAttr("start event"));
246 }
247
248 if (auto stopOp = dyn_cast<TraceStopEventOp>(op)) {
249 uint32_t stopEvent = 0;
250 if (stopOp.getBroadcast()) {
251 stopEvent = *stopOp.getBroadcast();
252 } else if (auto eventAttr = stopOp.getEvent()) {
253 // Use getEventName() helper and check for enum
254 std::string eventName = eventAttr->getEventName();
255
256 std::optional<uint32_t> eventNum;
257 if (auto enumVal = eventAttr->getEnumValue()) {
258 eventNum = static_cast<uint32_t>(*enumVal);
259 } else {
260 eventNum = targetModel.lookupEvent(eventName, tileID, isMem);
261 }
262
263 if (eventNum) {
264 stopEvent = *eventNum;
265 } else {
266 stopOp.emitError("unknown trace event '") << eventName << "'";
267 return signalPassFailure();
268 }
269 }
270
271 configBuilder.create<TraceRegOp>(
272 trace.getLoc(), builder.getStringAttr("Trace_Control0"),
273 builder.getStringAttr("Trace_Stop_Event"),
274 builder.getI32IntegerAttr(stopEvent),
275 /*mask=*/nullptr, builder.getStringAttr("stop event"));
276 }
277
278 // Emit mode if present.
279 // Memory trace does not expose Trace_Control0.Mode in the register DB.
280 if (auto modeOp = dyn_cast<TraceModeOp>(op); modeOp && !isMem) {
281 configBuilder.create<TraceRegOp>(
282 trace.getLoc(), builder.getStringAttr("Trace_Control0"),
283 builder.getStringAttr("Mode"),
284 builder.getI32IntegerAttr(
285 static_cast<uint32_t>(modeOp.getMode())),
286 /*mask=*/nullptr, builder.getStringAttr("trace mode"));
287 }
288
289 // Emit packet config if present
290 if (auto packetOp = dyn_cast<TracePacketOp>(op)) {
291 configBuilder.create<TraceRegOp>(
292 trace.getLoc(), builder.getStringAttr("Trace_Control1"),
293 builder.getStringAttr("ID"),
294 builder.getI32IntegerAttr(packetOp.getId()),
295 /*mask=*/nullptr, builder.getStringAttr("packet ID"));
296
297 configBuilder.create<TraceRegOp>(
298 trace.getLoc(), builder.getStringAttr("Trace_Control1"),
299 builder.getStringAttr("Packet_Type"),
300 builder.getI32IntegerAttr(
301 static_cast<uint32_t>(packetOp.getType())),
302 /*mask=*/nullptr, builder.getStringAttr("packet type"));
303 }
304 }
305
306 // 2. Emit port configurations (Stream_Switch_Event_Port_Selection_0/1)
307 for (auto &op : trace.getBody().getOps()) {
308 if (auto portOp = dyn_cast<TracePortOp>(op)) {
309 uint32_t slot = portOp.getSlot();
310
311 // Determine which register based on slot
312 StringRef registerName = (slot < 4)
313 ? "Stream_Switch_Event_Port_Selection_0"
314 : "Stream_Switch_Event_Port_Selection_1";
315
316 // Generate field names
317 std::string idFieldName = "Port_" + std::to_string(slot) + "_ID";
318 std::string masterSlaveFieldName =
319 "Port_" + std::to_string(slot) + "_Master_Slave";
320
321 // Generate port value string "PORT:CHANNEL"
322 std::string portValue = stringifyWireBundle(portOp.getPort()).str() +
323 ":" + std::to_string(portOp.getChannel());
324
325 // Convert DMAChannelDir to master flag: S2MM=master(1), MM2S=slave(0)
326 int masterSlaveValue =
327 (portOp.getDirection() == DMAChannelDir::S2MM) ? 1 : 0;
328
329 // Emit Port_N_ID field
330 configBuilder.create<TraceRegOp>(
331 portOp.getLoc(), builder.getStringAttr(registerName),
332 builder.getStringAttr(idFieldName),
333 builder.getStringAttr(portValue), // "NORTH:1" format
334 /*mask=*/nullptr,
335 builder.getStringAttr("port " + std::to_string(slot) + " ID"));
336
337 // Emit Port_N_Master_Slave field
338 configBuilder.create<TraceRegOp>(
339 portOp.getLoc(), builder.getStringAttr(registerName),
340 builder.getStringAttr(masterSlaveFieldName),
341 builder.getI32IntegerAttr(masterSlaveValue),
342 /*mask=*/nullptr,
343 builder.getStringAttr("port " + std::to_string(slot) +
344 " master/slave"));
345 }
346 }
347
348 // 3. Emit event slots (Trace_Event0 / Trace_Event1)
349 SmallVector<TraceEventOp> events;
350 for (auto &op : trace.getBody().getOps()) {
351 if (auto eventOp = dyn_cast<TraceEventOp>(op)) {
352 events.push_back(eventOp);
353 }
354 }
355
356 for (size_t i = 0; i < events.size() && i < 8; ++i) {
357 std::string eventName = events[i].getEvent().getEventName();
358
359 // If enum, use enum value directly; otherwise lookup by name
360 std::optional<uint32_t> eventNum;
361
362 if (auto enumVal = events[i].getEvent().getEnumValue()) {
363 eventNum = static_cast<uint32_t>(*enumVal);
364 } else {
365 eventNum = targetModel.lookupEvent(eventName, tileID, isMem);
366 }
367
368 if (!eventNum) {
369 events[i].emitError("unknown trace event '") << eventName << "'";
370 return signalPassFailure();
371 }
372
373 // Determine which register and field
374 StringRef registerName = (i < 4) ? "Trace_Event0" : "Trace_Event1";
375 std::string fieldName = "Trace_Event" + std::to_string(i);
376
377 // Emit register write with event number as integer
378 configBuilder.create<TraceRegOp>(
379 trace.getLoc(), builder.getStringAttr(registerName),
380 builder.getStringAttr(fieldName),
381 events[i].getEvent(), // builder.getI32IntegerAttr(*eventNum),
382 /*mask=*/nullptr, builder.getStringAttr(eventName));
383 }
384
385 // Add terminator
386 configBuilder.create<EndOp>(trace.getLoc());
387
388 // Update all trace.start_config references
389 device.walk([&](TraceStartConfigOp startConfig) {
390 if (startConfig.getTraceConfig() == trace.getSymName()) {
391 startConfig.setTraceConfigAttr(
392 SymbolRefAttr::get(builder.getContext(), configName));
393 }
394 });
395
396 // Remove original trace op
397 trace.erase();
398 }
399 }
400};

409//===----------------------------------------------------------------------===//
410// AIETraceRegPackWritesPass - Pack multiple register field writes
411//===----------------------------------------------------------------------===//
412
413namespace {
414
415struct AIETraceRegPackWritesPass
416 : xilinx::AIE::impl::AIETraceRegPackWritesBase<AIETraceRegPackWritesPass> {
417 void runOnOperation() override {
418 DeviceOp device = getOperation();
419 const auto &targetModel = device.getTargetModel();
420
421 // Process each trace config
422 device.walk([&](TraceConfigOp configOp) {
423 // Determine module based on tile type and packet type
424 auto tile = cast<TileOp>(configOp.getTile().getDefiningOp());
425
426 // Get packet type to determine if this is memory or core trace
427 bool isMem = false;
428 if (auto packetType = configOp.getPacketType()) {
429 isMem = (*packetType == TracePacketType::Mem);
430 }
431
432 // Phase 1: Convert field+value to mask+shifted_value
433 SmallVector<TraceRegOp> regsToConvert;
434 for (auto &op : configOp.getBody().front()) {
435 if (auto regOp = dyn_cast<TraceRegOp>(op)) {
436 if (regOp.getField() && !regOp.getMask()) {
437 regsToConvert.push_back(regOp);
438 }
439 }
440 }
441
442 OpBuilder builder(&configOp.getBody().front(),
443 configOp.getBody().front().begin());
444
445 for (auto regOp : regsToConvert) {
446 // Look up register and field information
447 TileID tileID = {tile.getCol(), tile.getRow()};
448 const RegisterInfo *regInfo =
449 targetModel.lookupRegister(regOp.getRegName(), tileID, isMem);
450
451 if (!regInfo) {
452 regOp.emitError("Register not found in database: ")
453 << regOp.getRegName();
454 return signalPassFailure();
455 }
456
457 const BitFieldInfo *fieldInfo = regInfo->getField(*regOp.getField());
458 if (!fieldInfo) {
459 regOp.emitError("Field not found in register: ")
460 << *regOp.getField() << " in " << regOp.getRegName();
461 return signalPassFailure();
462 }
463
464 // Get the value - handle both integers and port strings
465 uint32_t value = 0;
466 Attribute valAttr = regOp.getValue();
467 if (auto traceEventAttr = dyn_cast<TraceEventAttr>(valAttr)) {
468 if (auto enumVal = traceEventAttr.getEnumValue()) {
469 value = static_cast<uint32_t>(*enumVal);
470 valAttr = builder.getI32IntegerAttr(value);
471 } else {
472 std::string eventName = traceEventAttr.getEventName();
473 std::optional<uint32_t> eventNum =
474 targetModel.lookupEvent(eventName, tileID, isMem);
475 if (!eventNum) {
476 regOp.emitError("unknown trace event '") << eventName << "'";
477 return signalPassFailure();
478 }
479 value = *eventNum;
480 valAttr = builder.getI32IntegerAttr(value);
481 }
482 }
483 if (auto intAttr = dyn_cast<IntegerAttr>(valAttr)) {
484 // Integer value
485 value = intAttr.getInt();
486 } else if (auto strAttr = dyn_cast<StringAttr>(valAttr)) {
487 // String value - check if it's a port specification
488 StringRef valueStr = strAttr.getValue();
489
490 // Determine master/slave from field name
491 // If field name contains "Master_Slave", this is not a port ID field
492 // Port ID fields are named "Port_N_ID"
493 bool isMasterSlaveField =
494 fieldInfo->name.find("Master_Slave") != std::string::npos;
495
496 if (!isMasterSlaveField && valueStr.contains(':')) {
497 // This looks like "PORT:CHANNEL" format
498 // We need master/slave info - look for corresponding Master_Slave
499 // field
500 bool master = false; // Default to slave
501
502 // Derive the port slot prefix from the current field name, e.g.:
503 // "Port_1_ID" -> "Port_1"
504 StringRef fieldName(fieldInfo->name);
505 StringRef portSlotPrefix;
506 if (fieldName.starts_with("Port_")) {
507 size_t idSuffixPos = fieldName.find("_ID");
508 if (idSuffixPos != std::string::npos)
509 portSlotPrefix = fieldName.take_front(idSuffixPos);
510 }
511 // Search for companion Master_Slave field in same register and,
512 // when possible, for the same port slot.
513 for (auto &siblingOp : configOp.getBody().front()) {
514 if (auto siblingReg = dyn_cast<TraceRegOp>(siblingOp)) {
515 if (siblingReg.getRegName() != regOp.getRegName() ||
516 !siblingReg.getField())
517 continue;
518 StringRef siblingFieldName = *siblingReg.getField();
519 // If we could determine a slot prefix (e.g. "Port_1"), require
520 // the sibling to match that slot and contain "Master_Slave".
521 if (!portSlotPrefix.empty()) {
522 if (!siblingFieldName.starts_with(portSlotPrefix) ||
523 !siblingFieldName.contains("Master_Slave"))
524 continue;
525 } else {
526 // Fallback: match any Master_Slave field in this register.
527 if (!siblingFieldName.contains("Master_Slave"))
528 continue;
529 }
530 // Found companion field - extract master flag
531 if (auto siblingInt =
532 dyn_cast<IntegerAttr>(siblingReg.getValue())) {
533 master = (siblingInt.getInt() != 0);
534 }
535 break;
536 }
537 }
538
539 // Resolve port value
540 auto portIndex =
541 targetModel.resolvePortValue(valueStr, tileID, master);
542 if (!portIndex) {
543 regOp.emitError("Failed to resolve port value: ") << valueStr;
544 return signalPassFailure();
545 }
546 value = *portIndex;
547 } else {
548 regOp.emitError("Unsupported string value: ") << valueStr;
549 return signalPassFailure();
550 }
551 } else {
552 regOp.emitError("Unsupported value type in pack pass");
553 return signalPassFailure();
554 }
555
556 // Compute mask and shifted value
557 auto mask = targetModel.getFieldMask(*fieldInfo);
558 if (!mask) {
559 regOp.emitError("Invalid field bit range for register write: ")
560 << regOp.getRegName() << "." << fieldInfo->name << " ["
561 << fieldInfo->bit_start << ":" << fieldInfo->bit_end
562 << "], width=" << fieldInfo->getWidth();
563 return signalPassFailure();
564 }
565 uint32_t shiftedValue = targetModel.encodeFieldValue(*fieldInfo, value);
566 // Create new operation with mask
567 builder.setInsertionPoint(regOp);
568 builder.create<TraceRegOp>(regOp.getLoc(), regOp.getRegNameAttr(),
569 nullptr, // no field
570 builder.getI32IntegerAttr(shiftedValue),
571 builder.getI32IntegerAttr(*mask),
572 regOp.getCommentAttr());
573
574 // Remove old operation
575 regOp.erase();
576 }
577
578 // Phase 2: Merge writes to same register with non-overlapping masks
579 bool changed = true;
580 while (changed) {
581 changed = false;
582
583 // Collect all register writes
584 SmallVector<TraceRegOp> regWrites;
585 for (TraceRegOp op : configOp.getBody().front().getOps<TraceRegOp>()) {
586 if (op.getMask()) {
587 regWrites.push_back(op);
588 }
589 }
590
591 // Try to merge pairs
592 for (size_t i = 0; i < regWrites.size() && !changed; ++i) {
593 for (size_t j = i + 1; j < regWrites.size() && !changed; ++j) {
594 TraceRegOp reg1 = regWrites[i];
595 TraceRegOp reg2 = regWrites[j];
596
597 // Must be same register
598 if (reg1.getRegName() != reg2.getRegName())
599 continue;
600
601 auto mask1Attr = reg1.getMask();
602 auto mask2Attr = reg2.getMask();
603 if (!mask1Attr || !mask2Attr)
604 continue;
605
606 uint32_t mask1 = *mask1Attr;
607 uint32_t mask2 = *mask2Attr;
608
609 // Check for overlap
610 if (mask1 & mask2) {
611 reg1.emitError("Overlapping masks for register ")
612 << reg1.getRegName() << ": mask1=" << mask1
613 << " mask2=" << mask2;
614 return signalPassFailure();
615 }
616
617 // Merge the two writes
618 auto value1Attr = dyn_cast<IntegerAttr>(reg1.getValue());
619 auto value2Attr = dyn_cast<IntegerAttr>(reg2.getValue());
620 if (!value1Attr || !value2Attr) {
621 reg1.emitError(
622 "Expected integer values for packed register writes");
623 return signalPassFailure();
624 }
625 uint32_t value1 = value1Attr.getInt();
626 uint32_t value2 = value2Attr.getInt();
627 uint32_t mergedValue = value1 | value2;
628 uint32_t mergedMask = mask1 | mask2;
629
630 // Create merged operation
631 builder.setInsertionPoint(reg1);
632 std::string comment;
633 if (reg1.getComment())
634 comment += reg1.getComment()->str();
635 if (reg2.getComment()) {
636 if (!comment.empty())
637 comment += " + ";
638 comment += reg2.getComment()->str();
639 }
640
641 builder.create<TraceRegOp>(
642 reg1.getLoc(), reg1.getRegNameAttr(), nullptr,
643 builder.getI32IntegerAttr(mergedValue),
644 builder.getI32IntegerAttr(mergedMask),
645 comment.empty() ? nullptr : builder.getStringAttr(comment));
646
647 // Remove both old operations
648 reg1.erase();
649 reg2.erase();
650 changed = true;
651 }
652 }
653 }
654 });
655 }
656};
