24struct AIECanonicalizeDevicePass
25 : AIECanonicalizeDeviceBase<AIECanonicalizeDevicePass> {
26 void runOnOperation() override {
27
28 ModuleOp moduleOp = getOperation();
29
30 for (auto deviceOp : moduleOp.getOps<AIETarget>()) {
31 (void)deviceOp;
32 // We have a device.. Nothing to do.
33 return;
34 }
35
36 // Builder with no insertion point, because we don't want to insert
37 // the new op quite yet.
38 OpBuilder builder(moduleOp->getContext());
39
40 Location location = builder.getUnknownLoc();
41 auto deviceOp = builder.create<DeviceOp>(
42 location, AIEDeviceAttr::get(builder.getContext(), AIEDevice::xcvc1902),
43 StringAttr::get(builder.getContext(),
44 DeviceOp::getDefaultDeviceName()));
45
46 deviceOp.getRegion().takeBody(moduleOp.getBodyRegion());
47 new (&moduleOp->getRegion(0)) Region(moduleOp);
48 moduleOp->getRegion(0).emplaceBlock();
49
50 DeviceOp::ensureTerminator(deviceOp.getBodyRegion(), builder, location);
51 OpBuilder builder2 = OpBuilder::atBlockBegin(moduleOp.getBody());
52 builder2.insert(deviceOp);
53 }
54};
