30struct DMAConfigureTaskForOpPattern
31 : public mlir::OpRewritePattern<DMAConfigureTaskForOp> {
32
33 using OpRewritePattern::OpRewritePattern;
34
35 LogicalResult matchAndRewrite(DMAConfigureTaskForOp op,
36 PatternRewriter &rewriter) const override {
37 AIE::DeviceOp device = op->getParentOfType<AIE::DeviceOp>();
38
39 AIE::ShimDMAAllocationOp alloc_op = AIE::ShimDMAAllocationOp::getForSymbol(
40 device, op.getAlloc().getRootReference());
41 if (!alloc_op) {
42 return op.emitOpError("no shim DMA allocation found for symbol");
43 }
44
45 const int col = alloc_op.getCol();
46 AIE::TileOp tile = AIE::TileOp::getOrCreate(rewriter, device, col, 0);
47 DMAConfigureTaskOp new_op = rewriter.create<DMAConfigureTaskOp>(
48 op.getLoc(), rewriter.getIndexType(), tile.getResult(),
49 alloc_op.getChannelDir(), (int32_t)alloc_op.getChannelIndex(),
50 op.getIssueToken(), op.getRepeatCount(),
51 alloc_op.getPacket().value_or(nullptr));
52 rewriter.replaceAllUsesWith(op.getResult(), new_op.getResult());
53 rewriter.inlineRegionBefore(op.getBody(), new_op.getBody(),
54 new_op.getBody().begin());
55 rewriter.eraseOp(op);
56 return success();
57 }
58};

60struct AIESubstituteShimDMAAllocationsPass
61 : AIESubstituteShimDMAAllocationsBase<AIESubstituteShimDMAAllocationsPass> {
62
63 void runOnOperation() override {
64 AIE::DeviceOp device = getOperation();
65
66 // Convert DMAConfigureTaskForOps that reference shim DMA allocations
67 // to regular DMAConfigureTaskOps
68 RewritePatternSet patterns(&getContext());
69 patterns.insert<DMAConfigureTaskForOpPattern>(&getContext());
70
71 (void)applyPatternsGreedily(device, std::move(patterns));
72 }
73};
