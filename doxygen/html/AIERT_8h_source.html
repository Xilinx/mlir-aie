29// RAII helper. Constructs while AIERTControl is recording an aie-rt
30// transaction, captures the current XAie_TxnInst command count at entry, and
31// at scope end attributes every aie-rt command produced during the scope to
32// `loc`. The captured ranges are stored on AIERTControl::txnInstrLocs and read
33// out by the AIEToConfiguration round-trip so re-emitted aiex.npu.* ops carry
34// the source op's MLIR Location instead of the device's fallback location.
35class TxnLocBracket {
36public:
37 TxnLocBracket(AIERTControl &ctl, mlir::Location loc);
38 ~TxnLocBracket();
39 TxnLocBracket(const TxnLocBracket &) = delete;
40 TxnLocBracket &operator=(const TxnLocBracket &) = delete;
41
42private:
43 AIERTControl &ctl;
44 mlir::Location loc;
45 uint32_t startCmds;
46};
