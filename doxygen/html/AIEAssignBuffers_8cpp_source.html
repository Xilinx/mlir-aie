25bool checkAndPrintOverflow(TileOp tile, int address, int maxDataMemorySize,
26 int stacksize, SmallVector<BufferOp> &buffers) {
27 if (address > maxDataMemorySize) {
28 InFlightDiagnostic error =
29 tile.emitOpError("allocated buffers exceeded available memory\n");
30 auto &note = error.attachNote() << "MemoryMap:\n";
31 auto printbuffer = [&](StringRef name, int address, int size) {
32 note << "\t" << name << " \t"
33 << ": 0x" << llvm::utohexstr(address) << "-0x"
34 << llvm::utohexstr(address + size - 1) << " \t(" << size
35 << " bytes)\n";
36 };
37 if (stacksize > 0)
38 printbuffer("(stack)", 0, stacksize);
39 else
40 error << "(no stack allocated)\n";
41
42 for (auto buffer : buffers) {
43 assert(buffer.getAddress().has_value() &&
44 "buffer must have address assigned");
45 printbuffer(buffer.name(), buffer.getAddress().value(),
46 buffer.getAllocationSize());
47 }
48 return false;
49 }
50 return true;
51}

53bool basicAllocation(TileOp tile) {
54 auto device = tile->getParentOfType<AIE::DeviceOp>();
55 if (!device)
56 return false;
57
58 const auto &targetModel = getTargetModel(tile);
59 int maxDataMemorySize = 0;
60 if (tile.isMemTile())
61 maxDataMemorySize = targetModel.getMemTileSize();
62 else
63 maxDataMemorySize = targetModel.getLocalMemorySize();
64
65 SmallVector<BufferOp> buffers;
66 SmallVector<BufferOp> allocated_buffers;
67 // Collect all the buffers for this tile. If the buffer has an address, add
68 // it to allocated_buffers. Otherwise, add it to buffers.
69 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
70 if (buffer.getTileOp() == tile) {
71 if (buffer.getAddress())
72 allocated_buffers.push_back(buffer);
73 else
74 buffers.push_back(buffer);
75 }
76 });
77
78 // Sort buffers by allocation size.
79 std::sort(buffers.begin(), buffers.end(), [](BufferOp a, BufferOp b) {
80 return a.getAllocationSize() > b.getAllocationSize();
81 });
82
83 // Sort allocated_buffers by address
84 std::sort(allocated_buffers.begin(), allocated_buffers.end(),
85 [](BufferOp a, BufferOp b) {
86 return a.getAddress().value() < b.getAddress().value();
87 });
88
89 // Address range owned by the MemTile is 0x80000.
90 // Address range owned by the tile is 0x8000 in
91 // AIE1 and 0x10000 in AIE2, but we need room at
92 // the bottom for stack.
93 int stacksize = 0;
94 int address = 0;
95 if (auto core = tile.getCoreOp()) {
96 stacksize = core.getStackSize();
97 address += stacksize;
98 }
99
100 // As the next address to allocate is assigned, skip over any buffers
101 // from the allocated_buffers list.
102 auto current_alloc = allocated_buffers.begin();
103 for (auto buffer : buffers) {
104 assert(!buffer.getAddress());
105 while (current_alloc != allocated_buffers.end() &&
106 address + buffer.getAllocationSize() >
107 current_alloc->getAddress().value()) {
108 address = current_alloc->getAddress().value() +
109 current_alloc->getAllocationSize();
110 current_alloc++;
111 }
112 buffer.setAddress(address);
113 address += buffer.getAllocationSize();
114 }
115
116 // Sort by smallest address before printing memory map.
117 std::sort(buffers.begin(), buffers.end(), [](BufferOp a, BufferOp b) {
118 assert(a.getAddress().has_value() && "buffer must have address assigned");
119 assert(b.getAddress().has_value() && "buffer must have address assigned");
120 return a.getAddress().value() < b.getAddress().value();
121 });
122 // Check if memory was exceeded on any bank and print debug info.
123 return checkAndPrintOverflow(tile, address, maxDataMemorySize, stacksize,
124 buffers);
125}

130typedef struct BankLimits {
131 int64_t startAddr;
132 int64_t endAddr;
133} BankLimits;

135// Function that given a number of banks and their size, computes
136// the start and end addresses for each bank and fills in the entry
137// in the bankLimits vector.
138void fillBankLimits(int numBanks, int bankSize,
139 std::vector<BankLimits> &bankLimits) {
140 for (int i = 0; i < numBanks; i++) {
141 auto startAddr = bankSize * i;
142 auto endAddr = bankSize * (i + 1);
143 bankLimits.push_back({startAddr, endAddr});
144 }
145}

147// Function that sets the address attribute of the given buffer to
148// the given start_addr. It also updates the entry in the
149// nextAddrInBanks for the corresponding bank.
150void setAndUpdateAddressInBank(BufferOp buffer, int64_t start_addr,
151 int64_t end_addr,
152 std::vector<int64_t> &nextAddrInBanks) {
153 // Fixme: alignment
154 buffer.setAddress(start_addr);
155 nextAddrInBanks[buffer.getMemBank().value()] = end_addr;
156}

158// Function that checks whether the given buffer already has a set address
159// attribute. If it does, it finds in which bank the buffer is and checks
160// whether there is enough space left for it. If there is the function
161// returns true and if not, the function emits a warning that the address
162// will be overwritten and returns false (which will cause the buffer to be
163// added to the list of buffers without addresses, to be completed later on).
164FailureOr<bool>
165checkAndAddBufferWithAddress(BufferOp buffer, int numBanks,
166 std::vector<int64_t> &nextAddrInBanks,
167 std::vector<BankLimits> &bankLimits) {
168 auto addrAttr = buffer->getAttrOfType<IntegerAttr>("address");
169 if (!addrAttr)
170 return false;
171
172 int addr = addrAttr.getInt();
173 for (int i = 0; i < numBanks; i++) {
174 // if the address is not within the bank, continue
175 if (addr < bankLimits[i].startAddr || addr >= bankLimits[i].endAddr)
176 continue;
177
178 // if the allocator already allocated this address, fail
179 if (addr < nextAddrInBanks[i])
180 return buffer->emitOpError("would override allocated address");
181
182 // the allocator can accomadate this existing allocation
183 nextAddrInBanks[i] = addr + buffer.getAllocationSize();
184 buffer.setMemBank(i);
185 }
186 return true;
187}
188
189// Function that checks whether the given buffer already has a set mem_bank
190// attribute. If it does, it checks whether there is enough space left for
191// it. If there is, it sets the buffer's address field and if not, the
192// function emits a warning that the mem_bank will be overwritten and returns
193// false (which will cause the buffer to be added to the list of buffers
194// without addresses, to be completed later on).
195FailureOr<bool>
196checkAndAddBufferWithMemBank(BufferOp buffer, int numBanks,
197 std::vector<int64_t> &nextAddrInBanks,
198 std::vector<BankLimits> &bankLimits) {
199 auto memBankAttr = buffer->getAttrOfType<IntegerAttr>("mem_bank");
200 if (!memBankAttr)
201 return false;
202
203 int mem_bank = memBankAttr.getInt();
204 int64_t startAddr = nextAddrInBanks[mem_bank];
205 int64_t endAddr = startAddr + buffer.getAllocationSize();
206 if (endAddr > bankLimits[mem_bank].endAddr)
207 return buffer->emitOpError("would override existing mem_bank");
208 setAndUpdateAddressInBank(buffer, startAddr, endAddr, nextAddrInBanks);
209 return true;
210}

212// Prints the memory map across banks
213void printMemMap(TileOp tile, SmallVector<BufferOp> &allocatedBuffers,
214 SmallVector<BufferOp> &preAllocatedBuffers, int numBanks,
215 std::vector<BankLimits> &bankLimits, int stacksize) {
216 InFlightDiagnostic error = tile.emitWarning(
217 "Not all requested buffers fit in the available memory.\n");
218 auto &note = error.attachNote()
219 << "Current configuration of buffers in bank(s) : ";
220 note << "MemoryMap:\n";
221 auto printbuffer = [&](StringRef name, int address, int size) {
222 note << "\t"
223 << "\t" << name << " \t"
224 << ": 0x" << llvm::utohexstr(address) << "-0x"
225 << llvm::utohexstr(address + size - 1) << " \t(" << size
226 << " bytes)\n";
227 };
228 for (int i = 0; i < numBanks; i++) {
229 if (i == 0) {
230 if (stacksize > 0)
231 printbuffer("(stack)", 0, stacksize);
232 else
233 note << "(no stack allocated)\n";
234 }
235 note << "\t"
236 << "bank : " << i << "\t"
237 << "0x" << llvm::utohexstr(bankLimits[i].startAddr) << "-0x"
238 << llvm::utohexstr(bankLimits[i].endAddr - 1) << "\n";
239 for (auto buffer : preAllocatedBuffers) {
240 auto addr = buffer.getAddress().value();
241 auto mem_bank = buffer.getMemBank().value();
242 if (mem_bank == i)
243 printbuffer(buffer.name(), addr, buffer.getAllocationSize());
244 }
245 for (auto buffer : allocatedBuffers) {
246 auto addr = buffer.getAddress().value();
247 auto mem_bank = buffer.getMemBank().value();
248 if (mem_bank == i)
249 printbuffer(buffer.name(), addr, buffer.getAllocationSize());
250 }
251 }
252}

254// Function that given a buffer will iterate over all the memory banks
255// starting from the given index to try and find a bank with enough
256// space. If it does, it will set the buffer's address and mem_bank
257// attributes and update the nextAddrInBanks vector.
258// If it does not find one with enough space, it will throw an error.
259// Returns true if the buffer was successfully allocated, false otherwise.
260// If no bank has enough space to accommodate the buffer, an error is emitted.
261
262int setBufferAddress(BufferOp buffer, int numBanks, int &startBankIndex,
263 std::vector<int64_t> &nextAddrInBanks,
264 std::vector<BankLimits> &bankLimits) {
265 assert(startBankIndex < numBanks &&
266 "Unexpected input value for startBankIndex");
267 int bankIndex = startBankIndex;
268 bool allocated = false;
269 for (int i = 0; i < numBanks; i++) {
270 int64_t startAddr = nextAddrInBanks[bankIndex];
271 int64_t endAddr = startAddr + buffer.getAllocationSize();
272 if (endAddr <= bankLimits[bankIndex].endAddr) {
273 buffer.setMemBank(bankIndex);
274 setAndUpdateAddressInBank(buffer, startAddr, endAddr, nextAddrInBanks);
275 allocated = true;
276 bankIndex = (bankIndex + 1) % numBanks;
277 startBankIndex = bankIndex;
278 break;
279 }
280 // Move to the next bank
281 bankIndex = (bankIndex + 1) % numBanks;
282 }
283 // If no bank has enough space, throws error
284 if (!allocated) {
285 buffer.emitWarning("Failed to allocate buffer: ")
286 << buffer.name() << " with size: " << buffer.getAllocationSize()
287 << " bytes.";
288 return false;
289 }
290 return true;
291}

293bool checkAndPrintOverflow(TileOp tile, int numBanks, int stacksize,
294 SmallVector<BufferOp> &allBuffers,
295 std::vector<int64_t> &nextAddrInBanks,
296 std::vector<BankLimits> &bankLimits) {
297 bool foundOverflow = false;
298 std::vector<int> overflow_banks;
299 for (int i = 0; i < numBanks; i++) {
300 if (nextAddrInBanks[i] > bankLimits[i].endAddr) {
301 foundOverflow = true;
302 overflow_banks.push_back(i);
303 }
304 }
305 if (foundOverflow) {
306 InFlightDiagnostic error =
307 tile.emitWarning("allocated buffers exceeded available memory\n");
308 auto &note = error.attachNote() << "Error in bank(s) : ";
309 for (auto bank : overflow_banks)
310 note << bank << " ";
311 note << "\n";
312 note << "MemoryMap:\n";
313 auto printbuffer = [&](StringRef name, int address, int size) {
314 note << "\t"
315 << "\t" << name << " \t"
316 << ": 0x" << llvm::utohexstr(address) << "-0x"
317 << llvm::utohexstr(address + size - 1) << " \t(" << size
318 << " bytes)\n";
319 };
320 for (int i = 0; i < numBanks; i++) {
321 note << "\t"
322 << "bank : " << i << "\t"
323 << "0x" << llvm::utohexstr(bankLimits[i].startAddr) << "-0x"
324 << llvm::utohexstr(bankLimits[i].endAddr - 1) << "\n";
325 if (i == 0) {
326 if (stacksize > 0)
327 printbuffer("(stack)", 0, stacksize);
328 else
329 error << "(no stack allocated)\n";
330 }
331 for (auto buffer : allBuffers) {
332 auto addr = buffer.getAddress().value();
333 auto mem_bank = buffer.getMemBank().value();
334 if (mem_bank == i)
335 printbuffer(buffer.name(), addr, buffer.getAllocationSize());
336 }
337 }
338 return false;
339 }
340 return true;

351bool simpleBankAwareAllocation(TileOp tile) {
352 auto device = tile->getParentOfType<AIE::DeviceOp>();
353 if (!device)
354 return false;
355
356 std::vector<int64_t>
357 nextAddrInBanks; // each entry is the next address available for use
358 // for that bank
359 // (e.g. nextAddrInBanks[tile_0][1] = next available
360 // address in bank 1 for tile_0)
361 std::vector<BankLimits> bankLimits; // the entries contain pairs of start and
362 // end addresses for each bank
363
364 const auto &targetModel = getTargetModel(tile);
365 int maxDataMemorySize = 0;
366 if (tile.isMemTile())
367 maxDataMemorySize = targetModel.getMemTileSize();
368 else
369 maxDataMemorySize = targetModel.getLocalMemorySize();
370
371 int numBanks = targetModel.getNumBanks(tile.getCol(), tile.getRow());
372 int bankSize = maxDataMemorySize / numBanks;
373
374 // Address range owned by the MemTile is 0x80000.
375 // Address range owned by the tile is 0x8000 in
376 // AIE1 and 0x10000 in AIE2, but we need room at
377 // the bottom for stack.
378 int stacksize = 0;
379 for (int i = 0; i < numBanks; i++)
380 nextAddrInBanks.push_back(bankSize * i);
381 if (auto core = tile.getCoreOp()) {
382 stacksize = core.getStackSize();
383 nextAddrInBanks[0] += stacksize;
384 }
385 fillBankLimits(numBanks, bankSize, bankLimits);
386
387 SmallVector<BufferOp> preAllocatedBuffers;
388 SmallVector<BufferOp> buffersToAlloc;
389 SmallVector<BufferOp> allBuffers;
390 // Collect all the buffers for this tile.
391 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
392 if (buffer.getTileOp() == tile)
393 allBuffers.push_back(buffer);
394 });
395 // If possible, the buffers with an already specified address will not
396 // be overwritten (the available address range of the bank the buffers
397 // are in will start AFTER the specified adress + buffer size).
398 // Buffers with a specified mem_bank will be assigned first, after
399 // the above.
400 for (auto buffer : allBuffers) {
401 if (buffer.getTileOp() == tile) {
402 auto has_addr = checkAndAddBufferWithAddress(buffer, numBanks,
403 nextAddrInBanks, bankLimits);
404 auto has_bank = checkAndAddBufferWithMemBank(buffer, numBanks,
405 nextAddrInBanks, bankLimits);
406 if (failed(has_addr) || failed(has_bank))
407 return false;
408 if (!has_addr.value() && !has_bank.value())
409 buffersToAlloc.push_back(buffer);
410 else
411 preAllocatedBuffers.push_back(buffer);
412 }
413 }
414
415 // Sort by largest allocation size before allocating.
416 std::sort(buffersToAlloc.begin(), buffersToAlloc.end(),
417 [](BufferOp a, BufferOp b) {
418 return a.getAllocationSize() > b.getAllocationSize();
419 });
420
421 // Set addresses for remaining buffers.
422 SmallVector<BufferOp> allocatedBuffers;
423 int bankIndex = 0;
424 for (auto buffer : buffersToAlloc) {
425 // If the buffer doesn't fit in any of the bank space then
426 // it prints the current memory map of the banks,
427 // deallocates all the buffers, and
428 // returns a failure.
429 if (!setBufferAddress(buffer, numBanks, bankIndex, nextAddrInBanks,
430 bankLimits)) {
431
432 printMemMap(tile, allocatedBuffers, preAllocatedBuffers, numBanks,
433 bankLimits, stacksize);
434 deAllocationBuffers(allocatedBuffers);
435 return false;
436 } else {
437 allocatedBuffers.push_back(buffer);
438 }
439 }
440
441 // Sort by smallest address before printing memory map.
442 std::sort(allBuffers.begin(), allBuffers.end(), [](BufferOp a, BufferOp b) {
443 assert(a.getAddress().has_value() && "buffer must have address assigned");
444 assert(b.getAddress().has_value() && "buffer must have address assigned");
445 return a.getAddress().value() < b.getAddress().value();
446 });
447 // Check if memory was exceeded on any bank and print debug info.
448 return checkAndPrintOverflow(tile, numBanks, stacksize, allBuffers,
449 nextAddrInBanks, bankLimits);
450}

452LogicalResult checkBufferScope(BufferOp buffer, DeviceOp device) {
453 // Buffers are not allowed to be inside the core without being statically
454 // initialized.
455 Operation *parent = buffer->getParentOp();
456 // Allowed to be in MemTile
457 if (!isa<DeviceOp>(parent) && !isa<MemTileDMAOp>(parent) &&
458 !buffer.getInitialValue().has_value()) {
459 auto tile = buffer.getTileOp();
460 tile->emitOpError("Buffer '")
461 << buffer.name()
462 << "' must be defined directly under the device scope. Currently it "
463 "is nested inside a core tile.";
464 return failure();
465 }
466 return success();
467}

469struct AIEAssignBufferAddressesPass
470 : AIEAssignBufferAddressesBase<AIEAssignBufferAddressesPass> {
471
472 void getDependentDialects(DialectRegistry &registry) const override {
473 registry.insert<func::FuncDialect>();
474 registry.insert<AIEDialect>();
475 }
476
477 void runOnOperation() override {
478 DeviceOp device = getOperation();
479 OpBuilder builder = OpBuilder::atBlockTerminator(device.getBody());
480 // Ensure all BufferOps are globally defined at the device level.
481 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
482 if (failed(checkBufferScope(buffer, device)))
483 return signalPassFailure();
484 });
485 // Make sure all the buffers have a name
486 int counter = 0;
487 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
488 if (!buffer.hasName()) {
489 std::string name = "_anonymous";
490 name += std::to_string(counter++);
491 buffer->setAttr(SymbolTable::getSymbolAttrName(),
492 builder.getStringAttr(name));
493 }
494 });
495
496 // Select allocation scheme per tile
497 for (auto tile : device.getOps<TileOp>()) {
498 auto tileAllocationScheme = tile.getAllocationScheme();
499
500 if (!tileAllocationScheme)
501 tileAllocationScheme = clAllocScheme;
502
503 if (tileAllocationScheme == "basic-sequential") {
504 if (!basicAllocation(tile)) {
505 tile.emitOpError("Basic sequential allocation failed.");
506 return signalPassFailure();
507 }
508 } else if (tileAllocationScheme == "bank-aware") {
509 if (!simpleBankAwareAllocation(tile)) {
510 tile.emitOpError("Bank-aware allocation failed.");
511 return signalPassFailure();
512 }
513 } else {
514 if (!simpleBankAwareAllocation(tile)) {
515 tile.emitWarning("Bank-aware allocation failed, trying basic "
516 "sequential allocation.");
517 if (!basicAllocation(tile)) {
518 tile.emitOpError("Basic sequential allocation also failed.");
519 return signalPassFailure();
520 }
521 }
522 }
523 }
524 }
525};
