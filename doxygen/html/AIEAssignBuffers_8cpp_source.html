27static bool isBufferPreAllocated(BufferOp buffer) {
28 auto addr = buffer.getAddress();
29 auto memBank = buffer.getMemBank();
30 return (addr != std::nullopt || memBank != std::nullopt);
31}

33// Return an address that is aligned to tile's load/store bus
34// NOTE: assume address are byte address
35static int64_t getAlignedAddress(int64_t address, uint32_t alignBitWidth) {
36 assert(alignBitWidth != 0 && alignBitWidth % 8 == 0 &&
37 "alignBitWidth must be a non-zero multiple of 8");
38 uint32_t alignByteWidth = alignBitWidth / 8;
39 if (address % alignByteWidth == 0) {
40 return address;
41 } else {
42 return ((address / alignByteWidth) + 1) * alignByteWidth;
43 }
44}

46// Check that every buffer in the list is properly aligned (when its
47// `aligned` attribute is set) and that no two buffers overlap. The input
48// vector must be sorted by ascending address. Emits an error on the first
49// offending buffer and returns false; returns true otherwise.
50static bool checkAndPrintBufferOverlap(SmallVector<BufferOp> &sortedBuffers,
51 uint32_t tileAlignBitWidth) {
52 uint32_t alignByteWidth = tileAlignBitWidth / 8;
53 for (size_t i = 0; i < sortedBuffers.size(); ++i) {
54 auto cur = sortedBuffers[i];
55 assert(cur.getAddress().has_value() && "buffer must have address assigned");
56 int64_t curAddr = cur.getAddress().value();
57
58 // Alignment check.
59 if (cur.getAligned() && alignByteWidth != 0 &&
60 curAddr % alignByteWidth != 0) {
61 cur.emitOpError("buffer '")
62 << cur.name() << "' at address 0x" << llvm::utohexstr(curAddr)
63 << " is not aligned to tile load/store bus width ("
64 << tileAlignBitWidth << " bits)";
65 return false;
66 }
67
68 // Pairwise overlap check against the previous buffer in address order.
69 if (i == 0)
70 continue;
71 auto prev = sortedBuffers[i - 1];
72 assert(prev.getAddress().has_value() &&
73 "buffer must have address assigned");
74 int64_t prevEnd = prev.getAddress().value() + prev.getAllocationSize();
75 if (curAddr < prevEnd) {
76 cur.emitOpError("buffer '")
77 << cur.name() << "' at address 0x" << llvm::utohexstr(curAddr)
78 << " overlaps with '" << prev.name() << "' at address 0x"
79 << llvm::utohexstr(prev.getAddress().value())
80 << " (size: " << prev.getAllocationSize() << " bytes)";
81 return false;
82 }
83 }
84 return true;
85}

87// Check if there is any overlap between the stack and the allocated buffers.
88static bool checkAndPrintOverlapStackframe(int stacksize,
89 SmallVector<BufferOp> &buffers) {
90 for (auto buf : buffers) {
91 assert(buf.getAddress().has_value() && "buffer must have address assigned");
92 if (buf.getAddress().value() < stacksize) {
93 buf.emitOpError("buffer '")
94 << buf.name() << "' at address 0x"
95 << llvm::utohexstr(buf.getAddress().value())
96 << " overlaps with stack (size: " << stacksize << " bytes)";
97 return false;
98 }
99 }
100 return true;
101}

106static bool checkAndPrintOverflow(TileOp tile, int address,
107 int maxDataMemorySize, int stacksize,
108 SmallVector<BufferOp> &buffers) {
109 if (address > maxDataMemorySize) {
110 InFlightDiagnostic error =
111 tile.emitOpError("allocated buffers exceeded available memory\n");
112 auto &note = error.attachNote() << "MemoryMap:\n";
113 auto printbuffer = [&](StringRef name, int address, int size) {
114 note << "\t" << name << " \t"
115 << ": 0x" << llvm::utohexstr(address) << "-0x"
116 << llvm::utohexstr(address + size - 1) << " \t(" << size
117 << " bytes)\n";
118 };
119 if (stacksize > 0)
120 printbuffer("(stack)", 0, stacksize);
121 else
122 error << "(no stack allocated)\n";
123
124 for (auto buffer : buffers) {
125 assert(buffer.getAddress().has_value() &&
126 "buffer must have address assigned");
127 printbuffer(buffer.name(), buffer.getAddress().value(),
128 buffer.getAllocationSize());
129 }
130 return false;
131 }
132 return true;
133}

135static bool basicAllocation(TileOp tile) {
136 auto device = tile->getParentOfType<AIE::DeviceOp>();
137 if (!device)
138 return false;
139
140 const auto &targetModel = getTargetModel(tile);
141 int maxDataMemorySize = 0;
142 uint32_t tileAlignBitWidth = 0;
143 if (tile.isMemTile()) {
144 maxDataMemorySize = targetModel.getMemTileSize();
145 tileAlignBitWidth = targetModel.getMemTileLoadStoreBusWidth();
146 } else {
147 maxDataMemorySize = targetModel.getLocalMemorySize();
148 tileAlignBitWidth = targetModel.getComputeTileLoadStoreBusWidth();
149 }
150
151 SmallVector<BufferOp> buffers;
152 SmallVector<BufferOp> allocated_buffers;
153 SmallVector<BufferOp> allBuffers_on_tile;
154 // Collect all the buffers for this tile. If the buffer has an address, add
155 // it to allocated_buffers. Otherwise, add it to buffers.
156 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
157 if (buffer.getTileOp() == tile) {
158 if (buffer.getAddress()) {
159 allocated_buffers.push_back(
160 buffer); // TODO: Right now, this ignore all buffer with only
161 // mem_bank attribute.
162 } else {
163 buffers.push_back(buffer);
164 }
165 allBuffers_on_tile.push_back(buffer);
166 }
167 });
168
169 // Sort buffers by allocation size.
170 std::sort(buffers.begin(), buffers.end(), [](BufferOp a, BufferOp b) {
171 return a.getAllocationSize() > b.getAllocationSize();
172 });
173
174 // Sort allocated_buffers by address
175 std::sort(allocated_buffers.begin(), allocated_buffers.end(),
176 [](BufferOp a, BufferOp b) {
177 return a.getAddress().value() < b.getAddress().value();
178 });
179
180 // Address range owned by the MemTile is 0x80000.
181 // Address range owned by the tile is 0x8000 in
182 // AIE1 and 0x10000 in AIE2, but we need room at
183 // the bottom for stack.
184 int64_t stacksize = 0;
185 int64_t address = 0;
186 if (auto core = tile.getCoreOp()) {
187 stacksize = core.getStackSize();
188 address += stacksize;
189 }
190
191 // Ensure alignment of preallocated buffer
192 for (auto buffer : allocated_buffers) {
193 if (buffer.getAligned() &&
194 buffer.getAddress().value() % (tileAlignBitWidth / 8) != 0) {
195 buffer.emitOpError("pre-allocated address must be aligned to tile "
196 "load/store bus width when aligned attribute is set");
197 return false;
198 }
199 }
200
201 // As the next address to allocate is assigned, skip over any buffers
202 // from the allocated_buffers list.
203 // Note: alignment must be applied *before* (and after each skip in) the
204 // overlap-skip loop below, so the loop reasons about the buffer's actual
205 // placement. Otherwise an unaligned candidate address can appear to fit
206 // before the next pre-allocated buffer, but get bumped forward by
207 // getAlignedAddress and silently alias that pre-allocated buffer.
208 auto current_alloc = allocated_buffers.begin();
209 for (auto buffer : buffers) {
210 assert(!buffer.getAddress());
211 if (buffer.getAligned())
212 address = getAlignedAddress(address, tileAlignBitWidth);
213 while (current_alloc != allocated_buffers.end() &&
214 address + buffer.getAllocationSize() >
215 current_alloc->getAddress().value()) {
216 address = current_alloc->getAddress().value() +
217 current_alloc->getAllocationSize();
218 if (buffer.getAligned())
219 address = getAlignedAddress(address, tileAlignBitWidth);
220 current_alloc++;
221 }
222
223 buffer.setAddress(address);
224 address += buffer.getAllocationSize();
225 }
226
227 // Sort by smallest address before printing memory map and running the
228 // overlap / overflow checks below.
229 std::sort(allBuffers_on_tile.begin(), allBuffers_on_tile.end(),
230 [](BufferOp a, BufferOp b) {
231 assert(a.getAddress().has_value() &&
232 "buffer must have address assigned");
233 assert(b.getAddress().has_value() &&
234 "buffer must have address assigned");
235 return a.getAddress().value() < b.getAddress().value();
236 });
237
238 // Compute the true high-water mark across *all* buffers (including
239 // pre-allocated ones above the dynamic-allocation cursor) so that
240 // checkAndPrintOverflow sees a correct memory bound.
241 int64_t highWater = address;
242 if (!allBuffers_on_tile.empty()) {
243 auto &last = allBuffers_on_tile.back();
244 highWater = std::max<int64_t>(highWater, last.getAddress().value() +
245 last.getAllocationSize());
246 }
247
248 // Check if memory was exceeded or buffers overlap, and print debug info.
249 return (checkAndPrintOverlapStackframe(stacksize, allBuffers_on_tile) &&
250 checkAndPrintBufferOverlap(allBuffers_on_tile, tileAlignBitWidth) &&
251 checkAndPrintOverflow(tile, highWater, maxDataMemorySize, stacksize,
252 allBuffers_on_tile));
253}

258typedef struct BankLimits {
259 int64_t startAddr;
260 int64_t endAddr;
261} BankLimits;

263// Function that given a number of banks and their size, computes
264// the start and end addresses for each bank and fills in the entry
265// in the bankLimits vector.
266static void fillBankLimits(int numBanks, int bankSize,
267 std::vector<BankLimits> &bankLimits) {
268 for (int i = 0; i < numBanks; i++) {
269 auto startAddr = bankSize * i;
270 auto endAddr = bankSize * (i + 1);
271 bankLimits.push_back({startAddr, endAddr});
272 }
273}

275// Function that sets the address attribute of the given buffer to
276// the given start_addr. It also updates the entry in the
277// nextAddrInBanks for the corresponding bank.
278static void setAndUpdateAddressInBank(BufferOp buffer, int64_t start_addr,
279 int64_t end_addr,
280 std::vector<int64_t> &nextAddrInBanks) {
281
282 buffer.setAddress(start_addr);
283 nextAddrInBanks[buffer.getMemBank().value()] = end_addr;
284}

286// Function that checks whether the given buffer already has a set address
287// attribute. If it does, it finds in which bank the buffer is and checks
288// whether there is enough space left for it (and ensure the bank match to
289// the mem_bank attribute if given).
290// If there is the function
291// returns true and if not, the function emits a warning that the address
292// will be overwritten and returns false (which will cause the buffer to be
293// added to the list of buffers without addresses, to be completed later on).
294static FailureOr<bool>
295checkAndAddBufferWithAddress(BufferOp buffer, int numBanks,
296 uint32_t tileAlignBitWidth,
297 std::vector<int64_t> &nextAddrInBanks,
298 std::vector<BankLimits> &bankLimits) {
299 auto addrAttr = buffer->getAttrOfType<IntegerAttr>("address");
300 if (!addrAttr)
301 return false;
302 // it is fine if mem_bank is not set
303 auto memBankAttr = buffer->getAttrOfType<IntegerAttr>("mem_bank");
304
305 int addr = addrAttr.getInt();
306 if (buffer.getAligned() && addr % (tileAlignBitWidth / 8) != 0) {
307 return buffer->emitOpError(
308 "address attribute value must be aligned to tile load/store bus width "
309 "when aligned attribute is set");
310 }
311 for (int i = 0; i < numBanks; i++) {
312 // if the address is not within the bank, continue
313 if (addr < bankLimits[i].startAddr || addr >= bankLimits[i].endAddr)
314 continue;
315
316 // if the allocator already allocated this address, fail
317 if (addr < nextAddrInBanks[i])
318 return buffer->emitOpError("would override allocated address");
319
320 // the allocator can accomadate this existing allocation
321 nextAddrInBanks[i] = addr + buffer.getAllocationSize();
322 if (memBankAttr) {
323 // specified both mem_bank and address, check if they are consistent
324 int mem_bank = memBankAttr.getInt();
325 if (mem_bank != i)
326 return buffer->emitOpError(
327 "mem_bank attribute is inconsistent with address attribute");
328 }
329 buffer.setMemBank(i);
330 return true;
331 }
332 return buffer->emitOpError(
333 "address attribute does not fall within any bank range");
334}
335
336// Function that checks whether the given buffer already has a set mem_bank
337// attribute. If it does, it checks whether there is enough space left for
338// it. If there is, it sets the buffer's address field and if not, the
339// function emits a warning that the mem_bank will be overwritten and returns
340// false (which will cause the buffer to be added to the list of buffers
341// without addresses, to be completed later on).
342static FailureOr<bool>
343checkAndAddBufferWithMemBank(BufferOp buffer, int numBanks,
344 uint32_t tileAlignBitWidth,
345 std::vector<int64_t> &nextAddrInBanks,
346 std::vector<BankLimits> &bankLimits) {
347 auto memBankAttr = buffer->getAttrOfType<IntegerAttr>("mem_bank");
348 if (!memBankAttr)
349 return false;
350
351 int mem_bank = memBankAttr.getInt();
352 if (mem_bank < 0 || mem_bank >= numBanks) {
353 return buffer->emitOpError("mem_bank attribute value is out of range");
354 }
355
356 int64_t startAddr = nextAddrInBanks[mem_bank];
357 if (buffer.getAligned()) {
358 startAddr = getAlignedAddress(startAddr, tileAlignBitWidth);
359 }
360
361 int64_t endAddr = startAddr + buffer.getAllocationSize();
362 if (endAddr > bankLimits[mem_bank].endAddr)
363 return buffer->emitOpError("would override existing mem_bank");
364 setAndUpdateAddressInBank(buffer, startAddr, endAddr, nextAddrInBanks);
365 return true;
366}

368// Prints the memory map across banks
369static void printMemMap(TileOp tile, SmallVector<BufferOp> &allocatedBuffers,
370 SmallVector<BufferOp> &preAllocatedBuffers,
371 int numBanks, std::vector<BankLimits> &bankLimits,
372 int stacksize) {
373 InFlightDiagnostic error = tile.emitWarning(
374 "Not all requested buffers fit in the available memory.\n");
375 auto &note = error.attachNote()
376 << "Current configuration of buffers in bank(s) : ";
377 note << "MemoryMap:\n";
378 auto printbuffer = [&](StringRef name, int address, int size) {
379 note << "\t"
380 << "\t" << name << " \t"
381 << ": 0x" << llvm::utohexstr(address) << "-0x"
382 << llvm::utohexstr(address + size - 1) << " \t(" << size
383 << " bytes)\n";
384 };
385 for (int i = 0; i < numBanks; i++) {
386 if (i == 0) {
387 if (stacksize > 0)
388 printbuffer("(stack)", 0, stacksize);
389 else
390 note << "(no stack allocated)\n";
391 }
392 note << "\t"
393 << "bank : " << i << "\t"
394 << "0x" << llvm::utohexstr(bankLimits[i].startAddr) << "-0x"
395 << llvm::utohexstr(bankLimits[i].endAddr - 1) << "\n";
396 for (auto buffer : preAllocatedBuffers) {
397 auto addr = buffer.getAddress().value();
398 auto mem_bank = buffer.getMemBank().value();
399 if (mem_bank == i)
400 printbuffer(buffer.name(), addr, buffer.getAllocationSize());
401 }
402 for (auto buffer : allocatedBuffers) {
403 auto addr = buffer.getAddress().value();
404 auto mem_bank = buffer.getMemBank().value();
405 if (mem_bank == i)
406 printbuffer(buffer.name(), addr, buffer.getAllocationSize());
407 }
408 }
409}

411// Function that given a buffer will iterate over all the memory banks
412// starting from the given index to try and find a bank with enough
413// space. If it does, it will set the buffer's address and mem_bank
414// attributes and update the nextAddrInBanks vector.
415// If it does not find one with enough space, it will throw an error.
416// Returns true if the buffer was successfully allocated, false otherwise.
417// If no bank has enough space to accommodate the buffer, an error is emitted.
418
419static int setBufferAddress(BufferOp buffer, int numBanks,
420 uint32_t tileAlignBitWidth, int &startBankIndex,
421 std::vector<int64_t> &nextAddrInBanks,
422 std::vector<BankLimits> &bankLimits) {
423 assert(startBankIndex < numBanks &&
424 "Unexpected input value for startBankIndex");
425 int bankIndex = startBankIndex;
426 bool allocated = false;
427 for (int i = 0; i < numBanks; i++) {
428 int64_t startAddr = nextAddrInBanks[bankIndex];
429
430 if (buffer.getAligned()) {
431 startAddr = getAlignedAddress(startAddr, tileAlignBitWidth);
432 }
433
434 int64_t endAddr = startAddr + buffer.getAllocationSize();
435 if (endAddr <= bankLimits[bankIndex].endAddr) {
436 buffer.setMemBank(bankIndex);
437 setAndUpdateAddressInBank(buffer, startAddr, endAddr, nextAddrInBanks);
438 allocated = true;
439 bankIndex = (bankIndex + 1) % numBanks;
440 startBankIndex = bankIndex;
441 break;
442 }
443 // Move to the next bank
444 bankIndex = (bankIndex + 1) % numBanks;
445 }
446 // If no bank has enough space, throws error
447 if (!allocated) {
448 buffer.emitWarning("Failed to allocate buffer: ")
449 << buffer.name() << " with size: " << buffer.getAllocationSize()
450 << " bytes.";
451 return false;
452 }
453 return true;
454}

456static bool checkAndPrintOverflow(TileOp tile, int numBanks, int stacksize,
457 SmallVector<BufferOp> &allBuffers,
458 std::vector<int64_t> &nextAddrInBanks,
459 std::vector<BankLimits> &bankLimits) {
460 bool foundOverflow = false;
461 std::vector<int> overflow_banks;
462 for (int i = 0; i < numBanks; i++) {
463 if (nextAddrInBanks[i] > bankLimits[i].endAddr) {
464 foundOverflow = true;
465 overflow_banks.push_back(i);
466 }
467 }
468 if (foundOverflow) {
469 InFlightDiagnostic error =
470 tile.emitWarning("allocated buffers exceeded available memory\n");
471 auto &note = error.attachNote() << "Error in bank(s) : ";
472 for (auto bank : overflow_banks)
473 note << bank << " ";
474 note << "\n";
475 note << "MemoryMap:\n";
476 auto printbuffer = [&](StringRef name, int address, int size) {
477 note << "\t"
478 << "\t" << name << " \t"
479 << ": 0x" << llvm::utohexstr(address) << "-0x"
480 << llvm::utohexstr(address + size - 1) << " \t(" << size
481 << " bytes)\n";
482 };
483 for (int i = 0; i < numBanks; i++) {
484 note << "\t"
485 << "bank : " << i << "\t"
486 << "0x" << llvm::utohexstr(bankLimits[i].startAddr) << "-0x"
487 << llvm::utohexstr(bankLimits[i].endAddr - 1) << "\n";
488 if (i == 0) {
489 if (stacksize > 0)
490 printbuffer("(stack)", 0, stacksize);
491 else
492 error << "(no stack allocated)\n";
493 }
494 for (auto buffer : allBuffers) {
495 auto addr = buffer.getAddress().value();
496 auto mem_bank = buffer.getMemBank().value();
497 if (mem_bank == i)
498 printbuffer(buffer.name(), addr, buffer.getAllocationSize());
499 }
500 }
501 return false;
502 }
503 return true;
504}

514static bool simpleBankAwareAllocation(TileOp tile) {
515 auto device = tile->getParentOfType<AIE::DeviceOp>();
516 if (!device)
517 return false;
518
519 std::vector<int64_t>
520 nextAddrInBanks; // each entry is the next address available for use
521 // for that bank
522 // (e.g. nextAddrInBanks[tile_0][1] = next available
523 // address in bank 1 for tile_0)
524 std::vector<BankLimits> bankLimits; // the entries contain pairs of start and
525 // end addresses for each bank
526
527 const auto &targetModel = getTargetModel(tile);
528 int maxDataMemorySize = 0;
529 uint32_t tileAlignBitWidth = 0;
530 if (tile.isMemTile()) {
531 maxDataMemorySize = targetModel.getMemTileSize();
532 tileAlignBitWidth = targetModel.getMemTileLoadStoreBusWidth();
533 } else {
534 maxDataMemorySize = targetModel.getLocalMemorySize();
535 tileAlignBitWidth = targetModel.getComputeTileLoadStoreBusWidth();
536 }
537
538 int numBanks = targetModel.getNumBanks(tile.getCol(), tile.getRow());
539 int bankSize = maxDataMemorySize / numBanks;
540
541 // Address range owned by the MemTile is 0x80000.
542 // Address range owned by the tile is 0x8000 in
543 // AIE1 and 0x10000 in AIE2, but we need room at
544 // the bottom for stack.
545 int stacksize = 0;
546 for (int i = 0; i < numBanks; i++)
547 nextAddrInBanks.push_back(bankSize * i);
548 if (auto core = tile.getCoreOp()) {
549 stacksize = core.getStackSize();
550 nextAddrInBanks[0] += stacksize;
551 }
552 fillBankLimits(numBanks, bankSize, bankLimits);
553
554 SmallVector<BufferOp> preAllocatedBuffers;
555 SmallVector<BufferOp> buffersToAlloc;
556 SmallVector<BufferOp> allBuffers_on_tile;
557 // Collect all the buffers for this tile.
558 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
559 if (buffer.getTileOp() == tile) {
560 if (!isBufferPreAllocated(buffer)) {
561 buffersToAlloc.push_back(buffer);
562 } else {
563 preAllocatedBuffers.push_back(buffer);
564 }
565 allBuffers_on_tile.push_back(buffer);
566 }
567 });
568
569 // First, allocate the buffer with pre-allocated address
570 // Then, allocated the buffer with pre-allocated mem_bank t
571 // Do it by doing a sort preAllocatedBuffers to have buffer with address
572 // first, then buffer with only mem_bank
573 std::sort(
574 preAllocatedBuffers.begin(), preAllocatedBuffers.end(),
575 [](BufferOp a, BufferOp b) -> bool {
576 auto a_addr = a.getAddress();
577 auto b_addr = b.getAddress();
578 if (a_addr.has_value() && !b_addr.has_value()) {
579 return true; // address buffers before mem_bank-only buffers
580 } else if (!a_addr.has_value() && b_addr.has_value()) {
581 return false; // mem_bank-only buffers after address buffers
582 } else if (a_addr.has_value() && b_addr.has_value()) {
583 return a_addr.value() <
584 b_addr.value(); // ascending address order within same bank
585 } else {
586 return false;
587 }
588 });
589
590 for (auto buffer : preAllocatedBuffers) {
591
592 auto has_addr = checkAndAddBufferWithAddress(
593 buffer, numBanks, tileAlignBitWidth, nextAddrInBanks, bankLimits);
594 if (failed(has_addr))
595 return false;
596 if (has_addr.value())
597 continue;
598 auto has_bank = checkAndAddBufferWithMemBank(
599 buffer, numBanks, tileAlignBitWidth, nextAddrInBanks, bankLimits);
600 if (failed(has_bank))
601 return false;
602 }
603
604 // Sort by largest allocation size before allocating.
605 std::sort(buffersToAlloc.begin(), buffersToAlloc.end(),
606 [](BufferOp a, BufferOp b) {
607 return a.getAllocationSize() > b.getAllocationSize();
608 });
609
610 // Set addresses for remaining buffers.
611 SmallVector<BufferOp>
612 allocatedBuffers; // keep track of buffers allocated in this function to
613 // be able to deallocate in case of failure and print
614 // helpful debug info about them. This does not include
615 // the pre-allocated buffers.
616 int bankIndex = 0;
617 for (auto buffer : buffersToAlloc) {
618 // If the buffer doesn't fit in any of the bank space then
619 // it prints the current memory map of the banks,
620 // deallocates all the buffers, and
621 // returns a failure.
622 if (!setBufferAddress(buffer, numBanks, tileAlignBitWidth, bankIndex,
623 nextAddrInBanks, bankLimits)) {
624
625 printMemMap(tile, allocatedBuffers, preAllocatedBuffers, numBanks,
626 bankLimits, stacksize);
627 deAllocationBuffers(allocatedBuffers);
628 return false;
629 } else {
630 allocatedBuffers.push_back(buffer);
631 }
632 }
633 assert(allocatedBuffers.size() == buffersToAlloc.size());
634
635 // Sort by smallest address before printing memory map.
636 std::sort(allBuffers_on_tile.begin(), allBuffers_on_tile.end(),
637 [](BufferOp a, BufferOp b) {
638 assert(a.getAddress().has_value() &&
639 "buffer must have address assigned");
640 assert(b.getAddress().has_value() &&
641 "buffer must have address assigned");
642 return a.getAddress().value() < b.getAddress().value();
643 });
644 // Check if memory was exceeded on any bank and print debug info.
645 return checkAndPrintOverlapStackframe(stacksize, allBuffers_on_tile) &&

651static LogicalResult checkBufferScope(BufferOp buffer, DeviceOp device) {
652 // Buffers are not allowed to be inside the core without being statically
653 // initialized.
654 Operation *parent = buffer->getParentOp();
655 // Allowed to be in MemTile
656 if (!isa<DeviceOp>(parent) && !isa<MemTileDMAOp>(parent) &&
657 !buffer.getInitialValue().has_value()) {
658 auto tile = buffer.getTileOp();
659 tile->emitOpError("Buffer '")
660 << buffer.name()
661 << "' must be defined directly under the device scope. Currently it "
662 "is nested inside a core tile.";
663 return failure();
664 }
665 return success();
666}

683 void runOnOperation() override {
684 DeviceOp device = getOperation();
685 OpBuilder builder = OpBuilder::atBlockTerminator(device.getBody());
686 // Ensure all BufferOps are globally defined at the device level.
687 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
688 if (failed(checkBufferScope(buffer, device)))
689 return signalPassFailure();
690 });
691 // Make sure all the buffers have a name
692 int counter = 0;
693 device.walk<WalkOrder::PreOrder>([&](BufferOp buffer) {
694 if (!buffer.hasName()) {
695 std::string name = "_anonymous";
696 name += std::to_string(counter++);
697 buffer->setAttr(SymbolTable::getSymbolAttrName(),
698 builder.getStringAttr(name));
699 }
700 });
701
702 // Select allocation scheme per tile
703 for (auto tile : device.getOps<TileOp>()) {
704 auto tileAllocationScheme = tile.getAllocationScheme();
705
706 if (!tileAllocationScheme)
707 tileAllocationScheme = clAllocScheme;
708
709 if (tileAllocationScheme == "basic-sequential") {
710 if (!basicAllocation(tile)) {
711 tile.emitOpError("Basic sequential allocation failed.");
712 return signalPassFailure();
713 }
714 } else if (tileAllocationScheme == "bank-aware") {
715 if (!simpleBankAwareAllocation(tile)) {
716 tile.emitOpError("Bank-aware allocation failed.");
717 return signalPassFailure();
718 }
719 } else {
720 if (!simpleBankAwareAllocation(tile)) {
721 tile.emitWarning("Bank-aware allocation failed, trying basic "
722 "sequential allocation.");
723 if (!basicAllocation(tile)) {
724 tile.emitOpError("Basic sequential allocation also failed.");
725 return signalPassFailure();
726 }
727 }
728 }
729 }
730 }
