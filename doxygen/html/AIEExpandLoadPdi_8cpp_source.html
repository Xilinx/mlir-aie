37// Helper to transform a single load_pdi operation
38static LogicalResult transformLoadPdi(NpuLoadPdiOp loadPdiOp, ModuleOp moduleOp,
39 unsigned index) {
40 static unsigned long i = 0;
41 OpBuilder builder(loadPdiOp);
42
43 // Only process load_pdi ops that reference a device
44 auto deviceRefAttr = loadPdiOp.getDeviceRefAttr();
45 if (!deviceRefAttr) {
46 return success();
47 }
48
49 auto referencedDevice = moduleOp.lookupSymbol<AIE::DeviceOp>(deviceRefAttr);
50 if (!referencedDevice) {
51 loadPdiOp.emitError("Referenced symbol '")
52 << deviceRefAttr.getValue() << "' is not a device";
53 return failure();
54 }
55
56 // Create a unique empty device for this reset to avoid PDI address caching
57 OpBuilder::InsertionGuard guard(builder);
58 builder.setInsertionPointToStart(moduleOp.getBody());
59
60 // Find a unique name for the empty device
61 std::string emptyName = "empty_" + std::to_string(index % 2);
62
63 AIE::DeviceOp emptyDevice = moduleOp.lookupSymbol<AIE::DeviceOp>(emptyName);
64 if (!emptyDevice) {
65 auto deviceType = referencedDevice.getDevice();
66 auto loc = builder.getUnknownLoc();
67 emptyDevice = AIE::DeviceOp::create(builder, loc, deviceType,
68 builder.getStringAttr(emptyName));
69 emptyDevice.getRegion().emplaceBlock();
70 Block *deviceBlock = &emptyDevice.getRegion().front();
71 builder.setInsertionPointToEnd(deviceBlock);
72 AIE::EndOp::create(builder, loc);
73 }
74
75 builder.setInsertionPoint(loadPdiOp);
76
77 // Create new empty load_pdi operation; this triggers a device reset.
78 // This is needed even for the first device configuration; without it, the
79 // first iteration of the design would run, but subsequent ones might not.
80 NpuLoadPdiOp::create(builder, loadPdiOp.getLoc(),
81 FlatSymbolRefAttr::get(emptyDevice.getSymNameAttr()),
82 loadPdiOp.getIdAttr(), loadPdiOp.getSizeAttr(),
83 loadPdiOp.getAddressAttr());
84
85 // Generate and insert configuration operations
86 if (failed(xilinx::AIE::generateAndInsertConfigOps(
87 builder, referencedDevice, "",
88 AIEToConfigurationOutputType::Transaction,
89 "loadpdi_" + std::to_string(i)))) {
90 loadPdiOp.emitError("Failed to generate configuration operations");
91 return failure();
92 }
93
94 // Erase the original load_pdi operation
95 loadPdiOp.erase();
96
97 i++;
98
99 return success();
100}

102struct AIEExpandLoadPdiPass
103 : public AIEExpandLoadPdiBase<AIEExpandLoadPdiPass> {
104 void getDependentDialects(DialectRegistry &registry) const override {
105 registry
106 .insert<memref::MemRefDialect, AIE::AIEDialect, AIEX::AIEXDialect>();
107 }
108
109 void runOnOperation() override {
110 auto module = getOperation();
111
112 // Collect all load_pdi operations in program order;
113 // need to collect once, then transform all collected ops;
114 // since the transform inserts a new empty load_pdi, we can't transform as
115 // we walk or it'd infinitely recurse.
116 SmallVector<NpuLoadPdiOp> loadPdiOps;
117
118 module.walk(
119 [&](NpuLoadPdiOp loadPdiOp) { loadPdiOps.push_back(loadPdiOp); });
120
121 // Transform load_pdi ops
122 unsigned idx = 0;
123 for (auto loadPdiOp : loadPdiOps) {
124 if (failed(transformLoadPdi(loadPdiOp, module, idx))) {
125 signalPassFailure();
126 return;
127 }
128 idx++;
129 }
130 }
131};
