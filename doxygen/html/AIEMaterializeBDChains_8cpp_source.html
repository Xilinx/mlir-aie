27struct DMAStartBdChainForOpPattern : RewritePattern {
28
29 DMAStartBdChainForOpPattern(MLIRContext *ctx)
30 : RewritePattern(DMAStartBdChainForOp::getOperationName(),
31 PatternBenefit(1), ctx) {}
32
33 LogicalResult matchAndRewrite(Operation *op_any,
34 PatternRewriter &rewriter) const override {
35 DMAStartBdChainForOp op = llvm::dyn_cast<DMAStartBdChainForOp>(op_any);
36 if (!op) {
37 return failure();
38 }
39 AIE::DeviceOp device = op->getParentOfType<AIE::DeviceOp>();
40
41 AIE::ShimDMAAllocationOp alloc_op =
42 AIE::ShimDMAAllocationOp::getForSymbol(device, op.getAlloc());
43 if (!alloc_op) {
44 return op.emitOpError("no shim DMA allocation found for symbol");
45 }
46
47 AIE::TileOp tile = alloc_op.getTileOp();
48 if (!tile) {
49 return op.emitOpError(
50 "shim DMA allocation must reference a valid TileOp");
51 }
52
53 DMAStartBdChainOp new_op = DMAStartBdChainOp::create(
54 rewriter, op.getLoc(), rewriter.getIndexType(), op.getSymbol(),
55 op.getArgs(), tile.getResult(), alloc_op.getChannelDir(),
56 (int32_t)alloc_op.getChannelIndex(), op.getIssueToken(),
57 op.getRepeatCount());
58 rewriter.replaceAllUsesWith(op.getResult(), new_op.getResult());
59 rewriter.eraseOp(op);
60 return success();
61 }
62};

64struct DMAInlineBDChainPattern : RewritePattern {
65
66 DMAInlineBDChainPattern(MLIRContext *ctx)
67 : RewritePattern(DMAStartBdChainOp::getOperationName(), PatternBenefit(1),
68 ctx) {}
69
70 LogicalResult matchAndRewrite(Operation *op,
71 PatternRewriter &rewriter) const override {
72 DMAStartBdChainOp start_op = llvm::dyn_cast<DMAStartBdChainOp>(op);
73 if (!start_op) { // Not a match.
74 return failure();
75 }
76 rewriter.setInsertionPointAfter(start_op);
77
78 // Get referenced abstract BD chain
79 AIE::BDChainOp chain_def = start_op.getBDChainOp();
80 assert(chain_def);
81 Region &source_region = chain_def.getBody();
82
83 // Create BD op into which the result will be inlined
84 DMAConfigureTaskOp configure_op = DMAConfigureTaskOp::create(
85 rewriter, start_op.getLoc(), rewriter.getIndexType(),
86 start_op.getTile(), start_op.getDirection(), start_op.getChannel(),
87 start_op.getIssueToken(), start_op.getRepeatCount());
88 Region &target_region = configure_op.getBody();
89
90 // Clone BD definition into usage site, replacing abstract SSA values with
91 // concrete ones
92 IRMapping arg_map;
93 ValueRange values = start_op.getArgs();
94 for (unsigned i = 0, n = source_region.getNumArguments(); i < n; i++) {
95 BlockArgument arg = source_region.getArgument(i);
96 Value val = values[i];
97 assert(arg.getType() == val.getType());
98 arg_map.map(arg, val);
99 }
100 source_region.cloneInto(&target_region, arg_map);
101
102 // Replace result of dma start task with result of bd chain configuration
103 rewriter.replaceAllUsesWith(start_op.getResult(), configure_op.getResult());
104
105 // Add a start BDs instruction
106 DMAStartTaskOp::create(rewriter, start_op.getLoc(),
107 configure_op.getResult());
108
109 // After fully inlining, remove the original instruction
110 rewriter.eraseOp(start_op);
111
112 return success();
113 }
114};

116struct AIEMaterializeBDChainsPass
117 : AIEMaterializeBDChainsBase<AIEMaterializeBDChainsPass> {
118
119 void runOnOperation() override {
120 MLIRContext *ctx = &getContext();
121 AIE::DeviceOp device = getOperation();
122 GreedyRewriteConfig rewriter_config = GreedyRewriteConfig();
123 rewriter_config.setRegionSimplificationLevel(
124 GreedySimplifyRegionLevel::Disabled);
125
126 RewritePatternSet patterns_0(ctx);
127 patterns_0.insert<DMAStartBdChainForOpPattern>(ctx);
128 DMAConfigureTaskOp::getCanonicalizationPatterns(patterns_0, ctx);
129 if (failed(applyPatternsGreedily(device, std::move(patterns_0),
130 rewriter_config))) {
131 signalPassFailure();
132 }
133
134 RewritePatternSet patterns_1(ctx);
135 patterns_1.insert<DMAInlineBDChainPattern>(ctx);
136 rewriter_config.setRegionSimplificationLevel(
137 GreedySimplifyRegionLevel::Disabled);
138 DMAConfigureTaskOp::getCanonicalizationPatterns(patterns_1, ctx);
139 if (failed(applyPatternsGreedily(device, std::move(patterns_1),
140 rewriter_config))) {
141 signalPassFailure();
142 }
143 }
144};
