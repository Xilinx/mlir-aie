51llvm::json::Value attrToJSON(Attribute &attr) {
52 if (auto a = llvm::dyn_cast<StringAttr>(attr))
53 return {a.getValue().str()};
54
55 if (auto arrayAttr = llvm::dyn_cast<ArrayAttr>(attr)) {
56 llvm::json::Array arrayJSON;
57 for (auto a : arrayAttr)
58 arrayJSON.push_back(attrToJSON(a));
59 return llvm::json::Value(std::move(arrayJSON));
60 }
61
62 if (auto dictAttr = llvm::dyn_cast<DictionaryAttr>(attr)) {
63 llvm::json::Object dictJSON;
64 for (auto a : dictAttr) {
65 auto ident = a.getName();
66 auto attr = a.getValue();
67 dictJSON[ident.str()] = attrToJSON(attr);
68 }
69 return llvm::json::Value(std::move(dictJSON));
70 }
71
72 if (auto intAttr = llvm::dyn_cast<IntegerAttr>(attr))
73 return llvm::json::Value(intAttr.getInt());
74
75 return llvm::json::Value(std::string(""));
76}

100// Output the buffer map for the given buffer operations, with the given offset.
101// The offset is different depending on where the buffers are accessed from.
102void writeBufferMap(raw_ostream &output, BufferOp buf, int offset) {
103 std::string bufName(buf.name().getValue());
104 int bufferBaseAddr = getBufferBaseAddress(buf);
105 int numBytes = buf.getAllocationSize();
106 output << "_symbol " << bufName << " "
107 << "0x" << llvm::utohexstr(offset + bufferBaseAddr) << " " << numBytes
108 << '\n';
109}

111LogicalResult AIETranslateToTargetArch(ModuleOp module, raw_ostream &output,
112 llvm::StringRef deviceName) {
113 DeviceOp targetOp = AIE::DeviceOp::getForSymbolInModule(module, deviceName);
114 AIEArch arch = AIEArch::AIE1;
115 if (targetOp) {
116 arch = targetOp.getTargetModel().getTargetArch();
117 }
118 if (arch == AIEArch::AIE1)
119 output << "AIE\n";
120 else
121 output << stringifyEnum(arch) << "\n";
122 return success();
123}

168 TranslateFromMLIRRegistration registrationMMap(
169 "aie-generate-mmap", "Generate AIE memory map",
170 [](ModuleOp module, raw_ostream &output) {
171 DenseMap<TileID, Operation *> tiles;
172 DenseMap<Operation *, SmallVector<BufferOp, 4>> buffers;
173
174 DeviceOp targetOp =
175 AIE::DeviceOp::getForSymbolInModule(module, deviceName);
176 if (!targetOp) {
177 module.emitOpError("expected AIE.device operation at toplevel");
178 }
179
180 collectTiles(targetOp, tiles);
181 // sort the tiles for deterministic output
182 using tileType = std::pair<TileID, Operation *>;
183 struct tileCmp {
184 bool operator()(const tileType &lhs, const tileType &rhs) const {
185 return lhs.first < rhs.first;
186 }
187 };
188 std::set<tileType, tileCmp> sortedTiles;
189 for (auto tile : tiles)
190 sortedTiles.insert(tileType{tile.first, tile.second});
191
192 collectBuffers(targetOp, buffers);
193
194 for (auto tile : sortedTiles) {
195 Operation *srcTileOp = tile.second;
196 TileID srcCoord = cast<TileOp>(srcTileOp).getTileID();
197 int srcCol = srcCoord.col;
198 int srcRow = srcCoord.row;
199
200 output << "// Tile(" << srcCol << ", " << srcRow << ")\n";
201 output << "// Memory map: name base_address num_bytes\n";
202
203 auto doBuffer = [&](std::optional<TileID> tile, int offset) {
204 if (tiles.count(*tile))
205 for (auto buf : buffers[tiles[*tile]])
206 writeBufferMap(output, buf, offset);
207 };
208
209 const auto &targetModel = xilinx::AIE::getTargetModel(srcTileOp);
210
211 if (auto tile = targetModel.getMemSouth(srcCoord))
212 doBuffer(tile, targetModel.getMemSouthBaseAddress());
213 if (auto tile = targetModel.getMemWest(srcCoord))
214 doBuffer(tile, targetModel.getMemWestBaseAddress());
215 if (auto tile = targetModel.getMemNorth(srcCoord))
216 doBuffer(tile, targetModel.getMemNorthBaseAddress());
217 if (auto tile = targetModel.getMemEast(srcCoord))
218 doBuffer(tile, targetModel.getMemEastBaseAddress());
219 }
220 return success();
221 },
222 registerDialects);

224 TranslateFromMLIRRegistration registrationShimDMAToJSON(
225 "aie-generate-json", "Transform AIE shim DMA allocation info into JSON",
226 [](ModuleOp module, raw_ostream &output) {
227 DeviceOp d =
228 AIE::DeviceOp::getForSymbolInModuleOrError(module, deviceName);
229 if (!d) {
230 return failure();
231 }
232 llvm::json::Object moduleJSON;
233 for (auto shimDMAMeta : d.getOps<ShimDMAAllocationOp>()) {
234 llvm::json::Object shimJSON;
235 auto channelDir = shimDMAMeta.getChannelDirAttr();
236 shimJSON["channelDir"] = attrToJSON(channelDir);
237 auto channelIndex = shimDMAMeta.getChannelIndexAttr();
238 shimJSON["channelIndex"] = attrToJSON(channelIndex);
239 AIE::TileOp tile = shimDMAMeta.getTileOp();
240 if (!tile) {
241 shimDMAMeta.emitError(
242 "shim DMA allocation must reference a valid TileOp");
243 return failure();
244 }
245 auto col = tile.getColAttr();
246 shimJSON["col"] = attrToJSON(col);
247 moduleJSON[shimDMAMeta.getSymName()] =
248 llvm::json::Value(std::move(shimJSON));
249 }
250 llvm::json::Value topv(std::move(moduleJSON));
251 std::string ret;
252 llvm::raw_string_ostream ss(ret);
253 ss << llvm::formatv("{0:2}", topv) << "\n";
254 output << ss.str();
255 return success();
256 },
257 registerDialects);
