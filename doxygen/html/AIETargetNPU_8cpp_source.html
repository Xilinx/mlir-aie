38// Example:
39// - instructions = {3,4,5}
40// - tailSize = 2
41// instructions becomes {3,4,5,0,0} and
42// a mutable reference to the tail {0,0} is returned.
43llvm::MutableArrayRef<uint32_t>
44reserveAndGetTail(std::vector<uint32_t> &instructions, uint64_t tailSize) {
45 auto oldSize = instructions.size();
46 auto newSize = oldSize + tailSize;
47 instructions.resize(newSize, 0);
48 return llvm::MutableArrayRef<uint32_t>(instructions.data() + oldSize,
49 tailSize);
50}

52void appendSync(std::vector<uint32_t> &instructions, NpuSyncOp op) {
53
54 auto words = reserveAndGetTail(instructions, 4);
55
56 // XAIE_IO_CUSTOM_OP_TCT
57 words[0] = TXN_OPC_TCT;
58
59 words[1] = words.size() * sizeof(uint32_t); // Operation Size
60
61 words[2] |= static_cast<uint32_t>(op.getDirection()) & 0xff;
62 words[2] |= (op.getRow() & 0xff) << 8;
63 words[2] |= (op.getColumn() & 0xff) << 16;
64
65 words[3] |= (op.getRowNum() & 0xff) << 8;
66 words[3] |= (op.getColumnNum() & 0xff) << 16;
67 words[3] |= (op.getChannel() & 0xff) << 24;
68}

70void appendWrite32(std::vector<uint32_t> &instructions, NpuWrite32Op op) {
71
72 auto words = reserveAndGetTail(instructions, 3);
73
74 if (op.getBuffer()) {
75 op.emitOpError("Cannot translate symbolic address");
76 return;
77 }
78
79 // XAIE_IO_WRITE
80 words[0] = TXN_OPC_WRITE;
81 words[1] = op.getAddress();
82 auto col = op.getColumn();
83 auto row = op.getRow();
84 if (col && row) {
85 const AIETargetModel &tm = op->getParentOfType<DeviceOp>().getTargetModel();
86 words[1] = ((*col & 0xff) << tm.getColumnShift()) |
87 ((*row & 0xff) << tm.getRowShift()) | (words[1] & 0xFFFFF);
88 }
89 words[2] = op.getValue(); // Value
90}

116void appendAddressPatch(std::vector<uint32_t> &instructions,
117 NpuAddressPatchOp op) {
118
119 auto words = reserveAndGetTail(instructions, 6);
120
121 // XAIE_IO_CUSTOM_OP_DDR_PATCH
122 words[0] = TXN_OPC_DDR_PATCH;
123 words[1] = words.size() * sizeof(uint32_t); // Operation Size
124
125 words[2] = op.getAddr();
126
127 words[3] = op.getArgIdx();
128
129 words[4] = op.getArgPlus();
130 words[5] = 0;
131}

133void appendBlockWrite(std::vector<uint32_t> &instructions, NpuBlockWriteOp op) {
134
135 Value memref = op.getData();
136 int64_t width = cast<MemRefType>(memref.getType()).getElementTypeBitWidth();
137 if (width != 32) {
138 op.emitWarning("Only 32-bit data type is supported for now");
139 return;
140 }
141
142 memref::GetGlobalOp getGlobal = memref.getDefiningOp<memref::GetGlobalOp>();
143 if (!getGlobal) {
144 op.emitError("Only MemRefs from memref.get_global are supported");
145 return;
146 }
147
148 auto global = dyn_cast_if_present<memref::GlobalOp>(
149 op->getParentOfType<AIE::DeviceOp>().lookupSymbol(getGlobal.getName()));
150 if (!global) {
151 op.emitError("Global symbol not found");
152 return;
153 }
154
155 auto initVal = global.getInitialValue();
156 if (!initVal) {
157 op.emitError("Global symbol has no initial value");
158 return;
159 }
160
161 auto data = dyn_cast<DenseIntElementsAttr>(*initVal);
162 if (!data) {
163 op.emitError("Global symbol initial value is not a dense int array");
164 return;
165 }
166
167 auto words = reserveAndGetTail(instructions, data.size() + 3);
168
169 // XAIE_IO_BLOCKWRITE
170 words[0] = TXN_OPC_BLOCKWRITE;
171 words[1] = op.getAddress();
172 auto col = op.getColumn();
173 auto row = op.getRow();
174 if (col && row) {
175 const AIETargetModel &tm = op->getParentOfType<DeviceOp>().getTargetModel();
176 words[1] = ((*col & 0xff) << tm.getColumnShift()) |
177 ((*row & 0xff) << tm.getRowShift()) | (words[1] & 0xFFFFF);
178 }
179 words[2] = words.size() * sizeof(uint32_t); // Operation Size
180
181 unsigned i = 3;
182 for (auto d : data)
183 words[i++] = d.getZExtValue();
184}

188LogicalResult
189xilinx::AIE::AIETranslateNpuToBinary(ModuleOp module,
190 std::vector<uint32_t> &instructions,
191 StringRef sequenceName) {
192
193 auto words = reserveAndGetTail(instructions, 4);
194
195 DeviceOp deviceOp = *module.getOps<DeviceOp>().begin();
196 const AIETargetModel &tm = deviceOp.getTargetModel();
197
198 // setup txn header
199 uint8_t major = 1;
200 uint8_t minor = 0;
201 uint8_t devGen = 3;
202 uint8_t numRows = tm.rows();
203 uint8_t numCols = tm.columns();
204 uint8_t numMemTileRows = tm.getNumMemTileRows();
205 uint32_t count = 0;
206 words[0] = (numRows << 24) | (devGen << 16) | (minor << 8) | major;
207 words[1] = (numMemTileRows << 8) | numCols;
208
209 auto sequenceOps = deviceOp.getOps<AIEX::RuntimeSequenceOp>();
210 for (auto seq : sequenceOps) {
211 if (sequenceName.size() && sequenceName != seq.getSymName())
212 continue;
213 Block &entry = seq.getBody().front();
214 for (auto &o : entry) {
215 llvm::TypeSwitch<Operation *>(&o)
216 .Case<NpuSyncOp>([&](auto op) {
217 count++;
218 appendSync(instructions, op);
219 })
220 .Case<NpuWrite32Op>([&](auto op) {
221 count++;
222 appendWrite32(instructions, op);
223 })
224 .Case<NpuBlockWriteOp>([&](auto op) {
225 count++;
226 appendBlockWrite(instructions, op);
227 })
228 .Case<NpuMaskWrite32Op>([&](auto op) {
229 count++;
230 appendMaskWrite32(instructions, op);
231 })
232 .Case<NpuAddressPatchOp>([&](auto op) {
233 count++;
234 appendAddressPatch(instructions, op);
235 });
236 }
237 }
238
239 // write size fields of the txn header
240 instructions[2] = count;
241 instructions[3] = instructions.size() * sizeof(uint32_t); // size of the txn
242 return success();
243}

257LogicalResult xilinx::AIE::AIETranslateControlPacketsToUI32Vec(
258 ModuleOp module, std::vector<uint32_t> &instructions,
259 StringRef sequenceName) {
260 DeviceOp deviceOp = *module.getOps<DeviceOp>().begin();
261 auto sequenceOps = deviceOp.getOps<AIEX::RuntimeSequenceOp>();
262 for (auto seq : sequenceOps) {
263 if (sequenceName.size() && sequenceName != seq.getSymName())
264 continue;
265 Block &entry = seq.getBody().front();
266 for (auto &o : entry) {
267 llvm::TypeSwitch<Operation *>(&o).Case<NpuControlPacketOp>([&](auto op) {
268 uint32_t size = 0;
269 auto data = op.getData();
270 auto length = op.getLength();
271 if (data)
272 size = data->size();
273 auto words = reserveAndGetTail(instructions, 1 + size);
274 if (!data && length)
275 size = *length;
276 auto parity = [](uint32_t n) {
277 uint32_t p = 0;
278 while (n) {
279 p += n & 1;
280 n >>= 1;
281 }
282 return (p % 2) == 0;
283 };
284 uint32_t addr = op.getAddress() & 0xFFFFF;
285 uint32_t beats = size - 1;
286 uint32_t opc = op.getOpcode();
287 uint32_t id = op.getStreamId();
288 uint32_t hdr = id << 24 | opc << 22 | beats << 20 | addr;
289 words[0] = hdr | (0x1 & parity(hdr)) << 31;
290 if (opc == 0x0 || opc == 0x2)
291 for (unsigned i = 0; i < size; i++)
292 words[i + 1] = data.value()[i];
293 });
294 }
295 }
296 return success();
297}
