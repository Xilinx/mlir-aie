24 struct AIELocalizeLocksPass : AIELocalizeLocksBase<AIELocalizeLocksPass> {
25  void getDependentDialects(DialectRegistry &registry) const override {
26  registry.insert<arith::ArithDialect>();
27  }
28  void runOnOperation() override {
29 
30  DeviceOp deviceOp = getOperation();
31 
32  for (auto coreOp : deviceOp.getOps<CoreOp>()) {
33  // Collect the locks used in this core.
34  const auto &targetModel = getTargetModel(coreOp);
35 
36  auto thisTile = dyn_cast<TileOp>(coreOp.getTile().getDefiningOp());
37  int col = thisTile.colIndex();
38  int row = thisTile.rowIndex();
39 
40  // Find the neighboring tiles
41  SmallVector<TileOp, 4> accessibleTiles;
42  for (auto tile : deviceOp.getOps<TileOp>())
43  if (int dstRow = tile.rowIndex();
44  targetModel.isLegalMemAffinity(col, row, tile.colIndex(), dstRow))
45  accessibleTiles.push_back(tile);
46 
47  for (auto tile : accessibleTiles) {
48  int dstCol = tile.colIndex();
49  int dstRow = tile.rowIndex();
50  int cardinalMemOffset = 0;
51 
52  const auto &targetModel = getTargetModel(tile);
53  int numLocks = targetModel.getNumLocks(dstCol, dstRow);
54  for (auto user : tile.getResult().getUsers())
55  if (auto lock = dyn_cast<LockOp>(user)) {
56  if (targetModel.isMemSouth(col, row, dstCol, dstRow))
57  cardinalMemOffset = 0;
58  else if (targetModel.isMemWest(col, row, dstCol, dstRow))
59  cardinalMemOffset = numLocks;
60  else if (targetModel.isMemNorth(col, row, dstCol, dstRow))
61  cardinalMemOffset = 2 * numLocks;
62  else if (targetModel.isMemEast(col, row, dstCol, dstRow))
63  cardinalMemOffset = 3 * numLocks;
64  else
65  llvm_unreachable("Found illegal lock user!");
66 
67  int localLockIndex = cardinalMemOffset + lock.getLockIDValue();
68 
69  OpBuilder builder =
70  OpBuilder::atBlockBegin(&coreOp.getBody().front());
71 
72  Value coreLockIDValue = builder.create<arith::ConstantIndexOp>(
73  builder.getUnknownLoc(), localLockIndex);
74  lock.getResult().replaceUsesWithIf(
75  coreLockIDValue, [&](OpOperand &opOperand) {
76  return opOperand.getOwner()->getParentOp() == coreOp;
77  });
78  }
79  }
80  }
81  }
82 };
