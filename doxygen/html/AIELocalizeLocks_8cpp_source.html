24struct AIELocalizeLocksPass : AIELocalizeLocksBase<AIELocalizeLocksPass> {
25 void getDependentDialects(DialectRegistry &registry) const override {
26 registry.insert<arith::ArithDialect>();
27 }
28 void runOnOperation() override {
29
30 DeviceOp deviceOp = getOperation();
31
32 for (auto coreOp : deviceOp.getOps<CoreOp>()) {
33 // Collect the locks used in this core.
34 const auto &targetModel = getTargetModel(coreOp);
35
36 auto thisTile = dyn_cast<TileOp>(coreOp.getTile().getDefiningOp());
37 int col = thisTile.colIndex();
38 int row = thisTile.rowIndex();
39
40 // Find the neighboring tiles
41 SmallVector<TileOp, 4> accessibleTiles;
42 for (auto tile : deviceOp.getOps<TileOp>())
43 if (int dstRow = tile.rowIndex();
44 targetModel.isLegalMemAffinity(col, row, tile.colIndex(), dstRow))
45 accessibleTiles.push_back(tile);
46
47 for (auto tile : accessibleTiles) {
48 int dstCol = tile.colIndex();
49 int dstRow = tile.rowIndex();
50
51 const auto &targetModel = getTargetModel(tile);
52 for (auto user : tile.getResult().getUsers())
53 if (auto lock = dyn_cast<LockOp>(user)) {
54 // At this point, we are iterating over all locks that are
55 // accessible from within the current core coreOp. We only need to
56 // localize the current lock if it is used within the core. Note
57 // that this pass is to be applied after the scf-to-cf lowering, so
58 // it suffices to check if the parent of a UseLockOp is coreOp.
59 if (llvm::none_of(lock.getResult().getUsers(),
60 [&](Operation *user) {
61 return user->getParentOp() == coreOp;
62 }))
63 continue;
64
65 auto lockIndexOffset =
66 targetModel.getLockLocalBaseIndex(col, row, dstCol, dstRow);
67 if (!lockIndexOffset)
68 llvm_unreachable("Found illegal lock user!");
69
70 int localLockIndex =
71 lockIndexOffset.value() + lock.getLockIDValue();
72
73 OpBuilder builder =
74 OpBuilder::atBlockBegin(&coreOp.getBody().front());
75
76 Value coreLockIDValue = arith::ConstantIndexOp::create(
77 builder, builder.getUnknownLoc(), localLockIndex);
78 lock.getResult().replaceUsesWithIf(
79 coreLockIDValue, [&](OpOperand &opOperand) {
80 return opOperand.getOwner()->getParentOp() == coreOp;
81 });
82 }
83 }
84 }
85 }
86};
