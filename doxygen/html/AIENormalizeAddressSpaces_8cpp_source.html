26Type memRefToDefaultAddressSpace(Type t) {
27 if (auto memRefType = llvm::dyn_cast<MemRefType>(t);
28 memRefType && memRefType.getMemorySpace() != nullptr) {
29 // Preserve ptr::GenericSpaceAttr - it's needed for ptr dialect
30 // transformations
31 if (llvm::isa<ptr::GenericSpaceAttr>(memRefType.getMemorySpace()))
32 return t; // Keep as-is
33
34 // Remove other memory spaces
35 return MemRefType::get(memRefType.getShape(), memRefType.getElementType(),
36 memRefType.getLayout(), nullptr /* Address Space */);
37 }
38 return t;
39}

43struct AIENormalizeAddressSpacesPass
44 : AIENormalizeAddressSpacesBase<AIENormalizeAddressSpacesPass> {
45 void getDependentDialects(DialectRegistry &registry) const override {
46 registry.insert<func::FuncDialect>();
47 }
48 void runOnOperation() override {
49 DeviceOp device = getOperation();
50
51 TypeConverter converter;
52 converter.addConversion([&](Type type) -> std::optional<Type> {
53 return memRefToDefaultAddressSpace(type);
54 });
55
56 ConversionTarget target(getContext());
57 target.addDynamicallyLegalOp<memref::GlobalOp>([](memref::GlobalOp op) {
58 return op.getType().getMemorySpace() == nullptr;
59 });
60
61 target.addDynamicallyLegalOp<func::FuncOp>([&](func::FuncOp op) {
62 return converter.isSignatureLegal(op.getFunctionType());
63 });
64
65 RewritePatternSet patterns(&getContext());
66 populateWithGenerated(patterns);
67 populateFunctionOpInterfaceTypeConversionPattern<func::FuncOp>(patterns,
68 converter);
69
70 if (failed(applyPartialConversion(device, target, std::move(patterns))))
71 signalPassFailure();
72
73 // Convert any output types to have the default address space
74 device.walk([&](Operation *op) {
75 for (Value r : op->getResults())
76 r.setType(memRefToDefaultAddressSpace(r.getType()));
77 });
78 }
79};
