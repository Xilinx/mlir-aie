46/// Check if a value depends on the given loop induction variable
47/// Uses a cache to avoid exponential recursion on complex dependency chains
48static bool dependsOnLoopIVForHoist(Value val, Value loopIV,
49 DenseMap<Value, bool> &cache) {
50 // Check cache - return cached result if already computed
51 auto it = cache.find(val);
52 if (it != cache.end())
53 return it->second;
54
55 // Mark as being computed (assume false initially to handle recursion)
56 // This prevents infinite recursion in case of cycles (though SSA shouldn't
57 // have cycles)
58 cache[val] = false;
59
60 bool result = false;
61 if (val == loopIV) {
62 result = true;
63 } else if (auto defOp = val.getDefiningOp()) {
64 // Check for operations that use the loop IV in their operands
65 for (Value operand : defOp->getOperands()) {
66 if (dependsOnLoopIVForHoist(operand, loopIV, cache)) {
67 result = true;
68 break;
69 }
70 }
71 }
72
73 // Store the computed result in cache
74 cache[val] = result;
75 return result;
76}
77
78/// Wrapper for dependsOnLoopIVForHoist that manages the cache
79static bool dependsOnLoopIVForHoist(Value val, Value loopIV) {
80 DenseMap<Value, bool> cache;
81 return dependsOnLoopIVForHoist(val, loopIV, cache);
82}

84/// Clone an operation and its operands (recursively) that don't depend on the
85/// loop IV. Uses memoization via the mapping to avoid exponential recursion.
86static Value cloneOpAndOperands(Operation *op, Value loopIV, OpBuilder &builder,
87 IRMapping &mapping) {
88 // Only handle operations with exactly one result
89 if (op->getNumResults() != 1)
90 return Value();
91
92 // If we've already cloned this operation, return the mapped result
93 // This is critical for avoiding exponential recursion
94 if (mapping.contains(op->getResult(0)))
95 return mapping.lookup(op->getResult(0));
96
97 // Check if this operation depends on the loop IV before trying to clone
98 if (dependsOnLoopIVForHoist(op->getResult(0), loopIV))
99 return Value();
100
101 // Clone operands recursively
102 SmallVector<Value> newOperands;
103 for (Value operand : op->getOperands()) {
104 if (auto defOp = operand.getDefiningOp()) {
105 Value clonedOperand = cloneOpAndOperands(defOp, loopIV, builder, mapping);
106 if (!clonedOperand)
107 return Value(); // Failed to clone an operand
108 newOperands.push_back(clonedOperand);
109 } else {
110 // Operand is a block argument or constant (guaranteed not to be the
111 // loop IV due to the dependency check at line 91)
112 newOperands.push_back(operand);
113 }
114 }
115
116 // Clone the operation
117 Operation *clonedOp = builder.clone(*op);
118 clonedOp->setOperands(newOperands);
119
120 // Map the result to enable memoization
121 mapping.map(op->getResult(0), clonedOp->getResult(0));
122 return clonedOp->getResult(0);
123}

125/// Get the total number of elements in a vector type
126static int64_t getVectorNumElements(VectorType vectorType) {
127 int64_t numElements = 1;
128 for (int64_t dim : vectorType.getShape()) {
129 numElements *= dim;
130 }
131 return numElements;
132}

138/// Information about a vector transfer operation
139struct TransferOpInfo {
140 Operation *op;
141 Value base;
142 MemRefType memrefType;
143 VectorType vectorType;
144 SmallVector<Value> indices;
145 int64_t constantStride; // Total constant stride per iteration
146 bool hasIVDependentIndices;
147};

149/// Pattern to hoist vector transfer operations with IV-dependent pointers
150/// out of scf.for loops by using iter_args to track pointer updates
151struct HoistVectorTransferPointersPattern
152 : public OpRewritePattern<scf::ForOp> {
153 using OpRewritePattern<scf::ForOp>::OpRewritePattern;
154
155 LogicalResult matchAndRewrite(scf::ForOp forOp,
156 PatternRewriter &rewriter) const override {
157 Value loopIV = forOp.getInductionVar();
158 Location loc = forOp.getLoc();
159
160 // Collect all vector transfer operations with IV-dependent indices
161 SmallVector<TransferOpInfo> transferOps;
162
163 for (Operation &op : forOp.getBody()->without_terminator()) {
164 Value base;
165 VectorType vectorType;
166 SmallVector<Value> indices;
167
168 if (auto readOp = dyn_cast<vector::TransferReadOp>(&op)) {
169 base = readOp.getBase();
170 vectorType = readOp.getVectorType();
171 indices.assign(readOp.getIndices().begin(), readOp.getIndices().end());
172 } else if (auto writeOp = dyn_cast<vector::TransferWriteOp>(&op)) {
173 base = writeOp.getBase();
174 vectorType = writeOp.getVectorType();
175 indices.assign(writeOp.getIndices().begin(),
176 writeOp.getIndices().end());
177 } else {
178 continue;
179 }
180
181 auto memrefType = dyn_cast<MemRefType>(base.getType());
182 if (!memrefType)
183 continue;
184
185 // Check if any indices depend on loop IV and compute constant stride
186 bool hasIVDependentIndices = false;
187 int64_t constantStride = 0;
188
189 // Get the loop step to account for in stride calculation
190 auto stepCst = forOp.getConstantStep();
191 int64_t loopStep =
192 stepCst.has_value() ? stepCst.value().getSExtValue() : 1;
193
194 for (size_t dimIdx = 0; dimIdx < indices.size(); ++dimIdx) {
195 Value idx = indices[dimIdx];
196 if (dependsOnLoopIVForHoist(idx, loopIV)) {
197 hasIVDependentIndices = true;
198
199 // Calculate the stride for this dimension
200 int64_t dimStride = 1;
201 bool hasDynamicStride = false;
202 for (size_t j = dimIdx + 1;
203 j < static_cast<size_t>(memrefType.getRank()); ++j) {
204 int64_t dimSize = memrefType.getShape()[j];
205 if (dimSize == ShapedType::kDynamic) {
206 hasDynamicStride = true;
207 break;
208 }
209 dimStride *= dimSize;
210 }
211
212 // Multiply by loop step - the stride per iteration is:
213 // (elements per dimension) * (loop step)
214 if (!hasDynamicStride)
215 constantStride += dimStride * loopStep;
216 else
217 hasIVDependentIndices = false; // Can't hoist if stride is dynamic
218 }
219 }
220
221 transferOps.push_back({&op, base, memrefType, vectorType, indices,
222 constantStride, hasIVDependentIndices});
223 }
224
225 // If there are no transfer ops, don't modify
226 if (transferOps.empty())
227 return failure();
228
229 // Prepare to add iter_args for each transfer operation with IV-dependent
230 // indices
231 SmallVector<Value> newInitArgs;
232 SmallVector<Value> flatMemrefs;
233
234 for (const auto &info : transferOps) {
235 if (!info.hasIVDependentIndices)
236 continue;
237
238 // Flatten the memref if needed
239 rewriter.setInsertionPoint(forOp);
240 Value flatMemref = info.base;
241 if (info.memrefType.getRank() > 1) {
242 int64_t totalSize = 1;
243 for (int64_t dim : info.memrefType.getShape()) {
244 if (dim == ShapedType::kDynamic)
245 return failure(); // Dynamic memref shapes not supported
246 totalSize *= dim;
247 }
248
249 // Preserve strided layout if present
250 MemRefType flatMemrefType;
251 if (auto stridedLayout = dyn_cast_or_null<StridedLayoutAttr>(
252 info.memrefType.getLayout())) {
253 // The collapsed stride is the innermost stride (last element)
254 int64_t collapsedStride = stridedLayout.getStrides().back();
255 int64_t offset = stridedLayout.getOffset();
256
257 auto newLayout = StridedLayoutAttr::get(rewriter.getContext(), offset,
258 {collapsedStride});
259 flatMemrefType =
260 MemRefType::get({totalSize}, info.memrefType.getElementType(),
261 newLayout, info.memrefType.getMemorySpace());
262 } else {
263 flatMemrefType =
264 MemRefType::get({totalSize}, info.memrefType.getElementType(),
265 AffineMap(), info.memrefType.getMemorySpace());
266 }
267
268 SmallVector<ReassociationIndices> reassociation;
269 ReassociationIndices allDims;
270 for (size_t i = 0; i < static_cast<size_t>(info.memrefType.getRank());
271 ++i) {
272 allDims.push_back(i);
273 }
274 reassociation.push_back(allDims);
275
276 flatMemref = memref::CollapseShapeOp::create(
277 rewriter, loc, flatMemrefType, info.base, reassociation);
278 }
279 flatMemrefs.push_back(flatMemref);
280
281 // Compute base pointer (with zeros for IV-dependent parts)
282 int64_t rank = info.memrefType.getRank();
283 AffineExpr linearExpr = rewriter.getAffineConstantExpr(0);
284 int64_t stride = 1;
285 for (int64_t i = rank - 1; i >= 0; --i) {
286 linearExpr = linearExpr + rewriter.getAffineDimExpr(i) * stride;
287 if (i > 0)
288 stride *= info.memrefType.getShape()[i];
289 }
290 auto linearMap = AffineMap::get(rank, 0, linearExpr);
291
292 // For IV-dependent indices, evaluate them at the loop's lower bound
293 // to preserve constant offsets (e.g., %iv+1 becomes lowerBound+1)
294 SmallVector<Value> evaluatedIndices;
295 IRMapping indexMapping;
296 for (Value idx : info.indices) {
297 if (dependsOnLoopIVForHoist(idx, loopIV)) {
298 // Clone the computation with the IV replaced by lower bound
299 if (auto affineOp = idx.getDefiningOp<affine::AffineApplyOp>()) {
300 SmallVector<Value> mappedOperands;
301 for (Value operand : affineOp.getMapOperands()) {
302 if (operand == loopIV)
303 mappedOperands.push_back(forOp.getLowerBound());
304 else
305 mappedOperands.push_back(operand);
306 }
307 Value evaluatedIdx = affine::AffineApplyOp::create(
308 rewriter, loc, affineOp.getAffineMap(), mappedOperands);
309 evaluatedIndices.push_back(evaluatedIdx);
310 } else {
311 // Direct IV usage - just use lower bound
312 evaluatedIndices.push_back(forOp.getLowerBound());
313 }
314 } else {
315 // Index doesn't depend on IV, clone it
316 if (auto defOp = idx.getDefiningOp()) {
317 Value clonedIdx =
318 cloneOpAndOperands(defOp, loopIV, rewriter, indexMapping);
319 if (clonedIdx)
320 evaluatedIndices.push_back(clonedIdx);
321 else
322 evaluatedIndices.push_back(idx);
323 } else {
324 evaluatedIndices.push_back(idx);
325 }
326 }
327 }
328
329 Value basePointer = affine::AffineApplyOp::create(
330 rewriter, loc, linearMap, evaluatedIndices);
331
332 newInitArgs.push_back(basePointer);
333 }
334
335 // If there are no IV-dependent transfers, just process them to flatten
336 // vectors
337 if (newInitArgs.empty()) {
338 // Check if any transfer needs flattening (avoid infinite rewrites)
339 bool needsFlattening = false;
340 bool hasProcessableTransfers = false;
341 for (const auto &info : transferOps) {
342 // Skip if base is defined inside the loop (e.g., a subview)
343 // We can't hoist these
344 if (info.base.getDefiningOp() &&
345 forOp->isProperAncestor(info.base.getDefiningOp()))
346 continue;
347
348 hasProcessableTransfers = true;
349
350 // Check if this transfer has already been flattened
351 // (flattened transfers use 1D identity map)
352 if (auto readOp = dyn_cast<vector::TransferReadOp>(info.op)) {
353 if (readOp.getPermutationMap().getNumDims() != 1)
354 needsFlattening = true;
355 } else if (auto writeOp = dyn_cast<vector::TransferWriteOp>(info.op)) {
356 if (writeOp.getPermutationMap().getNumDims() != 1)
357 needsFlattening = true;
358 }
359 }
360
361 // If there are no processable transfers (all bases defined in loop)
362 // or nothing needs flattening, bail out
363 if (!hasProcessableTransfers || !needsFlattening)
364 return failure();
365
366 // First, create flattened memrefs outside the loop for bases not defined
367 // inside
368 DenseMap<Value, Value> baseFlatMemrefs;
369 rewriter.setInsertionPoint(forOp);
370 for (const auto &info : transferOps) {
371 if (baseFlatMemrefs.count(info.base))
372 continue;
373
374 // Skip if base is defined inside the loop (e.g., a subview)
375 if (info.base.getDefiningOp() &&
376 forOp->isProperAncestor(info.base.getDefiningOp()))
377 continue;
378
379 Value flatMemref = info.base;
380 if (info.memrefType.getRank() > 1) {
381 int64_t totalSize = 1;
382 for (int64_t dim : info.memrefType.getShape()) {
383 totalSize *= dim;
384 }
385
386 // Preserve strided layout if present
387 MemRefType flatMemrefType;
388 if (auto stridedLayout = dyn_cast_or_null<StridedLayoutAttr>(
389 info.memrefType.getLayout())) {
390 int64_t collapsedStride = stridedLayout.getStrides().back();
391 int64_t offset = stridedLayout.getOffset();
392
393 auto newLayout = StridedLayoutAttr::get(rewriter.getContext(),
394 offset, {collapsedStride});
395 flatMemrefType =
396 MemRefType::get({totalSize}, info.memrefType.getElementType(),
397 newLayout, info.memrefType.getMemorySpace());
398 } else {
399 flatMemrefType =
400 MemRefType::get({totalSize}, info.memrefType.getElementType(),
401 AffineMap(), info.memrefType.getMemorySpace());
402 }
403
404 SmallVector<ReassociationIndices> reassociation;
405 ReassociationIndices allDims;
406 for (size_t i = 0; i < static_cast<size_t>(info.memrefType.getRank());
407 ++i) {
408 allDims.push_back(i);
409 }
410 reassociation.push_back(allDims);
411 flatMemref = memref::CollapseShapeOp::create(
412 rewriter, loc, flatMemrefType, info.base, reassociation);
413 }
414 baseFlatMemrefs[info.base] = flatMemref;
415 }
416
417 // Process all transfers without using iter_args
418 bool madeChanges = false;
419 for (const auto &info : transferOps) {
420 // Skip if base is defined inside the loop
421 if (info.base.getDefiningOp() &&
422 forOp->isProperAncestor(info.base.getDefiningOp()))
423 continue;
424
425 // Skip if we don't have a flattened version
426 if (!baseFlatMemrefs.count(info.base))
427 continue;
428
429 rewriter.setInsertionPoint(info.op);
430
431 // Flatten vector type
432 int64_t numElements = getVectorNumElements(info.vectorType);
433 VectorType flatVectorType =
434 VectorType::get({numElements}, info.vectorType.getElementType());
435
436 // Get the flattened memref
437 Value flatMemref = baseFlatMemrefs[info.base];
438
439 // Compute pointer from indices
440 int64_t rank = info.memrefType.getRank();
441 AffineExpr linearExpr = rewriter.getAffineConstantExpr(0);
442 int64_t stride = 1;
443 for (int64_t i = rank - 1; i >= 0; --i) {
444 linearExpr = linearExpr + rewriter.getAffineDimExpr(i) * stride;
445 if (i > 0)
446 stride *= info.memrefType.getShape()[i];
447 }
448 auto linearMap = AffineMap::get(rank, 0, linearExpr);
449
450 Value currentPointer = affine::AffineApplyOp::create(
451 rewriter, loc, linearMap, info.indices);
452
453 // Transform the transfer operation
454 AffineMap identityMap1D = AffineMap::get(
455 1, 0, rewriter.getAffineDimExpr(0), rewriter.getContext());
456 auto inBoundsAttr = rewriter.getBoolArrayAttr({true});
457
458 if (auto readOp = dyn_cast<vector::TransferReadOp>(info.op)) {
459 Value flatRead = vector::TransferReadOp::create(
460 rewriter, loc, flatVectorType, flatMemref,
461 ValueRange{currentPointer}, AffineMapAttr::get(identityMap1D),
462 readOp.getPadding(),
463 /*mask=*/Value(), inBoundsAttr);
464 Value shapedRead = vector::ShapeCastOp::create(
465 rewriter, loc, info.vectorType, flatRead);
466 rewriter.replaceOp(readOp, shapedRead);
467 madeChanges = true;
468 } else if (auto writeOp = dyn_cast<vector::TransferWriteOp>(info.op)) {
469 Value flatValue = vector::ShapeCastOp::create(
470 rewriter, loc, flatVectorType, writeOp.getVector());
471 rewriter.replaceOpWithNewOp<vector::TransferWriteOp>(
472 writeOp, flatValue, flatMemref, ValueRange{currentPointer},
473 AffineMapAttr::get(identityMap1D), /*mask=*/Value(),
474 inBoundsAttr);
475 madeChanges = true;
476 }
477 }
478 return madeChanges ? success() : failure();
479 }
480
481 // Use replaceWithAdditionalYields to add pointer iter_args
482 auto yieldValuesFn =
483 [&](OpBuilder &b, Location yieldLoc,
484 ArrayRef<BlockArgument> newBbArgs) -> SmallVector<Value> {
485 SmallVector<Value> yieldValues;
486
487 // Process each transfer operation with IV-dependent indices
488 size_t iterArgIdx = 0;
489 for (size_t i = 0; i < transferOps.size(); ++i) {
490 const auto &info = transferOps[i];
491 if (!info.hasIVDependentIndices)
492 continue;
493
494 BlockArgument ptrIterArg =
495 newBbArgs[newBbArgs.size() - newInitArgs.size() + iterArgIdx];
496 Value flatMemref = flatMemrefs[iterArgIdx];
497
498 // Flatten vector type
499 int64_t numElements = getVectorNumElements(info.vectorType);
500 VectorType flatVectorType =
501 VectorType::get({numElements}, info.vectorType.getElementType());
502
503 // Transform the transfer operation to use the iter_arg pointer
504 b.setInsertionPoint(info.op);
505
506 AffineMap identityMap1D =
507 AffineMap::get(1, 0, b.getAffineDimExpr(0), b.getContext());
508 auto inBoundsAttr = b.getBoolArrayAttr({true});
509
510 if (auto readOp = dyn_cast<vector::TransferReadOp>(info.op)) {
511 Value flatRead = vector::TransferReadOp::create(
512 b, loc, flatVectorType, flatMemref, ValueRange{ptrIterArg},
513 AffineMapAttr::get(identityMap1D), readOp.getPadding(),
514 /*mask=*/Value(), inBoundsAttr);
515 Value shapedRead =
516 vector::ShapeCastOp::create(b, loc, info.vectorType, flatRead);
517 rewriter.replaceOp(readOp, shapedRead);
518 } else if (auto writeOp = dyn_cast<vector::TransferWriteOp>(info.op)) {
519 Value flatValue = vector::ShapeCastOp::create(b, loc, flatVectorType,
520 writeOp.getVector());
521 rewriter.replaceOpWithNewOp<vector::TransferWriteOp>(
522 writeOp, flatValue, flatMemref, ValueRange{ptrIterArg},
523 AffineMapAttr::get(identityMap1D), /*mask=*/Value(),
524 inBoundsAttr);
525 }
526
527 // Compute next pointer value: current_ptr + constant_stride
528 Value strideConst =
529 arith::ConstantIndexOp::create(b, yieldLoc, info.constantStride);
530 Value nextPtr =
531 arith::AddIOp::create(b, yieldLoc, ptrIterArg, strideConst);
532 yieldValues.push_back(nextPtr);
533
534 iterArgIdx++;
535 }
536
537 return yieldValues;
538 };
539
540 // Create new loop with additional iter_args for pointers
541 FailureOr<LoopLikeOpInterface> newLoopResult =
542 cast<LoopLikeOpInterface>(forOp.getOperation())
543 .replaceWithAdditionalYields(
544 rewriter, newInitArgs, // new init operands (base pointers)
545 true, // replace uses in loop
546 yieldValuesFn);
547
548 if (failed(newLoopResult))
549 return failure();
550
551 return success();
552 }

559struct AIEHoistVectorTransferPointersPass
560 : AIEHoistVectorTransferPointersBase<AIEHoistVectorTransferPointersPass> {
561 void getDependentDialects(DialectRegistry &registry) const override {
562 registry.insert<affine::AffineDialect, arith::ArithDialect,
563 memref::MemRefDialect, scf::SCFDialect,
564 vector::VectorDialect>();
565 }
566
567 void runOnOperation() override {
568 ModuleOp moduleOp = getOperation();
569 MLIRContext *context = &getContext();
570
571 RewritePatternSet patterns(context);
572 patterns.add<HoistVectorTransferPointersPattern>(context);
573
574 // Apply patterns to the entire module - the pattern will only match scf.for
575 // ops within aie.core regions
576 if (failed(applyPatternsGreedily(moduleOp, std::move(patterns))))
577 signalPassFailure();
578 }
