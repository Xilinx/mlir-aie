25std::string getModuleForTile(const AIETargetModel &model, TileID tile,
26 bool isMem) {
27 if (model.isShimNOCorPLTile(tile.col, tile.row))
28 return "shim";
29 if (model.isMemTile(tile.col, tile.row))
30 return "memory_tile";
31 return isMem ? std::string("memory") : std::string("core");
32}

148bool AIE1TargetModel::isLegalMemAffinity(int coreCol, int coreRow, int memCol,
149 int memRow) const {
150 bool IsEvenRow = ((coreRow % 2) == 0);
151
152 bool IsMemWest = (isWest(coreCol, coreRow, memCol, memRow) && !IsEvenRow) ||
153 (isInternal(coreCol, coreRow, memCol, memRow) && IsEvenRow);
154
155 bool IsMemEast = (isEast(coreCol, coreRow, memCol, memRow) && IsEvenRow) ||
156 (isInternal(coreCol, coreRow, memCol, memRow) && !IsEvenRow);
157
158 bool IsMemNorth = isNorth(coreCol, coreRow, memCol, memRow);
159 bool IsMemSouth = isSouth(coreCol, coreRow, memCol, memRow);
160
161 return IsMemSouth || IsMemNorth || IsMemWest || IsMemEast;
162}

164uint64_t AIE1TargetModel::getDmaBdAddress(int col, int row, uint32_t bd_id,
165 int channel,
166 AIE::DMAChannelDir direction) const {
167 uint32_t offset = 0;
168 if (isShimNOCTile(col, row)) {
169 offset = 0x0001D000 + (bd_id * 0x14);
170 } else if (isCoreTile(col, row)) {
171 offset = 0x0001D000 + (bd_id * 0x20);
172 } else {
173 llvm_unreachable(
174 "AIE1TargetModel::getDmaBdAddress called for non-DMA tile");
175 }
176 return ((col & 0xff) << getColumnShift()) | ((row & 0xff) << getRowShift()) |
177 offset;
178}

188uint32_t
189AIE1TargetModel::getDmaControlAddress(int col, int row, int channel,
190 AIE::DMAChannelDir direction) const {
191 uint32_t offset = 0;
192 if (isShimNOCTile(col, row))

204 offset;

208AIE1TargetModel::getNumDestSwitchboxConnections(int col, int row,
209 WireBundle bundle) const {
210 if (isShimNOCTile(col, row) || isShimPLTile(col, row))
211 switch (bundle) {
212 case WireBundle::FIFO:
213 return 2;
214 case WireBundle::North:
215 return 6;
216 case WireBundle::West: {

218 return 0;
219 return 4;
220 }

225 return 0;
226 return 4;

231 return 0;
232 }

235 case WireBundle::Core:
236 case WireBundle::DMA:
237 case WireBundle::FIFO:
238 return 2;
239 case WireBundle::North: {
240 if (row == rows() - 1)
241 return 0;
242 return 6;
243 }
244 case WireBundle::West: {
245 if (col == 0)
246 return 0;
247 return 4;
248 }
249 case WireBundle::South:
250 return 4;
251 case WireBundle::East: {
252 if (col == columns() - 1)
253 return 0;
254 return 4;
255 }
256 case WireBundle::TileControl:
257 return 1;
258 default:
259 return 0;
260 }

359bool AIE1TargetModel::isLegalTileConnection(int col, int row,
360 WireBundle srcBundle, int srcChan,
361 WireBundle dstBundle,
362 int dstChan) const {
363 // Check Channel Id within the range
364 if (srcChan >= int(getNumSourceSwitchboxConnections(col, row, srcBundle)))
365 return false;
366 if (dstChan >= int(getNumDestSwitchboxConnections(col, row, dstBundle)))
367 return false;
368
369 // Memtile
370 if (isMemTile(col, row)) {
371 return false;
372 }
373 // Shimtile
374 else if (isShimNOCorPLTile(col, row)) {
375 if (srcBundle == WireBundle::Trace)
376 return dstBundle == WireBundle::South;
377 else
378 return true;
379 }
380 // Coretile
381 else if (isCoreTile(col, row)) {
382 if (srcBundle == WireBundle::Trace)
383 return dstBundle == WireBundle::South;
384 else
385 return true;
386 }
387 return false;
388}

469bool AIE2TargetModel::isLegalMemAffinity(int coreCol, int coreRow, int memCol,
470 int memRow) const {
471
472 bool IsMemWest = isMemWest(coreCol, coreRow, memCol, memRow);
473 bool IsMemEast = isMemEast(coreCol, coreRow, memCol, memRow);
474 bool IsMemNorth = isMemNorth(coreCol, coreRow, memCol, memRow);
475 bool IsMemSouth = isMemSouth(coreCol, coreRow, memCol, memRow);
476
477 if (isMemTile(coreCol, coreRow))
478 return isEast(coreCol, coreRow, memCol, memRow) ||
479 isInternal(coreCol, coreRow, memCol, memRow) ||
480 isWest(coreCol, coreRow, memCol, memRow);
481 return (IsMemSouth && !isMemTile(memCol, memRow)) || IsMemNorth ||
482 IsMemWest || IsMemEast;
483}

724bool AIE2TargetModel::isLegalTileConnection(int col, int row,
725 WireBundle srcBundle, int srcChan,
726 WireBundle dstBundle,
727 int dstChan) const {
728 // Check Channel Id within the range
729 if (srcChan >= int(getNumSourceSwitchboxConnections(col, row, srcBundle)))
730 return false;
731 if (dstChan >= int(getNumDestSwitchboxConnections(col, row, dstBundle)))
732 return false;
733
734 // Lambda function to check if a bundle is in a list
735 auto isBundleInList = [](WireBundle bundle,
736 std::initializer_list<WireBundle> bundles) {
737 return std::find(bundles.begin(), bundles.end(), bundle) != bundles.end();
738 };
739
740 // Memtile
741 if (isMemTile(col, row)) {
742 if (srcBundle == WireBundle::DMA) {
743 if (dstBundle == WireBundle::DMA)
744 return srcChan == dstChan;
745 if (isBundleInList(dstBundle, {WireBundle::TileControl, WireBundle::South,
746 WireBundle::North}))
747 return true;
748 }
749 if (srcBundle == WireBundle::TileControl) {
750 if (dstBundle == WireBundle::DMA)
751 return dstChan == 5;
752 if (isBundleInList(dstBundle, {WireBundle::South, WireBundle::North}))
753 return true;
754 }
755 if (isBundleInList(srcBundle, {WireBundle::South, WireBundle::North})) {
756 if (isBundleInList(dstBundle, {WireBundle::DMA, WireBundle::TileControl}))
757 return true;
758 if (isBundleInList(dstBundle, {WireBundle::South, WireBundle::North}))
759 return srcChan == dstChan;
760 }
761 if (srcBundle == WireBundle::Trace) {
762 if (dstBundle == WireBundle::DMA)
763 return dstChan == 5;
764 if (dstBundle == WireBundle::South)
765 return true;
766 }
767 }
768 // Shimtile
769 else if (isShimNOCorPLTile(col, row)) {
770 if (srcBundle == WireBundle::TileControl)
771 return dstBundle != WireBundle::TileControl;
772 if (isBundleInList(srcBundle, {WireBundle::FIFO, WireBundle::South}))
773 return isBundleInList(dstBundle,
774 {WireBundle::TileControl, WireBundle::FIFO,
775 WireBundle::South, WireBundle::West,
776 WireBundle::North, WireBundle::East});
777 if (isBundleInList(srcBundle,
778 {WireBundle::West, WireBundle::North, WireBundle::East}))
779 return (srcBundle == dstBundle)
780 ? (srcChan == dstChan)
781 : isBundleInList(dstBundle,
782 {WireBundle::TileControl, WireBundle::FIFO,
783 WireBundle::South, WireBundle::West,
784 WireBundle::North, WireBundle::East});
785 if (srcBundle == WireBundle::Trace) {
786 if (isBundleInList(dstBundle, {WireBundle::FIFO, WireBundle::South}))
787 return true;
788 if (isBundleInList(dstBundle, {WireBundle::West, WireBundle::East}))
789 return dstChan == 0;
790 }
791 }
792 // Coretile
793 else if (isCoreTile(col, row)) {
794 if (isBundleInList(srcBundle,
795 {WireBundle::DMA, WireBundle::FIFO, WireBundle::South,
796 WireBundle::West, WireBundle::North, WireBundle::East}))
797 if (isBundleInList(dstBundle, {WireBundle::Core, WireBundle::DMA,
798 WireBundle::TileControl, WireBundle::FIFO,
799 WireBundle::South, WireBundle::West,
800 WireBundle::North, WireBundle::East}))
801 return (srcBundle == dstBundle) ? (srcChan == dstChan) : true;
802 if (srcBundle == WireBundle::Core)
803 return dstBundle != WireBundle::Core;
804 if (srcBundle == WireBundle::TileControl)
805 return dstBundle != WireBundle::TileControl &&
806 dstBundle != WireBundle::DMA;
807 if (srcBundle == WireBundle::Trace) {
808 if (dstBundle == WireBundle::DMA)
809 return dstChan == 0;
810 if (isBundleInList(dstBundle, {WireBundle::FIFO, WireBundle::South}))
811 return true;
812 }
813 }
814 return false;
815}

822std::optional<uint32_t>
823AIE2TargetModel::getLocalLockAddress(uint32_t lockId, TileID tile) const {
824 auto computeTileBaseAddress = 0x0001F000;
825 auto memTileBaseAddress = 0x000C0000;
826 auto shimTileBaseAddress = 0x00014000;
827 auto lockAddrOffset = 0x10;
828
829 if (isCoreTile(tile.col, tile.row) &&
830 lockId < getNumLocks(tile.col, tile.row))
831 return computeTileBaseAddress + lockAddrOffset * lockId;
832
833 if (isMemTile(tile.col, tile.row) && lockId < getNumLocks(tile.col, tile.row))
834 return memTileBaseAddress + lockAddrOffset * lockId;
835
836 if (isShimNOCorPLTile(tile.col, tile.row) &&
837 lockId < getNumLocks(tile.col, tile.row))
838 return shimTileBaseAddress + lockAddrOffset * lockId;
839
840 return std::nullopt;
841}

843void AIETargetModel::validate() const {
844 // Every tile in a shimtile row must be a shimtile, and can only be one type
845 // of shim tile.
846 for (int j = 0; j < columns(); j++) {
847 assert(!isMemTile(j, 0) && (isShimPLTile(j, 0) || isShimNOCTile(j, 0)) &&
848 !isCoreTile(j, 0));
849 assert(isShimPLTile(j, 0) ^ isShimNOCTile(j, 0));
850 }
851
852 // Every tile in a memtile row must be a memtile.
853 for (int i = 1; i < 1 + static_cast<int>(getNumMemTileRows()); i++)
854 for (int j = 0; j < columns(); j++)
855 assert(isMemTile(j, i) && !isShimPLTile(j, i) && !isShimNOCTile(j, i) &&
856 !isCoreTile(j, i));
857
858 // Every other tile is a coretile.
859 for (int i = 1 + getNumMemTileRows(); i < rows(); i++)
860 for (int j = 0; j < columns(); j++)
861 assert(!isMemTile(j, i) && !isShimPLTile(j, i) && !isShimNOCTile(j, i) &&
862 isCoreTile(j, i));
863
864 // Looking North, buses must match
865 for (int i = 0; i < rows() - 1; i++)
866 for (int j = 0; j < columns(); j++)
867 assert(getNumSourceSwitchboxConnections(j, i, WireBundle::North) ==
868 getNumDestSwitchboxConnections(j, i + 1, WireBundle::South));
869 // Looking South, buses must match
870 for (int i = 1; i < rows(); i++)
871 for (int j = 0; j < columns(); j++)
872 assert(getNumSourceSwitchboxConnections(j, i, WireBundle::South) ==
873 getNumDestSwitchboxConnections(j, i - 1, WireBundle::North));
874 // Looking East, buses must match
875 for (int i = 0; i < rows(); i++)
876 for (int j = 0; j < columns() - 1; j++)
877 assert(getNumSourceSwitchboxConnections(j, i, WireBundle::East) ==
878 getNumDestSwitchboxConnections(j + 1, i, WireBundle::West));
879 // Looking West, buses must match
880 for (int i = 0; i < rows(); i++)
881 for (int j = 1; j < columns(); j++)
882 assert(getNumSourceSwitchboxConnections(j, i, WireBundle::West) ==
883 getNumDestSwitchboxConnections(j - 1, i, WireBundle::East));
884 // Edges have no connections
885 for (int j = 0; j < columns(); j++)
886 assert(getNumSourceSwitchboxConnections(j, rows() - 1, WireBundle::North) ==
887 0);
888 for (int i = 0; i < rows(); i++)
889 assert(getNumSourceSwitchboxConnections(columns() - 1, i,
890 WireBundle::East) == 0);
891 for (int i = 0; i < rows(); i++)
892 assert(getNumSourceSwitchboxConnections(0, i, WireBundle::West) == 0);
893
894 // FIFOS are consistent
895 for (int i = 0; i < rows(); i++)
896 for (int j = 0; j < columns(); j++)
897 assert(getNumSourceSwitchboxConnections(j, i, WireBundle::FIFO) ==
898 getNumDestSwitchboxConnections(j, i, WireBundle::FIFO));
899}

901std::optional<uint32_t>
902AIETargetModel::getLockLocalBaseIndex(int localCol, int localRow, int lockCol,
903 int lockRow) const {
904 if (isCoreTile(localCol, localRow)) {
905 if (isMemSouth(localCol, localRow, lockCol, lockRow))
906 return 0;
907 if (isMemWest(localCol, localRow, lockCol, lockRow))
908 return getNumLocks(localCol, localRow);
909 if (isMemNorth(localCol, localRow, lockCol, lockRow))
910 return getNumLocks(localCol, localRow) * 2;
911 if (isMemEast(localCol, localRow, lockCol, lockRow))
912 return getNumLocks(localCol, localRow) * 3;
913 }
914
915 if (isMemTile(localCol, localRow)) {
916 if (isWest(localCol, localRow, lockCol, lockRow))
917 return 0;
918 if (isInternal(localCol, localRow, lockCol, lockRow))
919 return getNumLocks(localCol, localRow);
920 if (isEast(localCol, localRow, lockCol, lockRow))
921 return getNumLocks(localCol, localRow) * 2;
922 }
923
924 return std::nullopt;
925}

933 if (isMemWest(localCol, localRow, memCol, memRow))

936 return getMemNorthBaseAddress();
937 if (isMemEast(localCol, localRow, memCol, memRow))

939 }
940

942 if (isWest(localCol, localRow, memCol, memRow))
943 return 0;
944 if (isInternal(localCol, localRow, memCol, memRow))
945 return getMemTileSize();
946 if (isEast(localCol, localRow, memCol, memRow))
947 return getMemTileSize() * 2;
948 }
949
950 return std::nullopt;
