18memref::GlobalOp AIEX::getOrCreateDataMemref(OpBuilder &builder,
19 AIE::DeviceOp dev,
20 mlir::Location loc,
21 ArrayRef<uint32_t> words) {
22 uint32_t num_words = words.size();
23 MemRefType memrefType = MemRefType::get({num_words}, builder.getI32Type());
24 TensorType tensorType =
25 RankedTensorType::get({num_words}, builder.getI32Type());
26 memref::GlobalOp global = nullptr;
27 auto initVal = DenseElementsAttr::get<uint32_t>(tensorType, words);
28 auto otherGlobals = dev.getOps<memref::GlobalOp>();
29 for (auto g : otherGlobals) {
30 if (g.getType() != memrefType)
31 continue;
32 auto otherValue = g.getInitialValue();
33 if (!otherValue)
34 continue;
35 if (*otherValue != initVal)
36 continue;
37 global = g;
38 break;
39 }
40 if (!global) {
41 std::string name = "blockwrite_data_";
42 while (dev.lookupSymbol(name + std::to_string(cachedId)))
43 cachedId++;
44 name += std::to_string(cachedId);
45 global = builder.create<memref::GlobalOp>(
46 loc, name, builder.getStringAttr("private"), memrefType, initVal, true,
47 nullptr);
48 }
49 return global;
50}
