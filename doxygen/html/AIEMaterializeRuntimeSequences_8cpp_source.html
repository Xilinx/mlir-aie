33struct RuntimeCallGraphCyclicityAnalysis {
34 AnalysisManager &analysisManager;
35
36 // if invalid, analysis failed and results should not be considered
37 bool isValid = false;
38
39 // Call graph is cyclic
40 bool isCyclic = false;
41
42 RuntimeCallGraphCyclicityAnalysis(Operation *op, AnalysisManager &am)
43 : analysisManager(am) {
44 AIE::RuntimeSequenceOp runtimeSequenceOp =
45 llvm::dyn_cast<AIE::RuntimeSequenceOp>(op);
46 if (!runtimeSequenceOp) {
47 op->emitError("RuntimeCallGraphCyclicityAnalysis can only be called on "
48 "aiex.runtime_sequence operations.");
49 return;
50 }
51
52 // Use DFS with a stack to detect cycles
53 // A cycle exists if we encounter a sequence already on the current path
54 llvm::DenseSet<AIE::RuntimeSequenceOp> callStack;
55 llvm::DenseSet<AIE::RuntimeSequenceOp> visited;
56
57 std::function<bool(AIE::RuntimeSequenceOp)> hasCycle =
58 [&](AIE::RuntimeSequenceOp seq) -> bool {
59 if (callStack.contains(seq)) {
60 return true; // Found a cycle
61 }
62 if (visited.contains(seq)) {
63 return false; // Already checked this sequence
64 }
65
66 callStack.insert(seq);
67 visited.insert(seq);
68
69 // Check all sequences called by this one
70 bool foundCycle = false;
71 seq.walk([&](RunOp runOp) {
72 if (AIE::RuntimeSequenceOp callee =
73 runOp.getCalleeRuntimeSequenceOp()) {
74 if (hasCycle(callee)) {
75 foundCycle = true;
76 return WalkResult::interrupt();
77 }
78 }
79 return WalkResult::advance();
80 });
81
82 callStack.erase(seq);
83 return foundCycle;
84 };
85
86 if (hasCycle(runtimeSequenceOp)) {
87 isCyclic = true;
88 isValid = true;
89 return;
90 }
91 isCyclic = false;
92 isValid = true;
93 }
94};

96// Turn aie.configure @device into aie.run %.. @configure
97// TODO: add check that liveness of two aie.configures do not overlap
98// (i.e., when we configure A, then configure B, cannot call runtime sequence of
99// A after configuring B)
100// TODO: add code to remove repeated @configure ops
101struct InsertLoadPdiForConfigurePattern : RewritePattern {
102
103 InsertLoadPdiForConfigurePattern(MLIRContext *context,
104 PatternBenefit benefit = 1)
105 : RewritePattern(ConfigureOp::getOperationName(), benefit, context) {}
106
107 LogicalResult matchAndRewrite(Operation *op,
108 PatternRewriter &rewriter) const override {
109 ConfigureOp configureOp = llvm::dyn_cast<ConfigureOp>(op);
110 if (!configureOp) {
111 return failure();
112 }
113
114 // LoadPDI resets the whole device, hence cannot do partial reconfiguration;
115 // therefore, this only supports top-level configure ops
116 if (!llvm::isa<AIE::RuntimeSequenceOp>(configureOp->getParentOp())) {
117 return failure();
118 }
119
120 AIE::DeviceOp referencedDevice = configureOp.getReferencedDeviceOp();
121 if (!referencedDevice) {
122 configureOp.emitError("Referenced symbol is not a device");
123 return failure();
124 }
125
126 Block *configureBlock;
127 if (configureOp.getBody().empty()) {
128 configureBlock = rewriter.createBlock(&configureOp.getBody());
129 } else {
130 configureBlock = &configureOp.getBody().front();
131 }
132
133 rewriter.setInsertionPointToStart(configureBlock);
134 AIEX::NpuLoadPdiOp::create(
135 rewriter, configureOp.getLoc(),
136 FlatSymbolRefAttr::get(referencedDevice.getSymNameAttr()));
137
138 return success();
139 }
140};

142// Collects all external SSA values referenced by an operation (and its nested
143// operations).
144// 1. Collects SSA values from the operation's operands.
145// 2. Recursively walks through all operations in the operation's regions.
146// 3. For each nested operation, collects SSA values from its operands.
147// 4. Skips values that are already in argMap or defined within the operation.
148// 5. For memref.subview operations, traces to the root block argument
149static void
150collectReferencedSSAValues(Operation *op, const IRMapping &argMap,
151 llvm::SetVector<Value> &referencedValues) {
152
153 auto processValue = [&](Value operand) {
154 if (argMap.contains(operand)) {
155 return;
156 }
157
158 // If this is a subview, trace to the root block argument
159 if (auto traceResult = traceSubviewToBlockArgument(operand)) {
160 // Check if the root argument is already mapped
161 if (!argMap.contains(traceResult->rootArg)) {
162 referencedValues.insert(traceResult->rootArg);
163 }
164 return;
165 }
166
167 // Not a subview chain leading to block arg, add as-is
168 referencedValues.insert(operand);
169 };
170
171 // Collect SSA values from the operation's direct operands.
172 for (Value operand : op->getOperands()) {
173 processValue(operand);
174 }
175
176 // Recursively collect SSA values from nested operations in all regions.
177 for (Region &region : op->getRegions()) {
178 region.walk([&](Operation *nestedOp) {
179 for (Value operand : nestedOp->getOperands()) {
180 if (argMap.contains(operand)) {
181 return;
182 }
183
184 // Check if defined within the parent operation
185 Operation *defOp = operand.getDefiningOp();
186 if (defOp && op->isProperAncestor(defOp)) {
187 return;
188 }
189
190 processValue(operand);
191 }
192 });
193 }
194}

196// Copies SSA value definitions into the caller device.
197// Currently, only `aie.tile` operations are supported.
198// Updates argMap to map old values to new/existing values.
199static LogicalResult
200copyReferencedSSAValues(PatternRewriter &rewriter,
201 const llvm::SetVector<Value> &referencedValues,
202 AIE::DeviceOp callerDevice, IRMapping &argMap,
203 mlir::OpBuilder::InsertPoint &clonedSSAInsertPoint,
204 Operation *errorReportOp) {
205
206 llvm::SetVector<Value> referencedValuesToVisit = referencedValues;
207 std::vector<Operation *> referencedOpsToClone = {};
208 while (!referencedValuesToVisit.empty()) {
209 Value referencedValue = referencedValuesToVisit.pop_back_val();
210 Operation *definingOp = referencedValue.getDefiningOp();
211 if (!definingOp) {
212 return errorReportOp->emitError()
213 << "Referenced value is not defined by an operation";
214 }
215 if (std::find(referencedOpsToClone.begin(), referencedOpsToClone.end(),
216 definingOp) != referencedOpsToClone.end()) {
217 continue;
218 }
219
220 if (auto tileOp = llvm::dyn_cast<AIE::TileOp>(definingOp)) {
221 referencedOpsToClone.insert(referencedOpsToClone.begin(), definingOp);
222 } else if (auto lockOp = llvm::dyn_cast<AIE::LockOp>(definingOp)) {
223 Value lockTile = lockOp.getTile();
224 if (lockTile) {
225 referencedValuesToVisit.insert(lockTile);
226 }
227 referencedOpsToClone.push_back(definingOp);
228 } else {
229 return errorReportOp->emitError()
230 << "Referenced SSA value defined by unsupported operation type: "
231 << definingOp->getName().getStringRef()
232 << ". Currently only aie.tile and aie.lock operations are "
233 "supported.";
234 }
235 }
236
237 for (Operation *definingOp : referencedOpsToClone) {
238 if (auto tileOp = llvm::dyn_cast<AIE::TileOp>(definingOp)) {
239 int col = tileOp.getCol();
240 int row = tileOp.getRow();
241
242 rewriter.restoreInsertionPoint(clonedSSAInsertPoint);
243 mlir::Operation *clonedTile = nullptr;
244
245 // Check if a tile with matching col/row already exists in the caller
246 // device
247 AIE::TileOp existingTile = nullptr;
248 for (AIE::TileOp tile : callerDevice.getOps<AIE::TileOp>()) {
249 if (tile.getCol() == col && tile.getRow() == row) {
250 existingTile = tile;
251 break;
252 }
253 }
254
255 if (existingTile) {
256 clonedTile = existingTile.getOperation();
257 // Verify that all attributes match
258 if (tileOp->getAttrDictionary() != existingTile->getAttrDictionary()) {
259 // Filter out result type attributes and symbol attributes for
260 // comparison
261 auto filterAttrs = [](DictionaryAttr dict) -> DictionaryAttr {
262 SmallVector<NamedAttribute> filteredAttrs;
263 for (auto namedAttr : dict) {
264 StringRef name = namedAttr.getName().getValue();
265 if (name != "col" && name != "row") {
266 filteredAttrs.push_back(namedAttr);
267 }
268 }
269 return DictionaryAttr::get(dict.getContext(), filteredAttrs);
270 };
271
272 DictionaryAttr tileAttrs = filterAttrs(tileOp->getAttrDictionary());
273 DictionaryAttr existingAttrs =
274 filterAttrs(existingTile->getAttrDictionary());
275
276 if (tileAttrs != existingAttrs) {
277 return errorReportOp->emitError()
278 << "aie.tile(" << col << ", " << row
279 << ") already exists in the device with different "
280 "attributes";
281 }
282 }
283 } else {
284 // Clone the tile operation into the caller device
285 rewriter.restoreInsertionPoint(clonedSSAInsertPoint);
286 clonedTile = rewriter.clone(*tileOp);
287 clonedSSAInsertPoint = rewriter.saveInsertionPoint();
288 }
289
290 argMap.map(definingOp->getResult(0), clonedTile->getResult(0));
291 rewriter.replaceOpUsesWithIf(
292 definingOp, clonedTile->getResult(0), [&](OpOperand &operand) {
293 return operand.getOwner()->getParentOfType<AIE::DeviceOp>() ==
294 callerDevice;
295 });
296
297 } else if (auto lockOp = llvm::dyn_cast<AIE::LockOp>(definingOp)) {
298 rewriter.restoreInsertionPoint(clonedSSAInsertPoint);
299 Operation *clonedLock = rewriter.clone(*lockOp, argMap);
300 clonedSSAInsertPoint = rewriter.saveInsertionPoint();
301 rewriter.replaceOpUsesWithIf(
302 definingOp, clonedLock->getResult(0), [&](OpOperand &operand) {
303 return operand.getOwner()->getParentOfType<AIE::DeviceOp>() ==
304 callerDevice;
305 });
306 } else {
307 return errorReportOp->emitError()
308 << "Referenced SSA value defined by unsupported operation type: "
309 << definingOp->getName().getStringRef()
310 << ". Currently only aie.tile and aie.lock operations are "
311 "supported.";
312 }
313 }
314
315 return success();
316}

318// Inlines the definitions of all symbols referenced in the given operation
319// at the current insertion point in the given rewriter, unless the symbol
320// definition is in the "previouslyInlinedSymbolMap" map. While inlining,
321// symbols will be renamed to have a unique name.
322// Also copies in SSA values referenced by the inlined symbol definitions.
323static LogicalResult inlineReferencedSymbolDefinitions(
324 PatternRewriter &rewriter, Operation *op, Operation *lookupFrom,
325 IRMapping argMap,
326 llvm::DenseMap<SymbolRefAttr, SymbolRefAttr> &previouslyInlinedSymbolMap,
327 AIE::DeviceOp callerDevice,
328 mlir::OpBuilder::InsertPoint &clonedDefOpsInsertionPoint,
329 llvm::SetVector<SymbolRefAttr> &allSymbolNames) {
330 MLIRContext *ctx = op->getContext();
331 for (NamedAttribute namedAttr : op->getAttrs()) {
332 Attribute attr = namedAttr.getValue();
333 auto newAttr = attr.replace([&](SymbolRefAttr oldSymbolRef) {
334 SymbolRefAttr newSymbolRef;
335 if (!previouslyInlinedSymbolMap.count(oldSymbolRef)) {
336 llvm::StringRef oldName = oldSymbolRef.getRootReference().getValue();
337 std::string uniqueName = oldName.str();
338 unsigned uniquingCounter = 0;
339 while (allSymbolNames.count(SymbolRefAttr::get(ctx, uniqueName))) {
340 uniqueName = oldName.str() + "_" + std::to_string(uniquingCounter);
341 uniquingCounter++;
342 }
343 newSymbolRef = SymbolRefAttr::get(ctx, uniqueName);
344 allSymbolNames.insert(newSymbolRef);
345 previouslyInlinedSymbolMap[oldSymbolRef] = newSymbolRef;
346
347 // Add the new symbol definition
348 // First try to look up from the lookupFrom operation (e.g., within the
349 // callee device). If not found, try looking up from the module level
350 // (for cross-device references).
351 Operation *symbolDefOp =
352 SymbolTable::lookupNearestSymbolFrom(lookupFrom, oldSymbolRef);
353 if (!symbolDefOp) {
354 if (ModuleOp moduleOp = lookupFrom->getParentOfType<ModuleOp>()) {
355 symbolDefOp = SymbolTable::lookupSymbolIn(moduleOp, oldSymbolRef);
356 }
357 }
358 if (!symbolDefOp) {
359 return std::make_pair(newSymbolRef, WalkResult::interrupt());
360 }
361
362 // If the symbol is a device, don't clone it - keep the original
363 // reference. Device ops must stay at module level.
364 if (llvm::isa<AIE::DeviceOp>(symbolDefOp)) {
365 return std::make_pair(oldSymbolRef, WalkResult::advance());
366 }
367
368 // Collect SSA values referenced by the symbol definition operation
369 llvm::SetVector<Value> symbolReferencedValues;
370 collectReferencedSSAValues(symbolDefOp, argMap, symbolReferencedValues);
371
372 // Copy SSA values referenced by the symbol definition
373 // This updates clonedDefOpsInsertionPoint to be after the copied SSA
374 // values
375 if (failed(copyReferencedSSAValues(rewriter, symbolReferencedValues,
376 callerDevice, argMap,
377 clonedDefOpsInsertionPoint, op))) {
378 return std::make_pair(newSymbolRef, WalkResult::interrupt());
379 }
380
381 // Insert the cloned symbol at the device level, after its SSA
382 // dependencies
383 rewriter.restoreInsertionPoint(clonedDefOpsInsertionPoint);
384 Operation *clonedSymbolDefOp = rewriter.clone(*symbolDefOp, argMap);
385 clonedSymbolDefOp->setAttr(SymbolTable::getSymbolAttrName(),
386 StringAttr::get(ctx, uniqueName));
387 clonedDefOpsInsertionPoint = rewriter.saveInsertionPoint();
388 } else {
389 newSymbolRef = previouslyInlinedSymbolMap[oldSymbolRef];
390 }
391
392 return std::make_pair(newSymbolRef, WalkResult::advance());
393 });
394 if (!newAttr) {
395 return failure();
396 }
397 op->setAttr(namedAttr.getName(), newAttr);
398 }
399 return success();
400}

402struct InlineRuntimeCallsPattern : RewritePattern {
403
404 mlir::OpBuilder::InsertPoint &ssaDefInsertPoint;
405 mlir::OpBuilder::InsertPoint &symbolDefInsertPoint;
406 llvm::SetVector<SymbolRefAttr> &allSymbolNames;
407
408 InlineRuntimeCallsPattern(MLIRContext *ctx,
409 mlir::OpBuilder::InsertPoint &ssaDefInsertPoint,
410 mlir::OpBuilder::InsertPoint &symbolDefInsertPoint,
411 llvm::SetVector<SymbolRefAttr> &allSymbolNames)
412 : RewritePattern(RunOp::getOperationName(), PatternBenefit(1), ctx),
413 ssaDefInsertPoint(ssaDefInsertPoint),
414 symbolDefInsertPoint(symbolDefInsertPoint),
415 allSymbolNames(allSymbolNames) {}
416
417 LogicalResult matchAndRewrite(Operation *op,
418 PatternRewriter &rewriter) const override {
419 llvm::DenseMap<SymbolRefAttr, SymbolRefAttr> previouslyInlinedSymbolMap;
420
421 RunOp runOp = llvm::dyn_cast<RunOp>(op);
422 if (!runOp) {
423 return failure();
424 }
425
426 AIE::DeviceOp calleeDevice = runOp.getCalleeDeviceOp();
427 AIE::RuntimeSequenceOp calleeRuntimeSequence =
428 runOp.getCalleeRuntimeSequenceOp();
429 if (!calleeDevice || !calleeRuntimeSequence) {
430 return failure();
431 }
432
433 // rewrite logic
434
435 // Get caller and callee bodies. The callee body will be inlined into the
436 // caller body at the point of the RunOp.
437 Region &calleeBody = calleeRuntimeSequence.getBody();
438 AIE::DeviceOp callerDevice =
439 runOp.getOperation()->getParentOfType<AIE::DeviceOp>();
440 if (!callerDevice) {
441 runOp.emitError() << "needs to be in a DeviceOp";
442 return failure();
443 }
444
445 // The argMap maps callee arguments to caller SSA values.
446 IRMapping argMap;
447 ValueRange values = runOp.getArgs();
448 for (unsigned i = 0, n = calleeBody.getNumArguments(); i < n; i++) {
449 BlockArgument arg = calleeBody.getArgument(i);
450 Value val = values[i];
451 argMap.map(arg, val);
452 }
453
454 // The callee body may reference SSA values and symbols that are defined
455 // in the callee device (outside the callee runtime sequence). We will
456 // inline a supported set of these and error otherwise.
457
458 // Collect SSA values referenced in the callee not defined by the callee and
459 // not in the argMap.
460 llvm::SetVector<Value> referencedValues;
461 for (Operation &op : calleeBody.getOps()) {
462 collectReferencedSSAValues(&op, argMap, referencedValues);
463 }
464 llvm::SetVector<Value> filteredValues;
465 for (Value val : referencedValues) {
466 if (val.getParentRegion() != &calleeBody) {
467 filteredValues.insert(val);
468 }
469 }
470 referencedValues = std::move(filteredValues);
471
472 // Copy the operations that define these SSA values into the caller device
473 if (failed(copyReferencedSSAValues(rewriter, referencedValues, callerDevice,
474 argMap, ssaDefInsertPoint, runOp))) {
475 return failure();
476 }
477
478 // Now, also inline symbol definitions referenced in the callee body;
479 // this may pull in additional SSA values referenced by the symbol
480 // definitions.
481 rewriter.setInsertionPoint(runOp);
482 mlir::OpBuilder::InsertPoint clonedOpInsertionPoint =
483 rewriter.saveInsertionPoint();
484 for (Operation &op : calleeBody.getOps()) {
485 rewriter.restoreInsertionPoint(clonedOpInsertionPoint);
486 Operation *clonedOp = rewriter.clone(op, argMap);
487 clonedOpInsertionPoint = rewriter.saveInsertionPoint();
488
489 if (failed(inlineReferencedSymbolDefinitions(
490 rewriter, clonedOp, calleeRuntimeSequence.getOperation(), argMap,
491 previouslyInlinedSymbolMap, callerDevice, symbolDefInsertPoint,
492 allSymbolNames))) {
493 return failure();
494 }
495 }
496
497 // The aiex.run op has been inlined; erase it.
498 rewriter.eraseOp(runOp);
499
500 return success();
501 }
502};

504struct AIEMaterializeRuntimeSequencesPass
505 : AIEMaterializeRuntimeSequencesBase<AIEMaterializeRuntimeSequencesPass> {
506 void runOnOperation() override {
507 ModuleOp moduleOp = getOperation();
508
509 // Process each device in the module
510 for (AIE::DeviceOp deviceOp : moduleOp.getOps<AIE::DeviceOp>()) {
511
512 // Verify all runtime sequences before materialization
513 for (AIE::RuntimeSequenceOp runtimeSequenceOp :
514 deviceOp.getOps<AIE::RuntimeSequenceOp>()) {
515 if (failed(runtimeSequenceOp.verifyBeforeMaterialization())) {
516 return signalPassFailure();
517 }
518 }
519
520 // Check for cycles in runtime sequence calls
521 for (AIE::RuntimeSequenceOp runtimeSequenceOp :
522 deviceOp.getOps<AIE::RuntimeSequenceOp>()) {
523 AnalysisManager am =
524 getAnalysisManager().nest(deviceOp).nest(runtimeSequenceOp);
525 RuntimeCallGraphCyclicityAnalysis cyclicity =
526 am.getAnalysis<RuntimeCallGraphCyclicityAnalysis>();
527 if (!cyclicity.isValid) {
528 return signalPassFailure();
529 }
530 if (cyclicity.isCyclic) {
531 runtimeSequenceOp.emitError(
532 "Runtime sequence call graph contains a cycle");
533 return signalPassFailure();
534 }
535 }
536
537 // Greedily inline all runtime sequences that can be inlined;
538 // this will start with runtime sequences that do not call other runtime
539 // sequences (leaves); once their callers inline them, the callers can
540 // be inlined as well, and so on
541 mlir::Block &deviceBodyFirstBlock = deviceOp.getBodyRegion().front();
542 auto runtimeSequenceOps = deviceOp.getOps<AIE::RuntimeSequenceOp>();
543 if (runtimeSequenceOps.begin() == runtimeSequenceOps.end()) {
544 // No runtime sequences to materialize
545 continue;
546 }
547 AIE::RuntimeSequenceOp firstRuntimeSequenceOp =
548 *runtimeSequenceOps.begin();
549 mlir::OpBuilder::InsertPoint ssaDefInsertPoint(
550 &deviceBodyFirstBlock, deviceBodyFirstBlock.begin());
551 mlir::OpBuilder::InsertPoint symbolDefInsertPoint(
552 &deviceBodyFirstBlock, mlir::Block::iterator(firstRuntimeSequenceOp));
553 llvm::SetVector<SymbolRefAttr> allSymbolNames = {};
554 for (Operation &op : deviceBodyFirstBlock) {
555 if (auto symbolName = op.getAttrOfType<StringAttr>(
556 SymbolTable::getSymbolAttrName())) {
557 allSymbolNames.insert(SymbolRefAttr::get(symbolName));
558 }
559 }
560
561 MLIRContext *ctx = &getContext();
562 GreedyRewriteConfig rewriter_config = GreedyRewriteConfig();
563 rewriter_config.setRegionSimplificationLevel(
564 GreedySimplifyRegionLevel::Disabled);
565
566 RewritePatternSet patterns_0(ctx);
567 patterns_0.insert<InlineRuntimeCallsPattern>(
568 ctx, ssaDefInsertPoint, symbolDefInsertPoint, allSymbolNames);
569 if (failed(applyPatternsGreedily(deviceOp, std::move(patterns_0),
570 rewriter_config))) {
571 return signalPassFailure();
572 }
573
574 // Insert LoadPDI ops for each aiex.configure op
575 RewritePatternSet patterns_1(ctx);
576 patterns_1.insert<InsertLoadPdiForConfigurePattern>(ctx);
577 walkAndApplyPatterns(deviceOp, std::move(patterns_1));
578
579 // Canonicalize to remove duplicate back-to-back load_pdi ops
580 RewritePatternSet canonicalize_patterns(ctx);
581 AIEX::NpuLoadPdiOp::getCanonicalizationPatterns(canonicalize_patterns,
582 ctx);
583 if (failed(applyPatternsGreedily(
584 deviceOp, std::move(canonicalize_patterns), rewriter_config))) {
585 return signalPassFailure();
586 }
587
588 // Flatten the IR: hoist all operations inside aiex.configure to be direct
589 // children of the runtime sequence, preserving order
590 for (AIE::RuntimeSequenceOp runtimeSequenceOp :
591 deviceOp.getOps<AIE::RuntimeSequenceOp>()) {
592 SmallVector<ConfigureOp> configureOps;
593
594 for (ConfigureOp configureOp :
595 runtimeSequenceOp.getOps<ConfigureOp>()) {
596 configureOps.push_back(configureOp);
597 }
598
599 IRRewriter rewriter(ctx);
600 for (ConfigureOp configureOp : configureOps) {
601 Block &configureBlock = configureOp.getBody().front();
602
603 // Collect all operations in the configure block
604 SmallVector<Operation *> opsToHoist;
605 for (Operation &op : configureBlock) {
606 opsToHoist.push_back(&op);
607 }
608
609 // Hoist operations to be right before the configure op
610 rewriter.setInsertionPoint(configureOp);
611 for (Operation *op : opsToHoist) {
612 op->moveBefore(configureOp);
613 }
614
615 // Erase the now-empty configure op
616 rewriter.eraseOp(configureOp);
617 }
618 }
619
620 } // end for each device
621 }
622};
