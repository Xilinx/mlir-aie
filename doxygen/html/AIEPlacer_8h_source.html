146// Invoke `fn(peer, thisIsFirst)` for each edge in `adjacency` that
147// mentions `op`. `peer` is the OTHER endpoint of the edge; `thisIsFirst`
148// is true when `op` is `edge.first`.
149template <typename F>
150inline void forEachPeer(mlir::Operation *op, const Placer::Adjacency &adjacency,
151 F &&fn) {
152 auto it = adjacency.tileToEdges.find(op);
153 if (it == adjacency.tileToEdges.end())
154 return;
155 for (unsigned idx : it->second) {
156 auto [first, second] = adjacency.edges[idx];
157 bool thisIsFirst = first.getOperation() == op;
158 TileLike peer = thisIsFirst ? second : first;
159 fn(peer, thisIsFirst);
160 }
161}

163// Invoke `fn(TileID)` on every CoreTile neighbor of `at` that is a legal
164// shared-L1 mem-affinity neighbor per the target model.
165template <typename F>
166inline void forEachMemAffinityNeighbor(const AIETargetModel &targetModel,
167 TileID at, F &&fn) {
168 for (auto [dc, dr] :
169 {std::pair{0, -1}, std::pair{0, 1}, std::pair{-1, 0}, std::pair{1, 0}}) {
170 int nc = at.col + dc;
171 int nr = at.row + dr;
172 if (nc < 0 || nc >= targetModel.columns())
173 continue;
174 if (nr < 0 || nr >= targetModel.rows())
175 continue;
176 if (targetModel.getTileType(nc, nr) != AIETileType::CoreTile)
177 continue;
178 if (!targetModel.isLegalMemAffinity(at.col, at.row, nc, nr))
179 continue;
180 fn(TileID{nc, nr});
181 }
182}

184// Sequential placement algorithm
185//
186// Greedy, single-pass placer that maps each `aie.logical_tile` (LTO) to a
187// physical (col, row). Four phases:
188//
189// 1. Collect LogicalTileOp / ObjectFifo* / Cascade* / Flow* / PacketFlow*
190// from the device.
191// 2. Build per-LTO constraints: buffer/cascade/compute-peer adjacencies,
192// channel requirements, and needNeighborIn/Out (the minimum number of
193// compute peers per LTO that MUST land on a shared-L1 neighbor to fit
194// the per-tile DMA budget).
195// 3. Place compute tiles. Sort by constraint level (pinned -> partial ->
196// unpinned) then by descending placementPriority. For each LTO:
197// validate buffer/cascade/compute-peer constraints and channel budget,
198// then claim the tile. Unpinned candidates are tried in two passes:
199// first the slots not soft-reserved for another demand-bearing LTO's
200// peers, then a fallback pass that accepts reserved slots. Each
201// placement of a demand-bearing LTO soft-reserves its mem-affinity
202// neighbor slots for that LTO's compute peers.
203// 4. Place each remaining non-core (mem/shim) LTO near the column
204// centroid of its placed-core peers. With mergeLogicalTiles == true
205// (default) several non-core LTOs may share one physical tile when
206// DMA capacity allows; with mergeLogicalTiles == false each non-core
207// LTO claims its own physical tile.
208//
209// Greedy: once a tile is chosen the decision is final, with no backtracking.
210// If a later LTO's constraints become unsatisfiable given prior placements,
211// the placer emits an error pointing at the LTOs the user can manually pin
212// to break the deadlock.
213//
214// Compute-to-compute ObjectFifos consume no DMA channel when their endpoints
215// land on shared-L1 neighbor tiles, so they enter the placer as a separate
216// compute-peer adjacency rather than as channel demand.
217class SequentialPlacer : public Placer {
218public:
219 SequentialPlacer(std::optional<int> coresPerCol = std::nullopt,
220 bool mergeLogicalTiles = true)
221 : coresPerCol(coresPerCol), mergeLogicalTiles(mergeLogicalTiles) {}
222
223 void initialize(const AIETargetModel &targetModel) override;
224
225 mlir::LogicalResult place(DeviceOp device) override;
226
227 llvm::StringRef getName() const override { return "sequential_placer"; }
228
229private:
230 std::optional<int> coresPerCol;
231 bool mergeLogicalTiles;
232 // Physical tiles already assigned to a non-core aie.logical_tile. Used
233 // only when mergeLogicalTiles == false to forbid mapping a second
234 // non-core aie.logical_tile onto a tile that already hosts one.

236 int deviceCoresPerCol = 0; // Actual cores per column in device

435// Configuration for the SA placer's cost model and schedule.
436//
437// The SA placer minimizes a total cost with four components:
438//
439// totalCost = HPWL + resourcePenalty + cascadePenalty + memoryPressure
440//
441// 1. HPWL (half-perimeter wire length): sum of bounding-box perimeters
442// across all nets.
443//
444// 2. Resource penalty (hard, blocks legality): penalizes MemTile buffer
445// overflow, core tile overflow, DMA channel overuse, and BD count
446// overuse.
447//
448// 3. Cascade penalty (hard, blocks legality): penalizes cascade put/get
449// pairs that aren't adjacent.
450//
451// 4. Memory pressure (soft, guides optimization): quadratic penalty when
452// tile utilization exceeds a threshold.
453struct SAConfig {
454 // HPWL (cost 1)
455 int multicastThreshold = 4; // consumer count above which penalty applies
456 int multicastMultiplier = 2; // HPWL multiplier for multicast nets
457
458 // Resource penalty (cost 2)
459 int memPenaltyPerKB = 30; // per KB of unresolved memory overflow
460 int dmaPenaltyPerChannel = 20; // per DMA channel or BD over limit
461
462 // Cascade penalty (cost 3)
463 int cascadeWeightPerDist = 30; // per Manhattan distance to valid position
464
465 // Memory pressure (cost 4)
466 int pressurePerKB = 3; // quadratic pressure above threshold
467 double pressureThreshold = 0.75; // utilization fraction triggering pressure
468
469 // Adaptive cooling
470 double coolFast = 0.995; // accept ratio > 96%: cool faster
471 double coolSteady = 0.998; // accept ratio > 80%: steady progress
472 double coolBalanced = 0.999; // accept ratio > 15%: balanced exploration
473 double coolStuck = 0.9998; // accept ratio <= 15%: barely cool
474 double greedyThreshold = 0.005; // switch to greedy (T=0) below this
475
476 // Schedule sizing
477 int minMovesPerIter = 100;
478 int maxMovesPerIter = 2000;
479 int minMaxIters = 10000;
480 int greedyMultiplier = 50; // greedyIters = multiplier * numMovable
481 double tempScaleEstimate = 10.0; // initTemp cap: scale * estimated T
482 double tempScaleCost = 2.0; // initTemp cap: scale * totalCost
483
484 // Temperature estimation
485 int tempSearchIters = 100; // binary search iterations
486 double tempSearchCeiling = 5.0; // highT = ceiling * maxDelta
487 double tempSearchEpsilon = 1e-10; // zero guard in binary search
488 double tempSearchTolerance = 1e-6; // convergence tolerance
489
490 // SA loop intervals
491 int debugInterval = 5000; // iterations between debug log lines
492 int fullRecomputeInterval = 10000; // iterations between full cost recompute
493
494 // Move generation
495 int maxMoveAttempts = 20; // retries per move before giving up
496
497 // Acceptance tracking
498 int windowSize = 100; // sliding window for acceptance ratio
499};
