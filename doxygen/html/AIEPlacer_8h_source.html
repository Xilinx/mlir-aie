59// Sequential placement algorithm
60//
61// Places logical tiles to physical tiles using a simple strategy:
62// - Compute tiles: Sequential column-major placement (fill column before next)
63// - Memory/shim tiles: DMA Channel capacity placement near common column
64//
65// Core-to-core connections are NOT validated because SequentialPlacer
66// doesn't account for shared memory optimization.
67//
68// By default (mergeLogicalTiles == true), Shim/Mem logical tiles with
69// identical placement constraints and sufficient combined DMA capacity
70// are merged to the same physical tile via TileOp::getOrCreate during
71// the conversion pattern. CoreTile placement is column-major sequential
72// and never merges. Callers that pre-aggregate non-core logical tiles
73// can pass mergeLogicalTiles == false to make the placer pin each
74// non-core LTO to its own physical tile.
75class SequentialPlacer : public Placer {
76public:
77 SequentialPlacer(std::optional<int> coresPerCol = std::nullopt,
78 bool mergeLogicalTiles = true)
79 : coresPerCol(coresPerCol), mergeLogicalTiles(mergeLogicalTiles) {}
80
81 void initialize(const AIETargetModel &targetModel) override;
82
83 mlir::LogicalResult place(DeviceOp device) override;
84
85 llvm::StringRef getName() const override { return "sequential_placer"; }

87private:
88 std::optional<int> coresPerCol;
89 bool mergeLogicalTiles;

91 // only when mergeLogicalTiles == false to forbid mapping a second

121 // Per-LTO peer edges indexed by either endpoint. `tileToEdges` only
122 // indexes `LogicalTileOp` endpoints; `TileOp` peers carry their own coords.
123 struct Adjacency {
124 llvm::SmallVector<std::pair<TileLike, TileLike>, 4> edges;
125 llvm::DenseMap<mlir::Operation *, llvm::SmallVector<unsigned, 2>>
126 tileToEdges;
127
128 void addEdge(TileLike first, TileLike second) {
129 unsigned idx = edges.size();
130 edges.push_back({first, second});
131 if (mlir::isa<LogicalTileOp>(first.getOperation()))
132 tileToEdges[first.getOperation()].push_back(idx);
133 if (mlir::isa<LogicalTileOp>(second.getOperation()))
134 tileToEdges[second.getOperation()].push_back(idx);
135 }
136
137 // Convenience for IR walkers: skip if either Value isn't a TileLike.
138 void addEdgeFromValues(mlir::Value a, mlir::Value b) {
139 auto aT = mlir::dyn_cast_or_null<TileLike>(a.getDefiningOp());
140 auto bT = mlir::dyn_cast_or_null<TileLike>(b.getDefiningOp());
141 if (aT && bT)
142 addEdge(aT, bT);
143 }
144 };
