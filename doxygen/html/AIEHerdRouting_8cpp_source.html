27template <typename MyOp>
28struct AIEOpRemoval : OpConversionPattern<MyOp> {
29 using OpConversionPattern<MyOp>::OpConversionPattern;
30 using OpAdaptor = typename MyOp::Adaptor;
31
32 explicit AIEOpRemoval(MLIRContext *context, PatternBenefit benefit = 1)
33 : OpConversionPattern<MyOp>(context, benefit) {}
34
35 LogicalResult
36 matchAndRewrite(MyOp op, OpAdaptor operands,
37 ConversionPatternRewriter &rewriter) const override {
38 Operation *Op = op.getOperation();
39
40 rewriter.eraseOp(Op);
41 return success();
42 }
43};

45std::optional<int> getAvailableDestChannel(SmallVector<Connect, 8> &connects,
46 Port sourcePort,
47 WireBundle destBundle) {
48
49 if (connects.empty())
50 return {0};
51
52 int numChannels;
53
54 if (destBundle == WireBundle::North)
55 numChannels = 6;
56 else if (destBundle == WireBundle::South || destBundle == WireBundle::East ||
57 destBundle == WireBundle::West)
58 numChannels = 4;
59 else
60 numChannels = 2;
61
62 // look for existing connect
63 for (int i = 0; i < numChannels; i++) {
64 if (Port port = {destBundle, i};
65 std::find(connects.begin(), connects.end(),
66 Connect{sourcePort, port}) != connects.end())
67 return {i};
68 }
69
70 // if not, look for available destination port
71 for (int i = 0; i < numChannels; i++) {
72 Port port = {destBundle, i};
73 SmallVector<Port, 8> ports;
74 for (auto [src, dst] : connects)
75 ports.push_back(dst);
76
77 if (std::find(ports.begin(), ports.end(), port) == ports.end())
78 return {i};
79 }
80
81 return std::nullopt;
82}

84void buildRoute(int xSrc, int ySrc, int xDest, int yDest,
85 WireBundle sourceBundle, int sourceChannel,
86 WireBundle destBundle, int destChannel, Operation *herdOp,
87 DenseMap<std::pair<Operation *, TileID>,
88 SmallVector<Connect, 8>> &switchboxes) {
89
90 int xCur = xSrc;
91 int yCur = ySrc;
92 WireBundle curBundle = WireBundle::Core;
93 int curChannel = 0;
94 WireBundle lastBundle = WireBundle::Core;
95 Port lastPort = {sourceBundle, sourceChannel};
96
97 SmallVector<TileID, 4> congestion;
98
99 LLVM_DEBUG(llvm::dbgs() << "Build route: " << xSrc << " " << ySrc << " --> "
100 << xDest << " " << yDest << '\n');
101 // traverse horizontally, then vertically
102 while (!(xCur == xDest && yCur == yDest)) {
103 LLVM_DEBUG(llvm::dbgs() << "coord " << xCur << " " << yCur << '\n');
104 TileID curCoord = {xCur, yCur};
105 SmallVector<WireBundle, 4> moves;
106
107 if (xCur < xDest)
108 moves.push_back(WireBundle::East);
109 if (xCur > xDest)
110 moves.push_back(WireBundle::West);
111 if (yCur < yDest)
112 moves.push_back(WireBundle::North);
113 if (yCur > yDest)
114 moves.push_back(WireBundle::South);
115
116 if (std::find(moves.begin(), moves.end(), WireBundle::East) == moves.end())
117 moves.push_back(WireBundle::East);
118 if (std::find(moves.begin(), moves.end(), WireBundle::West) == moves.end())
119 moves.push_back(WireBundle::West);
120 if (std::find(moves.begin(), moves.end(), WireBundle::North) == moves.end())
121 moves.push_back(WireBundle::North);
122 if (std::find(moves.begin(), moves.end(), WireBundle::South) == moves.end())
123 moves.push_back(WireBundle::South);
124
125 for (auto move : moves) {
126 if (auto maybeDestChannel = getAvailableDestChannel(
127 switchboxes[std::make_pair(herdOp, curCoord)], lastPort, move))
128 curChannel = maybeDestChannel.value();
129 else
130 continue;
131
132 if (move == lastBundle)
133 continue;
134
135 if (move == WireBundle::East)
136 xCur = xCur + 1;
137 // yCur = yCur;
138 else if (move == WireBundle::West)
139 xCur = xCur - 1;
140 // yCur = yCur;
141 else if (move == WireBundle::North)
142 // xCur = xCur;
143 yCur = yCur + 1;
144 else if (move == WireBundle::South)
145 // xCur = xCur;
146 yCur = yCur - 1;
147
148 if (std::find(congestion.begin(), congestion.end(), TileID{xCur, yCur}) !=
149 congestion.end())
150 continue;
151
152 curBundle = move;
153 lastBundle = move == WireBundle::East ? WireBundle::West
154 : move == WireBundle::West ? WireBundle::East
155 : move == WireBundle::North ? WireBundle::South
156 : move == WireBundle::South ? WireBundle::North
157 : lastBundle;
158 break;
159 }
160
161 assert(curChannel >= 0 && "Could not find available destination port!");
162 LLVM_DEBUG(llvm::dbgs()
163 << "[" << stringifyWireBundle(lastPort.bundle) << " : "
164 << lastPort.channel << "], [" << stringifyWireBundle(curBundle)
165 << " : " << curChannel << "]\n");
166
167 Port curPort = {curBundle, curChannel};
168 Connect connect = {lastPort, curPort};
169 if (std::find(switchboxes[std::make_pair(herdOp, curCoord)].begin(),
170 switchboxes[std::make_pair(herdOp, curCoord)].end(),
171 connect) ==
172 switchboxes[std::make_pair(herdOp, curCoord)].end())
173 switchboxes[std::make_pair(herdOp, curCoord)].push_back(connect);
174 lastPort = {lastBundle, curChannel};
175 }
176
177 LLVM_DEBUG(llvm::dbgs() << "coord " << xCur << " " << yCur << '\n');
178 LLVM_DEBUG(llvm::dbgs() << "[" << stringifyWireBundle(lastPort.bundle)
179 << " : " << lastPort.channel << "], ["
180 << stringifyWireBundle(destBundle) << " : "
181 << destChannel << "]\n");
182
183 switchboxes[std::make_pair(herdOp, TileID{xCur, yCur})].push_back(
184 {lastPort, Port{destBundle, destChannel}});
185}

187struct AIEHerdRoutingPass : AIEHerdRoutingBase<AIEHerdRoutingPass> {
188 void runOnOperation() override {
189
190 DeviceOp device = getOperation();
191 OpBuilder builder(device.getBody()->getTerminator());
192
193 SmallVector<HerdOp, 4> herds;
194 SmallVector<Operation *, 4> placeOps;
195 SmallVector<Operation *, 4> routeOps;
196 DenseMap<std::pair<Operation *, Operation *>, std::pair<int, int>>
197 distances;
198 SmallVector<std::pair<std::pair<int, int>, std::pair<int, int>>, 4> routes;
199 DenseMap<std::pair<Operation *, TileID>, SmallVector<Connect, 8>>
200 switchboxes;
201
202 for (auto herd : device.getOps<HerdOp>())
203 herds.push_back(herd);
204
205 for (auto placeOp : device.getOps<PlaceOp>()) {
206 placeOps.push_back(placeOp);
207 Operation *sourceHerd = placeOp.getSourceHerd().getDefiningOp();
208 Operation *destHerd = placeOp.getDestHerd().getDefiningOp();
209 int distX = placeOp.getDistXValue();
210 int distY = placeOp.getDistYValue();
211 distances[std::make_pair(sourceHerd, destHerd)] =
212 std::make_pair(distX, distY);
213 }
214
215 // FIXME: multiple route ops with different sourceHerds does not seem to be
216 // aware of the routes done before
217 for (auto routeOp : device.getOps<RouteOp>()) {
218 routeOps.push_back(routeOp);
219
220 auto sourceHerds =
221 dyn_cast<SelectOp>(routeOp.getSourceHerds().getDefiningOp());
222 auto destHerds =
223 dyn_cast<SelectOp>(routeOp.getDestHerds().getDefiningOp());
224 WireBundle sourceBundle = routeOp.getSourceBundle();
225 WireBundle destBundle = routeOp.getDestBundle();
226 int sourceChannel = routeOp.getSourceChannelValue();
227 int destChannel = routeOp.getDestChannelValue();
228
229 HerdOp sourceHerd =
230 dyn_cast<HerdOp>(sourceHerds.getStartHerd().getDefiningOp());
231 IterOp sourceIterX =
232 dyn_cast<IterOp>(sourceHerds.getIterX().getDefiningOp());
233 IterOp sourceIterY =
234 dyn_cast<IterOp>(sourceHerds.getIterY().getDefiningOp());
235
236 HerdOp destHerd =
237 dyn_cast<HerdOp>(destHerds.getStartHerd().getDefiningOp());
238 IterOp destIterX = dyn_cast<IterOp>(destHerds.getIterX().getDefiningOp());
239 IterOp destIterY = dyn_cast<IterOp>(destHerds.getIterY().getDefiningOp());
240
241 int sourceStartX = sourceIterX.getStartValue();
242 int sourceEndX = sourceIterX.getEndValue();
243 int sourceStrideX = sourceIterX.getStrideValue();
244 int sourceStartY = sourceIterY.getStartValue();
245 int sourceEndY = sourceIterY.getEndValue();
246 int sourceStrideY = sourceIterY.getStrideValue();
247
248 int destStartX = destIterX.getStartValue();
249 int destEndX = destIterX.getEndValue();
250 int destStrideX = destIterX.getStrideValue();
251 int destStartY = destIterY.getStartValue();
252 int destEndY = destIterY.getEndValue();
253 int destStrideY = destIterY.getStrideValue();
254
255 assert(distances.count(std::make_pair(sourceHerd, destHerd)) == 1);
256
257 auto [distX, distY] = distances[std::make_pair(sourceHerd, destHerd)];
258 // FIXME: this looks like it can be improved further ...
259 for (int xSrc = sourceStartX; xSrc < sourceEndX; xSrc += sourceStrideX)
260 for (int ySrc = sourceStartY; ySrc < sourceEndY; ySrc += sourceStrideY)
261 for (int xDst = destStartX; xDst < destEndX; xDst += destStrideX)
262 for (int yDst = destStartY; yDst < destEndY; yDst += destStrideY) {
263 // Build route (x0, y0) --> (x1, y1)
264 int x0 = xSrc;
265 int y0 = ySrc;
266 int x1 = xDst;
267 int y1 = yDst;
268 if (destIterX == sourceIterX)
269 x1 = x0;
270 if (destIterY == sourceIterX)
271 y1 = x0;
272 if (destIterX == sourceIterY)
273 x1 = y0;
274 if (destIterY == sourceIterY)
275 y1 = y0;
276
277 auto route = std::make_pair(
278 std::make_pair(x0, y0),
279 std::make_pair(distX + x1 - x0, distY + y1 - y0));
280 if (std::find(routes.begin(), routes.end(), route) !=
281 routes.end())
282 continue;
283
284 buildRoute(x0, y0, x1 + distX, y1 + distY, sourceBundle,
285 sourceChannel, destBundle, destChannel, sourceHerd,
286 switchboxes);
287
288 routes.push_back(route);
289 }
290 }
291
292 for (const auto &swboxCfg : switchboxes) {
293 Operation *herdOp = swboxCfg.first.first;
294 int x = swboxCfg.first.second.col;
295 int y = swboxCfg.first.second.row;
296 auto connects = swboxCfg.second;
297 HerdOp herd = dyn_cast<HerdOp>(herdOp);
298
299 builder.setInsertionPoint(device.getBody()->getTerminator());
300
301 auto iterx =
302 IterOp::create(builder, builder.getUnknownLoc(), x, x + 1, 1);
303 auto itery =
304 IterOp::create(builder, builder.getUnknownLoc(), y, y + 1, 1);
305 auto sel = SelectOp::create(builder, builder.getUnknownLoc(), herd, iterx,
306 itery);
307 auto swbox = SwitchboxOp::create(builder, builder.getUnknownLoc(), sel);
308 SwitchboxOp::ensureTerminator(swbox.getConnections(), builder,
309 builder.getUnknownLoc());
310 Block &b = swbox.getConnections().front();
311 builder.setInsertionPoint(b.getTerminator());
312
313 for (auto [sourcePort, destPort] : connects) {
314 WireBundle sourceBundle = sourcePort.bundle;
315 int sourceChannel = sourcePort.channel;
316 WireBundle destBundle = destPort.bundle;
317 int destChannel = destPort.channel;
318
319 ConnectOp::create(builder, builder.getUnknownLoc(), sourceBundle,
320 sourceChannel, destBundle, destChannel);
321 }
322 }
323
324 ConversionTarget target(getContext());
325
326 RewritePatternSet patterns(&getContext());
327 patterns.insert<AIEOpRemoval<PlaceOp>, AIEOpRemoval<RouteOp>>(
328 device.getContext());
329
330 if (failed(applyPartialConversion(device, target, std::move(patterns))))
331 signalPassFailure();
332 }
333};
