100struct RtpToWrite32Pattern : OpConversionPattern<NpuWriteRTPOp> {
101 using OpConversionPattern::OpConversionPattern;
102
103 RtpToWrite32Pattern(MLIRContext *context, PatternBenefit benefit = 1)
104 : OpConversionPattern(context, benefit) {}
105
106 LogicalResult
107 matchAndRewrite(NpuWriteRTPOp op, OpAdaptor adaptor,
108 ConversionPatternRewriter &rewriter) const override {
109
110 auto device = op->getParentOfType<AIE::DeviceOp>();
111
112 auto buffer = device.lookupSymbol<AIE::BufferOp>(op.getBuffer());
113 if (!buffer) {
114 op->emitError("buffer '" + op.getBuffer() + "' not found in device");
115 return failure();
116 }
117
118 if (!buffer.getAddress()) {
119 op->emitError("buffer must have address assigned");
120 return failure();
121 }
122 AIE::TileOp tile = buffer.getTileOp();
123
124 uint32_t idx = op.getIndex() * sizeof(uint32_t);
125 uint32_t address = buffer.getAddress().value() + idx;
126
127 rewriter.create<NpuWrite32Op>(op->getLoc(), address, op.getValue(), nullptr,
128 rewriter.getI32IntegerAttr(tile.getCol()),
129 rewriter.getI32IntegerAttr(tile.getRow()));
130
131 rewriter.eraseOp(op);
132 return success();
133 }

136struct PushQueuetoWrite32Pattern : OpConversionPattern<NpuPushQueueOp> {
137
138public:
139 using OpConversionPattern::OpConversionPattern;
140
141 PushQueuetoWrite32Pattern(MLIRContext *context, PatternBenefit benefit = 1)
142 : OpConversionPattern(context, benefit) {}
143
144 LogicalResult
145 matchAndRewrite(NpuPushQueueOp op, OpAdaptor adaptor,
146 ConversionPatternRewriter &rewriter) const override {
147
148 const auto &tm = AIE::getTargetModel(op);
149 uint32_t ctrl_offset = tm.getDmaControlAddress(
150 op.getColumn(), op.getRow(), op.getChannel(), op.getDirection());
151
152 // control packet for issuing token
153 if (op.getIssueToken()) {
154 // set the task-complete-token controller ID field in the dma control
155 // register
156 AIE::TileOp shimTile = AIE::TileOp::getOrCreate(
157 rewriter, op->getParentOfType<AIE::DeviceOp>(), op.getColumn(), 0);
158 if (shimTile->hasAttr("controller_id")) {
159 AIE::PacketInfoAttr controller_id_attr =
160 shimTile->getAttrOfType<AIE::PacketInfoAttr>("controller_id");
161 uint32_t data = controller_id_attr.getPktId() << 8;
162 uint32_t mask = 0x00001F00;
163 rewriter.create<NpuMaskWrite32Op>(op->getLoc(), ctrl_offset, data, mask,
164 nullptr, nullptr, nullptr);
165 }
166 }
167
168 // the offset of the task queue register in the tile
169 uint32_t queue_offset = ctrl_offset + 0x4;
170
171 // the value to write
172 uint32_t bd_id = op.getBdId();
173 uint32_t repeat_cnt = op.getRepeatCount();
174 uint32_t cmd = 0;
175 cmd |= bd_id & 0xF;
176 cmd |= (repeat_cnt & 0xFF) << 16;
177 if (op.getIssueToken())
178 cmd |= 0x80000000;
179
180 rewriter.create<NpuWrite32Op>(op->getLoc(), queue_offset, cmd, nullptr,
181 nullptr, nullptr);
182 rewriter.eraseOp(op);
183 return success();
184 }

187struct DmaToNpuPattern : OpConversionPattern<NpuDmaMemcpyNdOp> {
188 using OpConversionPattern::OpConversionPattern;
189
190public:
191 DmaToNpuPattern(MLIRContext *context, PatternBenefit benefit = 1)
192 : OpConversionPattern(context, benefit) {}
193
194 LogicalResult
195 matchAndRewrite(NpuDmaMemcpyNdOp op, OpAdaptor adaptor,
196 ConversionPatternRewriter &rewriter) const override {
197 const auto &targetModel = AIE::getTargetModel(op);
198 BaseMemRefType bufferType = op.getMemref().getType();
199 auto *ctx = op->getContext();
200 auto i32ty = IntegerType::get(ctx, 32);
201 auto zero = IntegerAttr::get(i32ty, 0);
202 auto memref = adaptor.getMemref();
203
204 auto dev = op->getParentOfType<AIE::DeviceOp>();
205 if (!dev)
206 return failure();
207
208 auto infoOp = AIE::ShimDMAAllocationOp::getForSymbol(
209 dev, op.getMetadata().getRootReference());
210 if (!infoOp) {
211 return op->emitOpError("couldn't find shim_dma_allocation op.");
212 }
213
214 auto channelDir = infoOp.getChannelDir();
215 bool isMM2S = channelDir == AIE::DMAChannelDir::MM2S;
216 int col = infoOp.getCol();
217
218 // initialize fields to zero
219 auto column = zero;
220 auto bd_id = zero;
221 auto buffer_length = zero;
222 auto buffer_offset = zero;
223 auto enable_packet = zero;
224 auto out_of_order_id = zero;
225 auto packet_id = zero;
226 auto packet_type = zero;
227 auto d0_size = zero;
228 auto d0_stride = zero;
229 auto d1_size = zero;
230 auto d1_stride = zero;
231 auto d2_size = zero;
232 auto d2_stride = zero;
233 auto iteration_current = zero;
234 auto iteration_size = zero;
235 auto iteration_stride = zero;
236 auto next_bd = zero;
237 auto row = zero;
238 auto use_next_bd = zero;
239 auto valid_bd = zero;
240 auto lock_rel_val = zero;
241 auto lock_rel_id = zero;
242 auto lock_acq_enable = zero;
243 auto lock_acq_val = zero;
244 auto lock_acq_id = zero;
245 auto d0_zero_before = zero;
246 auto d1_zero_before = zero;
247 auto d2_zero_before = zero;
248 auto d0_zero_after = zero;
249 auto d1_zero_after = zero;
250 auto d2_zero_after = zero;
251 auto burst_length = zero;
252
253 auto issue_token = BoolAttr::get(ctx, false);
254 auto repeat_count = zero;
255 llvm::SmallVector<int64_t, 4> inputSizes = llvm::map_to_vector(
256 llvm::reverse(op.getMixedSizes()),
257 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
258 llvm::SmallVector<int64_t, 4> inputStrides = llvm::map_to_vector(
259 llvm::reverse(op.getMixedStrides()),
260 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
261 llvm::SmallVector<int64_t, 4> sizes(4);
262 llvm::SmallVector<int64_t, 4> strides(4);
263 getHardwareStridesWraps(targetModel, op, bufferType, inputSizes,
264 inputStrides, sizes, strides);
265 int64_t offset = op.getOffsetInBytes();
266
267 // column
268 column = IntegerAttr::get(i32ty, col);
269
270 // row
271 row = IntegerAttr::get(i32ty, 0);
272
273 bool skipTransformationChecks = op.isLinearTransferWithoutTransformation();
274 if (failed(verifyStridesWraps(op, bufferType, col, 0, inputSizes,
275 inputStrides, sizes, strides,
276 skipTransformationChecks))) {
277 return failure();
278 }
279
280 // arg_idx
281 AIE::RuntimeSequenceOp seq_op =
282 op->getParentOfType<AIE::RuntimeSequenceOp>();
283 if (!seq_op) {
284 op->emitOpError("NpuDmaMemcpyNdOps must have RuntimeSequenceOp parent at "
285 "time of lowering.");
286 return failure();
287 }
288 Block &entryBB = seq_op.getBody().front();
289 int arg_idx = -1;
290 for (int i = 0, e = entryBB.getNumArguments(); i < e; i++) {
291 if (entryBB.getArgument(i) == memref) {
292 arg_idx = i;
293 break;
294 }
295 }
296 if (arg_idx < 0)
297 return failure();
298
299 // bd_id
300 bd_id = IntegerAttr::get(i32ty, op.getId());
301
302 // buffer_length
303 uint64_t buffer_length_val = inputSizes[0] * op.getElementTypeBitwidth() /
304 targetModel.getAddressGenGranularity();
305 if (inputSizes.size() > 1) {
306 for (size_t i = 1; i < std::min(inputSizes.size(), (size_t)3); i++) {
307 buffer_length_val *= inputSizes[i];
308 }
309 }
310 buffer_length = IntegerAttr::get(i32ty, buffer_length_val);
311
312 // buffer_offset - zero because the complete address is set by the patch op
313 buffer_offset = IntegerAttr::get(i32ty, 0);
314
315 // enable_packet
316 if (auto packetInfo = op.getPacket()) {
317 enable_packet = IntegerAttr::get(i32ty, 1);
318 packet_type = IntegerAttr::get(i32ty, packetInfo->getPktType());
319 packet_id = IntegerAttr::get(i32ty, packetInfo->getPktId());
320 }
321
322 // out_of_order_id
323
324 if (!op.isLinearTransferWithoutTransformation()) {
325 // d0_size, d0_stride
326 d0_size = IntegerAttr::get(i32ty, sizes[0]);
327 d0_stride = IntegerAttr::get(i32ty, strides[0]);
328
329 // d1_size, d1_stride
330 d1_size = IntegerAttr::get(i32ty, sizes[1]);
331 d1_stride = IntegerAttr::get(i32ty, strides[1]);
332
333 // d2_stride
334 d2_stride = IntegerAttr::get(i32ty, strides[2]);
335
336 // d2_size
337 if (targetModel.isMemTile(col, 0)) // Need to be any row
338 d2_size = IntegerAttr::get(i32ty, sizes[2]);
339 else
340 d2_size = IntegerAttr::get(i32ty, 0);
341 }
342 // iteration_current, iteration_size, iteration_stride, repeat_count
343 if (inputSizes[3] > 1) {
344 if (inputStrides[3] > 0) {
345 iteration_size = IntegerAttr::get(i32ty, sizes[3]);
346 iteration_stride = IntegerAttr::get(i32ty, strides[3]);
347 } else {
348 // We allow users to encode the repeat_count as a dimension 3 stride
349 // of 0. This must lower to a iteration wrap of 0, so no stride is
350 // ever added. We then repeat the BD using the repeat_count in
351 // NpuPushQueueOp.
352 iteration_size = zero;
353 iteration_stride = zero;
354 }
355 }
356 repeat_count = IntegerAttr::get(i32ty, sizes[3]);
357
358 // next_bd
359
360 // use_next_bd
361
362 // valid_bd
363 valid_bd = IntegerAttr::get(i32ty, 1);
364
365 // lock_rel_val
366
367 // lock_rel_id
368
369 // lock_acq_enable
370
371 // lock_acq_val
372
373 // lock_acq_id
374
375 // d0_zero_before
376 d0_zero_before = IntegerAttr::get(i32ty, op.getD0ZeroBefore());
377
378 // d1_zero_before
379 d1_zero_before = IntegerAttr::get(i32ty, op.getD1ZeroBefore());
380
381 // d2_zero_before
382 d2_zero_before = IntegerAttr::get(i32ty, op.getD2ZeroBefore());
383
384 // d0_zero_after
385 d0_zero_after = IntegerAttr::get(i32ty, op.getD0ZeroAfter());
386
387 // d1_zero_after
388 d1_zero_after = IntegerAttr::get(i32ty, op.getD1ZeroAfter());
389
390 // d2_zero_after
391 d2_zero_after = IntegerAttr::get(i32ty, op.getD2ZeroAfter());
392
393 // burst_size
394 burst_length = IntegerAttr::get(i32ty, op.getBurstLength());
395
396 // Set the issue_token
397 issue_token = BoolAttr::get(ctx, op.getIssueToken());
398 // Earlier, all S2MM channels were implicitly assumed to issue a token.
399 // This logic is kept for now for backward compatibility.
400 if (!isMM2S)
401 issue_token = BoolAttr::get(ctx, true);
402
403 if (targetModel.isMemTile(col, 0) && (!isMM2S) &&
404 (op.getD0ZeroBefore() != 0 || op.getD0ZeroAfter() != 0 ||
405 op.getD1ZeroBefore() != 0 || op.getD1ZeroAfter() != 0 ||
406 op.getD2ZeroBefore() != 0 || op.getD2ZeroAfter() != 0))
407 op->emitOpError("MemTile supports zero padding only on MM2S direction");
408
409 // write the buffer descriptor to the array
410 rewriter.create<NpuWriteBdOp>(
411 op->getLoc(), column, bd_id, buffer_length, buffer_offset,
412 enable_packet, out_of_order_id, packet_id, packet_type, d0_size,
413 d0_stride, d1_size, d1_stride, d2_size, d2_stride, iteration_current,
414 iteration_size, iteration_stride, next_bd, row, use_next_bd, valid_bd,
415 lock_rel_val, lock_rel_id, lock_acq_enable, lock_acq_val, lock_acq_id,
416 d0_zero_before, d1_zero_before, d2_zero_before, d0_zero_after,
417 d1_zero_after, d2_zero_after, burst_length);
418
419 // compute the location of the address to patch in the bd and emit patch
420 // instruction to perform the patch.
421 uint64_t addr = targetModel.getDmaBdAddress(col, 0, op.getId()) +
422 targetModel.getDmaBdAddressOffset(col, 0);
423 rewriter.create<NpuAddressPatchOp>(op->getLoc(), addr, arg_idx, offset);
424
425 // push the patched bd onto the dma task queue
426 rewriter.create<NpuPushQueueOp>(
427 op->getLoc(), column, row, infoOp.getChannelDirAttr(),
428 infoOp.getChannelIndexAttr(), issue_token, repeat_count, bd_id);
429
430 rewriter.eraseOp(op);
431 return success();
432 }
433};

435/// Convert NpuDmaWaitOp into NpuSyncOp by retrieving the necessary
436/// information from the ShimDMAAllocationOp referenced through the
437/// symbol argument of this op.
438struct DmaWaitToSyncPattern : OpConversionPattern<NpuDmaWaitOp> {
439
440public:
441 using OpConversionPattern::OpConversionPattern;
442
443 DmaWaitToSyncPattern(MLIRContext *context, PatternBenefit benefit = 1)
444 : OpConversionPattern(context, benefit) {}
445
446 LogicalResult
447 matchAndRewrite(NpuDmaWaitOp op, OpAdaptor adaptor,
448 ConversionPatternRewriter &rewriter) const override {
449 AIE::DeviceOp dev = op->getParentOfType<AIE::DeviceOp>();
450 if (!dev)
451 return op->emitError("couldn't find parent of type DeviceOp");
452
453 AIE::ShimDMAAllocationOp shimDmaAllocOp =
454 AIE::ShimDMAAllocationOp::getForSymbol(dev, op.getSymbol());
455 if (!shimDmaAllocOp) {
456 return op->emitError("couldn't find shim_dma_allocation op");
457 }
458
459 // Create with `column_num == 1` and `row_num == 1` to check for a single
460 // column and row. Row is always 0 for shim tiles.
461 (void)rewriter.replaceOpWithNewOp<NpuSyncOp>(
462 op, shimDmaAllocOp.getCol(), /* row */ 0,
463 static_cast<uint32_t>(shimDmaAllocOp.getChannelDir()),
464 shimDmaAllocOp.getChannelIndex(), 1, 1);
465
466 return success();
467 }

470struct WriteBdToBlockWritePattern : OpConversionPattern<NpuWriteBdOp> {
471 using OpConversionPattern::OpConversionPattern;
472
473public:
474 WriteBdToBlockWritePattern(MLIRContext *context, PatternBenefit benefit = 1)
475 : OpConversionPattern(context, benefit) {}
476
477 LogicalResult
478 matchAndRewrite(NpuWriteBdOp op, OpAdaptor adaptor,
479 ConversionPatternRewriter &rewriter) const override {
480
481 AIE::DeviceOp dev = op->getParentOfType<AIE::DeviceOp>();
482 const AIE::AIETargetModel &tm = dev.getTargetModel();
483
484 int num_words = 0;
485 if (isa<AIE::AIE2TargetModel>(tm))
486 num_words = 8;
487 else
488 llvm_unreachable(
489 "Unsupported AIETargetModel in WriteBdToBlockWritePattern");
490
491 std::vector<uint32_t> words(num_words, 0);
492
493 uint32_t bd_id = op.getBdId();
494 int col = op.getColumn();
495 int row = op.getRow();
496 uint64_t bd_addr = tm.getDmaBdAddress(col, row, bd_id);
497 if (tm.isShimNOCTile(col, row)) {
498 // DMA_BDX_0
499 words[0] = op.getBufferLength();
500
501 // DMA_BDX_1
502 words[1] = op.getBufferOffset();
503
504 // DMA_BDX_2
505 // En Packet , OoO BD ID , Packet ID , Packet Type
506 words[2] |= (op.getEnablePacket() & 0x1) << 30;
507 words[2] |= (op.getOutOfOrderId() & 0x3f) << 24;
508 words[2] |= (op.getPacketId() & 0x1f) << 19;
509 words[2] |= (op.getPacketType() & 0x7) << 16;
510
511 // DMA_BDX_3
512 // TODO: Secure Access
513 words[3] |= (op.getD0Size() & 0x3ff) << 20;
514 words[3] |= op.getD0Stride() & 0xfffff;
515
516 // DMA_BDX_4
517 words[4] = (getShimBurstLengthEncoding(tm, op.getBurstLength()) & 0x3)
518 << 30;
519 words[4] |= (op.getD1Size() & 0x3ff) << 20;
520 words[4] |= op.getD1Stride() & 0xfffff;
521
522 // DMA_BDX_5
523 // TODO: SIMID, AXQoS
524 words[5] |= (2 & 0xf) << 24; // AXCache = 2 to enable upsizing in NoC
525 words[5] |= op.getD2Stride() & 0xfffff;
526
527 // DMA_BDX_6
528 words[6] |= (op.getIterationCurrent() & 0x3f) << 26;
529 words[6] |= (op.getIterationSize() & 0x3f) << 20;
530 words[6] |= op.getIterationStride() & 0xfffff;
531
532 // DMA_BDX_7
533 // TODO: TLAST Suppress
534 words[7] |= (op.getNextBd() & 0xf) << 27;
535 words[7] |= (op.getUseNextBd() & 0x1) << 26;
536 words[7] |= (op.getValidBd() & 0x1) << 25;
537 words[7] |= (op.getLockRelVal() & 0x7f) << 18;
538 words[7] |= (op.getLockRelId() & 0xf) << 13;
539 words[7] |= (op.getLockAcqEnable() & 0x1) << 12;
540 words[7] |= (op.getLockAcqVal() & 0x7f) << 5;
541 words[7] |= op.getLockAcqId() & 0xf;
542 if (op.getD0ZeroBefore() || op.getD1ZeroBefore() ||
543 op.getD2ZeroBefore() || op.getD0ZeroAfter() || op.getD1ZeroAfter() ||
544 op.getD2ZeroAfter()) {
545 op->emitError("Zero padding is only available on MemTile");
546 }
547 } else if (tm.isMemTile(op.getColumn(), op.getRow())) {
548
549 // DMA_BDX_0
550 words[0] |= (op.getEnablePacket() & 0x1) << 31;
551 words[0] |= (op.getPacketType() & 0x7) << 28;
552 words[0] |= (op.getPacketId() & 0x1f) << 23;
553 words[0] |= (op.getOutOfOrderId() & 0x3f) << 17;
554 words[0] |= op.getBufferLength() & 0x1ffff;
555
556 // DMA_BDX_1
557 words[1] |= (op.getD0ZeroBefore() & 0x3F) << 26;
558 words[1] |= (op.getNextBd() & 0x3f) << 20;
559 words[1] |= (op.getUseNextBd() & 0x1) << 19;
560 words[1] |= op.getBufferOffset() & 0x7ffff;
561
562 // DMA_BDX_2
563 words[2] |= (op.getD0Size() & 0x3ff) << 17;
564 words[2] |= op.getD0Stride() & 0x1ffff;
565
566 // DMA_BDX_3
567 // TODO: Secure Access
568 words[3] |= (op.getD1ZeroBefore() & 0x1F) << 27;
569 words[3] |= (op.getD1Size() & 0x3ff) << 17;
570 words[3] |= op.getD1Stride() & 0x1ffff;
571
572 // DMA_BDX_4
573 // TODO: D2Size
574 words[4] |= (op.getD2ZeroBefore() & 0xF) << 27;
575 words[4] |= op.getD2Stride() & 0x1ffff;
576
577 // DMA_BDX_5
578 // ToDO: D3Stride
579 words[5] |= (op.getD2ZeroAfter() & 0xF) << 28;
580 words[5] |= (op.getD1ZeroAfter() & 0x1F) << 23;
581 words[5] |= (op.getD0ZeroAfter() & 0x3F) << 17;
582
583 // DMA_BDX_6
584 words[6] |= (op.getIterationCurrent() & 0x3f) << 23;
585 words[6] |= (op.getIterationSize() & 0x3f) << 17;
586 words[6] |= op.getIterationStride() & 0x1ffff;
587
588 // DMA_BDX_7
589 words[7] |= (op.getValidBd() & 0x1) << 31;
590 words[7] |= (op.getLockRelVal() & 0x7f) << 24;
591 words[7] |= (op.getLockRelId() & 0xff) << 16;
592 words[7] |= (op.getLockAcqEnable() & 0x1) << 15;
593 words[7] |= (op.getLockAcqVal() & 0x7f) << 8;
594 words[7] |= op.getLockAcqId() & 0xff;
595 } else {
596 // TODO: DMA BD configuration for Compute Tiles
597 op->emitError("Run-time DMA configuration is supported only for "
598 "ShimTiles and MemTiles currently.");
599 return failure();
600 }
601
602 memref::GlobalOp global = nullptr;
603 {
604 OpBuilder::InsertionGuard guard(rewriter);
605 rewriter.setInsertionPoint(op->getParentOfType<AIE::RuntimeSequenceOp>());
606 global = getOrCreateDataMemref(rewriter, dev, op.getLoc(), words);
607 }
608 auto memref = rewriter.create<memref::GetGlobalOp>(
609 op.getLoc(), global.getType(), global.getName());
610
611 (void)rewriter.replaceOpWithNewOp<NpuBlockWriteOp>(
612 op, rewriter.getUI32IntegerAttr(bd_addr), memref.getResult(), nullptr,
613 nullptr, nullptr);
614 return success();
615 }
616};

618struct AIEDmaToNpuPass : AIEDmaToNpuBase<AIEDmaToNpuPass> {
619
620 void getDependentDialects(DialectRegistry &registry) const override {
621 registry.insert<memref::MemRefDialect>();
622 }
623
