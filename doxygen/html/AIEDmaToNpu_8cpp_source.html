101struct RtpToWrite32Pattern : OpConversionPattern<NpuWriteRTPOp> {
102 using OpConversionPattern::OpConversionPattern;
103
104 RtpToWrite32Pattern(MLIRContext *context, PatternBenefit benefit = 1)
105 : OpConversionPattern(context, benefit) {}
106
107 LogicalResult
108 matchAndRewrite(NpuWriteRTPOp op, OpAdaptor adaptor,
109 ConversionPatternRewriter &rewriter) const override {
110
111 auto device = op->getParentOfType<AIE::DeviceOp>();
112
113 auto buffer = device.lookupSymbol<AIE::BufferOp>(op.getBuffer());
114 if (!buffer) {
115 op->emitError("buffer '" + op.getBuffer() + "' not found in device");
116 return failure();
117 }
118
119 if (!buffer.getAddress()) {
120 op->emitError("buffer must have address assigned");
121 return failure();
122 }
123 AIE::TileOp tile = buffer.getTileOp();
124
125 uint32_t idx = op.getIndex() * sizeof(uint32_t);
126 uint32_t address = buffer.getAddress().value() + idx;
127
128 NpuWrite32Op::create(rewriter, op->getLoc(), address, op.getValue(),
129 nullptr, rewriter.getI32IntegerAttr(tile.getCol()),
130 rewriter.getI32IntegerAttr(tile.getRow()));
131
132 rewriter.eraseOp(op);
133 return success();
134 }

137struct PushQueuetoWrite32Pattern : OpConversionPattern<NpuPushQueueOp> {
138
139public:
140 using OpConversionPattern::OpConversionPattern;
141
142 PushQueuetoWrite32Pattern(MLIRContext *context, PatternBenefit benefit = 1)
143 : OpConversionPattern(context, benefit) {}
144
145 LogicalResult
146 matchAndRewrite(NpuPushQueueOp op, OpAdaptor adaptor,
147 ConversionPatternRewriter &rewriter) const override {
148
149 const auto &tm = AIE::getTargetModel(op);
150 uint32_t ctrl_offset = tm.getDmaControlAddress(
151 op.getColumn(), op.getRow(), op.getChannel(), op.getDirection());
152
153 // control packet for issuing token
154 if (op.getIssueToken()) {
155 // set the task-complete-token controller ID field in the dma control
156 // register
157 AIE::TileOp shimTile = AIE::TileOp::getOrCreate(
158 rewriter, op->getParentOfType<AIE::DeviceOp>(), op.getColumn(),
159 op.getRow());
160 if (shimTile->hasAttr("controller_id")) {
161 AIE::PacketInfoAttr controller_id_attr =
162 shimTile->getAttrOfType<AIE::PacketInfoAttr>("controller_id");
163 uint32_t data = controller_id_attr.getPktId() << 8;
164 uint32_t mask = 0x00001F00;
165 NpuMaskWrite32Op::create(rewriter, op->getLoc(), ctrl_offset, data,
166 mask, nullptr, nullptr, nullptr);
167 }
168 }
169
170 // the offset of the task queue register in the tile
171 uint32_t queue_offset = ctrl_offset + 0x4;
172
173 // the value to write
174 uint32_t bd_id = op.getBdId();
175 uint32_t repeat_cnt = op.getRepeatCount();
176 uint32_t cmd = 0;
177 cmd |= bd_id & 0xF;
178 cmd |= (repeat_cnt & 0xFF) << 16;
179 if (op.getIssueToken())
180 cmd |= 0x80000000;
181
182 NpuWrite32Op::create(rewriter, op->getLoc(), queue_offset, cmd, nullptr,
183 nullptr, nullptr);
184 rewriter.eraseOp(op);
185 return success();
186 }

189struct DmaToNpuPattern : OpConversionPattern<NpuDmaMemcpyNdOp> {
190 using OpConversionPattern::OpConversionPattern;
191
192public:
193 DmaToNpuPattern(MLIRContext *context, PatternBenefit benefit = 1)
194 : OpConversionPattern(context, benefit) {}
195
196 LogicalResult
197 matchAndRewrite(NpuDmaMemcpyNdOp op, OpAdaptor adaptor,
198 ConversionPatternRewriter &rewriter) const override {
199 const auto &targetModel = AIE::getTargetModel(op);
200 BaseMemRefType bufferType = op.getMemref().getType();
201 auto *ctx = op->getContext();
202 auto i32ty = IntegerType::get(ctx, 32);
203 auto zero = IntegerAttr::get(i32ty, 0);
204 auto memref = adaptor.getMemref();
205
206 auto dev = op->getParentOfType<AIE::DeviceOp>();
207 if (!dev)
208 return failure();
209
210 auto infoOp = AIE::ShimDMAAllocationOp::getForSymbol(
211 dev, op.getMetadata().getRootReference());
212 if (!infoOp) {
213 return op->emitOpError("couldn't find shim_dma_allocation op.");
214 }
215
216 AIE::TileOp shimTile = infoOp.getTileOp();
217 if (!shimTile) {
218 return op->emitOpError(
219 "shim_dma_allocation op must reference a valid TileOp.");
220 }
221
222 auto channelDir = infoOp.getChannelDir();
223 bool isMM2S = channelDir == AIE::DMAChannelDir::MM2S;
224 int tileCol = shimTile.getCol();
225 int tileRow = shimTile.getRow();
226
227 // initialize fields to zero
228 auto column = zero;
229 auto bd_id = zero;
230 auto buffer_length = zero;
231 auto buffer_offset = zero;
232 auto enable_packet = zero;
233 auto out_of_order_id = zero;
234 auto packet_id = zero;
235 auto packet_type = zero;
236 auto d0_size = zero;
237 auto d0_stride = zero;
238 auto d1_size = zero;
239 auto d1_stride = zero;
240 auto d2_size = zero;
241 auto d2_stride = zero;
242 auto iteration_current = zero;
243 auto iteration_size = zero;
244 auto iteration_stride = zero;
245 auto next_bd = zero;
246 auto row = zero;
247 auto use_next_bd = zero;
248 auto valid_bd = zero;
249 auto lock_rel_val = zero;
250 auto lock_rel_id = zero;
251 auto lock_acq_enable = zero;
252 auto lock_acq_val = zero;
253 auto lock_acq_id = zero;
254 auto d0_zero_before = zero;
255 auto d1_zero_before = zero;
256 auto d2_zero_before = zero;
257 auto d0_zero_after = zero;
258 auto d1_zero_after = zero;
259 auto d2_zero_after = zero;
260 auto burst_length = zero;
261
262 auto issue_token = BoolAttr::get(ctx, false);
263 auto repeat_count = zero;
264 llvm::SmallVector<int64_t, 4> inputSizes = llvm::map_to_vector(
265 llvm::reverse(op.getMixedSizes()),
266 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
267 llvm::SmallVector<int64_t, 4> inputStrides = llvm::map_to_vector(
268 llvm::reverse(op.getMixedStrides()),
269 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
270 llvm::SmallVector<int64_t, 4> sizes(4);
271 llvm::SmallVector<int64_t, 4> strides(4);
272 getHardwareStridesWraps(targetModel, op, bufferType, inputSizes,
273 inputStrides, sizes, strides);
274 int64_t offset = op.getOffsetInBytes();
275
276 // column
277 column = IntegerAttr::get(i32ty, tileCol);
278
279 // row
280 row = IntegerAttr::get(i32ty, tileRow);
281
282 bool skipTransformationChecks = op.isLinearTransferWithoutTransformation();
283 if (failed(verifyStridesWraps(op, bufferType, tileCol, tileRow, inputSizes,
284 inputStrides, sizes, strides,
285 skipTransformationChecks))) {
286 return failure();
287 }
288
289 // arg_idx and offset for block arguments
290 AIE::RuntimeSequenceOp seq_op =
291 op->getParentOfType<AIE::RuntimeSequenceOp>();
292 if (!seq_op) {
293 op->emitOpError("NpuDmaMemcpyNdOps must have RuntimeSequenceOp parent at "
294 "time of lowering.");
295 return failure();
296 }
297
298 mlir::Value rootMemref = memref;
299 int64_t subviewOffset = 0;
300
301 // Trace through memref.subview and memref.reinterpret_cast chain, if any,
302 // to find root block argument
303 auto traceResult = traceSubviewToBlockArgument(memref);
304 if (!traceResult) {
305 return op->emitOpError(
306 "memref must be a block argument or subview/cast/reinterpret_cast of "
307 "a block argument with static offsets, sizes, and strides");
308 }
309 rootMemref = traceResult->rootArg;
310 subviewOffset = traceResult->offsetInBytes;
311
312 // Find the argument index of the root memref
313 Block &entryBB = seq_op.getBody().front();
314 int arg_idx = -1;
315 for (int i = 0, e = entryBB.getNumArguments(); i < e; i++) {
316 if (entryBB.getArgument(i) == rootMemref) {
317 arg_idx = i;
318 break;
319 }
320 }
321 if (arg_idx < 0)
322 return failure();
323
324 offset += subviewOffset;
325
326 // bd_id
327 bd_id = IntegerAttr::get(i32ty, op.getId());
328
329 // buffer_length
330 uint64_t buffer_length_val = inputSizes[0] * op.getElementTypeBitwidth() /
331 targetModel.getAddressGenGranularity();
332 if (inputSizes.size() > 1) {
333 for (size_t i = 1; i < std::min(inputSizes.size(), (size_t)3); i++) {
334 buffer_length_val *= inputSizes[i];
335 }
336 }
337 buffer_length = IntegerAttr::get(i32ty, buffer_length_val);
338
339 // buffer_offset - zero because the complete address is set by the patch op
340 buffer_offset = IntegerAttr::get(i32ty, 0);
341
342 // enable_packet
343 if (auto packetInfo = op.getPacket()) {
344 enable_packet = IntegerAttr::get(i32ty, 1);
345 packet_type = IntegerAttr::get(i32ty, packetInfo->getPktType());
346 packet_id = IntegerAttr::get(i32ty, packetInfo->getPktId());
347 }
348
349 // out_of_order_id
350
351 if (!op.isLinearTransferWithoutTransformation()) {
352 // d0_size, d0_stride
353 d0_size = IntegerAttr::get(i32ty, sizes[0]);
354 d0_stride = IntegerAttr::get(i32ty, strides[0]);
355
356 // d1_size, d1_stride
357 d1_size = IntegerAttr::get(i32ty, sizes[1]);
358 d1_stride = IntegerAttr::get(i32ty, strides[1]);
359
360 // d2_stride
361 d2_stride = IntegerAttr::get(i32ty, strides[2]);
362
363 // d2_size
364 if (targetModel.isMemTile(tileCol, 0)) // Need to be any row
365 d2_size = IntegerAttr::get(i32ty, sizes[2]);
366 else
367 d2_size = IntegerAttr::get(i32ty, 0);
368 }
369 // iteration_current, iteration_size, iteration_stride, repeat_count
370 if (inputSizes[3] > 1) {
371 if (inputStrides[3] > 0) {
372 iteration_size = IntegerAttr::get(i32ty, sizes[3]);
373 iteration_stride = IntegerAttr::get(i32ty, strides[3]);
374 } else {
375 // We allow users to encode the repeat_count as a dimension 3 stride
376 // of 0. This must lower to a iteration wrap of 0, so no stride is
377 // ever added. We then repeat the BD using the repeat_count in
378 // NpuPushQueueOp.
379 iteration_size = zero;
380 iteration_stride = zero;
381 }
382 }
383 repeat_count = IntegerAttr::get(i32ty, sizes[3]);
384
385 // next_bd
386
387 // use_next_bd
388
389 // valid_bd
390 valid_bd = IntegerAttr::get(i32ty, 1);
391
392 // lock_rel_val
393
394 // lock_rel_id
395
396 // lock_acq_enable
397
398 // lock_acq_val
399
400 // lock_acq_id
401
402 // d0_zero_before
403 d0_zero_before = IntegerAttr::get(i32ty, op.getD0ZeroBefore());
404
405 // d1_zero_before
406 d1_zero_before = IntegerAttr::get(i32ty, op.getD1ZeroBefore());
407
408 // d2_zero_before
409 d2_zero_before = IntegerAttr::get(i32ty, op.getD2ZeroBefore());
410
411 // d0_zero_after
412 d0_zero_after = IntegerAttr::get(i32ty, op.getD0ZeroAfter());
413
414 // d1_zero_after
415 d1_zero_after = IntegerAttr::get(i32ty, op.getD1ZeroAfter());
416
417 // d2_zero_after
418 d2_zero_after = IntegerAttr::get(i32ty, op.getD2ZeroAfter());
419
420 // burst_size
421 burst_length = IntegerAttr::get(i32ty, op.getBurstLength());
422
423 // Set the issue_token
424 issue_token = BoolAttr::get(ctx, op.getIssueToken());
425 // Earlier, all S2MM channels were implicitly assumed to issue a token.
426 // This logic is kept for now for backward compatibility.
427 if (!isMM2S)
428 issue_token = BoolAttr::get(ctx, true);
429
430 if (targetModel.isMemTile(tileCol, tileRow) && (!isMM2S) &&
431 (op.getD0ZeroBefore() != 0 || op.getD0ZeroAfter() != 0 ||
432 op.getD1ZeroBefore() != 0 || op.getD1ZeroAfter() != 0 ||
433 op.getD2ZeroBefore() != 0 || op.getD2ZeroAfter() != 0)) {
434 op->emitOpError("MemTile supports zero padding only on MM2S direction");
435 return failure();
436 }
437
438 // write the buffer descriptor to the array
439 NpuWriteBdOp::create(
440 rewriter, op->getLoc(), column, bd_id, buffer_length, buffer_offset,
441 enable_packet, out_of_order_id, packet_id, packet_type, d0_size,
442 d0_stride, d1_size, d1_stride, d2_size, d2_stride, iteration_current,
443 iteration_size, iteration_stride, next_bd, row, use_next_bd, valid_bd,
444 lock_rel_val, lock_rel_id, lock_acq_enable, lock_acq_val, lock_acq_id,
445 d0_zero_before, d1_zero_before, d2_zero_before, d0_zero_after,
446 d1_zero_after, d2_zero_after, burst_length);
447
448 // compute the location of the address to patch in the bd and emit patch
449 // instruction to perform the patch.
450 uint64_t addr = targetModel.getDmaBdAddress(tileCol, tileRow, op.getId()) +
451 targetModel.getDmaBdAddressOffset(tileCol, tileRow);
452 NpuAddressPatchOp::create(rewriter, op->getLoc(), addr, arg_idx, offset);
453
454 // push the patched bd onto the dma task queue
455 NpuPushQueueOp::create(
456 rewriter, op->getLoc(), column, row, infoOp.getChannelDirAttr(),
457 infoOp.getChannelIndexAttr(), issue_token, repeat_count, bd_id);
458
459 rewriter.eraseOp(op);
460 return success();
461 }
462};

464/// Convert NpuDmaWaitOp into NpuSyncOp by retrieving the necessary
465/// information from the ShimDMAAllocationOp referenced through the
466/// symbol argument of this op.
467struct DmaWaitToSyncPattern : OpConversionPattern<NpuDmaWaitOp> {
468
469public:
470 using OpConversionPattern::OpConversionPattern;
471
472 DmaWaitToSyncPattern(MLIRContext *context, PatternBenefit benefit = 1)
473 : OpConversionPattern(context, benefit) {}
474
475 LogicalResult
476 matchAndRewrite(NpuDmaWaitOp op, OpAdaptor adaptor,
477 ConversionPatternRewriter &rewriter) const override {
478 AIE::DeviceOp dev = op->getParentOfType<AIE::DeviceOp>();
479 if (!dev)
480 return op->emitError("couldn't find parent of type DeviceOp");
481
482 AIE::ShimDMAAllocationOp shimDmaAllocOp =
483 AIE::ShimDMAAllocationOp::getForSymbol(dev, op.getSymbol());
484 if (!shimDmaAllocOp) {
485 return op->emitError("couldn't find shim_dma_allocation op");
486 }
487
488 AIE::TileOp shimTile = shimDmaAllocOp.getTileOp();
489 if (!shimTile) {
490 return op->emitError(
491 "shim_dma_allocation op must reference a valid TileOp");
492 }
493
494 // Create with `column_num == 1` and `row_num == 1` to check for a single
495 // column and row.
496 (void)rewriter.replaceOpWithNewOp<NpuSyncOp>(
497 op, shimTile.getCol(), shimTile.getRow(),
498 static_cast<uint32_t>(shimDmaAllocOp.getChannelDir()),
499 shimDmaAllocOp.getChannelIndex(), 1, 1);
500
501 return success();
502 }
503};

505struct WriteBdToBlockWritePattern : OpConversionPattern<NpuWriteBdOp> {
506 using OpConversionPattern::OpConversionPattern;
507
508public:
509 WriteBdToBlockWritePattern(MLIRContext *context, PatternBenefit benefit = 1)
510 : OpConversionPattern(context, benefit) {}
511
512 LogicalResult
513 matchAndRewrite(NpuWriteBdOp op, OpAdaptor adaptor,
514 ConversionPatternRewriter &rewriter) const override {
515
516 AIE::DeviceOp dev = op->getParentOfType<AIE::DeviceOp>();
517 const AIE::AIETargetModel &tm = dev.getTargetModel();
518 int col = op.getColumn();
519 int row = op.getRow();
520
521 int num_words = 0;
522 if (isa<AIE::AIE2TargetModel>(tm)) {
523 // Tile DMAs have 6 words, MemTile and Shim have 8 words
524 if (tm.isCoreTile(col, row))
525 num_words = 6;
526 else
527 num_words = 8;
528 } else {
529 llvm_unreachable(
530 "Unsupported AIETargetModel in WriteBdToBlockWritePattern");
531 }
532
533 std::vector<uint32_t> words(num_words, 0);
534
535 uint32_t bd_id = op.getBdId();
536 uint64_t bd_addr = tm.getDmaBdAddress(col, row, bd_id);
537 if (tm.isShimNOCTile(col, row)) {
538 // DMA_BDX_0
539 words[0] = op.getBufferLength();
540
541 // DMA_BDX_1
542 words[1] = op.getBufferOffset();
543
544 // DMA_BDX_2
545 // En Packet , OoO BD ID , Packet ID , Packet Type
546 words[2] |= (op.getEnablePacket() & 0x1) << 30;
547 words[2] |= (op.getOutOfOrderId() & 0x3f) << 24;
548 words[2] |= (op.getPacketId() & 0x1f) << 19;
549 words[2] |= (op.getPacketType() & 0x7) << 16;
550
551 // DMA_BDX_3
552 // TODO: Secure Access
553 words[3] |= (op.getD0Size() & 0x3ff) << 20;
554 words[3] |= op.getD0Stride() & 0xfffff;
555
556 // DMA_BDX_4
557 words[4] = (getShimBurstLengthEncoding(tm, op.getBurstLength()) & 0x3)
558 << 30;
559 words[4] |= (op.getD1Size() & 0x3ff) << 20;
560 words[4] |= op.getD1Stride() & 0xfffff;
561
562 // DMA_BDX_5
563 // TODO: SIMID, AXQoS
564 words[5] |= (2 & 0xf) << 24; // AXCache = 2 to enable upsizing in NoC
565 words[5] |= op.getD2Stride() & 0xfffff;
566
567 // DMA_BDX_6
568 words[6] |= (op.getIterationCurrent() & 0x3f) << 26;
569 words[6] |= (op.getIterationSize() & 0x3f) << 20;
570 words[6] |= op.getIterationStride() & 0xfffff;
571
572 // DMA_BDX_7
573 // TODO: TLAST Suppress
574 words[7] |= (op.getNextBd() & 0xf) << 27;
575 words[7] |= (op.getUseNextBd() & 0x1) << 26;
576 words[7] |= (op.getValidBd() & 0x1) << 25;
577 words[7] |= (op.getLockRelVal() & 0x7f) << 18;
578 words[7] |= (op.getLockRelId() & 0xf) << 13;
579 words[7] |= (op.getLockAcqEnable() & 0x1) << 12;
580 words[7] |= (op.getLockAcqVal() & 0x7f) << 5;
581 words[7] |= op.getLockAcqId() & 0xf;
582 if (op.getD0ZeroBefore() || op.getD1ZeroBefore() ||
583 op.getD2ZeroBefore() || op.getD0ZeroAfter() || op.getD1ZeroAfter() ||
584 op.getD2ZeroAfter()) {
585 op->emitError("Zero padding is only available on MemTile");
586 }
587 } else if (tm.isMemTile(op.getColumn(), op.getRow())) {
588
589 // DMA_BDX_0
590 words[0] |= (op.getEnablePacket() & 0x1) << 31;
591 words[0] |= (op.getPacketType() & 0x7) << 28;
592 words[0] |= (op.getPacketId() & 0x1f) << 23;
593 words[0] |= (op.getOutOfOrderId() & 0x3f) << 17;
594 words[0] |= op.getBufferLength() & 0x1ffff;
595
596 // DMA_BDX_1
597 words[1] |= (op.getD0ZeroBefore() & 0x3F) << 26;
598 words[1] |= (op.getNextBd() & 0x3f) << 20;
599 words[1] |= (op.getUseNextBd() & 0x1) << 19;
600 words[1] |= op.getBufferOffset() & 0x7ffff;
601
602 // DMA_BDX_2
603 words[2] |= (op.getD0Size() & 0x3ff) << 17;
604 words[2] |= op.getD0Stride() & 0x1ffff;
605
606 // DMA_BDX_3
607 // TODO: Secure Access
608 words[3] |= (op.getD1ZeroBefore() & 0x1F) << 27;
609 words[3] |= (op.getD1Size() & 0x3ff) << 17;
610 words[3] |= op.getD1Stride() & 0x1ffff;
611
612 // DMA_BDX_4
613 // TODO: D2Size
614 words[4] |= (op.getD2ZeroBefore() & 0xF) << 27;
615 words[4] |= op.getD2Stride() & 0x1ffff;
616
617 // DMA_BDX_5
618 // ToDO: D3Stride
619 words[5] |= (op.getD2ZeroAfter() & 0xF) << 28;
620 words[5] |= (op.getD1ZeroAfter() & 0x1F) << 23;
621 words[5] |= (op.getD0ZeroAfter() & 0x3F) << 17;
622
623 // DMA_BDX_6
624 words[6] |= (op.getIterationCurrent() & 0x3f) << 23;
625 words[6] |= (op.getIterationSize() & 0x3f) << 17;
626 words[6] |= op.getIterationStride() & 0x1ffff;
627
628 // DMA_BDX_7
629 words[7] |= (op.getValidBd() & 0x1) << 31;
630 words[7] |= (op.getLockRelVal() & 0x7f) << 24;
631 words[7] |= (op.getLockRelId() & 0xff) << 16;
632 words[7] |= (op.getLockAcqEnable() & 0x1) << 15;
633 words[7] |= (op.getLockAcqVal() & 0x7f) << 8;
634 words[7] |= op.getLockAcqId() & 0xff;
635 } else {
636 // AIE2 Tile DMA - 6 words
637 // DMA_BDX_0
638 // Base_Address [27:14], Buffer_Length [13:0]
639 words[0] = ((op.getBufferOffset() / 4) & 0x3fff) << 14;
640 words[0] |= op.getBufferLength() & 0x3fff;
641
642 // DMA_BDX_1
643 // Enable_Compression [31], Enable_Packet [30], Out_Of_Order_BD_ID
644 // [29:24], Packet_ID [23:19], Packet_Type [18:16]
645 words[1] = 0; // Enable_Compression
646 words[1] |= (op.getEnablePacket() & 0x1) << 30;
647 words[1] |= (op.getOutOfOrderId() & 0x3f) << 24;
648 words[1] |= (op.getPacketId() & 0x1f) << 19;
649 words[1] |= (op.getPacketType() & 0x7) << 16;
650
651 // DMA_BDX_2
652 // D1_Stepsize [25:13], D0_Stepsize [12:0]
653 words[2] = (op.getD1Stride() & 0x1fff) << 13;
654 words[2] |= op.getD0Stride() & 0x1fff;
655
656 // DMA_BDX_3
657 // D1_Wrap [28:21], D0_Wrap [20:13], D2_Stepsize [12:0]
658 words[3] = (op.getD1Size() & 0xff) << 21;
659 words[3] |= (op.getD0Size() & 0xff) << 13;
660 words[3] |= op.getD2Stride() & 0x1fff;
661
662 // DMA_BDX_4
663 // Iteration_Current [24:19], Iteration_Wrap [18:13], Iteration_Stepsize
664 // [12:0]
665 words[4] = (op.getIterationCurrent() & 0x3f) << 19;
666 words[4] |= (op.getIterationSize() & 0x3f) << 13;
667 words[4] |= op.getIterationStride() & 0x1fff;
668
669 // DMA_BDX_5
670 // TLAST_Suppress [31], Next_BD [30:27], Use_Next_BD [26], Valid_BD [25],
671 // Lock_Rel_Value [24:18], Lock_Rel_ID [16:13], Lock_Acq_Enable [12],
672 // Lock_Acq_Value [11:5], Lock_Acq_ID [3:0]
673 words[5] = 0; // TLAST_Suppress
674 words[5] |= (op.getNextBd() & 0xf) << 27;
675 words[5] |= (op.getUseNextBd() & 0x1) << 26;
676 words[5] |= (op.getValidBd() & 0x1) << 25;
677 words[5] |= (op.getLockRelVal() & 0x7f) << 18;
678 words[5] |= (op.getLockRelId() & 0xf) << 13;
679 words[5] |= (op.getLockAcqEnable() & 0x1) << 12;
680 words[5] |= (op.getLockAcqVal() & 0x7f) << 5;
681 words[5] |= op.getLockAcqId() & 0xf;
682 }
683
684 memref::GlobalOp global = nullptr;
685 {
686 OpBuilder::InsertionGuard guard(rewriter);
687 rewriter.setInsertionPoint(op->getParentOfType<AIE::RuntimeSequenceOp>());
688 global = getOrCreateDataMemref(rewriter, dev, op.getLoc(), words);
689 }
690 auto memref = memref::GetGlobalOp::create(
691 rewriter, op.getLoc(), global.getType(), global.getName());
692
693 (void)rewriter.replaceOpWithNewOp<NpuBlockWriteOp>(
694 op, rewriter.getUI32IntegerAttr(bd_addr), memref.getResult(), nullptr,
695 nullptr, nullptr);
696 return success();
697 }
698};
