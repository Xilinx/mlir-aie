301/*
302 * Tile address format:
303 * --------------------------------------------
304 * | 7 bits 5 bits 18 bits |
305 * --------------------------------------------
306 * | Array offset | Column | Row | Tile addr |
307 * --------------------------------------------
308 */
309class TileAddress {
310public:
311 TileAddress(uint8_t column, uint8_t row, uint64_t arrayOffset = 0x000u)
312 : arrayOffset{arrayOffset}, column{column}, row{row} {}
313
314 // SFINAE is used here to choose the copy constructor for `TileAddress`,
315 // and this constructor for all other classes.
316 template <typename Op,
317 std::enable_if_t<!std::is_same_v<Op, TileAddress>, bool> = true>
318 TileAddress(Op &op)
319 : TileAddress{static_cast<uint8_t>(op.colIndex()),
320 static_cast<uint8_t>(op.rowIndex())} {}
321
322 uint64_t fullAddress(uint64_t registerOffset) const {
323 return (arrayOffset << TILE_ADDR_ARR_SHIFT) |
324 (static_cast<uint64_t>(column) << TILE_ADDR_COL_SHIFT) |
325 (static_cast<uint64_t>(row) << TILE_ADDR_ROW_SHIFT) | registerOffset;
326 }
327
328 bool isShim() const { return row == 0; }
329
330 operator uint16_t() const {
331 return (static_cast<uint16_t>(column) << TILE_ADDR_ROW_WIDTH) | row;
332 }
333
334 uint8_t col() const { return column; }
335
336 void clearRange(uint32_t rangeStart, uint32_t length);
337
338private:
339 uint64_t arrayOffset : 34;
340 uint8_t column : TILE_ADDR_COL_WIDTH;
341 uint8_t row : TILE_ADDR_ROW_WIDTH;
342};

376// This template can be instantiated to represent a bitfield in a register.
377template <uint8_t highBit, uint8_t lowBit = highBit>
378class Field final {
379public:
380 static_assert(highBit >= lowBit,
381 "The high bit should be higher than the low bit");
382 static_assert(highBit < sizeof(uint32_t) * 8u,
383 "The field must live in a 32-bit register");
384
385 static constexpr auto NUM_BITS_USED = (highBit - lowBit) + 1u;
386 static constexpr auto UNSHIFTED_MASK = (1u << NUM_BITS_USED) - 1u;
387 static_assert((lowBit != highBit) ^ (UNSHIFTED_MASK == 1),
388 "1 is a valid mask iff the field is 1 bit wide");
389
390 static constexpr auto SHIFTED_MASK = UNSHIFTED_MASK << lowBit;
391
392 [[nodiscard]] constexpr uint32_t operator()(uint32_t value) const {
393 return (value << lowBit) & SHIFTED_MASK;
394 }
395};

397/*
398 Add or replace a register value in 'mem_writes'
399*/
400static void write32(Address addr, uint32_t value) {
401 if (addr.destTile().col() <= 0)
402 llvm::report_fatal_error(
403 llvm::Twine("address of destination tile <= 0 : ") +
404 std::to_string(addr.destTile().col()));
405
406 auto ret = memWrites.emplace(addr, value);
407 if (!ret.second)
408 ret.first->second = value;
409}
410
411/*
412 Look up a value for a given address
413
414 If the address is found return the value, otherwise 0
415*/
416static uint32_t read32(Address addr) {
417 auto ret = memWrites.find(addr);
418 if (ret != memWrites.end())
419 return ret->second;
420
421 return 0;
422}

440 write32(Address{*this, off}, 0);

443/*
444 Read the ELF produced by the AIE compiler and include its loadable
445 output in the airbin ELF
446*/
447static void loadElf(TileAddress tile, const std::string &filename) {
448 LLVM_DEBUG(llvm::dbgs() << "Reading ELF file " << filename << " for tile "
449 << tile << '\n');
450
451 int elfFd = open(filename.c_str(), O_RDONLY);
452 if (elfFd < 0)
453 llvm::report_fatal_error(llvm::Twine("Can't open elf file ") + filename);
454
455 elf_version(EV_CURRENT);
456 Elf *inElf = elf_begin(elfFd, ELF_C_READ, nullptr);
457
458 // check the characteristics
459 GElf_Ehdr *ehdr;
460 GElf_Ehdr ehdrMem;
461 ehdr = gelf_getehdr(inElf, &ehdrMem);
462 if (!ehdr)
463 llvm::report_fatal_error(llvm::Twine("cannot get ELF header: ") +
464 elf_errmsg(-1));
465
466 // Read data as 32-bit little endian
467 assert(ehdr->e_ident[EI_CLASS] == ELFCLASS32 &&
468 "(ehdr->e_ident[EI_CLASS] != ELFCLASS32");
469 assert(ehdr->e_ident[EI_DATA] == ELFDATA2LSB &&
470 "ehdr->e_ident[EI_DATA] != ELFDATA2LSB");
471
472 size_t phnum;
473 if (elf_getphdrnum(inElf, &phnum) != 0)
474 llvm::report_fatal_error(llvm::Twine("cannot get program header count: ") +
475 elf_errmsg(-1));
476
477 // iterate through all program headers
478 for (unsigned int ndx = 0; ndx < phnum; ndx++) {
479 GElf_Phdr phdrMem;
480 GElf_Phdr *phdr = gelf_getphdr(inElf, ndx, &phdrMem);
481 if (!phdr)
482 llvm::report_fatal_error(llvm::Twine("cannot get program header entry ") +
483 std::to_string(ndx) + ": " + elf_errmsg(-1));
484
485 // for each loadable program header
486 if (phdr->p_type != PT_LOAD)
487 continue;
488
489 // decide destination address based on header attributes
490 uint32_t dest;
491 if (phdr->p_flags & PF_X)
492 dest = ME_PROG_MEM_BASE + phdr->p_vaddr;
493 else
494 dest = ME_DATA_MEM_BASE + (phdr->p_vaddr & (DATA_MEM_SIZE - 1));
495
496 LLVM_DEBUG(llvm::dbgs()
497 << llvm::format("ELF flags=0x%x vaddr=0x%lx dest=0x%x\r\n",
498 phdr->p_flags, phdr->p_vaddr, dest));
499 // read data one word at a time and write it to the output list
500 // TODO since we know these are data and not registers, we could likely
501 // bypass the output list and write a section directly into the AIRBIN
502 size_t elfSize;
503 uint32_t offset;
504 char *raw = elf_rawfile(inElf, &elfSize);
505
506 for (offset = phdr->p_offset; offset < phdr->p_offset + phdr->p_filesz;
507 offset += 4) {
508 Address destAddr{tile, dest};
509 uint32_t data = *reinterpret_cast<uint32_t *>(raw + offset);
510 write32(destAddr, data);
511 dest += 4;
512 }
513 }
514
515 elf_end(inElf);
516 close(elfFd);
517}

540static void configMETile(TileOp tileOp, const std::string &coreFilesDir) {
541 TileAddress tileAddress{tileOp};
542 // Reset configuration
543
544 // clear program and data memory
545 tileAddress.clearRange(ME_PROG_MEM_BASE, PROG_MEM_SIZE);
546 tileAddress.clearRange(ME_DATA_MEM_BASE, DATA_MEM_SIZE);

555 tileAddress.clearRange(ME_SS_SLAVE_CFG_BASE, sizeof(MESSSlaveCfgBlock));
556 tileAddress.clearRange(ME_SS_SLAVE_SLOT_BASE, sizeof(MESSSlaveSlotBlock));
557
558 // NOTE: Here is usually where locking is done.
559 // However, the runtime will handle that when loading the airbin.
560
561 // read the AIE executable and copy the loadable parts
562 if (auto coreOp = tileOp.getCoreOp()) {
563 std::string fileName;
564 if (auto fileAttr = coreOp->getAttrOfType<StringAttr>("elf_file")) {
565 fileName = fileAttr.str();
566 } else {
567 return coreOp.emitOpError()
568 << "Expected lowered ELF file to be given as attribute "
569 "`elf_file` for this core. Compile cores first.";
570 }
571 loadElf(tileAddress, fileName);
572 }

612static void configureDMAs(DeviceOp &targetOp) {
613 Field<1> dmaChannelReset;
614 Field<0> dmaChannelEnable;
615
616 for (auto memOp : targetOp.getOps<MemOp>()) {
617 TileAddress tile{memOp};
618 LLVM_DEBUG(llvm::dbgs() << "DMA: tile=" << memOp.getTile());
619 // Clear the CTRL and QUEUE registers for the DMA channels.
620 for (auto chNum = 0u; chNum < DMA_S2MM_CHANNEL_COUNT; ++chNum) {
621 write32({tile, regDMAS2MMCtrl(chNum)},
622 dmaChannelReset(DISABLE) | dmaChannelEnable(DISABLE));
623 write32({tile, regDMAS2MMQueue(chNum)}, 0);
624 }
625 for (auto chNum = 0u; chNum < DMA_MM2S_CHANNEL_COUNT; ++chNum) {
626 write32({tile, regDMAMM2SCtrl(chNum)},
627 dmaChannelReset(DISABLE) | dmaChannelEnable(DISABLE));
628 write32({tile, regDMAMM2SQueue(chNum)}, 0);
629 }
630
631 DenseMap<Block *, int> blockMap;
632
633 {
634 // Assign each block a BD number
635 auto bdNum = 0;
636 for (auto &block : memOp.getBody()) {
637 if (!block.getOps<DMABDOp>().empty()) {
638 blockMap[&block] = bdNum;
639 bdNum++;
640 }
641 }
642 }
643
644 for (auto &block : memOp.getBody()) {
645 auto bdInfo = getBDInfo(block);
646
647 if (bdInfo.hasA and bdInfo.hasB) {
648 bdInfo.abMode = ENABLE;
649 if (bdInfo.lenA != bdInfo.lenB)
650 llvm::errs() << "ABmode must have matching lengths.\n";
651 if (bdInfo.bytesA != bdInfo.bytesB)
652 llvm::errs() << "ABmode must have matching element data types.\n";
653 }
654
655 int acqValue = 0, relValue = 0;
656 auto acqEnable = DISABLE;
657 auto relEnable = DISABLE;
658 std::optional<int> lockID = std::nullopt;
659
660 for (auto op : block.getOps<UseLockOp>()) {
661 LockOp lock = dyn_cast<LockOp>(op.getLock().getDefiningOp());
662 lockID = lock.getLockIDValue();
663 if (op.acquire()) {
664 acqEnable = ENABLE;
665 acqValue = op.getLockValue();
666 } else {
667 relEnable = ENABLE;
668 relValue = op.getLockValue();
669 }
670 }
671
672 // We either
673 // a. went thru the loop once (`lockID` should be something) xor
674 // b. did not enter the loop (the enables should be both disable)
675 assert(lockID.has_value() ^
676 (acqEnable == DISABLE and relEnable == DISABLE) &&
677 "lock invariants not satisfied");
678
679 for (auto op : block.getOps<DMABDPACKETOp>()) {
680 bdInfo.foundBDPacket = true;
681 bdInfo.packetType = op.getPacketType();
682 bdInfo.packetID = op.getPacketID();
683 }
684
685 auto bdNum = blockMap[&block];
686 MERegDMABD bdData;
687 if (bdInfo.foundBD) {
688 Field<25, 22> bdAddressLockID;
689 Field<21> bdAddressReleaseEnable;
690 Field<20> bdAddressReleaseValue;
691 Field<19> bdAddressReleaseValueEnable;
692 Field<18> bdAddressAcquireEnable;
693 Field<17> bdAddressAcquireValue;
694 Field<16> bdAddressAcquireValueEnable;
695
696 if (bdInfo.hasA) {
697 bdData.addrA = bdAddressLockID(lockID.value()) |
698 bdAddressReleaseEnable(relEnable) |
699 bdAddressAcquireEnable(acqEnable);
700 if (relValue != 0xFFu)
701 bdData.addrA |= bdAddressReleaseValueEnable(true) |
702 bdAddressReleaseValue(relValue);
703 if (acqValue != 0xFFu)
704 bdData.addrA |= bdAddressAcquireValueEnable(true) |
705 bdAddressAcquireValue(acqValue);
706 }
707 if (bdInfo.hasB)
708 llvm::report_fatal_error("bdInfo.hasB not supported");
709
710 auto addrA = bdInfo.baseAddrA + bdInfo.offsetA;
711 auto addrB = bdInfo.baseAddrB + bdInfo.offsetB;
712
713 Field<12, 0> bdAddressBase, bdControlLength;
714 Field<30> bdControlABMode;
715 Field<28> bdControlFifo;
716
717 bdData.addrA |= bdAddressBase(addrA >> 2u);
718 bdData.addrB |= bdAddressBase(addrB >> 2u);
719 bdData.control |= bdControlLength(bdInfo.lenA - 1) |
720 bdControlFifo(bdInfo.fifoMode) |
721 bdControlABMode(bdInfo.abMode);
722
723 if (block.getNumSuccessors() > 0) {
724 // should have only one successor block
725 assert(block.getNumSuccessors() == 1 &&
726 "block.getNumSuccessors() != 1");
727 auto *nextBlock = block.getSuccessors()[0];
728 auto nextBDNum = blockMap[nextBlock];
729
730 Field<16, 13> bdControlNextBD;
731 Field<17> bdControlEnableNextBD;
732
733 bdData.control |= bdControlEnableNextBD(nextBDNum != 0xFFu) |
734 bdControlNextBD(nextBDNum);
735 }
736
737 if (bdInfo.foundBDPacket) {
738 Field<14, 12> bdPacketType;
739 Field<4, 0> bdPacketID;
740 Field<27> bdControlEnablePacket;
741
742 bdData.packet =
743 bdPacketID(bdInfo.packetID) | bdPacketType(bdInfo.packetType);
744 bdData.control |= bdControlEnablePacket(ENABLE);
745 }
746
747 Field<31> bdControlValid;
748
749 assert(bdNum < ME_DMA_BD_COUNT && "bdNum >= ME_DMA_BD_COUNT");
750 uint64_t bdOffset = regDMAAddrABD(bdNum);
751
752 write32({tile, bdOffset}, bdData.addrA);
753 write32({tile, regDMAAddrBBD(bdNum)}, bdData.addrB);
754 write32({tile, regDMA2DXBD(bdNum)}, bdData.x2d);
755 write32({tile, regDMA2DYBD(bdNum)}, bdData.y2d);
756 write32({tile, regDMAPktBD(bdNum)}, bdData.packet);
757 write32({tile, regDMAIntStateBD(bdNum)}, bdData.interleave);
758 write32({tile, regDMACtrlBD(bdNum)},
759 bdData.control | bdControlValid(true));
760 }
761 }
762
763 for (auto &block : memOp.getBody()) {
764 for (auto op : block.getOps<DMAStartOp>()) {
765 auto bdNum = blockMap[op.getDest()];
766 if (bdNum != 0xFFU) {
767 Field<4, 0> dmaChannelQueueStartBd;
768
769 uint32_t chNum = op.getChannelIndex();
770 if (op.getChannelDir() == DMAChannelDir::MM2S) {
771 write32(Address{tile, regDMAMM2SQueue(chNum)},
772 dmaChannelQueueStartBd(bdNum));
773 write32({tile, regDMAMM2SCtrl(chNum)},
774 dmaChannelEnable(ENABLE) | dmaChannelReset(DISABLE));
775 } else {
776 write32(Address{tile, regDMAS2MMQueue(chNum)},
777 dmaChannelQueueStartBd(bdNum));
778 write32({tile, regDMAS2MMCtrl(chNum)},
779 dmaChannelEnable(ENABLE) | dmaChannelReset(DISABLE));
780 }
781 }
782 }
783 }
784 }

916 static constexpr auto STREAM_SWITCH_ARB_SHIFT = 0u;

919 auto config = streamMasterDropHeader(DROP_HEADER) |
920 (mask << STREAM_SWITCH_MSEL_SHIFT) |
921 (arbiter << STREAM_SWITCH_ARB_SHIFT);
922 Address dest{tile, regMESSMaster(masterPort)};
923 write32(dest, STREAM_ENABLE(ENABLE) | STREAM_PACKET_ENABLE(ENABLE) |
924 streamMasterDropHeader(DROP_HEADER) |
925 streamMasterConfig(config));

927 }
928 }

936 int msel = amselOp.getMselValue();
937
938 for (auto tile : switchboxSet) {
939 auto slavePort =
940 computeSlavePort(connectOp.getSourceBundle(),
941 connectOp.sourceIndex(), tile.isShim());

943 STREAM_ENABLE(ENABLE) | STREAM_PACKET_ENABLE(ENABLE));
944
945 Field<28, 24> streamSlotId;
946 Field<20, 16> streamSlotMask;
947 Field<8> streamSlotEnable;
948 Field<5, 4> streamSlotMSel;
949 Field<2, 0> streamSlotArbit;
950
951 auto config = streamSlotId(slotOp.valueInt()) |
952 streamSlotMask(slotOp.maskInt()) |
953 streamSlotEnable(ENABLE) | streamSlotMSel(msel) |
954 streamSlotArbit(arbiter);
955 write32({tile, regMESSSlaveSlot(slavePort, slot)}, config);
956 slot++;
957 }
958 }
959 }

1124static void groupSections(std::vector<Section *> &sections) {
1125 uint64_t lastAddr = 0;
1126 Section *section = nullptr;

1128 for (auto write : memWrites) {

1138 section->addData(write.second);
1139 lastAddr = write.first;
1140 }

1142 sections.push_back(section);

1183mlir::LogicalResult AIETranslateToAirbin(mlir::ModuleOp module,
1184 const std::string &outputFilename,
1185 const std::string &coreFilesDir,
1186 llvm::StringRef deviceName,
1187 bool testAirBin) {
1188 int tmpElfFD;
1189 Elf *outElf;
1190 GElf_Ehdr ehdrMem;
1191 GElf_Ehdr *ehdr;
1192 GElf_Shdr *shdr;
1193 GElf_Shdr shdrMem;
1194 char emptyStr[] = "";
1195 char strTabName[] = ".shstrtab";
1196 std::vector<Section *> sections;
1197
1198 DeviceOp targetOp = AIE::DeviceOp::getForSymbolInModule(module, deviceName);
1199 if (!targetOp) {
1200 LLVM_DEBUG(llvm::dbgs() << "no device ops found");
1201 return success();
1202 }
1203
1204 // Write the initial configuration for every tile specified in the MLIR.
1205 for (auto tileOp : targetOp.getOps<TileOp>()) {
1206 LLVM_DEBUG(llvm::dbgs() << "CC: tile=" << tileOp.getTileID());
1207 if (tileOp.isShimTile())
1208 configShimTile(tileOp);
1209 else
1210 configMETile(tileOp, coreFilesDir);
1211 }
1212
1213 configureSwitchBoxes(targetOp);
1214 configureCascade(targetOp);
1215 configureDMAs(targetOp);
1216 groupSections(sections);
1217
1218 LLVM_DEBUG(llvm::dbgs() << llvm::format("mem_writes: %lu in %lu sections\n",
1219 memWrites.size(), sections.size()));
1220
1221 elf_version(EV_CURRENT);
1222 tmpElfFD =
1223 open(outputFilename.c_str(), O_RDWR | O_CREAT | O_TRUNC, DEFFILEMODE);
1224 outElf = elf_begin(tmpElfFD, ELF_C_WRITE, nullptr);
1225
1226 if (!gelf_newehdr(outElf, ELFCLASS64))
1227 llvm::report_fatal_error(llvm::Twine("Error creating ELF64 header: ") +
1228 elf_errmsg(-1));
1229
1230 ehdr = gelf_getehdr(outElf, &ehdrMem);
1231 if (!ehdr)
1232 llvm::report_fatal_error(llvm::Twine("cannot get ELF header: ") +
1233 elf_errmsg(-1));
1234
1235 // Initialize header.
1236 ehdr->e_ident[EI_DATA] = ELFDATA2LSB;
1237 ehdr->e_ident[EI_OSABI] = ELFOSABI_GNU;
1238 ehdr->e_type = ET_NONE;
1239 ehdr->e_machine = EM_AMDAIR;
1240 ehdr->e_version = EV_CURRENT;
1241 if (gelf_update_ehdr(outElf, ehdr) == 0)
1242 llvm::report_fatal_error(llvm::Twine("cannot update ELF header: ") +
1243 elf_errmsg(-1));
1244
1245 // Create new section for the 'section header string table'
1246 Elf_Scn *shStrTabScn = elf_newscn(outElf);
1247 if (!shStrTabScn)
1248 llvm::report_fatal_error(
1249 llvm::Twine("cannot create new shstrtab section: ") + elf_errmsg(-1));
1250
1251 // the first entry in the string table must be a NULL string
1252 addString(shStrTabScn, emptyStr);
1253
1254 shdr = gelf_getshdr(shStrTabScn, &shdrMem);
1255 if (!shdr)
1256 llvm::report_fatal_error(
1257 llvm::Twine("cannot get header for sh_strings section: ") +
1258 elf_errmsg(-1));
1259
1260 shdr->sh_type = SHT_STRTAB;
1261 shdr->sh_flags = 0;
1262 shdr->sh_addr = 0;
1263 shdr->sh_link = SHN_UNDEF;
1264 shdr->sh_info = SHN_UNDEF;
1265 shdr->sh_addralign = 1;
1266 shdr->sh_entsize = 0;
1267 shdr->sh_name = addString(shStrTabScn, strTabName);
1268
1269 // add all the AIRBIN-specific section names up front and index them
1270 for (uint8_t secIdx = SEC_IDX_SSMAST; secIdx < SEC_IDX_MAX; secIdx++)
1271 secNameOffset[secIdx] = addString(shStrTabScn, secNameStr[secIdx]);
1272 secNameOffset[SEC_IDX_NULL] = 0;
1273
1274 // We have to store the section strtab index in the ELF header so sections
1275 // have actual names.
1276 int ndx = elf_ndxscn(shStrTabScn);
1277 ehdr->e_shstrndx = ndx;
1278
1279 if (!gelf_update_ehdr(outElf, ehdr))
1280 llvm::report_fatal_error(llvm::Twine("cannot update ELF header: ") +
