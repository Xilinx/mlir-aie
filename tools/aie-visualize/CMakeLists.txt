  ${dialect_libs}
  MLIRParser

  MLIRAffineToStandard
  MLIRAIEVecDialect
  MLIRAIEVecToLLVM
  MLIRAIEVecTransforms

  AIE
  AIETransforms
  AIETargets
  AIEX
  AIEXTransforms
  MLIRAIEVecDialect
  MLIRAIEVecAIE1Dialect
  MLIRXLLVMDialect)

  RUNTIME DESTINATION ${LLVM_TOOLS_INSTALL_DIR}
  COMPONENT aie-visualize)
