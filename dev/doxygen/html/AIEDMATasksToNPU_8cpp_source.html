35struct DMAStartTaskOpPattern : OpConversionPattern<DMAStartTaskOp> {
36 using OpConversionPattern::OpConversionPattern;
37
38 LogicalResult
39 matchAndRewrite(DMAStartTaskOp op, OpAdaptor adaptor,
40 ConversionPatternRewriter &rewriter) const override {
41 DMAConfigureTaskOp task_op = op.getTaskOp();
42 if (!task_op) {
43 // Cannot rewrite this; probably points to a DMAStartTaskForOp,
44 // which we will lower once it has been rewritten into a DMAStartTaskOp.
45 return failure();
46 }
47 AIE::TileOp tile = task_op.getTileOp();
48 Location loc = op.getLoc();
49
50 // The bd_id for the queue push: the runtime pool value (dynamic free-list)
51 // if the configure carries one, else the statically-assigned first BD id.
52 Value bdIdVal;
53 if (Value runtimeBdId = task_op.getBdIdVal()) {
54 bdIdVal = runtimeBdId;
55 } else {
56 std::optional<uint32_t> first_bd_id = task_op.getFirstBdId();
57 if (!first_bd_id) {
58 auto err = op.emitOpError(
59 "First buffer descriptor in chain has not been assigned an ID");
60 err.attachNote()
61 << "Run the `aie-assign-runtime-buffer-descriptor-ids` "
62 "pass first or manually assign an ID.";
63 return failure();
64 }
65 bdIdVal = createConstantI32(rewriter, loc, *first_bd_id);
66 }
67 // push_queue takes bd_id + repeat_count as SSA operands. repeat_count is a
68 // runtime operand when present (dynamic tile count), else the compile-time
69 // attribute materialized as a constant.
70 Value repeatCount = getAsValue(rewriter, loc, task_op.getRepeatCountValue(),
71 rewriter.getI32Type());
72 rewriter.replaceOpWithNewOp<NpuPushQueueOp>(
73 op, tile.getCol(), tile.getRow(), task_op.getDirection(),
74 task_op.getChannel(), task_op.getIssueToken(), repeatCount, bdIdVal);
75 return success();
76 }
77};

79// Resolve a task value to a configure that gives the sync its physical channel.
80// The value is usually a configure result, but under the dynamic BD pool path a
81// task threaded through runtime control flow surfaces as an scf result: a loop
82// result (the task carried across iterations) or an scf.if result (the phi of
83// the task in flight, whichever branch ran). Walk such a result back to a
84// configure via the yields (and, for a loop, the init). Every reachable
85// configure targets the same physical channel -- the pool pass verified both
86// branches of an scf.if agree -- so the first one found gives the right
87// channel.
88static DMAConfigureTaskOp resolveConfigureThroughCF(Value task) {
89 llvm::SmallPtrSet<Value, 8> seen;
90 SmallVector<Value> worklist{task};
91 while (!worklist.empty()) {
92 Value v = worklist.pop_back_val();
93 if (!v || !seen.insert(v).second)
94 continue;
95 if (auto cfg = v.getDefiningOp<DMAConfigureTaskOp>())
96 return cfg;
97 if (auto res = dyn_cast<OpResult>(v)) {
98 Operation *def = res.getOwner();
99 unsigned k = res.getResultNumber();
100 if (auto ifOp = dyn_cast<scf::IfOp>(def)) {
101 worklist.push_back(ifOp.thenBlock()->getTerminator()->getOperand(k));
102 worklist.push_back(ifOp.elseBlock()->getTerminator()->getOperand(k));
103 } else if (auto forOp = dyn_cast<scf::ForOp>(def)) {
104 worklist.push_back(forOp.getInitArgs()[k]);
105 worklist.push_back(forOp.getBody()->getTerminator()->getOperand(k));
106 }
107 }
108 }
109 return nullptr;
110}

112struct DMAAwaitTaskOpPattern : OpConversionPattern<DMAAwaitTaskOp> {
113 using OpConversionPattern::OpConversionPattern;
114
115 LogicalResult
116 matchAndRewrite(DMAAwaitTaskOp op, OpAdaptor adaptor,
117 ConversionPatternRewriter &rewriter) const override {
118 DMAConfigureTaskOp task_op = op.getTaskOp();
119 if (!task_op) {
120 // The awaited task threads through runtime control flow (an scf result),
121 // so it has no directly-defining configure; walk the SSA carry to one.
122 task_op = resolveConfigureThroughCF(op.getTask());
123 }
124 if (!task_op) {
125 return failure();
126 }
127 if (!task_op.getIssueToken()) {
128 auto err = op.emitOpError(
129 "Cannot wait on a BD that is not configured to issue a token.");
130 err.attachNote(task_op.getLoc())
131 << "Consider adding attribute `issue_token=true` here.";
132 return err;
133 }
134 AIE::TileOp tile = task_op.getTileOp();
135 Location loc = op.getLoc();
136 rewriter.replaceOpWithNewOp<NpuSyncOp>(
137 op, createConstantI32(rewriter, loc, tile.getCol()),
138 createConstantI32(rewriter, loc, tile.getRow()),
139 createConstantI32(rewriter, loc, (uint32_t)task_op.getDirection()),
140 createConstantI32(rewriter, loc, task_op.getChannel()),
141 createConstantI32(rewriter, loc, 1),
142 createConstantI32(rewriter, loc, 1));
143 return success();
144 }
145};

150 bool shouldSkipBlock(Block &block) {
151 // Allow blocks in the input IR that contain nothing but a next_bd operation
152 // as the entry block. We will skip these blocks and not lower them to
153 // anything.
154 auto it = block.without_terminator();
155 return block.isEntryBlock() && it.begin() == it.end();
156 }

158 LogicalResult verifyBdInBlock(Block &block, bool hasRuntimeBdId = false) {
159 auto bd_ops = block.getOps<AIE::DMABDOp>();
160 // Exactly one BD op per block
161 int n_bd_ops = std::distance(bd_ops.begin(), bd_ops.end());
162 if (n_bd_ops < 1) {
163 auto error = block.getTerminator()->emitError(
164 "Block ending in this terminator does not contain a required "
165 "aie.dma_bd operation.");
166 error.attachNote(block.getParentOp()->getLoc())
167 << "Error encountered while lowering this BD configuration.";
168 return failure();
169 } else if (n_bd_ops > 1) {
170 auto error = block.getTerminator()->emitOpError(
171 "This block contains multiple aie.dma_bd operations. Exactly one is "
172 "required.");
173 auto it = bd_ops.begin();
174 ++it;
175 for (; it != bd_ops.end(); ++it) {
176 error.attachNote((*it)->getLoc()) << "Extra aie.dma_bd operation here.";
177 }
178 return failure();
179 }
180 AIE::DMABDOp bd_op = *bd_ops.begin();
181 // A runtime bd_id (dynamic free-list pool, on the configure's bd_id_val)
182 // takes the place of the static attribute; only require the attribute when
183 // there is no runtime id.
184 if (!hasRuntimeBdId && !bd_op.getBdId().has_value()) {
185 auto error = bd_op.emitOpError(
186 "Cannot lower buffer descriptor without assigned ID.");
187 error.attachNote()
188 << "Run the `--aie-assign-runtime-sequence-bd-ids` pass first or "
189 "manually assign an ID to this buffer descriptor.";
190 error.attachNote(block.getParentOp()->getLoc())
191 << "Error encountered while lowering this BD configuration.";
192 return failure();
193 }
194 return success();
195 }
196
197 LogicalResult verifyOptionalLocksInBlock(Block &block) {
198 auto lock_ops = block.getOps<AIE::UseLockOp>();
199 int n_lock_ops = std::distance(lock_ops.begin(), lock_ops.end());
200 // Allow exactly 0 or 2 lock ops (acquire and release)
201 if (n_lock_ops != 0 && n_lock_ops != 2) {
202 AIE::UseLockOp lock_op = *lock_ops.begin();
203 lock_op.emitOpError(
204 "BD blocks must have either 0 or 2 lock operations (acquire and "
205 "release). Found ")
206 << n_lock_ops << " lock operations.";
207 return failure();
208 }
209 return success();
210 }
211
212 LogicalResult verifyNoUnsupportedOpsInBlock(Block &block) {
213 WalkResult unsupported_ops = block.walk([&](Operation *inner_op) {
214 return llvm::TypeSwitch<Operation *, WalkResult>(inner_op)
215 .Case<AIE::DMABDOp>(
216 [&](AIE::DMABDOp bd_op) { return WalkResult::advance(); })
217 .Case<AIE::UseLockOp>(
218 [&](AIE::UseLockOp lock_op) { return WalkResult::advance(); })
219 .Case<arith::ConstantOp>(
220 [&](arith::ConstantOp const_op) { return WalkResult::advance(); })
221 .Case<AIE::NextBDOp>(
222 [&](AIE::NextBDOp lock_op) { return WalkResult::advance(); })
223 .Case<AIE::EndOp>(
224 [&](AIE::EndOp lock_op) { return WalkResult::advance(); })
225 .Default([&](Operation *inner_op) {
226 auto error = block.getParentOp()->emitOpError(
227 "Unsupported operation within BD block.");
228 error.attachNote(inner_op->getLoc())
229 << "No lowering to NPU instructions available for this "
230 "operation.";
231 return WalkResult::interrupt();
232 });
233 });
234 if (unsupported_ops.wasInterrupted()) {
235 return failure();
236 }
237 return success();
238 }

240 AIE::DMABDOp getBdForBlock(Block &block) {
241 auto bd_ops = block.getOps<AIE::DMABDOp>();
242 AIE::DMABDOp bd_op = *bd_ops.begin(); // Dereference first (and only, after
243 // previous checks) bd op iterator
244 return bd_op;

247 // Returns pair of (acquire_lock_op, release_lock_op) if present
248 std::optional<std::pair<AIE::UseLockOp, AIE::UseLockOp>>
249 getOptionalLockOpsForBlock(Block &block) {
250 auto lock_ops = block.getOps<AIE::UseLockOp>();
251 int n_lock_ops = std::distance(lock_ops.begin(), lock_ops.end());
252 if (n_lock_ops != 2) {
253 return std::nullopt;
254 }
255
256 AIE::UseLockOp acquire_op = nullptr;
257 AIE::UseLockOp release_op = nullptr;
258
259 for (auto lock_op : lock_ops) {
260 if (lock_op.acquire() || lock_op.acquireGE()) {
261 acquire_op = lock_op;
262 } else if (lock_op.release()) {
263 release_op = lock_op;
264 }
265 }
266
267 if (acquire_op && release_op) {
268 return std::make_pair(acquire_op, release_op);
269 }
270 return std::nullopt;
271 }

273 LogicalResult setAddressForSingleBD(OpBuilder &builder, AIE::DMABDOp &bd_op,
274 AIE::TileOp &tile,
275 OpFoldResult bdId = {}) {
276 const AIE::AIETargetModel &target_model = AIE::getTargetModel(bd_op);
277 auto buf = bd_op.getBuffer();
278 auto col = tile.getCol();
279 auto row = tile.getRow();
280 // The static register address uses the pinned bd_id attribute; on the
281 // runtime pool path the attribute is absent and runtimeRegisterAddr (below)
282 // supplies the address instead, so fall back to bd 0 for the constant.
283 uint32_t bd_id = bd_op.getBdId().value_or(0);
284 uint64_t register_addr = target_model.getDmaBdAddress(col, row, bd_id) +
285 target_model.getDmaBdAddressOffset(col, row);
286 // On the runtime-bd_id path the patched register (BD buffer-address word)
287 // is itself runtime: getBdRegisterBase(bd_id) + getDmaBdAddressOffset.
288 // Emitted as an SSA operand on the address patch; null keeps the static
289 // constant.
290 Value runtimeRegisterAddr;
291 if (bdId && !getConstantIntValue(bdId)) {
292 Value base = getBdRegisterBase(builder, bd_op.getLoc(), target_model, col,
293 row, bdId);
294 runtimeRegisterAddr = arith::AddIOp::create(
295 builder, bd_op.getLoc(), base,
296 createConstantI32(builder, bd_op.getLoc(),
297 target_model.getDmaBdAddressOffset(col, row)));
298 }
299
300 // A buffer descriptor can refer to a statically allocated aie.buffer, or to
301 // a DDR buffer which will be passed as a runtime argument (block
302 // argument). Try to find the root block argument, either directly or
303 // through supported subview/view/cast chains.
304 mlir::BlockArgument buf_arg = nullptr;
305 int64_t offset = 0;
306
307 if (auto directArg = llvm::dyn_cast<mlir::BlockArgument>(buf)) {
308 buf_arg = directArg;
309 offset = 0;
310 } else if (auto traceResult = traceSubviewToBlockArgument(buf)) {
311 buf_arg = traceResult->rootArg;
312 offset = traceResult->offsetInBytes;
313 }
314
315 if (buf_arg) {
316 if (!target_model.isShimNOCTile(tile.getCol(), tile.getRow())) {
317 return bd_op->emitOpError("DDR memory (runtime input arguments) can "
318 "only be referred to on shim tiles.");
319 }
320
321 unsigned arg_idx = buf_arg.getArgNumber();
322 // arg_plus = buffer byte offset. The dma_bd offset is a single element
323 // offset (stride 1); a constant folds to a constant (byte-identical to
324 // before), a runtime offset operand is built with arith. `offset` here is
325 // the constant subview base in bytes.
326 OpFoldResult offsetOfr =
327 bd_op.getOffset() ? OpFoldResult(bd_op.getOffset())
328 : OpFoldResult(builder.getI32IntegerAttr(
329 bd_op.getConstantOffset().value_or(0)));
330 OpFoldResult oneStride = builder.getI32IntegerAttr(1);
331 Value argPlus =
332 buildArgPlusValue(builder, bd_op.getLoc(), {offsetOfr}, {oneStride},
333 bd_op.getBufferElementTypeWidthInBytes(), offset);
334 NpuAddressPatchOp::create(builder, bd_op.getLoc(),
335 /*addr*/ register_addr,
336 /*addr_val*/ runtimeRegisterAddr,
337 /*arg_idx*/ arg_idx, argPlus);
338 } else if (AIE::BufferOp buffer =
339 llvm::dyn_cast<AIE::BufferOp>(buf.getDefiningOp())) {
340 uint64_t buf_addr;
341 if (!buffer.getAddress().has_value()) {
342 return bd_op->emitOpError(
343 "Cannot lower buffer without associated address. Run pass "
344 "--aie-assign-buffer-addresses first or manually assign an "
345 "address.");
346 }
347 buf_addr = *buffer.getAddress();
348 buf_addr += bd_op.getOffsetInBytes();
349 if (target_model.isCoreTile(col, row)) {
350 NpuMaskWrite32Op::create(
351 builder, bd_op.getLoc(),
352 createConstantI32(builder, bd_op.getLoc(),
353 static_cast<uint32_t>(register_addr)),
354 createConstantI32(builder, bd_op.getLoc(),
355 static_cast<uint32_t>((buf_addr / 4) << 14)),
356 createConstantI32(builder, bd_op.getLoc(), 0x0fffc000), nullptr,
357 nullptr, nullptr);
358 } else if (target_model.isMemTile(col, row)) {
359 // On AIE2p (NPU2), memtile DMAs use an offset-based address
360 // space where the base depends on the relative position of the
361 // buffer's tile (west=0, internal=getMemTileSize, east=2x).
362 // On AIE2 (NPU1), memtile DMAs address local memory directly
363 // starting at 0. Only add the offset for AIE2p.
364 if (target_model.getTargetArch() == AIE::AIEArch::AIE2p) {
365 auto addrOffset = target_model.getMemLocalBaseAddress(
366 col, row, buffer.getTileOp().getCol(),
367 buffer.getTileOp().getRow());
368 if (addrOffset)
369 buf_addr += addrOffset.value();
370 }
371 NpuMaskWrite32Op::create(
372 builder, bd_op.getLoc(),
373 createConstantI32(builder, bd_op.getLoc(),
374 static_cast<uint32_t>(register_addr)),
375 createConstantI32(builder, bd_op.getLoc(),
376 static_cast<uint32_t>(buf_addr / 4)),
377 createConstantI32(builder, bd_op.getLoc(), 0x0007FFFF), nullptr,
378 nullptr, nullptr);
379 } else {
380 NpuWrite32Op::create(
381 builder, bd_op.getLoc(),
382 createConstantI32(builder, bd_op.getLoc(),
383 static_cast<uint32_t>(register_addr)),
384 createConstantI32(builder, bd_op.getLoc(),
385 static_cast<uint32_t>(buf_addr)),
386 nullptr, nullptr, nullptr);
387 }
388 } else {
389 return bd_op->emitOpError(
390 "Buffer argument must be a constant aie.buffer, a runtime sequence "
391 "input argument, or a supported chain of memref.subview, "
392 "memref.view, memref.cast, or memref.reinterpret_cast operations "
393 "rooted at a block argument. Subviews must be static and contiguous; "
394 "views must have a constant byte shift and no dynamic result sizes.");
395 }
396
397 // If this BD has an offset_state_table_idx, emit update_from_scratchpad to
398 // add the runtime offset to the BD address register. This is applied after
399 // the base address is set (by either NpuAddressPatchOp for DDR buffers or
400 // NpuMaskWrite32Op/NpuWrite32Op for on-chip buffers), since the hardware
401 // update_from_scratchpad instruction is additive -- it reads the existing
402 // register value and adds a computed delta to it.
403 if (bd_op.getOffsetStateTableIdxAttr()) {
404 auto bufType = llvm::cast<BaseMemRefType>(bd_op.getBuffer().getType());
405 if (failed(emitUpdateBdAddressFromOffsetParameter(builder, bd_op, bufType,
406 register_addr)))
407 return failure();
408 }
409
410 return success();
411 }

413 // Compile-time BD template fields shared by the static and dynamic lowering
414 // paths (only sizes/strides/len can be runtime on a dma_bd; everything here
415 // is always constant). The dynamic path bakes these into its zero-template
416 // NpuWriteBdOp exactly as the static path does, so a runtime BD keeps its
417 // locks, packet header and next_bd chaining.
418 struct BdTemplateFields {
419 uint32_t use_next_bd = 0, next_bd_id = 0;
420 int32_t enable_packet = 0, packet_id = 0, packet_type = 0;
421 int32_t lock_rel_val = 0, lock_rel_id = 0;
422 int32_t lock_acq_enable = 0, lock_acq_val = 0, lock_acq_id = 0;
423 };
424
425 // Gather the constant lock / packet / next_bd fields for a BD block. Returns
426 // failure if a lock carries a non-constant value.
427 FailureOr<BdTemplateFields>
428 gatherBdTemplateFields(Block &block, AIE::DMABDOp bd_op, AIE::TileOp &tile,
429 const AIE::AIETargetModel &target_model,
430 std::optional<xilinx::AIE::PacketInfoAttr> packet) {
431 BdTemplateFields f;
432 if (bd_op.getNextBdId().has_value()) {
433 f.next_bd_id = bd_op.getNextBdId().value();
434 f.use_next_bd = 1;
435 }
436
437 auto info = bd_op.getPacket().value_or(packet.value_or(nullptr));
438 if (info) {
439 f.enable_packet = 1;
440 f.packet_type = info.getPktType();
441 f.packet_id = info.getPktId();
442 }
443
444 auto lock_ops = getOptionalLockOpsForBlock(block);
445 if (lock_ops) {
446 auto [acquire_op, release_op] = *lock_ops;
447 AIE::LockOp acq_lock = acquire_op.getLockOp();
448 AIE::LockOp rel_lock = release_op.getLockOp();
449
450 if (acq_lock.getLockID().has_value()) {
451 f.lock_acq_id = acq_lock.getLockID().value();
452 auto value = acquire_op.getConstantValue();
453 if (failed(value))
454 return failure();
455 f.lock_acq_val = *value;
456 // For AcquireGreaterEqual, negate the value to signal the hardware to
457 // use >= comparison instead of == comparison.
458 if (acquire_op.acquireGE())
459 f.lock_acq_val = -f.lock_acq_val;
460 f.lock_acq_enable = 1;
461 }
462
463 if (rel_lock.getLockID().has_value()) {
464 f.lock_rel_id = rel_lock.getLockID().value();
465 auto value = release_op.getConstantValue();
466 if (failed(value))
467 return failure();
468 f.lock_rel_val = *value;
469 }
470
471 // For memtile, add lock offset using getLockLocalBaseIndex. This matches
472 // AIERT.cpp implementation.
473 if (target_model.isMemTile(tile.getCol(), tile.getRow())) {
474 auto lockOffset = target_model.getLockLocalBaseIndex(
475 tile.getCol(), tile.getRow(), acq_lock.colIndex(),
476 acq_lock.rowIndex());
477 if (lockOffset && acq_lock.getLockID().has_value())
478 f.lock_acq_id += lockOffset.value();
479 if (lockOffset && rel_lock.getLockID().has_value())
480 f.lock_rel_id += lockOffset.value();
481 }
482 }
483 return f;
484 }

486 // Dynamic (runtime SSA size/stride/len) shim-NOC BD lowering, the dma_task
487 // sibling of DmaToNpuPattern::lowerDynamic. Emits a zero-template
488 // NpuWriteBdOp (constant locks/packet/next_bd baked in, size/stride words
489 // zeroed) folded to one blockwrite, then per-word write32 overrides via the
490 // shared encoder. Scope (shim NOC, no padding, realizability) is enforced by
491 // the caller.
492 LogicalResult
493 rewriteSingleBDDynamic(OpBuilder &builder, Block &block, AIE::DMABDOp bd_op,
494 AIE::TileOp &tile,
495 std::optional<xilinx::AIE::PacketInfoAttr> packet,
496 Value runtimeBdId = nullptr) {
497 const auto &target_model = AIE::getTargetModel(bd_op);
498 Location loc = bd_op.getLoc();
499 auto i32ty = builder.getIntegerType(32);
500 int col = tile.getCol();
501 int row = tile.getRow();
502
503 auto fieldsOr =
504 gatherBdTemplateFields(block, bd_op, tile, target_model, packet);
505 if (failed(fieldsOr))
506 return failure();
507 BdTemplateFields f = *fieldsOr;
508
509 // The bd_id as an OpFoldResult: the runtime pool value if present, else the
510 // pinned constant attribute.
511 OpFoldResult bdIdOfr =
512 runtimeBdId
513 ? OpFoldResult(runtimeBdId)
514 : OpFoldResult(builder.getI32IntegerAttr(bd_op.getBdId().value()));
515
516 if (runtimeBdId) {
517 // A runtime bd_id makes the register block's address runtime, so the
518 // constant-address blockwrite path can't be used. Emit the template words
519 // as write32s instead -- register replay treats N write32s and an N-word
520 // blockwrite identically, matching the static BD.
521 if (failed(emitShimTemplateWordOverrides(builder, loc, target_model, col,
522 row, bdIdOfr, f,
523 bd_op.getBurstLength())))
524 return failure();
525 } else {
526 // Zero-template BD: constant fields baked in, size/stride words zeroed
527 // for the write32 overrides to fill. valid_bd = 1.
528 NpuWriteBdOp::create(
529 builder, loc, col, bd_op.getBdId().value(), /*buffer_length=*/0,
530 /*buffer_offset=*/0, f.enable_packet, /*out_of_order_id=*/0,
531 f.packet_id, f.packet_type, /*d0_size=*/0, /*d0_stride=*/0,
532 /*d1_size=*/0,
533 /*d1_stride=*/0, /*d2_size=*/0, /*d2_stride=*/0,
534 /*iteration_current=*/0,
535 /*iteration_size=*/0, /*iteration_stride=*/0, f.next_bd_id, row,
536 f.use_next_bd, /*valid_bd=*/1, f.lock_rel_val, f.lock_rel_id,
537 f.lock_acq_enable, f.lock_acq_val, f.lock_acq_id,
538 /*d0_zero_before=*/0,
539 /*d1_zero_before=*/0, /*d2_zero_before=*/0, /*d0_zero_after=*/0,
540 /*d1_zero_after=*/0, /*d2_zero_after=*/0, bd_op.getBurstLength());
541 }

543 // Normalize sizes/strides to a 4-element outermost-first mixed list (the
544 // shared emitter's contract, matching memcpy_nd's always-4D operands),
545 // padding absent leading dims with size 1 / stride 0. dma_bd's mixed lists
546 // are outermost-first and variable-length (0..4 dims).
547 SmallVector<OpFoldResult> sizes(bd_op.getMixedSizes());
548 SmallVector<OpFoldResult> strides(bd_op.getMixedStrides());
549 if (sizes.size() > 4 || strides.size() > 4)
550 return bd_op->emitOpError("At most four data layout transformation "
551 "dimensions may be provided.");
552 OpFoldResult one = builder.getI64IntegerAttr(1);
553 OpFoldResult zeroOfr = builder.getI64IntegerAttr(0);
554 SmallVector<OpFoldResult, 4> sizes4(4, one), strides4(4, zeroOfr);
555 for (size_t i = 0; i < sizes.size(); i++) {
556 sizes4[4 - sizes.size() + i] = sizes[i];
557 strides4[4 - strides.size() + i] = strides[i];
558 }
559
560 // buffer_length override = len (elements) * elemWidth / addressGranularity,
561 // as an SSA value. dma_task carries the transfer length explicitly (unlike
562 // memcpy_nd's size-product), and it may be runtime.
563 uint64_t elemWidth =
564 static_cast<uint64_t>(bd_op.getBufferElementTypeWidthInBytes()) * 8;
565 uint32_t gran = target_model.getAddressGenGranularity();
566 // len as OpFoldResult: the runtime operand if present, else the static_len
567 // attr (a constant BD with runtime sizes/strides still reaches here).
568 OpFoldResult lenOfr;
569 if (Value lenOperand = bd_op.getLen())
570 lenOfr = lenOperand;
571 else
572 lenOfr = builder.getI32IntegerAttr(bd_op.getConstantLen().value());
573 Value lenVal = getAsValue(builder, loc, lenOfr, i32ty);
574 Value bufLen = arith::DivUIOp::create(
575 builder, loc,
576 arith::MulIOp::create(builder, loc, lenVal,
577 createConstantI32(builder, loc, elemWidth)),
578 createConstantI32(builder, loc, gran));
579
580 // The BD-level repeat_count (encoder output) is unused here: the dma_task
581 // queue push is emitted separately by DMAStartTaskOpPattern from the task
582 // op's repeat_count, not the BD's outer dim.
583 Value bdRepeatCount;
584 if (failed(emitDynamicShimBdWordOverrides(
585 builder, loc, target_model, col, row, bdIdOfr, sizes4, strides4,
586 elemWidth, bd_op.getBurstLength(), bufLen, bdRepeatCount)))
587 return failure();
588 return setAddressForSingleBD(builder, bd_op, tile, bdIdOfr);
589 }
590
591 // Emit the shim BD template words (those the size/stride encoder doesn't own)
592 // as runtime-addressed write32s, for the runtime-bd_id path where a constant-
593 // address zero-template blockwrite can't be formed. Layout mirrors
594 // WriteBdToBlockWritePattern. Words 4/5 carry constant burst_length/AXCache
595 // bits the encoder's later ND write32 overwrites by last-write; 1/2/7 unused.
596 LogicalResult emitShimTemplateWordOverrides(
597 OpBuilder &builder, Location loc, const AIE::AIETargetModel &target_model,
598 int col, int row, OpFoldResult bdId, const BdTemplateFields &f,
599 uint32_t burstLength) {
600 Value bdBase =
601 getBdRegisterBase(builder, loc, target_model, col, row, bdId);
602 auto writeWord = [&](uint32_t wordIdx, uint32_t val) {
603 Value addr = arith::AddIOp::create(
604 builder, loc, bdBase, createConstantI32(builder, loc, wordIdx * 4));
605 NpuWrite32Op::create(builder, loc, addr,
606 createConstantI32(builder, loc, val), nullptr,
607 nullptr, nullptr);
608 };
609 // word[1] buffer_offset: 0 (the address patch supplies the buffer pointer).
610 writeWord(1, 0);
611 // word[2] enable_packet [30], out_of_order_id [29:24], packet_id [23:19],
612 // packet_type [18:16].
613 uint32_t w2 = ((f.enable_packet & 0x1) << 30) |
614 ((f.packet_id & 0x1f) << 19) | ((f.packet_type & 0x7) << 16);
615 writeWord(2, w2);
616 // word[4] burst_length [31:30] (constant); d1_size/stride overlaid by the
617 // encoder in ND mode.
618 writeWord(4,
619 (AIE::getShimBurstLengthEncoding(target_model, burstLength) & 0x3)
620 << 30);
621 // word[5] AXCache [27:24] = 2 (constant, enables NoC upsizing); d2_stride
622 // overlaid by the encoder in ND mode.
623 writeWord(5, (2u & 0xf) << 24);
624 // word[7] next_bd [30:27], use_next_bd [26], valid_bd [25], lock fields.
625 uint32_t w7 = ((f.next_bd_id & 0xf) << 27) | ((f.use_next_bd & 0x1) << 26) |
626 (1u << 25) | ((f.lock_rel_val & 0x7f) << 18) |
627 ((f.lock_rel_id & 0xf) << 13) |
628 ((f.lock_acq_enable & 0x1) << 12) |
629 ((f.lock_acq_val & 0x7f) << 5) | (f.lock_acq_id & 0xf);
630 writeWord(7, w7);
631 return success();
632 }

634 LogicalResult
635 rewriteSingleBD(OpBuilder &builder, Block &block, AIE::TileOp &tile,
636 AIE::DMAChannelDir channelDir,
637 std::optional<xilinx::AIE::PacketInfoAttr> packet,
638 Value runtimeBdId = nullptr) {
639 AIE::DMABDOp bd_op = getBdForBlock(block);
640 const auto &target_model = AIE::getTargetModel(bd_op);
641 auto buffer_type = llvm::cast<BaseMemRefType>(bd_op.getBuffer().getType());
642 uint32_t addr_granularity = target_model.getAddressGenGranularity();
643
644 // Runtime (SSA) sizes/strides/len/offset take the dynamic BD-word encoder
645 // path; a runtime bd_id (dynamic free-list pool) also forces it, since the
646 // BD register addresses are then runtime and cannot fold into a blockwrite.
647 // A fully-constant descriptor with a pinned bd_id takes the static path
648 // below unchanged. Only the shim-NOC layout is encodable this way (see
649 // rewriteSingleBDDynamic), so anything the dynamic path can't represent
650 // stays a clean diagnostic.
651 bool runtimeLen = bd_op.getLen() && !bd_op.getConstantLen();
652 bool runtimeOffset = bd_op.getOffset() && !bd_op.getConstantOffset();
653 bool runtimeDims =
654 llvm::any_of(bd_op.getMixedSizes(),
655 [](OpFoldResult s) { return !getConstantIntValue(s); }) ||
656 llvm::any_of(bd_op.getMixedStrides(),
657 [](OpFoldResult s) { return !getConstantIntValue(s); });
658 if (runtimeLen || runtimeDims || runtimeOffset || runtimeBdId) {
659 if (!target_model.isShimNOCTile(tile.getCol(), tile.getRow()))
660 return bd_op->emitOpError(
661 "runtime-valued BD size/stride/len/bd_id is only supported on shim "
662 "NOC tiles; use compile-time constants on other tiles.");
663 if (bd_op.getPadDimensions().has_value())
664 return bd_op->emitOpError(
665 "zero padding is not supported with runtime sizes/strides/len.");
666 // Realizability of the constant size/stride operands (runtime ones are
667 // guarded at lowering by the shared encoder). Mixed lists are
668 // outermost-first; the helper wants innermost-first.
669 uint64_t elemWidth =
670 static_cast<uint64_t>(bd_op.getBufferElementTypeWidthInBytes()) * 8;
671 SmallVector<OpFoldResult, 4> sizesRev(
672 llvm::reverse(bd_op.getMixedSizes()));
673 SmallVector<OpFoldResult, 4> stridesRev(
674 llvm::reverse(bd_op.getMixedStrides()));
675 if (failed(verifyConstBdRealizability(
676 bd_op, sizesRev, stridesRev, elemWidth,
677 target_model.getAddressGenGranularity())))
678 return failure();
679 return rewriteSingleBDDynamic(builder, block, bd_op, tile, packet,
680 runtimeBdId);
681 }
682
683 // Static path: bd_id is a pinned attribute (the dynamic/runtime-bd_id path
684 // returned above) and the offset is constant (runtime offset routed above).
685 uint32_t bd_id = bd_op.getBdId().value();
686 int64_t offset = bd_op.getOffsetInBytes();
687 uint64_t len = bd_op.getLenInBytes();
688 uint64_t len_addr_granularity = len * 8 / addr_granularity;
689
690 if (offset * 8 % addr_granularity != 0) {
691 return bd_op->emitOpError("Offset must be aligned to ")
692 << (addr_granularity / 8) << " byte boundary.";
693 }
694
695 if (len < addr_granularity / 8) {
696 return bd_op->emitOpError("Transfer size of ")
697 << len << " bytes falls below minimum hardware transfer unit of "
698 << (addr_granularity / 8) << " bytes.";
699 }
700 // The owning storage must outlive the ArrayRef view below.
701 std::optional<llvm::SmallVector<AIE::BDDimLayoutAttr>> dimsStorage =
702 bd_op.getConstantDimensions();
703 if (!dimsStorage)
704 return bd_op->emitOpError("internal error folding BD dimensions");
705 std::optional<llvm::ArrayRef<AIE::BDDimLayoutAttr>> dims;
706 if (!dimsStorage->empty())
707 dims = llvm::ArrayRef<AIE::BDDimLayoutAttr>(*dimsStorage);
708 llvm::SmallVector<int64_t, 4> sizes = llvm::SmallVector<int64_t, 4>(4, 0);
709 llvm::SmallVector<int64_t, 4> strides = llvm::SmallVector<int64_t, 4>(4, 0);
710
711 // Padding
712 std::optional<llvm::ArrayRef<AIE::BDPadLayoutAttr>> padDims =
713 bd_op.getPadDimensions();
714 llvm::SmallVector<int64_t, 4> padBefore =
715 llvm::SmallVector<int64_t, 4>(4, 0);
716 llvm::SmallVector<int64_t, 4> padAfter =
717 llvm::SmallVector<int64_t, 4>(4, 0);
718 std::fill(padBefore.begin(), padBefore.end(), 0);
719 std::fill(padAfter.begin(), padAfter.end(), 0);
720
721 auto out_of_order_id = 0;
722 auto d0size = 0;
723 auto d0stride = 0;
724 auto d1size = 0;
725 auto d1stride = 0;
726 auto d2size = 0;
727 auto d2stride = 0;
728 auto iteration_size = 0;
729 auto iteration_stride = 0;
730
731 if (dims && dims->size() > 0) {
732 llvm::SmallVector<int64_t, 4> input_sizes =
733 llvm::SmallVector<int64_t, 4>(4, 1);
734 llvm::SmallVector<int64_t, 4> input_strides =
735 llvm::SmallVector<int64_t, 4>(4, 0);
736 if (dims->size() > 4) {
737 return bd_op->emitOpError("At most four data layout transformation "
738 "dimensions may be provided.");
739 }
740
741 for (size_t i = 0; i < dims->size(); i++) {
742 // Pass down dimensions in reverse order; in the MLIR, this allows
743 // us to specify step sizes/wraps in the same order as we would
744 // access a multi-dim C array, with the highest dimension first.
745 int j = dims->size() - i - 1;
746 input_sizes[i] = (*dims)[j].getSize();
747 input_strides[i] = (*dims)[j].getStride();
748 }
749
750 // d3 (repeat) is excluded; a repeated linear transfer is still linear.
751 // A contiguous row-major ND access on a shim NOC tile is also lowered
752 // using the wide buffer_length register, exempt from the 10-bit ND
753 // wrap-size limit. Canonicalization zeroes size-1 strides before this
754 // pass runs, so isContiguousTransfer is sufficient.
755 bool treatAsLinear =
756 isLinearTransfer(input_sizes, input_strides) ||
757 (target_model.isShimNOCTile(tile.getCol(), tile.getRow()) &&
758 isContiguousTransfer(input_sizes, input_strides));
759
760 if (dims->size() > 2) {
761 d2size = (target_model.isMemTile(tile.getCol(), tile.getRow()))
762 ? (*dims)[2].getSize()
763 : 0;
764 }
765 if (padDims.has_value()) {
766 if (!target_model.isMemTile(tile.getCol(), tile.getRow()))
767 return bd_op->emitOpError()
768 << "Padding is only supported by memtile dma bds.";
769 if (padDims->size() > dims->size())
770 return bd_op->emitOpError()
771 << "Mismatch number of dimensions between padding(s)"
772 << " and wrap(s) and stride(s).";
773 if (channelDir == AIE::DMAChannelDir::MM2S) {
774 for (size_t i = 0; i < padDims->size(); i++) {
775 int j = padDims->size() - i - 1;
776 padBefore[i] = (*padDims)[j].getConstPadBefore();
777 padAfter[i] = (*padDims)[j].getConstPadAfter();
778 }
779 for (size_t i = padDims->size(); i < dims->size(); i++) {
780 padBefore[i] = 0;
781 padAfter[i] = 0;
782 }
783 } else
784 return bd_op->emitOpError()
785 << "supports padding only for MM2S direction on MemTiles.";
786 }
787 getHardwareStridesWraps(target_model, bd_op, buffer_type, input_sizes,
788 input_strides, sizes, strides);
789
790 if (failed(verifyStridesWraps(bd_op, buffer_type, tile.getCol(),
791 tile.getRow(), input_sizes, input_strides,
792 sizes, strides, treatAsLinear))) {
793 return failure();
794 }
795
796 iteration_size = sizes[3];
797 iteration_stride = strides[3];
798
799 if (!treatAsLinear) {
800 // d0_size, d0_stride
801 d0size = sizes[0];
802 d0stride = strides[0];
803
804 // d1_size, d1_stride
805 d1size = sizes[1];
806 d1stride = strides[1];
807
808 // d2_stride
809 d2stride = strides[2];
810 // d2_size set elsewhere
811 }
812 if (input_sizes[3] > 1 && input_strides[3] == 0) {
813 // We allow users to encode the repeat_count as a dimension 3 stride
814 // of 0. This must lower to a iteration wrap of 0, so no stride is
815 // ever added. We then repeat the BD using the repeat_count in
816 // NpuPushQueueOp.
817 iteration_size = 0;
818 iteration_stride = 0;
819 }
820
821 // Ensure the total transfer length and the length expressed in the lowest
822 // three dimensions of strides/wraps agree. (Fourth dimension is
823 // iteration/repeat count and repeats the whole BD, so should not be
824 // incorporated in length of a single BD invocation.)
825 uint64_t len_dims_addr_granularity = 1;
826 for (size_t i = 0; i < 3; i++) {
827 len_dims_addr_granularity *= sizes[i];
828 }
829 if (len_dims_addr_granularity != len_addr_granularity) {
830 auto err =
831 bd_op->emitOpError(
832 "Buffer descriptor length does not match length of transfer "
833 "expressed by lowest three dimensions of data layout "
834 "transformation strides/wraps. ")
835 << "BD length is " << (len_addr_granularity * addr_granularity / 8)
836 << " bytes. "
837 << "Lowest three dimensions of data layout transformation would "
838 "result in transfer of "
839 << (len_dims_addr_granularity * addr_granularity / 8) << " bytes. ";
840 err.attachNote() << "Do not include the highest dimension size in "
841 "transfer length, as this is the BD repeat count.";
842 return failure();
843 }
844 } else {
845 if (padDims && target_model.isMemTile(tile.getCol(), tile.getRow()) &&
846 channelDir == AIE::DMAChannelDir::MM2S) {
847 return bd_op->emitOpError()
848 << "Padding requires n-d data layouts expressed as "
849 << "wrap(s) and stride(s).";
850 } else if (padDims) {
851 return bd_op->emitOpError() << "Padding is supported only on MemTiles.";
852 }
853 }
854 auto fieldsOr =
855 gatherBdTemplateFields(block, bd_op, tile, target_model, packet);
856 if (failed(fieldsOr))
857 return failure();
858 BdTemplateFields f = *fieldsOr;
859
860 NpuWriteBdOp::create(
861 builder, bd_op.getLoc(), tile.getCol(), bd_id, len_addr_granularity,
862 offset,
863 /*enable_packet=*/f.enable_packet,
864 /*out_of_order_id=*/out_of_order_id,
865 /*packet_id=*/f.packet_id,
866 /*packet_type=*/f.packet_type,
867 /*d0_size=*/d0size, /*d0_stride=*/d0stride,
868 /*d1_size=*/d1size, /*d1_stride=*/d1stride,
869 /*d2_size=*/d2size, /*d2_stride=*/d2stride,
870 /*iteration_current=*/0, /*iteration_size=*/iteration_size,
871 /*iteration_stride=*/iteration_stride,
872 /*next_bd=*/f.next_bd_id,
873 /*row=*/tile.getRow(),
874 /*use_next_bd=*/f.use_next_bd,
875 /*valid_bd=*/1,
876 /*lock_rel_val=*/f.lock_rel_val, /*lock_rel_id=*/f.lock_rel_id,
877 /*lock_acq_enable=*/f.lock_acq_enable,
878 /*lock_acq_val=*/f.lock_acq_val, /*lock_acq_id=*/f.lock_acq_id,
879 /*d0_zero_before=*/padBefore[0],
880 /*d1_zero_before=*/padBefore[1], /*d2_zero_before=*/padBefore[2],
881 /*d0_zero_after=*/padAfter[0], /*d1_zero_after=*/padAfter[1],
882 /*d2_zero_after=*/padAfter[2],
883 /*burst_length=*/bd_op.getBurstLength());
884 return setAddressForSingleBD(builder, bd_op, tile);
885 }

888 Region &body = op.getBody();
889 for (auto it = body.begin(); it != body.end(); ++it) {
890 Block &block = *it;
891 if (shouldSkipBlock(block)) {
892 continue;
893 }
894 AIE::DMABDOp bd_op = getBdForBlock(block);
895 if (AIE::NextBDOp next_bd_op =
896 llvm::dyn_cast<AIE::NextBDOp>(block.getTerminator())) {
897 if (bd_op.getNextBdId().has_value()) {

899 bd_op.emitOpError("Cannot specify both next_bd_id attribute and "
900 "aie.next_bd operation.");
901 error.attachNote(next_bd_op.getLoc())
902 << "Potentially conflicting next buffer descriptor ID specified "
903 "here.";
904 return failure();
905 }
906 Block &next_bd_block = *next_bd_op.getDest();
907 AIE::DMABDOp next_dma_bd_op = getBdForBlock(next_bd_block);
908 assert(next_dma_bd_op.getBdId()
909 .has_value()); // Next BD should have assigned ID, and this
910 // should have been checked by earlier
911 // verifyBdInBlock() call
912 bd_op.setNextBdId(next_dma_bd_op.getBdId().value());
913 OpBuilder builder(next_bd_op);
914 AIE::EndOp::create(builder, next_bd_op.getLoc());
915 next_bd_op.erase();
916 }
917 }
918 return success();

921 LogicalResult rewriteSingleDMAConfigureTaskOp(DMAConfigureTaskOp op) {
922 OpBuilder builder(op);
923 AIE::TileOp tile = op.getTileOp();
924
925 if (!op.use_empty()) {
926 auto err = op.emitOpError("Cannot lower while op still has uses.");
927 mlir::Operation::use_range uses = op.getOperation()->getUses();
928 for (auto it = uses.begin(); it != uses.end(); ++it) {
929 err.attachNote(it->getOwner()->getLoc()) << "Used here.";
930 }
931 return failure();
932 }
933
934 Region &body = op.getBody();
935 bool hasRuntimeBdId = op.getBdIdVal() != nullptr;
936
937 // Verify each BD block first; subsequent functions rely on them being
938 // well-formed
939 for (auto it = body.begin(); it != body.end(); ++it) {
940 if (shouldSkipBlock(*it)) {
941 continue;
942 }
943 if (failed(verifyNoUnsupportedOpsInBlock(*it))) {
944 return failure();
945 }
946 if (failed(verifyBdInBlock(*it, hasRuntimeBdId))) {
947 return failure();
948 }
949 if (failed(verifyOptionalLocksInBlock(*it))) {
950 return failure();
951 }
952 }
953
954 // Hoist next_bd operations into next_bd_id attribute of the dma_bd
955 if (failed(hoistNextBdOpsIntoAttrs(op))) {
956 return failure();
957 }
958
959 auto channelDir = op.getDirection();
960 auto packet = op.getPacket();
961 // A runtime bd_id (dynamic free-list pool) supplied by
962 // aie-lower-dynamic-bd-pool; null on the static/pinned path.
963 Value runtimeBdId = op.getBdIdVal();
964
965 // Lower all BDs
966 for (auto it = body.begin(); it != body.end(); ++it) {
967 Block &block = *it;
968 if (shouldSkipBlock(block)) {
969 continue;
970 }
971 if (failed(rewriteSingleBD(builder, block, tile, channelDir, packet,
972 runtimeBdId))) {
973 return failure();
974 }
975 }
976
977 op.erase();
978
979 return success();
980 }

983 WalkResult result = device.walk([&](DMAConfigureTaskOp op) {
984 if (failed(rewriteSingleDMAConfigureTaskOp(op))) {
985 return WalkResult::interrupt();
986 }

988 });

995 // Drop the dead task-index carries left by the dynamic BD pool path once
996 // awaits have lowered to npu.sync. scf.for/scf.if carry the task value as a
997 // result purely to hold the await's data dependence on the configure; with
998 // the await gone that carry is dead but still counts as a use of the
999 // branch-local configure, blocking its use_empty lowering. scf's own
1000 // canonicalizations remove dead iter-args/results and prune the yields
1001 // feeding them. Run them device-wide but with folding and constant-CSE
1002 // DISABLED, so the static path's byte-golden constant emission is untouched
1003 // -- only the dead scf carries go.
1004 LogicalResult dropDeadTaskCarries(AIE::DeviceOp device) {
1005 RewritePatternSet patterns(&getContext());
1006 scf::ForOp::getCanonicalizationPatterns(patterns, &getContext());
1007 scf::IfOp::getCanonicalizationPatterns(patterns, &getContext());
1008 GreedyRewriteConfig config;
1009 config.enableFolding(false).enableConstantCSE(false);
1010 return applyPatternsGreedily(device, std::move(patterns), config);
1011 }

1013 void runOnOperation() override {
1014 AIE::DeviceOp device = getOperation();
1015
1016 // Convert DMAStartBD and DMAAwaitBD ops
1017 ConversionTarget target(getContext());
1018 target.addLegalDialect<AIEXDialect>();
1019 target.addLegalDialect<arith::ArithDialect>();
1020 target.addIllegalOp<DMAStartTaskOp>();
1021 target.addIllegalOp<DMAAwaitTaskOp>();
1022 RewritePatternSet patterns(&getContext());
1023 patterns.insert<DMAStartTaskOpPattern>(&getContext());
1024 patterns.insert<DMAAwaitTaskOpPattern>(&getContext());
1025 if (failed(applyPartialConversion(device, target, std::move(patterns)))) {
1026 signalPassFailure();
1027 }
1028
1029 // Drop the now-dead task-index carries the awaits held, so the branch-local
1030 // configures they used can reach use_empty and lower below.
1031 if (failed(dropDeadTaskCarries(device)))
1032 signalPassFailure();
1033
1034 // Lower the configuration for the BDs
1035 if (failed(rewriteDMAConfigureTaskOp(device))) {
1036 signalPassFailure();
1037 }
1038 }
