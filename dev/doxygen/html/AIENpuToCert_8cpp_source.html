33// slightly smaller than the actual page size to account for overheads and
34// estimation errors
35static constexpr uint32_t cert_page_size = 8000;
36
37// Returns the PDI id to use for a cert.load_pdi that references `deviceSymName`
38// within `parentDevice`. If a load_pdi for that symbol already exists, its id
39// is reused so repeated loads of the same PDI share an id; otherwise a fresh id
40// (1-based, one past the current maximum) is assigned.
41static uint32_t getOrAssignPdiId(AIE::DeviceOp parentDevice,
42 StringRef deviceSymName) {
43 uint32_t pdiId = 1;
44 bool foundExisting = false;
45 parentDevice.walk([&](AIEX::CertLoadPdiOp loadPdiOp) {
46 if (loadPdiOp.getSymbol() == deviceSymName) {
47 pdiId = loadPdiOp.getPdiId();
48 foundExisting = true;
49 } else if (!foundExisting) {
50 pdiId = std::max(pdiId, loadPdiOp.getPdiId() + 1);
51 }
52 });
53 return pdiId;
54}

56struct RuntimeSequenceToCertJob : OpConversionPattern<AIE::RuntimeSequenceOp> {
57 using OpConversionPattern::OpConversionPattern;
58
59 RuntimeSequenceToCertJob(MLIRContext *context, PatternBenefit benefit = 1)
60 : OpConversionPattern(context, benefit) {}
61
62 LogicalResult
63 matchAndRewrite(AIE::RuntimeSequenceOp op, OpAdaptor adaptor,
64 ConversionPatternRewriter &rewriter) const override {
65
66 auto symName = op.getSymName();
67 uint32_t newJobId = 1;
68 if (symName != "configure") {
69 uint32_t maxJobId = 1;
70 op->getParentOp()->walk([&](AIEX::CertJobOp certJobOp) {
71 maxJobId = std::max(maxJobId, certJobOp.getJobId());
72 });
73 newJobId = maxJobId + 1;
74 }
75
76 // Create page wrapper at the same location as the runtime sequence
77 rewriter.setInsertionPoint(op);
78 auto pageOp = rewriter.create<AIEX::CertPageOp>(op.getLoc());
79 Block *pageBlock = new Block();
80 pageOp.getBody().push_back(pageBlock);
81 rewriter.setInsertionPointToStart(pageBlock);
82
83 // Create job inside page
84 auto jobOp = rewriter.create<AIEX::CertJobOp>(op.getLoc(), newJobId);
85
86 // Clone runtime sequence body into job
87 // Note: This preserves block arguments from the runtime sequence, which
88 // will be present in the MLIR IR but are not emitted in the final assembly
89 IRMapping remap;
90 op.getRegion().cloneInto(&jobOp.getBody(), remap);
91 AIEX::CertJobOp::ensureTerminator(jobOp.getBody(), rewriter, op->getLoc());
92 AIEX::CertPageOp::ensureTerminator(pageOp.getBody(), rewriter,
93 op->getLoc());
94
95 // Erase the original runtime sequence
96 rewriter.eraseOp(op);
97
98 return success();
99 }

102struct NpuWrite32ToCertWrite32 : OpConversionPattern<AIEX::NpuWrite32Op> {
103 using OpConversionPattern::OpConversionPattern;
104
105 LogicalResult
106 matchAndRewrite(AIEX::NpuWrite32Op op, OpAdaptor adaptor,
107 ConversionPatternRewriter &rewriter) const override {
108
109 std::optional<uint32_t> address =
110 AIEX::getConstantIntOperand(op.getAddress());
111 std::optional<uint32_t> value = AIEX::getConstantIntOperand(op.getValue());
112 if (!address || !value)
113 return op.emitOpError(
114 "cannot lower to cert.write32 with non-constant address or value");
115
116 // Get the absolute address, which properly handles row/col if present
117 std::optional<uint32_t> absAddress = op.getAbsoluteAddress();
118 if (!absAddress)
119 return failure();
120
121 uint32_t absoluteAddr = *absAddress;
122
123 // If row and col are specified, validate that the address upper bits match
124 std::optional<uint32_t> col = op.getColumn();
125 std::optional<uint32_t> row = op.getRow();
126 if (col && row) {
127 const auto &tm = AIE::getTargetModel(op);
128 uint32_t expectedUpperBits = ((*col & 0xff) << tm.getColumnShift()) |
129 ((*row & 0xff) << tm.getRowShift());
130
131 // Warn if the original address had non-zero upper bits that don't match
132 uint32_t origAddress = *address;
133 uint32_t origUpperBits = origAddress & ~0xfffff;
134 if (origUpperBits != 0 && origUpperBits != expectedUpperBits) {
135 op.emitWarning() << "address upper bits (0x"
136 << llvm::utohexstr(origUpperBits)
137 << ") don't match row=" << *row << " col=" << *col
138 << " computed bits (0x"
139 << llvm::utohexstr(expectedUpperBits) << ")";
140 }
141 }
142
143 rewriter.replaceOpWithNewOp<AIEX::CertWrite32Op>(op, absoluteAddr, *value);
144 return success();
145 }

198struct NpuBlockWriteToCertUcDma : OpConversionPattern<AIEX::NpuBlockWriteOp> {
199 using OpConversionPattern::OpConversionPattern;
200
201 LogicalResult
202 matchAndRewrite(AIEX::NpuBlockWriteOp op, OpAdaptor adaptor,
203 ConversionPatternRewriter &rewriter) const override {
204
205 memref::GetGlobalOp dataOperand =
206 dyn_cast_or_null<memref::GetGlobalOp>(op.getData().getDefiningOp());
207 if (!dataOperand)
208 return failure();
209 MemRefType dataType = cast<MemRefType>(dataOperand.getResult().getType());
210 uint32_t dataSize = dataType.getNumElements();
211
212 int id = 0;
213 std::string symbolName = "chain_" + std::to_string(id);
214 while (op->getParentOfType<AIE::DeviceOp>().lookupSymbol(symbolName))
215 symbolName = "chain_" + std::to_string(++id);
216
217 // Create a new uc_dma_write_des_sync operation
218 rewriter.replaceOpWithNewOp<AIEX::CertUcDmaWriteDesSyncOp>(op, symbolName);
219
220 // Create the uc_dma_chain operation
221 // Find the nearest device to insert the chain
222 auto parentDevice = op->getParentOfType<AIE::DeviceOp>();
223 if (!parentDevice) {
224 // No parent device - this shouldn't happen but handle gracefully
225 return failure();
226 }
227
228 // Insert after the last existing uc_dma_chain (before any pages/jobs),
229 // preserving the order of blockwrite ops. Globals are allowed to precede
230 // the chains, so skip over them too.
231 Block *deviceBody = parentDevice.getBody();
232 Operation *insertAfter = nullptr;
233 for (Operation &bodyOp : *deviceBody) {
234 if (isa<AIEX::CertUcDmaChainOp>(bodyOp) || isa<memref::GlobalOp>(bodyOp))
235 insertAfter = &bodyOp;
236 else
237 break;
238 }
239 if (insertAfter)
240 rewriter.setInsertionPointAfter(insertAfter);
241 else
242 rewriter.setInsertionPointToStart(deviceBody);
243
244 auto symbolAttr = rewriter.getStringAttr(symbolName);
245 auto chainOp =
246 AIEX::CertUcDmaChainOp::create(rewriter, op.getLoc(), symbolAttr);
247
248 Block *bb = new Block();
249 chainOp.getRegion().push_back(bb);
250 rewriter.setInsertionPointToStart(bb);
251 AIEX::CertUcDmaBdOp::create(rewriter, op.getLoc(), dataOperand.getName(),
252 op.getAddress(), dataSize, false);
253
254 AIEX::CertUcDmaChainOp::ensureTerminator(chainOp.getBody(), rewriter,
255 op->getLoc());
256 return success();
257 }

260struct NpuSyncToCertWaitTCTS : OpConversionPattern<AIEX::NpuSyncOp> {
261 using OpConversionPattern::OpConversionPattern;
262
263 LogicalResult
264 matchAndRewrite(AIEX::NpuSyncOp op, OpAdaptor adaptor,
265 ConversionPatternRewriter &rewriter) const override {
266 std::optional<uint32_t> rowOpt = AIEX::getConstantIntOperand(op.getRow());
267 std::optional<uint32_t> colOpt =
268 AIEX::getConstantIntOperand(op.getColumn());
269 std::optional<uint32_t> channelOpt =
270 AIEX::getConstantIntOperand(op.getChannel());
271 std::optional<uint32_t> directionOpt =
272 AIEX::getConstantIntOperand(op.getDirection());
273 if (!rowOpt || !colOpt || !channelOpt || !directionOpt)
274 return op.emitOpError(
275 "cannot lower to cert.wait_tcts with non-constant sync parameters");
276 uint32_t row = *rowOpt;
277 uint32_t col = *colOpt;
278
279 // These are the shift amounts from the tct packet format in the
280 // architecture spec. The firmware expects the row and column packed and
281 // shifted down to zero.
282 const int row_id_shift = 16;
283 const int col_id_shift = 21;
284 uint16_t tile_id = col << (col_id_shift - row_id_shift) | row;
285 uint32_t channel = *channelOpt;
286 uint32_t direction = *directionOpt;
287
288 const std::vector<int> chan2actor_shim_s2mm = {0, 2, 3, 4};
289 const std::vector<int> chan2actor_shim_mm2s = {6, 7, 8, 9, 10, 11, 12, 13};
290 const std::vector<int> chan2actor_mem_s2mm = {1, 2, 3, 4, 5, 6, 7};
291 const std::vector<int> chan2actor_mem_mm2s = {16, 17, 18, 19, 20,
292 22, 23, 24, 25, 26};
293 const std::vector<int> chan2actor_tile_s2mm = {0, 1};
294 const std::vector<int> chan2actor_tile_mm2s = {6};
295
296 const auto &tm = AIE::getTargetModel(op);
297 const bool isS2MM =
298 direction == static_cast<std::underlying_type_t<AIE::DMAChannelDir>>(
299 AIE::DMAChannelDir::S2MM);
300
301 const std::vector<int> *chan2actor = nullptr;
302 if (tm.isCoreTile(col, row))
303 chan2actor = isS2MM ? &chan2actor_tile_s2mm : &chan2actor_tile_mm2s;
304 else if (tm.isMemTile(col, row))
305 chan2actor = isS2MM ? &chan2actor_mem_s2mm : &chan2actor_mem_mm2s;
306 else
307 chan2actor = isS2MM ? &chan2actor_shim_s2mm : &chan2actor_shim_mm2s;
308
309 size_t chanIdx = static_cast<size_t>(channel);
310 if (chanIdx >= chan2actor->size()) {
311 op.emitError("invalid DMA channel ")
312 << channel << " for " << (isS2MM ? "S2MM" : "MM2S")
313 << " direction in NpuSyncToCertWaitTCTS conversion";
314 return failure();
315 }
316
317 uint8_t actor_id = static_cast<uint8_t>((*chan2actor)[chanIdx]);
318 uint8_t num_tcts = 1;
319 rewriter.replaceOpWithNewOp<AIEX::CertWaitTCTSOp>(op, tile_id, actor_id,
320 num_tcts);
321 return success();
322 }

355struct RunOpInlining : OpRewritePattern<AIEX::RunOp> {
356 using OpRewritePattern::OpRewritePattern;
357
358 LogicalResult matchAndRewrite(AIEX::RunOp op,
359 PatternRewriter &rewriter) const override {
360 // Get the callee runtime sequence
361 AIE::RuntimeSequenceOp calleeRuntimeSequence =
362 op.getCalleeRuntimeSequenceOp();
363 if (!calleeRuntimeSequence)
364 return failure();
365
366 // Get the callee body region
367 Region &calleeBody = calleeRuntimeSequence.getBody();
368
369 // Create argument mapping from run op arguments to callee parameters
370 IRMapping argMap;
371 ValueRange values = op.getArgs();
372 if (values.size() != calleeBody.getNumArguments()) {
373 op.emitError("number of run op arguments (")
374 << values.size()
375 << ") does not match number of callee runtime sequence "
376 "arguments ("
377 << calleeBody.getNumArguments() << ")";
378 return failure();
379 }
380 for (unsigned i = 0, n = calleeBody.getNumArguments(); i < n; i++) {
381 BlockArgument arg = calleeBody.getArgument(i);
382 Value val = values[i];
383 argMap.map(arg, val);
384 }
385
386 // Clone operations from callee into current location
387 rewriter.setInsertionPoint(op);
388 for (Operation &o : calleeBody.front().getOperations()) {
389 // Skip the terminator
390 if (isa<AIE::EndOp>(o))
391 continue;
392 rewriter.clone(o, argMap);
393 }
394
395 // Erase the run op
396 rewriter.eraseOp(op);
397
398 return success();
399 }

402struct ConfigureOpToCertSection : OpRewritePattern<AIEX::ConfigureOp> {
403 using OpRewritePattern::OpRewritePattern;
404
405 LogicalResult matchAndRewrite(AIEX::ConfigureOp op,
406 PatternRewriter &rewriter) const override {
407 // Get the referenced device
408 AIE::DeviceOp referencedDevice = op.getReferencedDeviceOp();
409 if (!referencedDevice)
410 return failure();
411
412 // Get the device's symbol name for the section
413 StringRef deviceSymName = referencedDevice.getSymName();
414
415 // Find the parent DeviceOp to insert the section into
416 auto parentDevice = op->getParentOfType<AIE::DeviceOp>();
417 if (!parentDevice)
418 return failure();
419
420 // Check if section already exists (avoid creating duplicates)
421 auto existingSection = dyn_cast_if_present<AIEX::CertSectionOp>(
422 parentDevice.lookupSymbol(deviceSymName));
423 if (existingSection) {
424 // Section already exists, just create load_pdi at call site, reusing the
425 // PDI ID already assigned to this section (or assigning a new one).
426 uint32_t pdiId = getOrAssignPdiId(parentDevice, deviceSymName);
427
428 rewriter.setInsertionPoint(op);
429 rewriter.create<AIEX::CertLoadPdiOp>(
430 op.getLoc(), rewriter.getUI32IntegerAttr(pdiId),
431 FlatSymbolRefAttr::get(rewriter.getContext(), deviceSymName));
432
433 // Clone configure body operations after the load_pdi
434 IRMapping bodyMapper;
435 for (Operation &o : op.getRegion().front().getOperations()) {
436 if (isa<AIE::EndOp>(o))
437 continue;
438 rewriter.clone(o, bodyMapper);
439 }
440
441 rewriter.eraseOp(op);
442 return success();
443 }
444
445 // Create cert.section with the device symbol name
446 rewriter.setInsertionPoint(parentDevice.getBody()->getTerminator());
447 auto sectionOp = rewriter.create<AIEX::CertSectionOp>(
448 op.getLoc(), rewriter.getStringAttr(deviceSymName));
449
450 // Create the section body with a page containing a job
451 Block *sectionBlock = new Block();
452 sectionOp.getBody().push_back(sectionBlock);
453 rewriter.setInsertionPointToStart(sectionBlock);
454
455 // Create page within section
456 auto pageOp = rewriter.create<AIEX::CertPageOp>(op.getLoc());
457 Block *pageBlock = new Block();
458 pageOp.getBody().push_back(pageBlock);
459 rewriter.setInsertionPointToStart(pageBlock);
460
461 // Create job within page - assign unique job ID across all devices
462 // Find the maximum job ID in the parent device
463 uint32_t maxJobId = 0;
464 parentDevice.walk([&](AIEX::CertJobOp certJobOp) {
465 maxJobId = std::max(maxJobId, certJobOp.getJobId());
466 });
467 uint32_t sectionJobId = maxJobId + 1;
468
469 auto jobOp = rewriter.create<AIEX::CertJobOp>(op.getLoc(), sectionJobId);
470 Block *jobBlock = new Block();
471 jobOp.getBody().push_back(jobBlock);
472 rewriter.setInsertionPointToStart(jobBlock);
473
474 // Find the "configure" sequence content from the referenced device
475 // It could be either:
476 // 1. Still a RuntimeSequenceOp named "configure" (if that device hasn't
477 // been processed yet)
478 // 2. Already converted to cert.page/cert.job (if that device was processed
479 // first)
480
481 AIE::RuntimeSequenceOp configureSeq = nullptr;
482 referencedDevice.walk([&](AIE::RuntimeSequenceOp seq) {
483 if (seq.getSymName() == "configure") {
484 configureSeq = seq;
485 }
486 });
487
488 if (configureSeq) {
489 // Clone operations from the "configure" runtime_sequence
490 IRMapping mapper;
491 for (Operation &o : configureSeq.getRegion().front().getOperations()) {
492 if (isa<AIE::EndOp>(o))
493 continue;
494 rewriter.clone(o, mapper);
495 }
496 } else {
497 // "configure" was already converted - find the first cert.page
498 // (configuration page)
499 AIEX::CertPageOp configPage = nullptr;
500 for (auto &op : referencedDevice.getBody()->getOperations()) {
501 if (auto page = dyn_cast<AIEX::CertPageOp>(op)) {
502 configPage = page;
503 break; // Take first page (configure)
504 }
505 }
506
507 if (configPage) {
508 // Clone operations from the config page's job
509 IRMapping mapper;
510 for (auto &op : configPage.getBody().front().getOperations()) {
511 if (auto job = dyn_cast<AIEX::CertJobOp>(op)) {
512 for (Operation &jobOp : job.getBody().front().getOperations()) {
513 if (isa<AIE::EndOp>(jobOp))
514 continue;
515 rewriter.clone(jobOp, mapper);
516 }
517 break; // Only clone first job
518 }
519 }
520 }
521 }
522
523 // Ensure terminators
524 AIEX::CertJobOp::ensureTerminator(jobOp.getBody(), rewriter, op.getLoc());
525 AIEX::CertPageOp::ensureTerminator(pageOp.getBody(), rewriter, op.getLoc());
526 AIEX::CertSectionOp::ensureTerminator(sectionOp.getBody(), rewriter,
527 op.getLoc());
528
529 // At the configure call site, replace with load_pdi, reusing the PDI ID
530 // already assigned to this device if present (or assigning a new one).
531 uint32_t pdiId = getOrAssignPdiId(parentDevice, deviceSymName);
532
533 rewriter.setInsertionPoint(op);
534 rewriter.create<AIEX::CertLoadPdiOp>(
535 op.getLoc(), rewriter.getUI32IntegerAttr(pdiId),
536 FlatSymbolRefAttr::get(rewriter.getContext(), deviceSymName));
537
538 // Clone configure body operations after the load_pdi
539 // Operations that reference external values were not cloned into the
540 // section, so we clone them here into the main control flow
541 IRMapping bodyMapper;
542 for (Operation &o : op.getRegion().front().getOperations()) {
543 if (isa<AIE::EndOp>(o))
544 continue;
545 rewriter.clone(o, bodyMapper);
546 }
547
548 // Erase the original configure op
549 rewriter.eraseOp(op);
550 return success();
551 }

554struct NpuAddressPatchToCertApplyOffset57
555 : OpConversionPattern<AIEX::NpuAddressPatchOp> {
556 using OpConversionPattern::OpConversionPattern;
557
558 LogicalResult
559 matchAndRewrite(AIEX::NpuAddressPatchOp op, OpAdaptor adaptor,
560 ConversionPatternRewriter &rewriter) const override {
561 // find the previous blockwrite operation
562 Block::iterator it(op);
563 while (it != op->getBlock()->begin()) {
564 --it;
565 auto blockWriteOp = dyn_cast<AIEX::NpuBlockWriteOp>(*it);
566 if (!blockWriteOp)
567 continue;
568
569 const auto &tm = AIE::getTargetModel(op);
570 uint32_t addr = op.getAddr();
571 int col = (addr >> tm.getColumnShift()) & 0x1f;
572 int row = (addr >> tm.getRowShift()) & 0x1f;
573 if (!tm.isValidTile({col, row}))
574 return failure();
575
576 // if it's not a matching blockwrite, give up.
577 if (blockWriteOp.getAddress() + tm.getDmaBdAddressOffset(col, row) !=
578 addr)
579 break;
580
581 Value data = blockWriteOp.getData();
582 auto getGlobalOp = dyn_cast<memref::GetGlobalOp>(data.getDefiningOp());
583 if (!getGlobalOp)
584 break;
585
586 // replace the address with the new address
587 rewriter.setInsertionPoint(blockWriteOp);
588 rewriter.replaceOpWithNewOp<AIEX::CertApplyOffset57Op>(
589 op, getGlobalOp.getName(), 1, op.getArgIdx());
590 return success();
591 }
592
593 return failure();
594 }

597struct MergeConsecutiveCertUcDmaWriteDesSyncOps
598 : OpRewritePattern<AIEX::CertUcDmaWriteDesSyncOp> {
599 using OpRewritePattern::OpRewritePattern;
600
601 LogicalResult matchAndRewrite(AIEX::CertUcDmaWriteDesSyncOp op,
602 PatternRewriter &rewriter) const override {
603 // Get the previous operation in the block
604 Block::iterator it(op);
605 AIEX::CertUcDmaWriteDesSyncOp prevWriteDesSync = nullptr;
606 while (it != op->getBlock()->begin() && !prevWriteDesSync) {
607 --it;
608 Operation *prevOp = &*it;
609 if (isa<AIEX::CertWrite32Op, AIEX::CertMaskWrite32Op,
610 AIEX::CertApplyOffset57Op, AIEX::CertWaitTCTSOp>(prevOp))
611 return failure();
612 prevWriteDesSync = dyn_cast<AIEX::CertUcDmaWriteDesSyncOp>(prevOp);
613 }
614 if (!prevWriteDesSync)
615 return failure();
616
617 // find the uc_dma_chain
618 StringRef sym_name = op.getSymbol();
619 StringRef prev_sym_name = prevWriteDesSync.getSymbol();
620 auto chain = dyn_cast_if_present<AIEX::CertUcDmaChainOp>(
621 op->getParentOfType<AIE::DeviceOp>().lookupSymbol(sym_name));
622 auto prevChain = dyn_cast_if_present<AIEX::CertUcDmaChainOp>(
623 prevWriteDesSync->getParentOfType<AIE::DeviceOp>().lookupSymbol(
624 prev_sym_name));
625 if (!chain || !prevChain)
626 return failure();
627
628 // Compute the size of the current and previous chains. If their combined
629 // data size is greater than the cert page size, then we cannot merge them.
630 uint32_t prevChainSize = 0;
631 for (auto &o : prevChain.getBody().front().getOperations()) {
632 auto bdOp = dyn_cast<AIEX::CertUcDmaBdOp>(o);
633 if (!bdOp)
634 continue;
635 prevChainSize += bdOp.getLength() * sizeof(int);
636 }
637 uint32_t currChainSize = 0;
638 for (auto &o : chain.getBody().front().getOperations()) {
639 auto bdOp = dyn_cast<AIEX::CertUcDmaBdOp>(o);
640 if (!bdOp)
641 continue;
642 currChainSize += bdOp.getLength() * sizeof(int);
643 }
644 if ((currChainSize + prevChainSize) >= cert_page_size)
645 return failure();
646
647 IRMapping map;
648 rewriter.setInsertionPointToStart(&chain.getBody().front());
649 for (auto &o : prevChain.getBody().front().getOperations()) {
650 auto bdOp = dyn_cast<AIEX::CertUcDmaBdOp>(o);
651 if (!bdOp)
652 continue;
653 AIEX::CertUcDmaBdOp::create(
654 rewriter, bdOp.getLoc(), bdOp.getRemoteAddress(),
655 bdOp.getLocalAddress(), bdOp.getLength(), true);
656 }
657 rewriter.eraseOp(prevChain);
658 rewriter.eraseOp(prevWriteDesSync);
659 return success();
660 }
661};

663struct SplitNpuBlockWriteOpPattern : OpRewritePattern<AIEX::NpuBlockWriteOp> {
664 using OpRewritePattern::OpRewritePattern;
665
666 LogicalResult matchAndRewrite(AIEX::NpuBlockWriteOp op,
667 PatternRewriter &rewriter) const override {
668
669 memref::GetGlobalOp dataOperand =
670 dyn_cast_or_null<memref::GetGlobalOp>(op.getData().getDefiningOp());
671 if (!dataOperand)
672 return failure();
673
674 MemRefType dataType = cast<MemRefType>(dataOperand.getResult().getType());
675 uint32_t dataSize = dataType.getNumElements();
676
677 uint32_t dataSizeBytes = dataSize * sizeof(int);
678 if (dataSizeBytes < cert_page_size)
679 return failure();
680
681 auto loc = op.getLoc();
682
683 // Calculate split point (split roughly in half)
684 uint32_t splitElements = dataSize / 2;
685 uint32_t firstChunkSize = splitElements;
686 uint32_t secondChunkSize = dataSize - splitElements;
687
688 // Find the original memref.global operation
689 auto deviceOp = op->getParentOfType<AIE::DeviceOp>();
690 auto originalGlobal = dyn_cast_if_present<memref::GlobalOp>(
691 deviceOp.lookupSymbol(dataOperand.getName()));
692 if (!originalGlobal)
693 return failure();
694
695 // Get the original data attribute
696 auto originalData = originalGlobal.getInitialValue();
697 if (!originalData)
698 return failure();
699
700 auto denseData = dyn_cast<DenseIntElementsAttr>(*originalData);
701 if (!denseData)
702 return failure();
703
704 // Split the data into two chunks
705 auto dataValues = denseData.getValues<APInt>();
706 std::vector<APInt> firstChunkData(dataValues.begin(),
707 dataValues.begin() + firstChunkSize);
708 std::vector<APInt> secondChunkData(dataValues.begin() + firstChunkSize,
709 dataValues.end());
710
711 // Create new global operations for the split data
712 auto elementType = rewriter.getI32Type();
713 auto firstChunkType = MemRefType::get({firstChunkSize}, elementType);
714 auto secondChunkType = MemRefType::get({secondChunkSize}, elementType);
715 TensorType firstTensorType =
716 RankedTensorType::get({firstChunkSize}, elementType);
717 TensorType secondTensorType =
718 RankedTensorType::get({secondChunkSize}, elementType);
719
720 auto firstChunkAttr =
721 DenseIntElementsAttr::get(firstTensorType, firstChunkData);
722 auto secondChunkAttr =
723 DenseIntElementsAttr::get(secondTensorType, secondChunkData);
724
725 // Generate unique names for the new globals
726 std::string firstName = dataOperand.getName().str() + "_split_0";
727 std::string secondName = dataOperand.getName().str() + "_split_1";
728
729 // Ensure unique names
730 int counter = 0;
731 while (deviceOp.lookupSymbol(firstName)) {
732 firstName =
733 dataOperand.getName().str() + "_split_0_" + std::to_string(counter++);
734 }
735 counter = 0;
736 while (deviceOp.lookupSymbol(secondName)) {
737 secondName =
738 dataOperand.getName().str() + "_split_1_" + std::to_string(counter++);
739 }
740
741 // Create the new global operations
742 rewriter.setInsertionPoint(originalGlobal);
743 memref::GlobalOp::create(rewriter, loc, firstName,
744 rewriter.getStringAttr("private"), firstChunkType,
745 firstChunkAttr, true, nullptr);
746
747 memref::GlobalOp::create(rewriter, loc, secondName,
748 rewriter.getStringAttr("private"), secondChunkType,
749 secondChunkAttr, true, nullptr);
750
751 // Create get_global operations for the new data
752 rewriter.setInsertionPoint(op);
753
754 auto firstGetGlobal =
755 memref::GetGlobalOp::create(rewriter, loc, firstChunkType, firstName);
756 auto secondGetGlobal =
757 memref::GetGlobalOp::create(rewriter, loc, secondChunkType, secondName);
758
759 uint32_t baseAddr = op.getAddress();
760
761 AIEX::NpuBlockWriteOp::create(rewriter, loc, baseAddr,
762 firstGetGlobal.getResult(), nullptr, nullptr,
763 nullptr);
764
765 AIEX::NpuBlockWriteOp::create(rewriter, loc, baseAddr + firstChunkSize * 4,
766 secondGetGlobal.getResult(), nullptr, nullptr,
767 nullptr);
768
769 // Replace the original operation
770 rewriter.eraseOp(op);
771
772 LLVM_DEBUG(llvm::outs()
773 << "Split NpuBlockWriteOp with data size: " << dataSizeBytes
774 << " bytes into chunks of " << firstChunkSize << " and "
775 << secondChunkSize << " elements\n");
776
777 return success();
778 }
779};

781struct AIENpuToCertPass
782 : xilinx::AIEX::impl::AIENpuToCertBase<AIENpuToCertPass> {
783 void runOnOperation() override {
784 auto moduleOp = getOperation();
785
786 // Collect all devices
787 llvm::SmallVector<AIE::DeviceOp, 4> devices;
788 for (auto &op : moduleOp.getBody()->getOperations()) {
789 if (auto deviceOp = dyn_cast<AIE::DeviceOp>(op)) {
790 devices.push_back(deviceOp);
791 }
792 }
793
794 // Process main device first (named "main" or specified by --device-name)
795 // so referenced devices' "configure" sequences can be found before they
796 // get converted to cert.job
797 AIE::DeviceOp mainDevice = nullptr;
798 for (auto dev : devices) {
799 if (dev.getSymName() == deviceName) {
800 mainDevice = dev;
801 processDevice(dev);
802 break;
803 }
804 }
805
806 if (!mainDevice) {
807 moduleOp.emitError("no device found matching --device-name \"")
808 << deviceName << "\"";
809 signalPassFailure();
810 return;
811 }
812
813 // Then process other devices
814 for (auto dev : devices) {
815 if (dev != mainDevice) {
816 processDevice(dev);
817 }
818 }
819
820 // Remove absorbed devices (keep only main device)
821 // Referenced devices have been absorbed as cert.sections, so remove them
822 llvm::SmallVector<AIE::DeviceOp, 4> devicesToRemove;
823 for (auto dev : devices) {
824 if (dev != mainDevice) {
825 devicesToRemove.push_back(dev);
826 }
827 }
828
829 for (auto dev : devicesToRemove) {
830 dev.erase();
831 }
832 }
833
834 void processDevice(AIE::DeviceOp currentDevice) {
835 auto moduleOp = currentDevice->getParentOfType<ModuleOp>();
836
837 // Identify and convert referenced "configure" sequences early, before the
838 // RuntimeSequenceOp conversion runs on other devices.
839 if (moduleOp) {
840
841 // Collect npu.load_pdi operations (before they are converted below)
842 llvm::SmallVector<StringRef, 4> referencedDeviceSyms;
843 currentDevice.walk([&](AIEX::NpuLoadPdiOp loadPdiOp) {
844 if (loadPdiOp.getDeviceRef())
845 referencedDeviceSyms.push_back(*loadPdiOp.getDeviceRef());
846 });
847
848 if (!referencedDeviceSyms.empty()) {
849 OpBuilder builder(&getContext());
850
851 for (StringRef refSymName : referencedDeviceSyms) {
852 // Skip if section already exists
853 if (currentDevice.lookupSymbol(refSymName))
854 continue;
855
856 // Find referenced device
857 AIE::DeviceOp refDevice = nullptr;
858 for (auto &op : moduleOp.getBody()->getOperations()) {
859 if (auto dev = dyn_cast<AIE::DeviceOp>(op)) {
860 if (dev.getSymName() == refSymName) {
861 refDevice = dev;
862 break;
863 }
864 }
865 }
866
867 if (!refDevice)
868 continue;
869
870 // Find "configure" runtime sequence
871 AIE::RuntimeSequenceOp configureSeq = nullptr;
872 refDevice.walk([&](AIE::RuntimeSequenceOp seq) {
873 if (seq.getSymName() == "configure") {
874 configureSeq = seq;
875 }
876 });
877
878 if (!configureSeq)
879 continue;
880
881 // Create cert.section in current device
882 builder.setInsertionPoint(currentDevice.getBody()->getTerminator());
883 auto sectionOp = builder.create<AIEX::CertSectionOp>(
884 configureSeq.getLoc(), builder.getStringAttr(refSymName));
885
886 Block *sectionBlock = new Block();
887 sectionOp.getBody().push_back(sectionBlock);
888 builder.setInsertionPointToStart(sectionBlock);
889
890 // Create page
891 auto pageOp = builder.create<AIEX::CertPageOp>(configureSeq.getLoc());
892 Block *pageBlock = new Block();
893 pageOp.getBody().push_back(pageBlock);
894 builder.setInsertionPointToStart(pageBlock);
895
896 // Get unique job ID
897 uint32_t maxJobId = 0;
898 currentDevice.walk([&](AIEX::CertJobOp certJobOp) {
899 maxJobId = std::max(maxJobId, certJobOp.getJobId());
900 });
901
902 // Create job
903 auto jobOp = builder.create<AIEX::CertJobOp>(configureSeq.getLoc(),
904 maxJobId + 1);
905
906 // Clone configure sequence body into job
907 IRMapping mapper;
908 configureSeq.getRegion().cloneInto(&jobOp.getBody(), mapper);
909
910 // Ensure terminators
911 AIEX::CertJobOp::ensureTerminator(jobOp.getBody(), builder,
912 configureSeq.getLoc());
913 AIEX::CertPageOp::ensureTerminator(pageOp.getBody(), builder,
914 configureSeq.getLoc());
915 AIEX::CertSectionOp::ensureTerminator(sectionOp.getBody(), builder,
916 configureSeq.getLoc());
917 }
918 }
919 }
920
921 // Inline RunOps first
922 RewritePatternSet p_run_inline(&getContext());
923 p_run_inline.insert<RunOpInlining>(&getContext());
924 if (failed(applyPatternsGreedily(currentDevice, std::move(p_run_inline))))
925 return signalPassFailure();
926
927 // Then convert ConfigureOps to cert.section + cert.load_pdi
928 RewritePatternSet p_configure(&getContext());
929 p_configure.insert<ConfigureOpToCertSection>(&getContext());
930 if (failed(applyPatternsGreedily(currentDevice, std::move(p_configure))))
931 return signalPassFailure();
932
933 ConversionTarget target(getContext());
934 target.addIllegalOp<AIE::RuntimeSequenceOp>();
935
936 target.addLegalOp<AIEX::CertApplyOffset57Op>();
937 target.addLegalOp<AIEX::CertJobOp>();
938 target.addLegalOp<AIEX::CertPageOp>();
939 target.addLegalOp<AIEX::CertSectionOp>();
940 target.addLegalOp<AIEX::CertLoadPdiOp>();
941 target.addLegalOp<AIEX::CertMaskWrite32Op>();
942 target.addLegalOp<AIEX::CertUcDmaWriteDesSyncOp>();
943 target.addLegalOp<AIEX::CertUcDmaChainOp>();
944 target.addLegalOp<AIEX::CertUcDmaBdOp>();
945 target.addLegalOp<AIEX::CertWrite32Op>();
946 target.addLegalOp<AIEX::CertWaitTCTSOp>();
947 target.addLegalOp<AIEX::ConfigureOp>(); // TODO: Convert in separate pass
948 target.addLegalDialect<AIE::AIEDialect>();
949
950 RewritePatternSet p0(&getContext());
951 p0.insert<RuntimeSequenceToCertJob>(&getContext());
952
953 if (failed(applyPartialConversion(currentDevice, target, std::move(p0))))
954 return signalPassFailure();
955
956 target.addIllegalOp<AIEX::NpuAddressPatchOp>();
957
958 // patch conversion must come before blockwrite conversion
959 RewritePatternSet p1(&getContext());
960 p1.insert<NpuAddressPatchToCertApplyOffset57>(&getContext());
961
962 if (failed(applyPartialConversion(currentDevice, target, std::move(p1))))
963 return signalPassFailure();
964
965 // Split oversized NpuBlockWriteOps before lowering them to cert ops
966 {
967 RewritePatternSet p(&getContext());
968 p.insert<SplitNpuBlockWriteOpPattern>(&getContext());
969 if (failed(applyPatternsGreedily(getOperation(), std::move(p))))
970 signalPassFailure();
971 }
972
973 target.addIllegalOp<AIEX::NpuBlockWriteOp>();
974 target.addIllegalOp<AIEX::NpuMaskWrite32Op>();
975 target.addIllegalOp<AIEX::NpuSyncOp>();
976 target.addIllegalOp<AIEX::NpuWrite32Op>();
977 target.addIllegalOp<AIEX::NpuLoadPdiOp>();
978
979 RewritePatternSet p2(&getContext());
980 p2.insert<NpuBlockWriteToCertUcDma>(&getContext());
981 p2.insert<NpuMaskWrite32ToCertMaskWrite32>(&getContext());
982 p2.insert<NpuWrite32ToCertWrite32>(&getContext());
983 p2.insert<NpuSyncToCertWaitTCTS>(&getContext());
984 p2.insert<NpuLoadPdiToCertLoadPdi>(&getContext());
985
986 if (failed(applyPartialConversion(currentDevice, target, std::move(p2))))
987 return signalPassFailure();
988
989 // Add the merge pattern for CertUcDmaWriteDesSyncOps
990 RewritePatternSet p3(&getContext());
991 p3.insert<MergeConsecutiveCertUcDmaWriteDesSyncOps>(&getContext());
992 if (failed(applyPatternsGreedily(currentDevice, std::move(p3))))
993 return signalPassFailure();
994
995 // Convert referenced devices to cert.sections
996 // Now that cert.load_pdi ops exist, find referenced devices and convert
997 // their cert jobs into sections in this device
998 {
999 OpBuilder builder(&getContext());
1000
1001 // Find all cert.load_pdi in current device
1002 llvm::SetVector<StringRef> referencedDeviceNames;
1003 currentDevice.walk([&](AIEX::CertLoadPdiOp loadPdiOp) {
1004 StringRef refSymName = loadPdiOp.getSymbol();
1005 if (!currentDevice.lookupSymbol(refSymName))
1006 referencedDeviceNames.insert(refSymName);
1007 });
1008
1009 // For each referenced device, create a section with its cert jobs
1010 for (StringRef refSymName : referencedDeviceNames) {
1011 // Find the referenced device
1012 AIE::DeviceOp refDevice = nullptr;
1013 for (auto &op : moduleOp.getBody()->getOperations()) {
1014 if (auto dev = dyn_cast<AIE::DeviceOp>(op)) {
1015 if (dev.getSymName() == refSymName) {
1016 refDevice = dev;
1017 break;
1018 }
1019 }
1020 }
1021
1022 if (!refDevice)
1023 continue;
1024
1025 // Find all cert.page ops in the referenced device that are direct
1026 // children
1027 llvm::SmallVector<AIEX::CertPageOp, 4> certPages;
1028 for (auto &op : refDevice.getBody()->getOperations()) {
1029 if (auto pageOp = dyn_cast<AIEX::CertPageOp>(op)) {
1030 certPages.push_back(pageOp);
1031 }
1032 }
1033
1034 if (certPages.empty())
1035 continue;
1036
1037 // Create cert.section in current device
1038 builder.setInsertionPoint(currentDevice.getBody()->getTerminator());
1039 auto sectionOp = builder.create<AIEX::CertSectionOp>(
1040 refDevice.getLoc(), builder.getStringAttr(refSymName));
1041
1042 Block *sectionBlock = new Block();
1043 sectionOp.getBody().push_back(sectionBlock);
1044
1045 // Clone all cert.page operations into the section
1046 IRMapping mapper;
1047 for (auto pageOp : certPages) {
1048 builder.setInsertionPointToEnd(sectionBlock);
1049 builder.clone(*pageOp.getOperation(), mapper);
1050 }
1051
1052 // Ensure section terminator
1053 AIEX::CertSectionOp::ensureTerminator(sectionOp.getBody(), builder,
1054 refDevice.getLoc());
1055 }
1056 }
1057
1058 // Clean up unused block arguments from cert.job operations
1059 currentDevice.walk([&](AIEX::CertJobOp jobOp) {
1060 Block &jobBlock = jobOp.getBody().front();
1061
1062 // Remove unused block arguments (in reverse order)
1063 for (int i = jobBlock.getNumArguments() - 1; i >= 0; --i) {
1064 BlockArgument arg = jobBlock.getArgument(i);
1065 if (arg.use_empty()) {
1066 jobBlock.eraseArgument(i);
1067 }
1068 }
1069 });
1070 }
1071};

1075static void updateCostForOp(Operation &o, AIE::DeviceOp deviceOp,
1076 uint32_t &text_cost, uint32_t &data_cost) {
1077 if (isa<AIEX::CertLocalBarrierOp>(o)) {
1078 text_cost += 8; // local barrier
1079 } else if (isa<AIEX::CertRemoteBarrierOp>(o)) {
1080 text_cost += 8; // remote barrier
1081 } else if (isa<AIEX::CertWaitTCTSOp>(o)) {
1082 text_cost += 8; // wait tct
1083 } else if (isa<AIEX::CertMaskWrite32Op>(o)) {
1084 text_cost += 16; // mask write
1085 } else if (isa<AIEX::CertWrite32Op>(o)) {
1086 text_cost += 12; // write
1087 } else if (isa<AIEX::CertApplyOffset57Op>(o)) {
1088 text_cost += 16; // apply offset
1089 } else if (auto syncOp = dyn_cast<AIEX::CertUcDmaWriteDesSyncOp>(o)) {
1090 text_cost += 16; // write des sync
1091 // find the uc_dma_chain
1092 StringRef sym_name = syncOp.getSymbol();
1093 auto chain = dyn_cast_if_present<AIEX::CertUcDmaChainOp>(
1094 deviceOp.lookupSymbol(sym_name));
1095 if (!chain)
1096 return;
1097 for (auto bdOp : chain.getBody().front().getOps<AIEX::CertUcDmaBdOp>()) {
1098 data_cost += 16; // bd op
1099 StringRef data_sym_name = bdOp.getRemoteAddress();
1100 auto global = dyn_cast_if_present<memref::GlobalOp>(
1101 deviceOp.lookupSymbol(data_sym_name));
1102 if (!global)
1103 continue;
1104 auto initVal = global.getInitialValue();
1105 if (!initVal)
1106 continue;
1107 auto data = dyn_cast<DenseIntElementsAttr>(*initVal);
1108 if (!data)
1109 continue;
1110 data_cost += data.getNumElements() * 4; // 4 bytes per element
1111 }
1112 }
1113}

1115static uint32_t estimateCost(AIEX::CertPageOp op, uint32_t split_target,
1116 AIEX::CertJobOp &split_job,
1117 Block::iterator &split_iter,
1118 bool &found_split_point) {
1119 uint32_t text_cost = 32; // page header
1120 uint32_t data_cost = 0;
1121 found_split_point = false;
1122 AIE::DeviceOp deviceOp = op->getParentOfType<AIE::DeviceOp>();
1123
1124 for (auto job : op.getBody().front().getOps<AIEX::CertJobOp>()) {
1125 for (auto &o : job.getBody().front().getOperations()) {
1126 Block::iterator current(&o);
1127 if (!found_split_point && !isa<AIE::EndOp>(o) &&
1128 current != job.getBody().front().begin() &&
1129 (text_cost + data_cost) >= split_target) {
1130 split_job = job;
1131 split_iter = current;
1132 found_split_point = true;
1133 }
1134
1135 updateCostForOp(o, deviceOp, text_cost, data_cost);
1136
1137 if (!found_split_point && (text_cost + data_cost) >= split_target) {
1138 Block::iterator next = current;
1139 ++next;
1140 if (next != job.getBody().front().end() && !isa<AIE::EndOp>(*next)) {
1141 split_job = job;
1142 split_iter = next;
1143 found_split_point = true;
1144 }
1145 }
1146 }
1147 }
1148 return text_cost + data_cost;
1149}

1175// Pattern to isolate load_pdi and preempt operations into their own job and
1176// page According to CERT spec: "load_pdi and preempt should take one whole job
1177// which in turn should take one whole page"
1178struct IsolateFullPageOpsPattern : OpRewritePattern<AIEX::CertJobOp> {
1179 using OpRewritePattern::OpRewritePattern;
1180
1181 LogicalResult matchAndRewrite(AIEX::CertJobOp jobOp,
1182 PatternRewriter &rewriter) const override {
1183
1184 // Find first load_pdi or preempt in this job
1185 Operation *fullPageOp = nullptr;
1186 Block::iterator fullPageOpIter;
1187 for (Block::iterator it = jobOp.getBody().front().begin();
1188 it != jobOp.getBody().front().end(); ++it) {
1189 if (isa<AIEX::CertLoadPdiOp, AIEX::CertPreemptOp>(*it)) {
1190 fullPageOp = &*it;
1191 fullPageOpIter = it;
1192 break;
1193 }
1194 }
1195
1196 if (!fullPageOp)
1197 return failure(); // No full-page op in this job
1198
1199 // Check if this job ONLY contains the full-page op (and terminator)
1200 size_t opCount = 0;
1201 for (Operation &op : jobOp.getBody().front().getOperations()) {
1202 if (!isa<AIE::EndOp>(op))
1203 opCount++;
1204 }
1205
1206 if (opCount == 1) {
1207 // Job only contains full-page op - check if it's in its own page
1208 auto parentPage = jobOp->getParentOfType<AIEX::CertPageOp>();
1209 if (!parentPage)
1210 return failure(); // No parent page (unusual but skip)
1211
1212 // Count jobs in parent page
1213 size_t jobCount = 0;
1214 for (Operation &op : parentPage.getBody().front().getOperations()) {
1215 if (isa<AIEX::CertJobOp>(op))
1216 jobCount++;
1217 }
1218
1219 if (jobCount == 1)
1220 return failure(); // Already properly isolated
1221
1222 // Job is isolated but shares page - need to move to own page
1223 auto loc = jobOp.getLoc();
1224 rewriter.setInsertionPointAfter(parentPage);
1225
1226 // Create new page for this job
1227 auto newPageOp = AIEX::CertPageOp::create(rewriter, loc);
1228 Block *newPageBlock = new Block();
1229 newPageOp.getBody().push_back(newPageBlock);
1230
1231 // Move the job to the new page
1232 rewriter.setInsertionPointToStart(newPageBlock);
1233 jobOp->moveBefore(newPageBlock, newPageBlock->begin());
1234
1235 AIEX::CertPageOp::ensureTerminator(newPageOp.getBody(), rewriter, loc);
1236
1237 return success();
1238 }
1239
1240 // Job contains full-page op mixed with other operations - need to split
1241 auto loc = jobOp.getLoc();
1242 auto parentDevice = jobOp->getParentOfType<AIE::DeviceOp>();
1243
1244 // Assign new job IDs
1245 uint32_t maxJobId = 0;
1246 parentDevice.walk([&](AIEX::CertJobOp certJobOp) {
1247 maxJobId = std::max(maxJobId, certJobOp.getJobId());
1248 });
1249
1250 uint32_t beforeJobId = jobOp.getJobId();
1251 uint32_t fullPageJobId = maxJobId + 1;
1252 uint32_t afterJobId = maxJobId + 2;
1253
1254 // Collect operations before full-page op
1255 SmallVector<Operation *> beforeOps;
1256 for (Block::iterator it = jobOp.getBody().front().begin();
1257 it != fullPageOpIter; ++it) {
1258 if (!isa<AIE::EndOp>(*it))
1259 beforeOps.push_back(&*it);
1260 }
1261
1262 // Collect operations after full-page op
1263 SmallVector<Operation *> afterOps;
1264 Block::iterator afterStart = fullPageOpIter;
1265 ++afterStart; // Skip the full-page op itself
1266 for (Block::iterator it = afterStart; it != jobOp.getBody().front().end();
1267 ++it) {
1268 if (!isa<AIE::EndOp>(*it))
1269 afterOps.push_back(&*it);
1270 }
1271
1272 // Get parent page to insert new pages after it
1273 auto parentPage = jobOp->getParentOfType<AIEX::CertPageOp>();
1274 rewriter.setInsertionPoint(parentPage);
1275
1276 // Create first page with operations before full-page op (if any)
1277 if (!beforeOps.empty()) {
1278 auto page1 = AIEX::CertPageOp::create(rewriter, loc);
1279 Block *page1Block = new Block();
1280 page1.getBody().push_back(page1Block);
1281 rewriter.setInsertionPointToStart(page1Block);
1282
1283 auto job1 = AIEX::CertJobOp::create(rewriter, loc, beforeJobId);
1284 Block *job1Block = new Block();
1285 job1.getBody().push_back(job1Block);
1286 rewriter.setInsertionPointToStart(job1Block);
1287
1288 for (Operation *op : beforeOps) {
1289 op->moveBefore(job1Block, job1Block->end());
1290 }
1291
1292 AIEX::CertJobOp::ensureTerminator(job1.getBody(), rewriter, loc);
1293 AIEX::CertPageOp::ensureTerminator(page1.getBody(), rewriter, loc);
1294 }
1295
1296 // Create page with full-page op in its own job
1297 rewriter.setInsertionPointAfter(parentPage);
1298 auto page2 = AIEX::CertPageOp::create(rewriter, loc);
1299 Block *page2Block = new Block();
1300 page2.getBody().push_back(page2Block);
1301 rewriter.setInsertionPointToStart(page2Block);
1302
1303 auto job2 = AIEX::CertJobOp::create(rewriter, loc, fullPageJobId);
1304 Block *job2Block = new Block();
1305 job2.getBody().push_back(job2Block);
1306 rewriter.setInsertionPointToStart(job2Block);
1307
1308 fullPageOp->moveBefore(job2Block, job2Block->end());
1309
1310 AIEX::CertJobOp::ensureTerminator(job2.getBody(), rewriter, loc);
1311 AIEX::CertPageOp::ensureTerminator(page2.getBody(), rewriter, loc);
1312
1313 // Create third page with operations after full-page op (if any)
1314 if (!afterOps.empty()) {
1315 rewriter.setInsertionPointAfter(page2);
1316 auto page3 = AIEX::CertPageOp::create(rewriter, loc);
1317 Block *page3Block = new Block();
1318 page3.getBody().push_back(page3Block);
1319 rewriter.setInsertionPointToStart(page3Block);
1320
1321 auto job3 = AIEX::CertJobOp::create(rewriter, loc, afterJobId);
1322 Block *job3Block = new Block();
1323 job3.getBody().push_back(job3Block);
1324 rewriter.setInsertionPointToStart(job3Block);
1325
1326 for (Operation *op : afterOps) {
1327 op->moveBefore(job3Block, job3Block->end());
1328 }
1329
1330 AIEX::CertJobOp::ensureTerminator(job3.getBody(), rewriter, loc);
1331 AIEX::CertPageOp::ensureTerminator(page3.getBody(), rewriter, loc);
1332 }
1333
1334 // Erase the original job and page
1335 rewriter.eraseOp(jobOp);
1336 if (parentPage.getBody().front().empty() ||
1337 llvm::all_of(parentPage.getBody().front(),
1338 [](Operation &op) { return isa<AIE::EndOp>(op); })) {
1339 rewriter.eraseOp(parentPage);
1340 }
1341
1342 return success();
1343 }

1346struct SplitCertPageOpPattern : OpRewritePattern<AIEX::CertPageOp> {
1347 using OpRewritePattern::OpRewritePattern;
1348
1349 LogicalResult matchAndRewrite(AIEX::CertPageOp op,
1350 PatternRewriter &rewriter) const override {
1351
1352 constexpr uint32_t split_threshold = cert_page_size;
1353
1354 AIEX::CertJobOp split_job;
1355 Block::iterator split_iter;
1356 bool found_split_point = false;
1357 uint32_t cost = estimateCost(op, cert_page_size / 2, split_job, split_iter,
1358 found_split_point);
1359 LLVM_DEBUG(llvm::outs() << "Estimate cost for page: "
1360 << " is " << cost << "\n");
1361
1362 if (cost < split_threshold || !found_split_point)
1363 return failure();
1364
1365 auto loc = op.getLoc();
1366 op->getParentOfType<AIE::DeviceOp>().walk([&](AIEX::CertJobOp certJobOp) {
1367 if (certJobOp.getJobId() > split_job.getJobId())
1368 certJobOp.setJobId(certJobOp.getJobId() + 1);
1369 });
1370
1371 auto cloneJobRange = [&](AIEX::CertJobOp sourceJob, uint32_t jobId,
1372 Block::iterator begin, Block::iterator end) {
1373 auto newJobOp = AIEX::CertJobOp::create(rewriter, loc, jobId);
1374 Block *newJobBlock = new Block();
1375 newJobOp.getBody().push_back(newJobBlock);
1376
1377 IRMapping mapper;
1378 Block &sourceBlock = sourceJob.getBody().front();
1379 for (BlockArgument arg : sourceBlock.getArguments()) {
1380 BlockArgument newArg =
1381 newJobBlock->addArgument(arg.getType(), arg.getLoc());
1382 mapper.map(arg, newArg);
1383 }
1384
1385 rewriter.setInsertionPointToStart(newJobBlock);
1386 for (Block::iterator oi = begin; oi != end; ++oi) {
1387 if (!isa<AIE::EndOp>(*oi))
1388 rewriter.clone(*oi, mapper);
1389 }
1390 AIEX::CertJobOp::ensureTerminator(newJobOp.getBody(), rewriter, loc);
1391 };
1392
1393 rewriter.setInsertionPoint(op);
1394 auto newPageOp0 = AIEX::CertPageOp::create(rewriter, loc);
1395 Block *newPageBlock0 = new Block();
1396 newPageOp0.getBody().push_back(newPageBlock0);
1397
1398 auto newPageOp1 = AIEX::CertPageOp::create(rewriter, loc);
1399 Block *newPageBlock1 = new Block();
1400 newPageOp1.getBody().push_back(newPageBlock1);
1401
1402 for (auto job : op.getBody().front().getOps<AIEX::CertJobOp>()) {
1403 if (job == split_job) {
1404 rewriter.setInsertionPointToEnd(newPageBlock0);
1405 cloneJobRange(job, job.getJobId(), job.getBody().front().begin(),
1406 split_iter);
1407
1408 rewriter.setInsertionPointToEnd(newPageBlock1);
1409 cloneJobRange(job, job.getJobId() + 1, split_iter,
1410 job.getBody().front().end());
1411 continue;
1412 }
1413
1414 if (job->isBeforeInBlock(split_job)) {
1415 rewriter.setInsertionPointToEnd(newPageBlock0);
1416 cloneJobRange(job, job.getJobId(), job.getBody().front().begin(),
1417 job.getBody().front().end());
1418 } else {
1419 rewriter.setInsertionPointToEnd(newPageBlock1);
1420 cloneJobRange(job, job.getJobId(), job.getBody().front().begin(),
1421 job.getBody().front().end());
1422 }
1423 }
1424
1425 AIEX::CertPageOp::ensureTerminator(newPageOp0.getBody(), rewriter, loc);
1426 AIEX::CertPageOp::ensureTerminator(newPageOp1.getBody(), rewriter, loc);
1427
1428 rewriter.eraseOp(op);
1429 return success();
1430 }
1431};
