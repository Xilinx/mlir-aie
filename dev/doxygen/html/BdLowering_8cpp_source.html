22// SsaStridePolicy: arith-emitting mirror of ConstStridePolicy.
23//
24// Arithmetic is i32, matching the BD word fields. This agrees with
25// ConstStridePolicy's int64 because every value is bounded by a BD field or the
26// 32-bit buffer_length, so no intermediate product overflows (a stride big
27// enough to overflow exceeds a single shim BD's extent).
28//===----------------------------------------------------------------------===//
29
30Value SsaStridePolicy::cst(int64_t c) const {
31 auto i32ty = builder.getIntegerType(32);
32 return arith::ConstantOp::create(builder, loc, IntegerAttr::get(i32ty, c));
33}
34
35Value SsaStridePolicy::mul(Value v, int64_t c) const {
36 return arith::MulIOp::create(builder, loc, v, cst(c));
37}
38
39Value SsaStridePolicy::div(Value v, int64_t c) const {
40 // Hardware granularity scaling is exact (verifier enforces divisibility);
41 // unsigned division matches the constant policy on the non-negative values
42 // that reach here.
43 return arith::DivUIOp::create(builder, loc, v, cst(c));
44}
45
46Value SsaStridePolicy::sub(Value v, int64_t c) const {
47 return arith::SubIOp::create(builder, loc, v, cst(c));
48}

50Value SsaStridePolicy::selectGT1(Value cond, Value t, Value e) const {
51 Value one = cst(1);
52 Value gt =
53 arith::CmpIOp::create(builder, loc, arith::CmpIPredicate::sgt, cond, one);
54 return arith::SelectOp::create(builder, loc, gt, t, e);
55}

57Value SsaStridePolicy::selectGT0(Value cond, Value t, Value e) const {
58 Value zero = cst(0);
59 Value gt = arith::CmpIOp::create(builder, loc, arith::CmpIPredicate::sgt,
60 cond, zero);
61 return arith::SelectOp::create(builder, loc, gt, t, e);

74Value getAsValue(OpBuilder &builder, Location loc, OpFoldResult ofr,
75 Type intType) {
76 if (auto constVal = getConstantIntValue(ofr))
77 return arith::ConstantOp::create(builder, loc,
78 IntegerAttr::get(intType, *constVal));
79 Value val = cast<Value>(ofr);
80 if (val.getType() != intType) {
81 unsigned valBits = val.getType().getIntOrFloatBitWidth();
82 unsigned tgtBits = intType.getIntOrFloatBitWidth();
83 if (valBits > tgtBits)
84 val = arith::TruncIOp::create(builder, loc, intType, val);
85 else
86 val = arith::ExtUIOp::create(builder, loc, intType, val);
87 }
88 return val;
89}

91Value buildArgPlusValue(OpBuilder &builder, Location loc,
92 ArrayRef<OpFoldResult> elementOffsets,
93 ArrayRef<OpFoldResult> strides, int64_t elemWidthBytes,
94 int64_t baseByteOffset) {
95 auto i32ty = builder.getIntegerType(32);
96
97 // Fast path: everything constant -> fold to one arith.constant, matching the
98 // static lowering byte-for-byte.
99 bool allConst = llvm::all_of(elementOffsets,
100 [](OpFoldResult o) {
101 return getConstantIntValue(o).has_value();
102 }) &&
103 llvm::all_of(strides, [](OpFoldResult s) {
104 return getConstantIntValue(s).has_value();
105 });
106 if (allConst) {
107 int64_t bytes = baseByteOffset;
108 for (auto [o, s] : llvm::zip(elementOffsets, strides))
109 bytes +=
110 *getConstantIntValue(o) * *getConstantIntValue(s) * elemWidthBytes;
111 return arith::ConstantOp::create(builder, loc,
112 IntegerAttr::get(i32ty, bytes));
113 }
114
115 // Runtime path: sum(offset[i] * stride[i]) * elemWidthBytes + base, as arith.
116 Value acc =
117 arith::ConstantOp::create(builder, loc, IntegerAttr::get(i32ty, 0));
118 for (auto [o, s] : llvm::zip(elementOffsets, strides)) {
119 Value ov = getAsValue(builder, loc, o, i32ty);
120 Value sv = getAsValue(builder, loc, s, i32ty);
121 Value prod = arith::MulIOp::create(builder, loc, ov, sv);
122 acc = arith::AddIOp::create(builder, loc, acc, prod);
123 }
124 Value width = arith::ConstantOp::create(
125 builder, loc, IntegerAttr::get(i32ty, elemWidthBytes));
126 acc = arith::MulIOp::create(builder, loc, acc, width);
127 if (baseByteOffset != 0) {
128 Value base = arith::ConstantOp::create(
129 builder, loc, IntegerAttr::get(i32ty, baseByteOffset));
130 acc = arith::AddIOp::create(builder, loc, acc, base);
131 }
132 return acc;
133}

135Value buildBdWord(OpBuilder &builder, Location loc,
136 ArrayRef<std::tuple<Value, uint32_t, uint32_t>> fields) {
137 auto i32ty = builder.getIntegerType(32);
138 Value result =
139 arith::ConstantOp::create(builder, loc, IntegerAttr::get(i32ty, 0));
140 for (auto &[val, mask, shift] : fields) {
141 Value masked = val;
142 if (mask != 0xFFFFFFFF) {
143 auto maskConst = arith::ConstantOp::create(
144 builder, loc, IntegerAttr::get(i32ty, (int64_t)(int32_t)mask));
145 masked = arith::AndIOp::create(builder, loc, masked, maskConst);
146 }
147 if (shift > 0) {
148 auto shiftConst = arith::ConstantOp::create(
149 builder, loc, IntegerAttr::get(i32ty, shift));
150 masked = arith::ShLIOp::create(builder, loc, masked, shiftConst);
151 }
152 result = arith::OrIOp::create(builder, loc, result, masked);
153 }
154 return result;
155}

157Value getBdRegisterBase(OpBuilder &builder, Location loc,
158 const AIE::AIETargetModel &targetModel, int tileCol,
159 int tileRow, OpFoldResult bdId) {
160 auto i32ty = builder.getIntegerType(32);
161 if (auto c = getConstantIntValue(bdId))
162 return createConstantI32(builder, loc,
163 static_cast<uint32_t>(targetModel.getDmaBdAddress(
164 tileCol, tileRow, *c)));
165 // Runtime bd_id: base + bd_id * bdStride, with base/stride from the (linear)
166 // target-model address function.
167 uint64_t addrForId0 = targetModel.getDmaBdAddress(tileCol, tileRow, 0);
168 uint64_t bdStride =
169 targetModel.getDmaBdAddress(tileCol, tileRow, 1) - addrForId0;
170 Value bdIdVal = getAsValue(builder, loc, bdId, i32ty);
171 return arith::AddIOp::create(
172 builder, loc, createConstantI32(builder, loc, addrForId0),
173 arith::MulIOp::create(builder, loc, bdIdVal,
174 createConstantI32(builder, loc, bdStride)));
175}

177LogicalResult emitDynamicShimBdWordOverrides(
178 OpBuilder &builder, Location loc, const AIE::AIETargetModel &targetModel,
179 int tileCol, int tileRow, OpFoldResult bdId,
180 ArrayRef<OpFoldResult> mixedSizes, ArrayRef<OpFoldResult> mixedStrides,
181 uint64_t elemWidth, uint32_t burstLength, Value bufLenOverride,
182 Value &repeatCountOut) {
183 auto i32ty = builder.getIntegerType(32);
184
185 // Compute the hardware sizes/strides as SSA values via the shared encoder.
186 uint32_t gran = targetModel.getAddressGenGranularity();
187 SmallVector<OpFoldResult, 4> sizesRev(llvm::reverse(mixedSizes));
188 SmallVector<OpFoldResult, 4> stridesRev(llvm::reverse(mixedStrides));
189 Value inS[4], inT[4], hwS[4], hwT[4];
190 for (int i = 0; i < 4; i++) {
191 inS[i] = getAsValue(builder, loc, sizesRev[i], i32ty);
192 inT[i] = getAsValue(builder, loc, stridesRev[i], i32ty);
193 }
194 SsaStridePolicy policy(builder, loc);
195 encodeHardwareStridesWraps(policy, elemWidth, gran, inS, inT, hwS, hwT);
196
197 // buffer_length (word[0]): the caller's runtime len if supplied (dma_task),
198 // else the d0*d1*d2 hardware-unit size-product (dma_memcpy_nd). hwS[0]
199 // already carries the elemWidth/gran scaling; d1/d2 are element counts.
200 Value bufLen = bufLenOverride;
201 if (!bufLen)
202 bufLen = arith::MulIOp::create(
203 builder, loc, arith::MulIOp::create(builder, loc, hwS[0], inS[1]),
204 inS[2]);
205
206 // Linear mode: contiguous transfer (each outer stride == product of inner
207 // sizes) folds d0/d1 into buffer_length, dodging the 10-bit d0_size limit. A
208 // runtime operand needed for the test falls back to ND mode.
209 auto cst = [&](OpFoldResult v) { return getConstantIntValue(v); };
210 auto knownContiguous = [&]() -> bool {
211 auto s0 = cst(stridesRev[0]);
212 if (!s0 || *s0 != 1)
213 return false;
214 auto sz0 = cst(sizesRev[0]);
215 auto d1sz = cst(sizesRev[1]);
216 auto d1st = cst(stridesRev[1]);
217 if (!(d1sz && *d1sz == 1))
218 if (!sz0 || !d1st || *d1st != *sz0)
219 return false;
220 auto d2sz = cst(sizesRev[2]);
221 auto d2st = cst(stridesRev[2]);
222 if (!(d2sz && *d2sz == 1)) {
223 auto prod01 =
224 (sz0 && d1sz) ? std::optional<int64_t>(*sz0 * *d1sz) : std::nullopt;
225 if (!prod01 || !d2st || *d2st != *prod01)
226 return false;
227 }
228 return true;
229 };
230 bool isLinear = knownContiguous();
231
232 // Guard a RUNTIME size against its narrow BD field (masking would silently
233 // truncate); constants are verifier-checked. d0/d1 wrap (10-bit) only in ND
234 // mode, iteration wrap (6-bit) always. Guard is on the hardware value.
235 auto guardField = [&](OpFoldResult inSize, Value hwVal, int64_t fieldMax) {
236 if (getConstantIntValue(inSize))
237 return; // constant: verifier already enforced the bound.
238 NpuAssertBdFieldOp::create(builder, loc, hwVal,
239 builder.getI32IntegerAttr(fieldMax));
240 };
241 if (!isLinear) {
242 guardField(sizesRev[0], hwS[0], ShimBdFieldWidths::d0WrapMax());
243 guardField(sizesRev[1], hwS[1], ShimBdFieldWidths::d1WrapMax());
244 }
245 guardField(sizesRev[3], hwS[3], ShimBdFieldWidths::iterWrapMax());
246
247 // Guard a RUNTIME size/stride whose byte extent must be a whole number of
248 // granules (mirrors verifyStridesWraps). Guard is on the input element count;
249 // constants are verifier-checked.
250 int64_t divisor = bdGranuleDivisor(elemWidth, gran);
251 auto guardDivisible = [&](OpFoldResult in, Value inVal, bool allowUnit) {
252 if (divisor <= 1 || getConstantIntValue(in))
253 return;
254 NpuAssertBdDivisibleOp::create(builder, loc, inVal, (uint32_t)divisor,
255 /*allow_unit=*/allowUnit);
256 };
257 guardDivisible(sizesRev[0], inS[0], /*allowUnit=*/false);
258 // The innermost stride collapses to hardware 0 when contiguous or
259 // granule-aligned; guard with the unit-stride exemption. Outer strides don't
260 // collapse.
261 guardDivisible(stridesRev[0], inT[0], /*allowUnit=*/true);
262 for (int i = 1; i < 4; i++)
263 guardDivisible(stridesRev[i], inT[i], /*allowUnit=*/false);
264
265 // The shim BD register block is at getDmaBdAddress(col,row,bd_id). A constant
266 // bd_id folds to the literal the static path uses; a runtime bd_id (dynamic
267 // free-list pool) yields an arith expression base + bd_id*bdStride, so each
268 // override targets that base + wordIdx*4.
269 Value bdBase =
270 getBdRegisterBase(builder, loc, targetModel, tileCol, tileRow, bdId);
271 std::optional<int64_t> constBase = getConstantIntValue(bdBase);
272 auto writeWord = [&](uint32_t wordIdx, Value val) {
273 // Fold the address to a literal for a constant base, so the static/pinned
274 // path emits the same single constant it always has.
275 Value addr =
276 constBase
277 ? createConstantI32(builder, loc,
278 static_cast<uint32_t>(*constBase + wordIdx * 4))
279 : arith::AddIOp::create(
280 builder, loc, bdBase,
281 createConstantI32(builder, loc, wordIdx * 4));
282 NpuWrite32Op::create(builder, loc, addr, val, nullptr, nullptr, nullptr);
283 };
284
285 // word[0] buffer_length is always overridden (it carries the runtime element
286 // count in both linear and ND modes).
287 writeWord(0, bufLen);
288
289 // Linear mode leaves the d0/d1/d2 size/stride words at their zero template
290 // (only buffer_length + iteration matter); ND mode overrides them.
291 if (!isLinear) {
292 // word[3]: d0_size [29:20], d0_stride [19:0].
293 writeWord(3, buildBdWord(builder, loc,
294 {{hwS[0], 0x3FF, 20}, {hwT[0], 0xFFFFF, 0}}));
295 // word[4]: burst_length [31:30] (static), d1_size [29:20], d1_stride
296 // [19:0].
297 Value burst = createConstantI32(
298 builder, loc,
299 (AIE::getShimBurstLengthEncoding(targetModel, burstLength) & 0x3)
300 << 30);
301 writeWord(4, arith::OrIOp::create(
302 builder, loc, burst,
303 buildBdWord(builder, loc,
304 {{hwS[1], 0x3FF, 20}, {hwT[1], 0xFFFFF, 0}})));
305 // word[5]: AXCache [27:24] (static), d2_stride [19:0]. Shim d2_size is
306 // always 0 (the template already has it), carried by bufLen.
307 Value axcache = createConstantI32(builder, loc, (2u & 0xf) << 24);
308 writeWord(5, arith::OrIOp::create(
309 builder, loc, axcache,
310 buildBdWord(builder, loc, {{hwT[2], 0xFFFFF, 0}})));
311 }
312 // word[6]: iteration_size [25:20], iteration_stride [19:0]. A zero outer
313 // stride is a pure repeat (carried by repeat_count), so both fields must be 0
314 // like AIEDmaToNpu; gate iteration_size on the stride while leaving hwS[3]
315 // for repeatCountOut. hwT[3] already collapses to 0.
316 Value zeroI32 = createConstantI32(builder, loc, 0);
317 Value iterStridePos = arith::CmpIOp::create(
318 builder, loc, arith::CmpIPredicate::sgt, inT[3], zeroI32);
319 Value iterSizeField =
320 arith::SelectOp::create(builder, loc, iterStridePos, hwS[3], zeroI32);
321 writeWord(6, buildBdWord(builder, loc,
322 {{iterSizeField, 0x3F, 20}, {hwT[3], 0xFFFFF, 0}}));
323
324 // repeat_count for the queue push is the biased outer size (N > 1 ? N - 1 :
325 // 0), matching the static path. A constant folds so the static push_queue
326 // lowering can consume it; a runtime size yields the SSA hwS[3].
327 if (auto outerConst = getConstantIntValue(sizesRev[3])) {
328 int64_t r = *outerConst > 1 ? *outerConst - 1 : 0;
329 repeatCountOut =
330 arith::ConstantOp::create(builder, loc, IntegerAttr::get(i32ty, r));
331 } else {
332 repeatCountOut = hwS[3];
333 }
334 return success();
335}
