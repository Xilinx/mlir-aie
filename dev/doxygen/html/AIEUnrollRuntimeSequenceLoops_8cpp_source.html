8// Fully unrolls constant-bound scf.for loops inside aie.runtime_sequence
9// bodies. Running this before aie-assign-runtime-sequence-bd-ids leaves the
10// allocator with straight-line IR (no back-edges), so a task freed in a later
11// iteration than it is configured becomes several distinct configures whose BD
12// ids ordinary liveness-based allocation recycles -- no rolled-loop rotation
13// analysis is needed.
14//
15// Runtime-bound loops (non-constant trip count) are left in place; they will be
16// handled by the dynamic EmitC path in a later phase.
17//
18//===----------------------------------------------------------------------===//

41struct AIEUnrollRuntimeSequenceLoopsPass
42 : xilinx::AIEX::impl::AIEUnrollRuntimeSequenceLoopsBase<
43 AIEUnrollRuntimeSequenceLoopsPass> {
44
45 void runOnOperation() override {
46 DeviceOp device = getOperation();
47
48 device.walk([&](RuntimeSequenceOp seq) {
49 // Collect constant-trip loops in post-order (innermost first), then
50 // unroll in that order. loopUnrollFull unrolls only the given loop,
51 // cloning any nested loops as-is -- so nested loops need their own calls.
52 // Post-order guarantees an inner loop is unrolled (and gone) before its
53 // enclosing loop, so every collected handle is still valid when reached
54 // and no re-walk is needed. Reaches loops anywhere in the sequence,
55 // including inside scf.if arms.
56 SmallVector<scf::ForOp> loops;
57 seq.walk([&](scf::ForOp forOp) {
58 if (forOp.getStaticTripCount().has_value())
59 loops.push_back(forOp);
60 });
61 for (scf::ForOp forOp : loops)
62 (void)loopUnrollFull(forOp);
63 });
64 }
65};
66
67} // namespace
