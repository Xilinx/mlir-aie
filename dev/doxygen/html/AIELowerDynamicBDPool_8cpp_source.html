7//
8// The dynamic counterpart to AIEAssignRuntimeSequenceBDIDs: where that pass
9// rejects a runtime-bound scf.for, this keeps the loop rolled and draws bd_ids
10// from a per-tile runtime free-list pool. Each configure gets a dma_bd_pool_pop
11// (its SSA id feeding bd_id_val); each free/await gets a dma_bd_pool_push.
12//
13// A popped id must be reachable at its push. Since a task value can cross a
14// loop back edge and exit as a result, the id is carried in lockstep: every
15// task iter_arg gets a parallel i32 id iter_arg, and a push looks up the paired
16// id.
17//
18// v1 is single-BD only; multi-BD chains would need runtime next_bd (follow-up).
19//
20//===----------------------------------------------------------------------===//

43struct AIELowerDynamicBDPoolPass
44 : xilinx::AIEX::impl::AIELowerDynamicBDPoolBase<AIELowerDynamicBDPoolPass> {
45
46 // Maps a task value (the Index result of a configure, or any Index the carry
47 // propagates it into) to the i32 pool id available at that same program
48 // point.
49 llvm::DenseMap<Value, Value> pairedId;
50 // The (col,row) of the tile each task's id belongs to, so a push names the
51 // right pool even when the id is a loop result rather than a direct pop.
52 llvm::DenseMap<Value, std::pair<int, int>> tileForTask;
53 // A representative (loop-invariant) configure per task, so a post-loop await
54 // on a loop result -- which has no defining configure -- can be redirected to
55 // one with the right (tile,dir,channel).
56 llvm::DenseMap<Value, DMAConfigureTaskOp> originConfigure;
57
58 // Count the BD ops in a configure's body (across all its blocks).
59 static unsigned countBds(DMAConfigureTaskOp cfg) {
60 unsigned n = 0;
61 cfg.walk([&](AIE::DMABDOp) { ++n; });
62 return n;
63 }
64
65 LogicalResult lowerConfigure(DMAConfigureTaskOp cfg) {
66 if (cfg.getBdIdVal())
67 return success(); // already lowered
68
69 if (countBds(cfg) != 1)
70 return cfg.emitOpError(
71 "dynamic BD pool lowering supports single-BD tasks only; a multi-BD "
72 "chain under a runtime-bound loop needs runtime next_bd chaining "
73 "(not yet implemented)");
74
75 // The pool owns all id allocation here, so a hand-pinned bd_id would
76 // collide with a runtime pop. Allocation is all-pool or all-static (for a
77 // straight-line sequence), never mixed.
78 WalkResult pinned = cfg.walk([&](AIE::DMABDOp bd) {
79 if (bd.getBdId().has_value()) {
80 bd.emitOpError(
81 "pins a buffer descriptor ID inside a runtime-bound "
82 "sequence that draws IDs from the dynamic pool; a pinned "
83 "ID would collide with a runtime-allocated one. Allocate "
84 "every BD in this sequence from the pool (drop the "
85 "bd_id), or make the sequence straight-line so the static "
86 "allocator assigns all IDs.");
87 return WalkResult::interrupt();
88 }
89 return WalkResult::advance();
90 });
91 if (pinned.wasInterrupted())
92 return failure();
93
94 AIE::TileOp tile = cfg.getTileOp();
95 OpBuilder b(cfg);
96 Value bdId = DMABdPoolPopOp::create(b, cfg.getLoc(), b.getI32Type(),
97 tile.getCol(), tile.getRow())
98 .getBdId();
99 cfg.getBdIdValMutable().assign(bdId);
100 pairedId[cfg.getResult()] = bdId;
101 tileForTask[cfg.getResult()] = {tile.getCol(), tile.getRow()};
102 originConfigure[cfg.getResult()] = cfg;
103 return success();
104 }
105
106 // Add a parallel i32 iter_arg for every task-carrying iter_arg of `forOp`, so
107 // the popped id flows in lockstep with the task Index. Returns the new loop.
108 // Updates `pairedId` for the new region iter_args and results.
109 scf::ForOp carryIdsThroughLoop(scf::ForOp forOp) {
110 SmallVector<unsigned> carried; // iter_arg positions carrying a paired task
111 SmallVector<Value> newInits;
112 for (unsigned k = 0; k < forOp.getInitArgs().size(); ++k) {
113 Value init = forOp.getInitArgs()[k];
114 auto it = pairedId.find(init);
115 if (it == pairedId.end())
116 continue;
117 carried.push_back(k);
118 newInits.push_back(it->second);
119 }
120 if (carried.empty())
121 return forOp;
122
123 unsigned nOrig = forOp.getInitArgs().size();
124 IRRewriter rewriter(forOp);
125 auto newYields = [&](OpBuilder &b, Location loc,
126 ArrayRef<BlockArgument> newBBArgs) {
127 // The value yielded for each new id iter_arg is the paired id of the task
128 // that carried position yields. For a rotating ping-pong the body yields
129 // a fresh configure result, whose id is already in the map.
130 auto *yield = forOp.getBody()->getTerminator();
131 SmallVector<Value> yielded;
132 for (unsigned k : carried) {
133 Value bodyTask = yield->getOperand(k);
134 Value id = pairedId.lookup(bodyTask);
135 assert(id && "carried task yields an unpaired id");
136 yielded.push_back(id);
137 }
138 return yielded;
139 };
140
141 FailureOr<LoopLikeOpInterface> newLoop = forOp.replaceWithAdditionalYields(
142 rewriter, newInits,
143 /*replaceInitOperandUsesInLoop=*/false, newYields);
144 assert(succeeded(newLoop) && "scf.for additional-yields rewrite failed");
145 auto newFor = cast<scf::ForOp>(newLoop->getOperation());
146
147 // id iter_args are appended after the original nOrig; pair each carried
148 // task iter_arg/result with its new id, propagating tile and origin from
149 // the init.
150 for (auto [i, k] : llvm::enumerate(carried)) {
151 Value init = forOp.getInitArgs()[k];
152 auto tile = tileForTask.lookup(init);
153 DMAConfigureTaskOp origin = originConfigure.lookup(init);
154 pairedId[newFor.getRegionIterArg(k)] = newFor.getRegionIterArg(nOrig + i);
155 tileForTask[newFor.getRegionIterArg(k)] = tile;
156 originConfigure[newFor.getRegionIterArg(k)] = origin;
157 pairedId[newFor.getResult(k)] = newFor.getResult(nOrig + i);
158 tileForTask[newFor.getResult(k)] = tile;
159 originConfigure[newFor.getResult(k)] = origin;
160 }
161 return newFor;
162 }
163
164 LogicalResult lowerRelease(Value task, Operation *op,
165 llvm::DenseSet<Value> &released) {
166 Value id = pairedId.lookup(task);
167 if (!id)
168 return op->emitOpError(
169 "does not resolve to a task allocated from the runtime pool; cannot "
170 "return its buffer descriptor ID");
171 // await returns the id; a later free of the same task is the "wait then
172 // release" idiom, not a double free -- push once.
173 if (!released.insert(task).second)
174 return success();
175 auto tile = tileForTask.lookup(task);
176 OpBuilder b(op);
177 DMABdPoolPushOp::create(b, op->getLoc(), tile.first, tile.second, id);
178 return success();
179 }
180
181 void runOnOperation() override {
182 AIE::DeviceOp device = getOperation();
183
184 WalkResult wr = device.walk([&](AIE::RuntimeSequenceOp seq) -> WalkResult {
185 bool hasRuntimeCF = false;
186 seq.walk([&](Operation *op) {
187 if (isa<scf::ForOp, scf::IfOp, scf::WhileOp>(op))
188 hasRuntimeCF = true;
189 });
190 if (!hasRuntimeCF)
191 return WalkResult::advance();
192
193 pairedId.clear();
194 tileForTask.clear();
195 originConfigure.clear();
196
197 // 1. Every configure pops an id (program order: a pop dominates every use
198 // of its id).
199 WalkResult r = seq.walk([&](DMAConfigureTaskOp cfg) -> WalkResult {
200 return failed(lowerConfigure(cfg)) ? WalkResult::interrupt()
201 : WalkResult::advance();
202 });
203 if (r.wasInterrupted())
204 return WalkResult::interrupt();
205
206 // 2. Carry the ids through loops so a push sees the id at its own scope.
207 // Innermost-first: a body's yielded ids must be paired before its
208 // enclosing loop consumes them.
209 SmallVector<scf::ForOp> loops;
210 seq.walk([&](scf::ForOp f) { loops.push_back(f); });
211 for (scf::ForOp f : llvm::reverse(loops))
212 carryIdsThroughLoop(f);
213
214 // 3. free/await -> push (await keeps its sync, adds a push). An await on
215 // a
216 // loop result lacks a defining configure, so redirect it to the
217 // loop-invariant origin configure.
218 llvm::DenseSet<Value> released;
219 SmallVector<Operation *> toErase;
220 r = seq.walk([&](Operation *op) -> WalkResult {
221 if (auto freeOp = dyn_cast<DMAFreeTaskOp>(op)) {
222 if (failed(lowerRelease(freeOp.getTask(), freeOp, released)))
223 return WalkResult::interrupt();
224 toErase.push_back(freeOp);
225 } else if (auto await = dyn_cast<DMAAwaitTaskOp>(op)) {
226 if (failed(lowerRelease(await.getTask(), await, released)))
227 return WalkResult::interrupt();
228 if (!await.getTask().getDefiningOp<DMAConfigureTaskOp>()) {
229 DMAConfigureTaskOp origin = originConfigure.lookup(await.getTask());
230 if (!origin)
231 return await.emitOpError(
232 "awaits a task whose configure cannot be resolved for "
233 "the dynamic pool path"),
234 WalkResult::interrupt();
235 await.getTaskMutable().assign(origin.getResult());
236 }
237 }
238 return WalkResult::advance();
239 });
240 if (r.wasInterrupted())
241 return WalkResult::interrupt();
242 for (Operation *op : toErase)
243 op->erase();
244 return WalkResult::advance();
245 });
246 if (wr.wasInterrupted())
247 return signalPassFailure();
248 }
249};
