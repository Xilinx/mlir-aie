7//
8// The dynamic counterpart to AIEAssignRuntimeSequenceBDIDs: where that pass
9// rejects a runtime-bound scf.for, this keeps the loop rolled and draws bd_ids
10// from a per-tile runtime free-list pool. Each configure gets a dma_bd_pool_pop
11// (its SSA id feeding bd_id_val); each free gets a dma_bd_pool_push. An await
12// is only a completion sync (npu_sync), never a push.
13//
14// A popped id must be reachable at its push. Since a task value can cross a
15// loop back edge and exit as a result, the id is carried in lockstep: every
16// task iter_arg gets a parallel i32 id iter_arg, and a push looks up the paired
17// id.
18//
19// v1 is single-BD only; multi-BD chains would need runtime next_bd (follow-up).
20//
21//===----------------------------------------------------------------------===//

44struct AIELowerDynamicBDPoolPass
45 : xilinx::AIEX::impl::AIELowerDynamicBDPoolBase<AIELowerDynamicBDPoolPass> {

47 // Maps a task value (the Index result of a configure, or any Index the carry
48 // propagates it into) to the i32 pool id available at that same program
49 // point.
50 llvm::DenseMap<Value, Value> pairedId;
51 // The (col,row) of the tile each task's id belongs to, so a push names the
52 // right pool even when the id is a loop result rather than a direct pop.
53 llvm::DenseMap<Value, std::pair<int, int>> tileForTask;
54 // A representative (loop-invariant) configure per task, so a post-loop await
55 // on a loop result -- which has no defining configure -- can be redirected to
56 // one with the right (tile,dir,channel).
57 llvm::DenseMap<Value, DMAConfigureTaskOp> originConfigure;
58
59 // Every SSA value that carries a DMA task: configure results plus everything
60 // the task flows into (for iter_args/results, if results). Computed by
61 // forward closure from configures, so it is independent of the order in which
62 // control flow is later rewritten.
63 llvm::DenseSet<Value> taskValues;
64
65 // A grown scf.for/scf.if whose appended i32 yield operands are still
66 // placeholders. carried[i] is the original task position that the appended id
67 // at nOrig+i shadows; the yields are wired once every position is paired.
68 struct Fixup {
69 Operation *op;
70 SmallVector<unsigned> carried;
71 unsigned nOrig;
72 SmallVector<Operation *> placeholders; // dead consts to erase (scf.if only)
73 };
74 SmallVector<Fixup> fixups;

76 // Count the BD ops in a configure's body (across all its blocks).
77 static unsigned countBds(DMAConfigureTaskOp cfg) {
78 unsigned n = 0;
79 cfg.walk([&](AIE::DMABDOp) { ++n; });
80 return n;
81 }

83 void markTask(Value v, SmallVectorImpl<Value> &worklist) {
84 if (taskValues.insert(v).second)
85 worklist.push_back(v);
86 }
87
88 // Forward-propagate task-ness from every configure result through scf.for
89 // (init -> iter_arg, yield -> result) and scf.if (yield -> result). A value
90 // that is never derived from a configure (e.g. an unrelated index yielded on
91 // one branch) stays out of the set, which is what lets the divergent-yield
92 // check below reject it.
93 void computeTaskValues(AIE::RuntimeSequenceOp seq) {
94 taskValues.clear();
95 SmallVector<Value> worklist;
96 seq.walk([&](DMAConfigureTaskOp c) { markTask(c.getResult(), worklist); });
97 while (!worklist.empty()) {
98 Value v = worklist.pop_back_val();
99 for (OpOperand &use : v.getUses()) {
100 Operation *user = use.getOwner();
101 if (auto f = dyn_cast<scf::ForOp>(user)) {
102 for (unsigned k = 0; k < f.getInitArgs().size(); ++k)
103 if (f.getInitArgs()[k] == v)
104 markTask(f.getRegionIterArg(k), worklist);
105 } else if (isa<scf::YieldOp>(user)) {
106 unsigned k = use.getOperandNumber();
107 Operation *parent = user->getParentOp();
108 if (auto pf = dyn_cast<scf::ForOp>(parent))
109 markTask(pf.getResult(k), worklist);
110 else if (auto pi = dyn_cast<scf::IfOp>(parent))
111 markTask(pi.getResult(k), worklist);
112 }
113 }
114 }
115 }

117 LogicalResult lowerConfigure(DMAConfigureTaskOp cfg) {
118 if (cfg.getBdIdVal())
119 return success(); // already lowered
120
121 if (countBds(cfg) != 1)
122 return cfg.emitOpError(
123 "dynamic BD pool lowering supports single-BD tasks only; a multi-BD "
124 "chain under a runtime-bound loop needs runtime next_bd chaining "
125 "(not yet implemented)");
126
127 // The pool owns all id allocation here, so a hand-pinned bd_id would
128 // collide with a runtime pop. Allocation is all-pool or all-static (for a
129 // straight-line sequence), never mixed.
130 WalkResult pinned = cfg.walk([&](AIE::DMABDOp bd) {
131 if (bd.getBdId().has_value()) {
132 bd.emitOpError(
133 "pins a buffer descriptor ID inside a runtime-bound "
134 "sequence that draws IDs from the dynamic pool; a pinned "
135 "ID would collide with a runtime-allocated one. Allocate "
136 "every BD in this sequence from the pool (drop the "
137 "bd_id), or make the sequence straight-line so the static "
138 "allocator assigns all IDs.");
139 return WalkResult::interrupt();
140 }
141 return WalkResult::advance();
142 });
143 if (pinned.wasInterrupted())
144 return failure();
145
146 AIE::TileOp tile = cfg.getTileOp();
147 OpBuilder b(cfg);
148 Value bdId = DMABdPoolPopOp::create(b, cfg.getLoc(), b.getI32Type(),
149 tile.getCol(), tile.getRow())
150 .getBdId();
151 cfg.getBdIdValMutable().assign(bdId);
152 pairedId[cfg.getResult()] = bdId;
153 return success();
154 }
155
156 // Grow an scf.for with a parallel i32 iter_arg for every task-carrying
157 // iter_arg, so the popped id can flow in lockstep with the task Index. The
158 // yields are left as placeholders (the new id block arg itself) and wired to
159 // the real per-iteration id later, once every position is paired -- a body
160 // may yield an id defined by a not-yet-grown inner op. Pairs the new
161 // iter_args/results in `pairedId` and records a Fixup.
162 scf::ForOp carryIdsThroughLoop(scf::ForOp forOp) {
163 SmallVector<unsigned> carried; // iter_arg positions carrying a task
164 SmallVector<Value> newInits;
165 for (unsigned k = 0; k < forOp.getInitArgs().size(); ++k) {
166 Value init = forOp.getInitArgs()[k];
167 if (!taskValues.contains(init))
168 continue;
169 Value id = pairedId.lookup(init);
170 assert(id && "for init carries a task but has no paired id");
171 carried.push_back(k);
172 newInits.push_back(id);
173 }
174 if (carried.empty())
175 return forOp;
176
177 unsigned nOrig = forOp.getInitArgs().size();
178 IRRewriter rewriter(forOp);
179 auto newYields = [&](OpBuilder &b, Location loc,
180 ArrayRef<BlockArgument> newBBArgs) {
181 // Placeholder: yield each new id iter_arg straight back. Same type,
182 // dominates the yield, and is a no-op for an identity (task-reuse) carry.
183 // wireFixups overwrites it with the real per-iteration id.
184 return SmallVector<Value>(newBBArgs.begin(), newBBArgs.end());
185 };
186
187 FailureOr<LoopLikeOpInterface> newLoop = forOp.replaceWithAdditionalYields(
188 rewriter, newInits,
189 /*replaceInitOperandUsesInLoop=*/false, newYields);
190 assert(succeeded(newLoop) && "scf.for additional-yields rewrite failed");
191 auto newFor = cast<scf::ForOp>(newLoop->getOperation());
192
193 for (auto [i, k] : llvm::enumerate(carried)) {
194 // The rewrite gave the loop new block args/results; migrate task-ness
195 // onto them so a later-grown consumer of this loop's result still sees a
196 // task.
197 taskValues.insert(newFor.getRegionIterArg(k));
198 taskValues.insert(newFor.getResult(k));
199 pairedId[newFor.getRegionIterArg(k)] = newFor.getRegionIterArg(nOrig + i);
200 pairedId[newFor.getResult(k)] = newFor.getResult(nOrig + i);
201 }
202 fixups.push_back({newFor, carried, nOrig, {}});
203 return newFor;
204 }
205
206 // Grow an scf.if with a parallel i32 result per task it yields out (its
207 // branch-local id), so a task freed after the if finds its id. Yields start
208 // as placeholders, wired later. A result carries a task iff BOTH branches
209 // yield one; a divergent yield (only one branch) is rejected -- no id to push
210 // on the other path.
211 LogicalResult carryIdsThroughIf(scf::IfOp ifOp) {
212 if (ifOp.getNumResults() == 0 || ifOp.elseBlock() == nullptr)
213 return success();
214
215 auto thenYield = cast<scf::YieldOp>(ifOp.thenBlock()->getTerminator());
216 auto elseYield = cast<scf::YieldOp>(ifOp.elseBlock()->getTerminator());
217
218 SmallVector<unsigned> carried; // result positions carrying a task
219 for (unsigned k = 0; k < ifOp.getNumResults(); ++k) {
220 bool thenT = taskValues.contains(thenYield.getOperand(k));
221 bool elseT = taskValues.contains(elseYield.getOperand(k));
222 if (thenT && elseT)
223 carried.push_back(k);
224 else if (thenT || elseT)
225 return ifOp.emitOpError(
226 "yields a pool-allocated task on only one branch of an scf.if; "
227 "both "
228 "branches must yield the task (and its buffer descriptor id) so it "
229 "can be freed after the if");
230 }
231 if (carried.empty())
232 return success();
233
234 unsigned nOrig = ifOp.getNumResults();
235 OpBuilder b(ifOp);
236 SmallVector<Type> resultTypes(ifOp.getResultTypes());
237 for (unsigned k : carried)
238 (void)k, resultTypes.push_back(b.getI32Type());
239
240 auto newIf =
241 scf::IfOp::create(b, ifOp.getLoc(), resultTypes, ifOp.getCondition(),
242 /*withElseRegion=*/true);
243 newIf.getThenRegion().takeBody(ifOp.getThenRegion());
244 newIf.getElseRegion().takeBody(ifOp.getElseRegion());
245
246 // Append a placeholder i32 to each branch's yield; wireFixups replaces it
247 // with the branch's real id (which may be defined by a not-yet-grown inner
248 // op). The placeholder is a dead constant erased during wiring.
249 SmallVector<Operation *> placeholders;
250 auto appendPlaceholders = [&](scf::YieldOp y) {
251 OpBuilder yb(y);
252 SmallVector<Value> ids;
253 for (unsigned k : carried) {
254 (void)k;
255 auto c =
256 arith::ConstantOp::create(yb, y.getLoc(), yb.getI32IntegerAttr(0));
257 placeholders.push_back(c);
258 ids.push_back(c);
259 }
260 y->insertOperands(y->getNumOperands(), ids);
261 };
262 appendPlaceholders(cast<scf::YieldOp>(newIf.thenBlock()->getTerminator()));
263 appendPlaceholders(cast<scf::YieldOp>(newIf.elseBlock()->getTerminator()));
264
265 for (auto [i, k] : llvm::enumerate(carried)) {
266 // Migrate task-ness onto the rebuilt result so a later-grown consumer of
267 // this if's result still sees a task.
268 taskValues.insert(newIf.getResult(k));
269 pairedId[newIf.getResult(k)] = newIf.getResult(nOrig + i);
270 }
271 for (unsigned k = 0; k < nOrig; ++k)
272 ifOp.getResult(k).replaceAllUsesWith(newIf.getResult(k));
273 ifOp.erase();
274 fixups.push_back({newIf, carried, nOrig, placeholders});
275 return success();
276 }
277
278 // Sweep 2: with every position paired, set each grown op's appended yield
279 // operand(s) to the real id yielded at that position. Runs after all growth,
280 // so an id defined by an inner op (grown later than its enclosing op) is
281 // resolvable, and an identity carry resolves to its own iter_arg's id.
282 void wireFixups() {
283 for (Fixup &fx : fixups) {
284 if (auto forOp = dyn_cast<scf::ForOp>(fx.op)) {
285 auto yield = cast<scf::YieldOp>(forOp.getBody()->getTerminator());
286 for (auto [i, k] : llvm::enumerate(fx.carried)) {
287 Value id = pairedId.lookup(yield.getOperand(k));
288 assert(id && "carried task yields an unpaired id");
289 yield.setOperand(fx.nOrig + i, id);
290 }
291 } else {
292 auto ifOp = cast<scf::IfOp>(fx.op);
293 for (Block *blk : {ifOp.thenBlock(), ifOp.elseBlock()}) {
294 auto yield = cast<scf::YieldOp>(blk->getTerminator());
295 for (auto [i, k] : llvm::enumerate(fx.carried)) {
296 Value id = pairedId.lookup(yield.getOperand(k));
297 assert(id && "carried task yields an unpaired id");
298 yield.setOperand(fx.nOrig + i, id);
299 }
300 }
301 }
302 }
303 for (Fixup &fx : fixups)
304 for (Operation *p : fx.placeholders)
305 p->erase();
306 }
307
308 // Sweep 3: over the final IR, give every task value its tile (for the push)
309 // and a representative origin configure (for an await that names a loop
310 // result with no defining configure -- a loop's pre-loop init, which
311 // dominates a post-loop await). Order-independent forward closure:
312 // - scf.for: the init's tile/origin flow to the iter_arg AND the result, so
313 // a result takes the loop-invariant init configure (which dominates a
314 // post-loop await), not the per-iteration body configure.
315 // - scf.if: the then-branch yield's tile/origin flow to the result.
316 // A push only reads the tile, so both branches must agree; a genuinely
317 // cross-tile scf.if is rejected.
318 LogicalResult computeMetadata(AIE::RuntimeSequenceOp seq) {
319 tileForTask.clear();
320 originConfigure.clear();
321 SmallVector<Value> worklist;
322 auto setMeta = [&](Value v, std::pair<int, int> tile,
323 DMAConfigureTaskOp origin) {
324 if (tileForTask.count(v))
325 return;
326 tileForTask[v] = tile;
327 originConfigure[v] = origin;
328 worklist.push_back(v);
329 };
330 seq.walk([&](DMAConfigureTaskOp c) {
331 AIE::TileOp t = c.getTileOp();
332 setMeta(c.getResult(), {t.getCol(), t.getRow()}, c);
333 });
334 while (!worklist.empty()) {
335 Value v = worklist.pop_back_val();
336 auto tile = tileForTask[v];
337 DMAConfigureTaskOp origin = originConfigure[v];
338 for (OpOperand &use : v.getUses()) {
339 Operation *user = use.getOwner();
340 if (auto f = dyn_cast<scf::ForOp>(user)) {
341 for (unsigned k = 0; k < f.getInitArgs().size(); ++k)
342 if (f.getInitArgs()[k] == v) {
343 setMeta(f.getRegionIterArg(k), tile, origin);
344 setMeta(f.getResult(k), tile, origin);
345 }
346 } else if (isa<scf::YieldOp>(user)) {
347 auto ifOp = dyn_cast<scf::IfOp>(user->getParentOp());
348 if (ifOp && user->getBlock() == ifOp.thenBlock())
349 setMeta(ifOp.getResult(use.getOperandNumber()), tile, origin);
350 }
351 }
352 }
353 // A task carried out of an scf.if and freed after it resolves to one BD id
354 // and one push, so both branches must agree on the full physical channel
355 // (tile,dir,channel) -- otherwise the single push would be ambiguous.
356 WalkResult wr = seq.walk([&](scf::IfOp ifOp) -> WalkResult {
357 if (ifOp.getNumResults() == 0 || ifOp.elseBlock() == nullptr)
358 return WalkResult::advance();
359 auto thenYield = cast<scf::YieldOp>(ifOp.thenBlock()->getTerminator());
360 auto elseYield = cast<scf::YieldOp>(ifOp.elseBlock()->getTerminator());
361 for (unsigned k = 0; k < ifOp.getNumResults(); ++k) {
362 DMAConfigureTaskOp to = originConfigure.lookup(thenYield.getOperand(k));
363 DMAConfigureTaskOp eo = originConfigure.lookup(elseYield.getOperand(k));
364 if (to && eo && syncSig(to) != syncSig(eo)) {
365 ifOp.emitOpError("yields tasks on different physical channels "
366 "(tile/direction/channel) from its two branches at "
367 "the same result; a pooled buffer descriptor id and "
368 "its completion sync belong to one channel");
369 return WalkResult::interrupt();
370 }
371 }
372 return WalkResult::advance();
373 });
374 return wr.wasInterrupted() ? failure() : success();
375 }

377 // The physical channel a configure targets: (col, row, direction, channel).
378 static std::tuple<int, int, int, int> syncSig(DMAConfigureTaskOp cfg) {
379 AIE::TileOp t = cfg.getTileOp();
380 return {t.getCol(), t.getRow(), (int)cfg.getDirection(), cfg.getChannel()};
381 }

383 LogicalResult lowerRelease(Value task, Operation *op) {
384 Value id = pairedId.lookup(task);
385 if (!id)
386 return op->emitOpError(
387 "does not resolve to a task allocated from the runtime pool; cannot "
388 "return its buffer descriptor ID");
389 auto tile = tileForTask.lookup(task);
390 OpBuilder b(op);
391 DMABdPoolPushOp::create(b, op->getLoc(), tile.first, tile.second, id);
392 return success();
393 }

395 void runOnOperation() override {
396 AIE::DeviceOp device = getOperation();
397
398 WalkResult wr = device.walk([&](AIE::RuntimeSequenceOp seq) -> WalkResult {
399 bool hasRuntimeCF = false;
400 seq.walk([&](Operation *op) {
401 if (isa<scf::ForOp, scf::IfOp, scf::WhileOp>(op))
402 hasRuntimeCF = true;
403 });
404 if (!hasRuntimeCF)
405 return WalkResult::advance();
406
407 // Idempotency: a sequence already carrying pool pops has been lowered by
408 // a prior run of this pass. Its awaits may have dropped their task
409 // operand (sync_* attrs) and its scf ops already carry i32 ids, so
410 // re-running the carry logic would misfire. Skip it.
411 bool alreadyLowered = false;
412 seq.walk([&](DMABdPoolPopOp) { alreadyLowered = true; });
413 if (alreadyLowered)
414 return WalkResult::advance();
415
416 // scf.while has no bounded trip count, so a BD id popped in its body has
417 // no place to be pushed back deterministically; only scf.for and scf.if
418 // are supported.
419 WalkResult whileCheck = seq.walk([&](scf::WhileOp w) {
420 w.emitOpError("scf.while in a runtime sequence is not supported by the "
421 "dynamic BD pool path; only scf.for and scf.if are "
422 "supported");
423 return WalkResult::interrupt();
424 });
425 if (whileCheck.wasInterrupted())
426 return WalkResult::interrupt();
427
428 pairedId.clear();
429 tileForTask.clear();
430 originConfigure.clear();
431 fixups.clear();
432
433 // 1. Every configure pops an id (program order: a pop dominates every use
434 // of its id).
435 WalkResult r = seq.walk([&](DMAConfigureTaskOp cfg) -> WalkResult {
436 return failed(lowerConfigure(cfg)) ? WalkResult::interrupt()
437 : WalkResult::advance();
438 });
439 if (r.wasInterrupted())
440 return WalkResult::interrupt();
441
442 // 2. Thread the id alongside the task everywhere the task flows. The task
443 // Index already threads (it is valid IR); shadowing every position it
444 // occupies with a parallel i32 makes the id in-scope at every push.
445 // a. Task closure: which SSA values carry a task (order-independent).
446 // b. Grow structure outermost-first (pre-order): an enclosing op's new
447 // iter_arg/result exists before an inner op yields through it. A
448 // divergent scf.if is rejected here.
449 // c. Wire the placeholder yields now that every position is paired --
450 // handles both siblings (producer grown before consumer) and
451 // parent-defines/child-consumes (inner id resolvable after growth).
452 computeTaskValues(seq);
453 SmallVector<Operation *> cf;
454 seq.walk<WalkOrder::PreOrder>([&](Operation *o) {
455 if (isa<scf::ForOp, scf::IfOp>(o))
456 cf.push_back(o);
457 });
458 for (Operation *o : cf) {
459 if (auto f = dyn_cast<scf::ForOp>(o)) {
460 carryIdsThroughLoop(f);
461 } else if (failed(carryIdsThroughIf(cast<scf::IfOp>(o)))) {
462 return WalkResult::interrupt();
463 }
464 }
465 wireFixups();
466
467 // 3. Metadata over the final IR: each task's tile (for the push) and a
468 // dominating origin configure (for an await on a loop/if result).
469 if (failed(computeMetadata(seq)))
470 return WalkResult::interrupt();
471
472 // 4. Return ids to the pool. A free is the ONLY push: it returns the id
473 // to the free-list. An await is a TCT sync (npu_sync), never a release
474 // -- it emits no push. In the rolled pool one physical id is live
475 // across iterations, so the static path's "await = release" (safe only
476 // because unrolling gives each await a distinct id) would push the
477 // same live id once per iteration and corrupt the free-list. Leaving a
478 // task unfreed is valid: the pool is stack-local and re-initialized
479 // every invocation, so an unfreed id merely holds a slot for the rest
480 // of this one run (the await-only idiom, live BDs <= pool size).
481 //
482 // The await keeps its natural SSA operand -- the task value it names,
483 // whether that is a direct configure, a loop result, or an scf.if
484 // result (the phi of the task in flight, whichever branch ran). The
485 // npu_sync lowering (AIEDMATasksToNPU) walks that value to the
486 // physical channel; both branches of an scf.if were verified to agree
487 // on it (computeMetadata). This preserves the sync's data dependence
488 // on the task rather than breaking it.
489 SmallVector<Operation *> toErase;
490 r = seq.walk([&](Operation *op) -> WalkResult {
491 if (auto freeOp = dyn_cast<DMAFreeTaskOp>(op)) {
492 if (failed(lowerRelease(freeOp.getTask(), freeOp)))
493 return WalkResult::interrupt();
494 toErase.push_back(freeOp);
495 }
496 return WalkResult::advance();
497 });
498 if (r.wasInterrupted())
499 return WalkResult::interrupt();
500 for (Operation *op : toErase)
501 op->erase();
502 return WalkResult::advance();
503 });
504 if (wr.wasInterrupted())
505 return signalPassFailure();
506 }
