8// This pass expands `npu.load_pdi` operations that reference a device. There
9// are two output modes, controlled by the `ctrl-pkt` option:
10//
11// 1. Default (ctrl-pkt=false): replaces each `load_pdi @device` with
12// a. an empty device PDI load (`load_pdi @empty_N`), which causes the
13// firmware to reset the device, and
14// b. explicit `aiex.npu.write32`/`aiex.npu.blockwrite` configuration ops.
15// 2. With ctrl-pkt=true: replaces each `load_pdi @device` with
16// a. a `load_pdi @ctrl_pkt_overlay`, which configures the NPU to stream
17// further configuration as control packets, and
18// b. a sequence of `aiex.npu.control_packet` ops carrying the device's
19// configuration.
20//
21//===----------------------------------------------------------------------===//

48// Name of the overlay device loaded ahead of streaming configuration as
49// control packets. The device is expected to exist in the module (e.g.,
50// emitted by the `aie-generate-column-control-overlay` pass).
51static constexpr llvm::StringLiteral kCtrlPktOverlayName = "ctrl_pkt_overlay";
52// Name of an alternating copy of the overlay device, used to avoid PDI
53// address caching when the same overlay would otherwise be loaded twice in a
54// row (the firmware would treat the second load as a no-op).
55static constexpr llvm::StringLiteral kCtrlPktOverlayCopyName =
56 "ctrl_pkt_overlay_copy";
57
58// Look up `kCtrlPktOverlayName` and return a clone of it named
59// `kCtrlPktOverlayCopyName`, creating it once if needed. Returns nullptr on
60// error.
61static AIE::DeviceOp getOrCreateCtrlPktOverlayCopy(ModuleOp moduleOp,
62 Operation *errLocOp) {
63 if (auto existing =
64 moduleOp.lookupSymbol<AIE::DeviceOp>(kCtrlPktOverlayCopyName))
65 return existing;
66
67 auto orig = moduleOp.lookupSymbol<AIE::DeviceOp>(kCtrlPktOverlayName);
68 if (!orig) {
69 errLocOp->emitError("ctrl-pkt mode requires a `@")
70 << kCtrlPktOverlayName << "` device in the module";
71 return nullptr;
72 }
73
74 OpBuilder builder(moduleOp.getContext());
75 builder.setInsertionPointAfter(orig);
76 auto *cloned = builder.clone(*orig.getOperation());
77 auto clonedDev = cast<AIE::DeviceOp>(cloned);
78 clonedDev.setSymName(kCtrlPktOverlayCopyName);
79 return clonedDev;
80}

82// Helper to transform a single load_pdi operation
83static LogicalResult transformLoadPdi(NpuLoadPdiOp loadPdiOp, ModuleOp moduleOp,
84 unsigned index,
85 AIEX::ExpandMode defaultMode) {
86 static unsigned long i = 0;
87 OpBuilder builder(loadPdiOp);
88
89 // Only process load_pdi ops that reference a device
90 auto deviceRefAttr = loadPdiOp.getDeviceRefAttr();
91 if (!deviceRefAttr) {
92 return success();
93 }
94
95 // Per-op annotation takes precedence; fall back to pass default
96 AIEX::ExpandMode mode = loadPdiOp.getExpandMode().value_or(defaultMode);
97 if (mode == AIEX::ExpandMode::none)
98 return success();
99 bool ctrlPkt = (mode == AIEX::ExpandMode::ctrlpkt);
100
101 auto referencedDevice = moduleOp.lookupSymbol<AIE::DeviceOp>(deviceRefAttr);
102 if (!referencedDevice) {
103 loadPdiOp.emitError("Referenced symbol '")
104 << deviceRefAttr.getValue() << "' is not a device";
105 return failure();
106 }
107
108 FlatSymbolRefAttr preloadRef;
109 if (ctrlPkt) {
110 // Overlay device PDI
111 // Alternate between the original overlay and a clone of it on every
112 // other load. Loading the same PDI twice in a row gets cached by the
113 // firmware (the second load becomes a no-op), so we need two distinct
114 // PDI addresses that carry the same overlay configuration.
115 StringRef overlayName =
116 (index % 2 == 0) ? kCtrlPktOverlayName : kCtrlPktOverlayCopyName;
117 if (index % 2 != 0) {
118 AIE::DeviceOp copy =
119 getOrCreateCtrlPktOverlayCopy(moduleOp, loadPdiOp.getOperation());
120 if (!copy)
121 return failure();
122 } else if (!moduleOp.lookupSymbol<AIE::DeviceOp>(kCtrlPktOverlayName)) {
123 loadPdiOp.emitError("ctrl-pkt mode requires a `@")
124 << kCtrlPktOverlayName << "` device in the module";
125 return failure();
126 }
127 preloadRef = FlatSymbolRefAttr::get(builder.getContext(), overlayName);
128 } else {
129 // Empty device PDI (triggers firmware reset)
130 OpBuilder::InsertionGuard guard(builder);
131 builder.setInsertionPointToStart(moduleOp.getBody());
132
133 std::string emptyName = "empty_" + std::to_string(index % 2);
134 AIE::DeviceOp emptyDevice = moduleOp.lookupSymbol<AIE::DeviceOp>(emptyName);
135 if (!emptyDevice) {
136 auto deviceType = referencedDevice.getDevice();
137 auto loc = builder.getUnknownLoc();
138 emptyDevice = AIE::DeviceOp::create(builder, loc, deviceType,
139 builder.getStringAttr(emptyName));
140 emptyDevice.getRegion().emplaceBlock();
141 Block *deviceBlock = &emptyDevice.getRegion().front();
142 builder.setInsertionPointToEnd(deviceBlock);
143 AIE::EndOp::create(builder, loc);
144 }
145 preloadRef = FlatSymbolRefAttr::get(emptyDevice.getSymNameAttr());
146 }
147
148 builder.setInsertionPoint(loadPdiOp);
149
150 // Emit the preload load_pdi (either empty-device reset or ctrl_pkt_overlay).
151 if (ctrlPkt) {
152 NpuLoadPdiOp::create(builder, loadPdiOp.getLoc(), preloadRef,
153 /*id=*/nullptr, /*size=*/nullptr,
154 /*address=*/nullptr,
155 /*expand_mode=*/
156 AIEX::ExpandModeAttr::get(builder.getContext(),
157 AIEX::ExpandMode::none));
158 } else {
159 NpuLoadPdiOp::create(builder, loadPdiOp.getLoc(), preloadRef,
160 loadPdiOp.getIdAttr(), loadPdiOp.getSizeAttr(),
161 loadPdiOp.getAddressAttr(),
162 /*expand_mode=*/
163 AIEX::ExpandModeAttr::get(builder.getContext(),
164 AIEX::ExpandMode::none));
165 }
166
167 // Step 2: generate and insert configuration ops.
168 auto outputType = ctrlPkt ? AIEToConfigurationOutputType::ControlPacket
169 : AIEToConfigurationOutputType::Transaction;
170 std::string prefix = ctrlPkt ? ("loadpdi_ctrlpkt_" + std::to_string(i) + "_")
171 : ("loadpdi_" + std::to_string(i));
172 if (failed(xilinx::AIE::generateAndInsertConfigOps(
173 builder, referencedDevice, /*clElfDir=*/"", outputType, prefix,
174 /*skipCtrlPktOverlay=*/ctrlPkt))) {
175 loadPdiOp.emitError("Failed to generate configuration operations");
176 return failure();
177 }
178
179 // Erase the original load_pdi operation
180 loadPdiOp.erase();
181
182 i++;
183
184 return success();
185}

187struct AIEExpandLoadPdiPass
188 : public xilinx::AIEX::impl::AIEExpandLoadPdiBase<AIEExpandLoadPdiPass> {
189 using AIEExpandLoadPdiBase::AIEExpandLoadPdiBase;
190
191 void getDependentDialects(DialectRegistry &registry) const override {
192 registry
193 .insert<memref::MemRefDialect, AIE::AIEDialect, AIEX::AIEXDialect>();
194 }
195
196 void runOnOperation() override {
197 auto module = getOperation();
198
199 // Collect all load_pdi operations in program order;
200 // need to collect once, then transform all collected ops;
201 // since the transform inserts a new preload load_pdi, we can't transform
202 // as we walk or it'd infinitely recurse.
203 SmallVector<NpuLoadPdiOp> loadPdiOps;
204
205 module.walk(
206 [&](NpuLoadPdiOp loadPdiOp) { loadPdiOps.push_back(loadPdiOp); });
207
208 // Map the legacy bool option to the new ExpandMode enum.
209 AIEX::ExpandMode defaultMode =
210 clCtrlPkt ? AIEX::ExpandMode::ctrlpkt : AIEX::ExpandMode::write32;
211
212 // Transform load_pdi ops
213 unsigned idx = 0;
214 for (auto loadPdiOp : loadPdiOps) {
215 if (failed(transformLoadPdi(loadPdiOp, module, idx, defaultMode))) {
216 signalPassFailure();
217 return;
218 }
219 idx++;
220 }
221 }
222};
