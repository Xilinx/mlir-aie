25// Transaction opcodes for the firmware TXN format the compiler currently
26// targets.
27//
28// These DO NOT match the third_party/aie-rt xaie_txn.h enum, which is an older
29// layout (CONFIG_SHIMDMA_BD=5, no NOOP/PREEMPT/LOADPDI block). They match the
30// newer firmware opcodes the compiler emits (e.g. CREATE_SCRATCHPAD=0x0A,
31// UPDATE_REG=0x0C).
32//
33// DRIFT WARNING: when the firmware TXN format changes, update these values
34// here. Do not "correct" them against the submodule's xaie_txn.h — it tracks a
35// different (older) numbering and reconciling to it would break the emitted
36// transactions.
37enum TxnOpcode : uint32_t {
38 TXN_OPC_WRITE = 0,
39 TXN_OPC_BLOCKWRITE = 1,
40 TXN_OPC_BLOCKSET = 2,
41 TXN_OPC_MASKWRITE = 3,
42 TXN_OPC_MASKPOLL = 4,
43 TXN_OPC_NOOP = 5,
44 TXN_OPC_PREEMPT = 6,
45 TXN_OPC_MASKPOLL_BUSY = 7,
46 TXN_OPC_LOADPDI = 8,
47 TXN_OPC_LOAD_PM_START = 9,
48 TXN_OPC_CREATE_SCRATCHPAD = 10,
49 TXN_OPC_UPDATE_STATE_TABLE = 11,
50 TXN_OPC_UPDATE_REG = 12,
51 TXN_OPC_UPDATE_SCRATCH = 13,
52 TXN_OPC_CONFIG_SHIMDMA_BD = 14,
53 TXN_OPC_CONFIG_SHIMDMA_DMABUF_BD = 15,
54 TXN_OPC_CUSTOM_OP_BEGIN = 1U << 7U,
55 TXN_OPC_TCT = TXN_OPC_CUSTOM_OP_BEGIN,
56 TXN_OPC_DDR_PATCH = TXN_OPC_CUSTOM_OP_BEGIN + 1,
57 TXN_OPC_READ_REGS = TXN_OPC_CUSTOM_OP_BEGIN + 2,
58 TXN_OPC_RECORD_TIMER = TXN_OPC_CUSTOM_OP_BEGIN + 3,

65// Upper bound on a tile's buffer-descriptor count across all tile types, sizing
66// the pool's fixed storage. This is a standalone header (no MLIR / target model
67// here), so the ACTUAL per-tile count is supplied at runtime by the compiler
68// via bd_pool_init(getNumBDs(col,row)) -- this constant only bounds the array.
69// AIE2 memtiles have the most BDs (48); shim/core have 16. Keep this >= the
70// largest getNumBDs any target model returns.
71constexpr uint32_t kMaxBDsPerTile = 48;
72
73// Runtime buffer-descriptor free-list pool, for dynamic runtime sequences whose
74// BD IDs are chosen at runtime (a rolled scf.for the compiler cannot unroll).
75// The static allocator assigns BD IDs at compile time; this is its runtime
76// counterpart. `head` is the number of currently-free IDs at the front of
77// `free_ids`; pop takes from the top, push returns to it.
78struct BdPool {
79 uint32_t free_ids[kMaxBDsPerTile];
80 int head;
81};
82
83// Initialize a pool with `n` free IDs. `n` is the tile's BD count, which the
84// compiler reads from the target model (getNumBDs) and passes in. IDs are
85// stacked so that pop() hands out the LOWEST free id first (id 0, then 1, ...),
86// matching the static allocator's lowest-free-first order -- so a first pop
87// from a fresh pool equals a pinned bd_id = 0.
88inline BdPool bd_pool_init(uint32_t n) {
89 BdPool p;
90 p.head = 0;
91 uint32_t count = n < kMaxBDsPerTile ? n : kMaxBDsPerTile;
92 for (uint32_t i = 0; i < count; ++i)
93 p.free_ids[p.head++] = count - 1 - i; // top of stack is id 0
94 return p;
95}
96
97// Pop a free BD ID into `out`. Returns false if the pool is empty -- the
98// generated builder turns that into a `return std::nullopt`, so a runtime
99// working set that exceeds the tile's BD count yields no stream rather than a
100// silently-corrupt one.
101inline bool bd_pool_pop(BdPool &p, uint32_t &out) {
102 if (p.head == 0)
103 return false;
104 out = p.free_ids[--p.head];
105 return true;
106}
107
108// Return a BD ID to the pool for reuse.
109inline void bd_pool_push(BdPool &p, uint32_t id) {
110 if (p.head < static_cast<int>(kMaxBDsPerTile))
111 p.free_ids[p.head++] = id;
112}

124// Append a 6-word write32 instruction.
125inline void txn_append_write32(std::vector<uint32_t> &txn, uint32_t addr,
126 uint32_t val) {
127 size_t pos = txn.size();
128 txn.resize(pos + 6, 0);
129 txn[pos + 0] = TXN_OPC_WRITE;
130 // txn[pos + 1] is reserved (0)
131 txn[pos + 2] = addr;
132 txn[pos + 3] = 0; // extra bits for reg offset
133 txn[pos + 4] = val;
134 txn[pos + 5] = 6 * sizeof(uint32_t); // operation size
135}

151// Append a 4-word sync (TCT) instruction.
152inline void txn_append_sync(std::vector<uint32_t> &txn, uint32_t col,
153 uint32_t row, uint32_t dir, uint32_t chan,
154 uint32_t ncol, uint32_t nrow) {
155 size_t pos = txn.size();
156 txn.resize(pos + 4, 0);
157 txn[pos + 0] = TXN_OPC_TCT;
158 txn[pos + 1] = 4 * sizeof(uint32_t); // operation size
159 txn[pos + 2] = (dir & 0xff) | ((row & 0xff) << 8) | ((col & 0xff) << 16);
160 txn[pos + 3] =
161 ((nrow & 0xff) << 8) | ((ncol & 0xff) << 16) | ((chan & 0xff) << 24);
162}

164// Append a variable-length blockwrite instruction.
165// `data` points to `count` uint32_t words of payload.
166//
167// The destination tile is encoded twice in the firmware blockwrite format: once
168// in the upper bits of the absolute `addr` (column/row folded in by the
169// compiler before this point) and once in the dedicated col/row field at
170// word[1]. Callers that build an absolute address from a tile pass that same
171// (col, row) here so both representations agree; callers with a purely flat
172// address pass 0/0.
173inline void txn_append_blockwrite(std::vector<uint32_t> &txn, uint32_t addr,
174 const uint32_t *data, size_t count,
175 uint32_t col = 0, uint32_t row = 0) {
176 const unsigned headerSize = 4;
177 size_t pos = txn.size();
178 txn.resize(pos + headerSize + count, 0);
179 txn[pos + 0] = TXN_OPC_BLOCKWRITE;
180 txn[pos + 1] = (col & 0xff) | ((row & 0xff) << 8);
181 txn[pos + 2] = addr;
182 txn[pos + 3] = static_cast<uint32_t>((headerSize + count) * sizeof(uint32_t));
183 for (size_t i = 0; i < count; ++i)
184 txn[pos + headerSize + i] = data[i];
185}

187// Append a 12-word address_patch (DDR_PATCH) instruction.
188inline void txn_append_address_patch(std::vector<uint32_t> &txn, uint32_t addr,
189 int32_t arg_idx, uint32_t arg_plus) {
190 size_t pos = txn.size();
191 txn.resize(pos + 12, 0);
192 txn[pos + 0] = TXN_OPC_DDR_PATCH; // opcode
193 txn[pos + 1] = 12 * sizeof(uint32_t); // operation size
194 // pos+2..4 are reserved (zero)
195 txn[pos + 5] = 0; // action (0 = patch)
196 txn[pos + 6] = addr; // register address to patch
197 // pos+7 is reserved (zero)
198 txn[pos + 8] = static_cast<uint32_t>(arg_idx); // buffer argument index
199 // pos+9 is reserved (zero)
200 txn[pos + 10] = arg_plus; // byte offset into buffer
201 // pos+11 is reserved (zero)
202}

204// Append a 4-word loadpdi instruction.
205inline void txn_append_loadpdi(std::vector<uint32_t> &txn, uint32_t id,
206 uint32_t size, uint64_t addr) {
207 size_t pos = txn.size();
208 txn.resize(pos + 4, 0);
209 txn[pos + 0] = TXN_OPC_LOADPDI | (id << 16);
210 txn[pos + 1] = size;
211 txn[pos + 2] = static_cast<uint32_t>(addr);
212 txn[pos + 3] = static_cast<uint32_t>(addr >> 32);
213}

215// Append a 1-word preempt instruction.
216inline void txn_append_preempt(std::vector<uint32_t> &txn, uint32_t level) {
217 txn.push_back(TXN_OPC_PREEMPT | (level << 8));
218}

220// Reserve 4 placeholder words for the TXN header. Call this BEFORE appending
221// any instructions, so that the header space is already allocated.
222inline void txn_init(std::vector<uint32_t> &txn) { txn.resize(4, 0); }
223
224// Finalize the 4-word TXN header in-place. Call this AFTER all instructions
225// are appended. The first 4 words must have been reserved by txn_init().
226// `op_count` is the number of operations appended.
227inline void txn_prepend_header(std::vector<uint32_t> &txn, uint32_t op_count,
228 TxnDeviceInfo info = {}) {
229 if (txn.size() < 4)
230 txn.resize(4, 0);
231 txn[0] = (static_cast<uint32_t>(info.numRows) << 24) |
232 (static_cast<uint32_t>(info.devGen) << 16) |
233 (static_cast<uint32_t>(info.minor) << 8) |
234 static_cast<uint32_t>(info.major);
235 txn[1] = (static_cast<uint32_t>(info.numMemTileRows) << 8) |
236 static_cast<uint32_t>(info.numCols);
237 txn[2] = op_count;
238 txn[3] = static_cast<uint32_t>(txn.size() * sizeof(uint32_t));
239}
