28// CORE_CONTROL register, tile-local offset. Uniform across the AIE2 family:
29// XAIE2PGBL_CORE_MODULE_CORE_CONTROL (aie2p) and
30// XAIEMLGBL_CORE_MODULE_CORE_CONTROL (aie2) are both 0x00032000. The op only
31// lowers on core tiles (the verifier rejects mem/shim), which exist on aie2 and
32// aie2p; the npu runtime sequence path is AIE2-only.
33static constexpr uint32_t kCoreCtrlAddr = 0x00032000;
34
35// CORE_CONTROL reset bit (bit 1). Matches
36// XAIE2PGBL_CORE_MODULE_CORE_CONTROL_RESET_MASK. Bit 0 of the same word is
37// ENABLE, which the mask preserves.
38static constexpr uint32_t kCoreCtrlResetMask = 0x2;
39
40struct CoreResetToMaskWrite32Pattern : OpConversionPattern<CoreResetOp> {
41 using OpConversionPattern<CoreResetOp>::OpConversionPattern;
42
43 CoreResetToMaskWrite32Pattern(MLIRContext *context)
44 : OpConversionPattern(context) {}
45
46 LogicalResult
47 matchAndRewrite(CoreResetOp op, OpAdaptor adaptor,
48 ConversionPatternRewriter &rewriter) const override {
49 AIE::TileOp tile = op.getTileOp();
50 int col = tile.getCol();
51 int row = tile.getRow();
52
53 Location loc = op.getLoc();
54 IntegerAttr colAttr = rewriter.getI32IntegerAttr(col);
55 IntegerAttr rowAttr = rewriter.getI32IntegerAttr(row);
56
57 // NpuMaskWrite32Op re-folds col/row from the attributes, so the address
58 // operand is the tile-local CORE_CONTROL offset, mirroring how
59 // AIELowerSetLock passes the local lock address with col/row.
60 //
61 // Reset pulse: assert the reset bit, then clear it. Both writes mask to the
62 // reset bit only, so the pulse preserves the ENABLE field packed in the
63 // same CORE_CONTROL word instead of clobbering it. This mirrors aie-rt's
64 // XAie_CoreReset/XAie_CoreUnreset, which drive the reset bit with a
65 // MaskWrite32. Constants are materialized in named locals so the emitted IR
66 // order does not depend on unspecified C++ argument-evaluation order.
67 Value assertAddr = createConstantI32(rewriter, loc, kCoreCtrlAddr);
68 Value assertVal = createConstantI32(rewriter, loc, kCoreCtrlResetMask);
69 Value assertMask = createConstantI32(rewriter, loc, kCoreCtrlResetMask);
70 rewriter.create<NpuMaskWrite32Op>(loc, assertAddr, assertVal, assertMask,
71 nullptr, colAttr, rowAttr);
72
73 Value clearAddr = createConstantI32(rewriter, loc, kCoreCtrlAddr);
74 Value clearVal = createConstantI32(rewriter, loc, 0u);
75 Value clearMask = createConstantI32(rewriter, loc, kCoreCtrlResetMask);
76 rewriter.replaceOpWithNewOp<NpuMaskWrite32Op>(
77 op, clearAddr, clearVal, clearMask, nullptr, colAttr, rowAttr);
78
79 return success();
80 };
81};

83struct AIELowerCoreResetPass
84 : public xilinx::AIEX::impl::AIELowerCoreResetBase<AIELowerCoreResetPass> {
85 void runOnOperation() override {
86 DeviceOp device = getOperation();
87
88 ConversionTarget target(getContext());
89 target.addLegalOp<NpuMaskWrite32Op>();
90 target.addLegalDialect<arith::ArithDialect>();
91 target.addIllegalOp<CoreResetOp>();

97 signalPassFailure();
