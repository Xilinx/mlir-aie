56uint32_t getParity(uint32_t n) {
57 int count = 0;
58 while (n > 0) {
59 if (n & 1) { // Check if the least significant bit is 1
60 count++;
61 }
62 n >>= 1; // Right shift to check the next bit
63 }
64 return (count % 2 == 0) ? 0 : 1; // 0 for even parity, 1 for odd parity
65}

67uint32_t create_ctrl_pkt(int operation, int beats, int addr,
68 int ctrl_pkt_read_id = 28) {
69 uint32_t ctrl_pkt = ((ctrl_pkt_read_id & 0xFF) << 24) |
70 ((operation & 0x3) << 22) | ((beats & 0x3) << 20) |
71 (addr & 0x7FFFF);
72 ctrl_pkt |= (0x1 ^ getParity(ctrl_pkt)) << 31;
73 return ctrl_pkt;
74}

76/*
77 ******************************************************************************
78 * Typed buffer descriptors for the variadic XRT test wrapper
79 ******************************************************************************
80 *
81 * The host buffer ABI is the runtime_sequence operand list: the kernel takes
82 * its data buffers in order, then -- only when enabled -- a control-packet
83 * buffer, then a trace buffer, each appended at the tail. The wrapper builds
84 * the kernel argument list to match exactly, so it works for any number of
85 * inputs and never passes a buffer the kernel did not declare.
86 */
87
88// A typed input buffer: element type, element count, and initializer.
89template <typename T>
90struct InBuf {
91 using elem_type = T;
92 int volume;
93 void (*init)(T *, int);
94};
95
96template <typename T>
97InBuf<T> make_in(int volume, void (*init)(T *, int)) {
98 return InBuf<T>{volume, init};
99}
100
101// A typed output buffer.
102template <typename T>
103struct OutBuf {
104 using elem_type = T;
105 int volume;
106 void (*init)(T *, int);
107};
108
109template <typename T>
110OutBuf<T> make_out(int volume, void (*init)(T *, int)) {
111 return OutBuf<T>{volume, init};
112}

114/*
115 ******************************************************************************
116 * XRT based test wrapper for N inputs and 1 output
117 ******************************************************************************
118 *
119 * Inputs are passed as InBuf<T> descriptors (any number, any types). The
120 * verifier is called as verify(in0*, in1*, ..., out*, out_volume, verbosity)
121 * and returns an error count.
122 *
123 * Host buffer indices follow the runtime_sequence operand list:
124 * group_id(3 + i) for data buffer i, then (if enabled) control packets, then
125 * (if trace_size > 0) the trace buffer -- each appended at the tail.
126 */
127template <typename Verify, typename TOut, typename... TIns>
128int setup_and_run_aie(Verify verify_results, std::tuple<InBuf<TIns>...> inputs,
129 OutBuf<TOut> output, struct args myargs,
130 bool enable_ctrl_pkts = false,
131 int num_trailing_params = 0) {
132
133 srand(time(NULL));
134
135 constexpr int num_inputs = sizeof...(TIns);
136 // Some designs place scalar/param buffers AFTER the output in the host ABI.
137 // The IRON `transform` runtime sequence, for example, is ordered
138 // [inputs, output, params]. `num_trailing_params` counts how many of the
139 // trailing entries in `inputs` are actually such params: they are bound to
140 // kernel arg slots *after* the output instead of before it. The verifier
141 // still receives all buffers in declared order (inputs..., params..., out).
142 const int num_real_inputs = num_inputs - num_trailing_params;
143
144 // Load instruction sequence
145 std::vector<uint32_t> instr_v = test_utils::load_instr_binary(myargs.instr);
146 if (myargs.verbosity >= 1)
147 std::cout << "Sequence instr count: " << instr_v.size() << "\n";
148
149 // Start the XRT context and load the kernel
150 xrt::device device;
151 xrt::kernel kernel;
152
153 test_utils::init_xrt_load_kernel(device, kernel, myargs.verbosity,
154 myargs.xclbin, myargs.kernel);
155
156 // The fixed scalar args occupy kernel arg slots 0..2 (opcode, instr, ninstr).
157 // Data buffers start at slot 3, i.e. group_id(3 + data_index).
158 constexpr int kFirstDataArg = 3;
159 // Output slot follows the *real* inputs; trailing params come after it.
160 const int out_arg = kFirstDataArg + num_real_inputs;
161 // Kernel arg slot for the idx-th declared `inputs` entry (real input or
162 // trailing param).
163 auto data_arg_slot = [&](int idx) {
164 return (idx < num_real_inputs) ? (kFirstDataArg + idx)
165 : (out_arg + 1 + (idx - num_real_inputs));
166 };
167
168 // Instruction buffer (kernel arg slot 1).
169 auto bo_instr = xrt::bo(device, instr_v.size() * sizeof(int),
170 XCL_BO_FLAGS_CACHEABLE, kernel.group_id(1));
171 void *bufInstr = bo_instr.map<void *>();
172 memcpy(bufInstr, instr_v.data(), instr_v.size() * sizeof(int));
173 bo_instr.sync(XCL_BO_SYNC_BO_TO_DEVICE);
174
175 if (myargs.verbosity >= 1)
176 std::cout << "Writing data into buffer objects.\n";
177
178 // Allocate + initialize the input buffer objects, in order. Each input lands
179 // at the next data arg slot.
180 std::vector<xrt::bo> in_bos;
181 in_bos.reserve(num_inputs);
182 std::tuple<TIns *...> in_ptrs;
183
184 // The verifier's SIZE argument is the first input's element count, matching
185 // the historical wrapper contract (verify iterates the primary input).
186 int first_in_volume = 0;
187
188 auto alloc_inputs = [&](auto &...inb) {
189 int idx = 0;
190 auto one = [&](auto &desc, auto &ptr_slot) {
191 using T = typename std::remove_reference_t<decltype(desc)>::elem_type;
192 auto bo = xrt::bo(device, desc.volume * sizeof(T), XRT_BO_FLAGS_HOST_ONLY,
193 kernel.group_id(data_arg_slot(idx)));
194 T *mapped = bo.template map<T *>();
195 desc.init(mapped, desc.volume);
196 bo.sync(XCL_BO_SYNC_BO_TO_DEVICE);
197 ptr_slot = mapped;
198 in_bos.push_back(std::move(bo));
199 if (idx == 0)
200 first_in_volume = desc.volume;
201 idx++;
202 };
203 std::apply([&](auto &...slots) { (one(inb, slots), ...); }, in_ptrs);
204 };
205 std::apply(alloc_inputs, inputs);
206
207 // Output buffer follows the real inputs (any params come after it).
208 auto bo_out = xrt::bo(device, output.volume * sizeof(TOut),
209 XRT_BO_FLAGS_HOST_ONLY, kernel.group_id(out_arg));
210 TOut *bufOut = bo_out.map<TOut *>();
211 output.init(bufOut, output.volume);
212 bo_out.sync(XCL_BO_SYNC_BO_TO_DEVICE);
213
214 // Control-packet and trace buffers are appended (in that order) only when
215 // enabled, matching what trace lowering appends to the runtime_sequence.
216 // Control packets are a trace-side feature, so they are only part of the ABI
217 // when tracing is also enabled.
218 const bool use_ctrl_pkts = enable_ctrl_pkts && myargs.trace_size > 0;
219 int next_arg = out_arg + 1 + num_trailing_params;
220
221 xrt::bo bo_ctrlpkts;
222 uint32_t *bufCtrlPkts = nullptr;
223 int ctrlpkt_arg = -1;
224 if (use_ctrl_pkts) {
225 ctrlpkt_arg = next_arg++;
226 bo_ctrlpkts = xrt::bo(device, 8, XRT_BO_FLAGS_HOST_ONLY,
227 kernel.group_id(ctrlpkt_arg));
228 bufCtrlPkts = bo_ctrlpkts.map<uint32_t *>();
229 }
230
231 xrt::bo bo_trace;
232 char *bufTrace = nullptr;
233 int trace_arg = -1;
234 if (myargs.trace_size > 0) {
235 trace_arg = next_arg++;
236 // Allocate a larger trace buffer (driver workaround) that also holds the
237 // 8-byte control packet response when control packets are enabled.
238 int alloc_trace_size = myargs.trace_size * 4;
239 bo_trace = xrt::bo(device, alloc_trace_size, XRT_BO_FLAGS_HOST_ONLY,
240 kernel.group_id(trace_arg));
241 bufTrace = bo_trace.map<char *>();
242 memset(bufTrace, 0, myargs.trace_size);
243 }
244
245 // Set control packet values.
246 if (use_ctrl_pkts) {
247 bufCtrlPkts[0] = create_ctrl_pkt(1, 0, 0x32004); // core status
248 bufCtrlPkts[1] = create_ctrl_pkt(1, 0, 0x320D8); // trace status
249 if (myargs.verbosity >= 1) {
250 std::cout << "bufCtrlPkts[0]:" << std::hex << bufCtrlPkts[0] << std::endl;
251 std::cout << "bufCtrlPkts[1]:" << std::hex << bufCtrlPkts[1] << std::endl;
252 }
253 }
254
255 // Sync trace/ctrlpkt buffers to device.
256 if (myargs.trace_size > 0) {
257 bo_trace.sync(XCL_BO_SYNC_BO_TO_DEVICE);
258 if (use_ctrl_pkts)
259 bo_ctrlpkts.sync(XCL_BO_SYNC_BO_TO_DEVICE);
260 }
261
262 // ------------------------------------------------------
263 // Initialize run configs
264 // ------------------------------------------------------
265 unsigned num_iter = myargs.n_iterations + myargs.n_warmup_iterations;
266 float npu_time_total = 0;
267 float npu_time_min = 9999999;
268 float npu_time_max = 0;
269
270 int errors = 0;
271
272 // ------------------------------------------------------
273 // Main run loop
274 // ------------------------------------------------------
275 for (unsigned iter = 0; iter < num_iter; iter++) {
276
277 if (myargs.verbosity >= 1)
278 std::cout << "Running Kernel.\n";
279
280 auto start = std::chrono::high_resolution_clock::now();
281 unsigned int opcode = 3;
282
283 // Build the kernel argument list to match the declared host ABI exactly:
284 // opcode, instr, ninstr, data buffers..., [ctrlpkt], [trace].
285 xrt::run run(kernel);
286 run.set_arg(0, opcode);
287 run.set_arg(1, bo_instr);
288 run.set_arg(2, instr_v.size());
289 for (int i = 0; i < num_inputs; i++)
290 run.set_arg(data_arg_slot(i), in_bos[i]);
291 run.set_arg(out_arg, bo_out);
292 if (ctrlpkt_arg >= 0)
293 run.set_arg(ctrlpkt_arg, bo_ctrlpkts);
294 if (trace_arg >= 0)
295 run.set_arg(trace_arg, bo_trace);
296 run.start();
297 run.wait();
298
299 auto stop = std::chrono::high_resolution_clock::now();
300 bo_out.sync(XCL_BO_SYNC_BO_FROM_DEVICE);
301 if (myargs.trace_size > 0)
302 bo_trace.sync(XCL_BO_SYNC_BO_FROM_DEVICE);
303
304 if (iter < myargs.n_warmup_iterations)
305 /* Warmup iterations do not count towards average runtime. */
306 continue;
307
308 // Copy output results and verify they are correct
309 if (myargs.do_verify) {
310 if (myargs.verbosity >= 1)
311 std::cout << "Verifying results ..." << std::endl;
312 auto vstart = std::chrono::system_clock::now();
313
314 // verify(in0*, in1*, ..., out*, first_input_volume, verbosity)
315 errors += std::apply(
316 [&](auto *...in_ptr) {
317 return verify_results(in_ptr..., bufOut, first_in_volume,
318 myargs.verbosity);
319 },
320 in_ptrs);
321
322 auto vstop = std::chrono::system_clock::now();
323 float vtime =
324 std::chrono::duration_cast<std::chrono::seconds>(vstop - vstart)
325 .count();
326 if (myargs.verbosity >= 1)
327 std::cout << "Verify time: " << vtime << "secs." << std::endl;
328 } else {
329 if (myargs.verbosity >= 1)
330 std::cout << "WARNING: results not verified." << std::endl;
331 }
332
333 // Write trace values if trace_size > 0 and first non-warmup iteration
334 if (myargs.trace_size > 0 && iter == myargs.n_warmup_iterations) {
335 test_utils::write_out_trace(((char *)bufTrace), myargs.trace_size,
336 myargs.trace_file);
337 }
338
339 // Write out control packet outputs
340 if (use_ctrl_pkts) {
341 uint32_t *ctrl_pkt_out =
342 (uint32_t *)(((char *)bufTrace) + myargs.trace_size);
343 if (myargs.verbosity >= 1) {
344 std::cout << "ctrl_pkt_out[0]:" << std::hex << ctrl_pkt_out[0]
345 << std::endl;
346 std::cout << "ctrl_pkt_out[1]:" << std::hex << ctrl_pkt_out[1]
347 << std::endl;
348 }
349 int col = (ctrl_pkt_out[0] >> 21) & 0x7F;
350 int row = (ctrl_pkt_out[0] >> 16) & 0x1F;
351 if ((ctrl_pkt_out[1] >> 8) == 3)
352 std::cout << "WARNING: Trace overflow detected in tile(" << row << ","
353 << col << ". Trace results may be invalid." << std::endl;
354 }
355
356 // Accumulate run times
357 float npu_time =
358 std::chrono::duration_cast<std::chrono::microseconds>(stop - start)
359 .count();
360
361 npu_time_total += npu_time;
362 npu_time_min = (npu_time < npu_time_min) ? npu_time : npu_time_min;
363 npu_time_max = (npu_time > npu_time_max) ? npu_time : npu_time_max;
364 }
365
366 // ------------------------------------------------------
367 // Print verification and timing results
368 // ------------------------------------------------------
369
370 // TODO - Mac count to guide gflops
371 float macs = 0;
372
373 std::cout << std::endl
374 << "Avg NPU time: " << npu_time_total / myargs.n_iterations << "us."
375 << std::endl;
376 if (macs > 0)
377 std::cout << "Avg NPU gflops: "
378 << macs / (1000 * npu_time_total / myargs.n_iterations)
379 << std::endl;
380
381 std::cout << std::endl
382 << "Min NPU time: " << npu_time_min << "us." << std::endl;
383 if (macs > 0)
384 std::cout << "Max NPU gflops: " << macs / (1000 * npu_time_min)
385 << std::endl;
386
387 std::cout << std::endl
388 << "Max NPU time: " << npu_time_max << "us." << std::endl;
389 if (macs > 0)
390 std::cout << "Min NPU gflops: " << macs / (1000 * npu_time_max)
391 << std::endl;
392
393 if (!errors) {
394 std::cout << "\nPASS!\n\n";
395 return 0;
396 } else {
397 std::cout << "\nError count: " << errors << "\n\n";
398 std::cout << "\nFailed.\n\n";
399 return 1;
400 }
