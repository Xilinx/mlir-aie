146struct PushQueuetoWrite32Pattern : OpConversionPattern<NpuPushQueueOp> {
147
148public:
149 using OpConversionPattern::OpConversionPattern;
150
151 PushQueuetoWrite32Pattern(MLIRContext *context, PatternBenefit benefit = 1)
152 : OpConversionPattern(context, benefit) {}
153
154 LogicalResult
155 matchAndRewrite(NpuPushQueueOp op, OpAdaptor adaptor,
156 ConversionPatternRewriter &rewriter) const override {
157
158 const auto &tm = AIE::getTargetModel(op);
159 uint32_t ctrl_offset = tm.getDmaControlAddress(
160 op.getColumn(), op.getRow(), op.getChannel(), op.getDirection());
161
162 // control packet for issuing token
163 if (op.getIssueToken()) {
164 // set the task-complete-token controller ID field in the dma control
165 // register
166 AIE::TileOp shimTile = AIE::TileOp::getOrCreate(
167 rewriter, op->getParentOfType<AIE::DeviceOp>(), op.getColumn(),
168 op.getRow());
169 if (shimTile->hasAttr("controller_id")) {
170 AIE::PacketInfoAttr controller_id_attr =
171 shimTile->getAttrOfType<AIE::PacketInfoAttr>("controller_id");
172 uint32_t data = controller_id_attr.getPktId() << 8;
173 uint32_t mask = 0x00001F00;
174 NpuMaskWrite32Op::create(
175 rewriter, op->getLoc(),
176 createConstantI32(rewriter, op->getLoc(), ctrl_offset),
177 createConstantI32(rewriter, op->getLoc(), data),
178 createConstantI32(rewriter, op->getLoc(), mask), nullptr, nullptr,
179 nullptr);
180 }
181 }
182
183 // the offset of the task queue register in the tile
184 uint32_t queue_offset = ctrl_offset + 0x4;
185
186 // Command word: bd_id [3:0], repeat_count [23:16], issue-token bit [31].
187 // bd_id is always compile-time; repeat_count may be a runtime SSA value, in
188 // which case the word is built with arith over a constant base.
189 std::optional<uint32_t> bd_id = getConstantIntOperand(op.getBdId());
190 if (!bd_id)
191 return op.emitOpError("cannot lower push_queue with non-constant bd_id");
192 uint32_t cmdBase = (*bd_id & 0xF) | (op.getIssueToken() ? 0x80000000 : 0);
193
194 Value cmdVal;
195 if (std::optional<uint32_t> repeat_cnt =
196 getConstantIntOperand(op.getRepeatCount())) {
197 cmdVal = createConstantI32(rewriter, op->getLoc(),
198 cmdBase | ((*repeat_cnt & 0xFF) << 16));
199 } else {
200 Location loc = op->getLoc();
201 Value repeat = op.getRepeatCount();
202 Value masked = arith::AndIOp::create(
203 rewriter, loc, repeat, createConstantI32(rewriter, loc, 0xFF));
204 Value shifted = arith::ShLIOp::create(
205 rewriter, loc, masked, createConstantI32(rewriter, loc, 16));
206 cmdVal = arith::OrIOp::create(
207 rewriter, loc, createConstantI32(rewriter, loc, cmdBase), shifted);
208 }
209
210 NpuWrite32Op::create(
211 rewriter, op->getLoc(),
212 createConstantI32(rewriter, op->getLoc(), queue_offset), cmdVal,
213 nullptr, nullptr, nullptr);
214 rewriter.eraseOp(op);
215 return success();
216 }
217};

219struct DmaToNpuPattern : OpConversionPattern<NpuDmaMemcpyNdOp> {
220 using OpConversionPattern::OpConversionPattern;
221
222public:
223 DmaToNpuPattern(MLIRContext *context, PatternBenefit benefit = 1)
224 : OpConversionPattern(context, benefit) {}
225
226 LogicalResult
227 matchAndRewrite(NpuDmaMemcpyNdOp op, OpAdaptor adaptor,
228 ConversionPatternRewriter &rewriter) const override {
229 // Any runtime (SSA) offset/size/stride takes the dynamic path; a
230 // fully-constant descriptor takes the static path below. The op verifier
231 // has already enforced the supported scope for the dynamic case (shim NOC,
232 // no padding, realizable/in-range constants).
233 bool allOffsetsConstant =
234 llvm::all_of(op.getMixedOffsets(),
235 [](OpFoldResult s) { return getConstantIntValue(s); });
236 bool allSizesConstant =
237 llvm::all_of(op.getMixedSizes(),
238 [](OpFoldResult s) { return getConstantIntValue(s); });
239 bool allStridesConstant =
240 llvm::all_of(op.getMixedStrides(),
241 [](OpFoldResult s) { return getConstantIntValue(s); });
242 if (!allOffsetsConstant || !allSizesConstant || !allStridesConstant)
243 return lowerDynamic(op, adaptor, rewriter);
244
245 const auto &targetModel = AIE::getTargetModel(op);
246 BaseMemRefType bufferType = op.getMemref().getType();
247 auto *ctx = op->getContext();
248 auto i32ty = IntegerType::get(ctx, 32);
249 auto zero = IntegerAttr::get(i32ty, 0);
250
251 auto dev = op->getParentOfType<AIE::DeviceOp>();
252 if (!dev)
253 return failure();
254
255 auto infoOp = AIE::ShimDMAAllocationOp::getForSymbol(
256 dev, op.getMetadata().getRootReference());
257 if (!infoOp) {
258 return op->emitOpError("couldn't find shim_dma_allocation op.");
259 }
260
261 AIE::TileOp shimTile = infoOp.getTileOp();
262 if (!shimTile) {
263 return op->emitOpError(
264 "shim_dma_allocation op must reference a valid TileOp.");
265 }
266
267 auto channelDir = infoOp.getChannelDir();
268 bool isMM2S = channelDir == AIE::DMAChannelDir::MM2S;
269 int tileCol = shimTile.getCol();
270 int tileRow = shimTile.getRow();
271
272 // initialize fields to zero
273 auto column = zero;
274 auto bd_id = zero;
275 auto buffer_length = zero;
276 auto buffer_offset = zero;
277 auto enable_packet = zero;
278 auto out_of_order_id = zero;
279 auto packet_id = zero;
280 auto packet_type = zero;
281 auto d0_size = zero;
282 auto d0_stride = zero;
283 auto d1_size = zero;
284 auto d1_stride = zero;
285 auto d2_size = zero;
286 auto d2_stride = zero;
287 auto iteration_current = zero;
288 auto iteration_size = zero;
289 auto iteration_stride = zero;
290 auto next_bd = zero;
291 auto row = zero;
292 auto use_next_bd = zero;
293 auto valid_bd = zero;
294 auto lock_rel_val = zero;
295 auto lock_rel_id = zero;
296 auto lock_acq_enable = zero;
297 auto lock_acq_val = zero;
298 auto lock_acq_id = zero;
299 auto d0_zero_before = zero;
300 auto d1_zero_before = zero;
301 auto d2_zero_before = zero;
302 auto d0_zero_after = zero;
303 auto d1_zero_after = zero;
304 auto d2_zero_after = zero;
305 auto burst_length = zero;
306
307 auto issue_token = BoolAttr::get(ctx, false);
308 auto repeat_count = zero;
309 llvm::SmallVector<int64_t, 4> inputSizes = llvm::map_to_vector(
310 llvm::reverse(op.getMixedSizes()),
311 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
312 llvm::SmallVector<int64_t, 4> inputStrides = llvm::map_to_vector(
313 llvm::reverse(op.getMixedStrides()),
314 [](OpFoldResult s) { return getConstantIntValue(s).value(); });
315 llvm::SmallVector<int64_t, 4> sizes(4);
316 llvm::SmallVector<int64_t, 4> strides(4);
317 getHardwareStridesWraps(targetModel, op, bufferType, inputSizes,
318 inputStrides, sizes, strides);
319
320 // column
321 column = IntegerAttr::get(i32ty, tileCol);
322
323 // row
324 row = IntegerAttr::get(i32ty, tileRow);
325
326 // A contiguous row-major ND access on a shim NOC tile is lowered to linear
327 // mode (d0_size=d1_size=0) just like an already-canonical linear transfer.
328 // This allows naturally-expressed multidimensional transfers (e.g., a 2D
329 // image as [height, width]) without hitting the 10-bit ND wrap-size limit.
330 bool isLinear = op.isLinearTransferWithoutTransformation() ||
331 (targetModel.isShimNOCTile(tileCol, tileRow) &&
332 isContiguousTransfer(inputSizes, inputStrides));
333 if (failed(verifyStridesWraps(op, bufferType, tileCol, tileRow, inputSizes,
334 inputStrides, sizes, strides, isLinear))) {
335 return failure();
336 }
337
338 // bd_id
339 bd_id = IntegerAttr::get(i32ty, op.getId());
340
341 // buffer_length
342 uint64_t buffer_length_val = inputSizes[0] * op.getElementTypeBitwidth() /
343 targetModel.getAddressGenGranularity();
344 if (inputSizes.size() > 1) {
345 for (size_t i = 1; i < std::min(inputSizes.size(), (size_t)3); i++) {
346 buffer_length_val *= inputSizes[i];
347 }
348 }
349 buffer_length = IntegerAttr::get(i32ty, buffer_length_val);
350
351 // buffer_offset - zero because the complete address is set by the patch op
352 buffer_offset = IntegerAttr::get(i32ty, 0);
353
354 // enable_packet
355 if (auto packetInfo = op.getPacket()) {
356 enable_packet = IntegerAttr::get(i32ty, 1);
357 packet_type = IntegerAttr::get(i32ty, packetInfo->getPktType());
358 packet_id = IntegerAttr::get(i32ty, packetInfo->getPktId());
359 }
360
361 // out_of_order_id
362
363 if (!isLinear) {
364 // d0_size, d0_stride
365 d0_size = IntegerAttr::get(i32ty, sizes[0]);
366 d0_stride = IntegerAttr::get(i32ty, strides[0]);
367
368 // d1_size, d1_stride
369 d1_size = IntegerAttr::get(i32ty, sizes[1]);
370 d1_stride = IntegerAttr::get(i32ty, strides[1]);
371
372 // d2_stride
373 d2_stride = IntegerAttr::get(i32ty, strides[2]);
374
375 // d2_size
376 if (targetModel.isMemTile(tileCol, 0)) // Need to be any row
377 d2_size = IntegerAttr::get(i32ty, sizes[2]);
378 else
379 d2_size = IntegerAttr::get(i32ty, 0);
380 }
381 // iteration_current, iteration_size, iteration_stride, repeat_count
382 if (inputSizes[3] > 1) {
383 if (inputStrides[3] > 0) {
384 iteration_size = IntegerAttr::get(i32ty, sizes[3]);
385 iteration_stride = IntegerAttr::get(i32ty, strides[3]);
386 } else {
387 // We allow users to encode the repeat_count as a dimension 3 stride
388 // of 0. This must lower to a iteration wrap of 0, so no stride is
389 // ever added. We then repeat the BD using the repeat_count in
390 // NpuPushQueueOp.
391 iteration_size = zero;
392 iteration_stride = zero;
393 }
394 }
395 repeat_count = IntegerAttr::get(i32ty, sizes[3]);
396
397 // next_bd
398
399 // use_next_bd
400
401 // valid_bd
402 valid_bd = IntegerAttr::get(i32ty, 1);
403
404 // lock_rel_val
405
406 // lock_rel_id
407
408 // lock_acq_enable
409
410 // lock_acq_val
411
412 // lock_acq_id
413
414 // d0_zero_before
415 d0_zero_before = IntegerAttr::get(i32ty, op.getD0ZeroBefore());
416
417 // d1_zero_before
418 d1_zero_before = IntegerAttr::get(i32ty, op.getD1ZeroBefore());
419
420 // d2_zero_before
421 d2_zero_before = IntegerAttr::get(i32ty, op.getD2ZeroBefore());
422
423 // d0_zero_after
424 d0_zero_after = IntegerAttr::get(i32ty, op.getD0ZeroAfter());
425
426 // d1_zero_after
427 d1_zero_after = IntegerAttr::get(i32ty, op.getD1ZeroAfter());
428
429 // d2_zero_after
430 d2_zero_after = IntegerAttr::get(i32ty, op.getD2ZeroAfter());
431
432 // burst_size
433 burst_length = IntegerAttr::get(i32ty, op.getBurstLength());
434
435 // Set the issue_token
436 issue_token = BoolAttr::get(ctx, op.getIssueToken());
437 // Earlier, all S2MM channels were implicitly assumed to issue a token.
438 // This logic is kept for now for backward compatibility.
439 if (!isMM2S)
440 issue_token = BoolAttr::get(ctx, true);
441
442 if (targetModel.isMemTile(tileCol, tileRow) && (!isMM2S) &&
443 (op.getD0ZeroBefore() != 0 || op.getD0ZeroAfter() != 0 ||
444 op.getD1ZeroBefore() != 0 || op.getD1ZeroAfter() != 0 ||
445 op.getD2ZeroBefore() != 0 || op.getD2ZeroAfter() != 0)) {
446 op->emitOpError("MemTile supports zero padding only on MM2S direction");
447 return failure();
448 }
449
450 // write the buffer descriptor to the array
451 NpuWriteBdOp::create(
452 rewriter, op->getLoc(), column, bd_id, buffer_length, buffer_offset,
453 enable_packet, out_of_order_id, packet_id, packet_type, d0_size,
454 d0_stride, d1_size, d1_stride, d2_size, d2_stride, iteration_current,
455 iteration_size, iteration_stride, next_bd, row, use_next_bd, valid_bd,
456 lock_rel_val, lock_rel_id, lock_acq_enable, lock_acq_val, lock_acq_id,
457 d0_zero_before, d1_zero_before, d2_zero_before, d0_zero_after,
458 d1_zero_after, d2_zero_after, burst_length);
459
460 // Resolve the buffer's runtime-sequence arg and emit the address patch
461 // (plus any offset-state update).
462 int arg_idx = -1;
463 if (failed(emitBufferAddressPatch(op, adaptor, rewriter, tileCol, tileRow,
464 arg_idx)))
465 return failure();
466
467 // push the patched bd onto the dma task queue. bd_id and repeat_count are
468 // SSA operands; materialize them as constants here (the static path).
469 NpuPushQueueOp::create(
470 rewriter, op->getLoc(), column, row, infoOp.getChannelDirAttr(),
471 infoOp.getChannelIndexAttr(), issue_token,
472 createConstantI32(rewriter, op->getLoc(),
473 static_cast<uint32_t>(repeat_count.getInt())),
474 createConstantI32(rewriter, op->getLoc(),
475 static_cast<uint32_t>(bd_id.getInt())));
476
477 rewriter.eraseOp(op);
478 return success();
479 }

481 // Resolve the runtime-sequence argument index the descriptor's buffer traces
482 // back to, and emit the address-patch (plus an offset-state update, if the op
483 // carries one) that binds the runtime buffer pointer into the BD. Shared by
484 // the static and dynamic lowering paths, which are otherwise identical here.
485 // On success `argIdx` receives the resolved index.
486 LogicalResult emitBufferAddressPatch(NpuDmaMemcpyNdOp op, OpAdaptor adaptor,
487 ConversionPatternRewriter &rewriter,
488 int tileCol, int tileRow,
489 int &argIdx) const {
490 AIE::RuntimeSequenceOp seqOp =
491 op->getParentOfType<AIE::RuntimeSequenceOp>();
492 if (!seqOp)
493 return op->emitOpError("NpuDmaMemcpyNdOps must have RuntimeSequenceOp "
494 "parent at time of lowering.");
495 auto traceResult = traceSubviewToBlockArgument(adaptor.getMemref());
496 if (!traceResult)
497 return op->emitOpError(
498 "memref must be a block argument or subview/cast/reinterpret_cast of "
499 "a block argument with static offsets, sizes, and strides");
500 argIdx = -1;
501 Block &entryBB = seqOp.getBody().front();
502 for (int i = 0, e = entryBB.getNumArguments(); i < e; i++)
503 if (entryBB.getArgument(i) == traceResult->rootArg) {
504 argIdx = i;
505 break;
506 }
507 if (argIdx < 0)
508 return failure();
509
510 const auto &targetModel = AIE::getTargetModel(op);
511 uint64_t patchAddr =
512 targetModel.getDmaBdAddress(tileCol, tileRow, op.getId()) +
513 targetModel.getDmaBdAddressOffset(tileCol, tileRow);
514
515 // arg_plus is the byte offset added to the runtime buffer base:
516 // sum(offset[i] * stride[i]) * elemBytes + constBase
517 // where constBase is the subview/cast trace offset. When every offset is
518 // constant this folds to a single constant (byte-identical to before); a
519 // runtime offset (or a runtime stride paired with a non-zero offset) is
520 // built with arith and flows through the SSA arg_plus operand.
521 Location loc = op->getLoc();
522 int64_t elemBytes = op.getElementTypeBitwidth() / 8;
523 auto offsets = llvm::to_vector<4>(llvm::reverse(op.getMixedOffsets()));
524 auto strides = llvm::to_vector<4>(llvm::reverse(op.getMixedStrides()));
525 // The constant fast-path is only valid when every term of
526 // sum(offset[i] * stride[i]) is compile-time known, i.e. each non-zero
527 // offset has both a constant offset and a constant stride (a zero offset
528 // contributes nothing regardless of its stride). Otherwise build arg_plus
529 // with arith. This mirrors getOffsetInBytes(), which reads stride[i] only
530 // where offset[i] != 0.
531 bool offsetFoldsToConstant = true;
532 for (size_t i = 0; i < offsets.size(); i++) {
533 auto off = getConstantIntValue(offsets[i]);
534 if (!off || (*off != 0 && !getConstantIntValue(strides[i]))) {
535 offsetFoldsToConstant = false;
536 break;
537 }
538 }
539
540 Value argPlus;
541 if (offsetFoldsToConstant) {
542 argPlus =
543 createConstantI32(rewriter, loc,
544 static_cast<uint32_t>(op.getOffsetInBytes() +
545 traceResult->offsetInBytes));
546 } else {
547 // Element-count offset = sum(offset[i] * stride[i]), then scale to bytes.
548 auto i32ty = rewriter.getIntegerType(32);
549 Value elems = createConstantI32(rewriter, loc, 0);
550 for (size_t i = 0; i < offsets.size(); i++) {
551 if (auto c = getConstantIntValue(offsets[i]); c && *c == 0)
552 continue;
553 Value off = getAsValue(rewriter, loc, offsets[i], i32ty);
554 Value str = getAsValue(rewriter, loc, strides[i], i32ty);
555 Value term = arith::MulIOp::create(rewriter, loc, off, str);
556 elems = arith::AddIOp::create(rewriter, loc, elems, term);
557 }
558 Value bytes = arith::MulIOp::create(
559 rewriter, loc, elems, createConstantI32(rewriter, loc, elemBytes));
560 argPlus = arith::AddIOp::create(
561 rewriter, loc, bytes,
562 createConstantI32(rewriter, loc,
563 static_cast<uint32_t>(traceResult->offsetInBytes)));
564 }
565 NpuAddressPatchOp::create(rewriter, loc, patchAddr, argIdx, argPlus);
566
567 // If this DMA op has an offset_state_table_idx, emit an
568 // update_from_scratchpad to add the runtime offset to the BD address
569 // register (additive; applied after the base patch above).
570 if (op.getOffsetStateTableIdxAttr()) {
571 auto bufType = cast<BaseMemRefType>(op.getMemref().getType());
572 if (failed(emitUpdateBdAddressFromOffsetParameter(rewriter, op, bufType,
573 patchAddr)))
574 return failure();
575 }
576 return success();
577 }

579 // Lower a shim-NOC dma_memcpy_nd carrying runtime (SSA) sizes/strides. Emit
580 // the same NpuWriteBdOp as the static path but with zeros in every
581 // size/stride-bearing field (still folded to one blockwrite), then override
582 // each size/stride word with an npu.write32 from the shared encoder -- same
583 // arithmetic as the static path, so a runtime value equal to a constant
584 // reproduces the same word. The write32s follow in program order at fixed
585 // addresses, so ordering needs no grouping.
586 LogicalResult lowerDynamic(NpuDmaMemcpyNdOp op, OpAdaptor adaptor,
587 ConversionPatternRewriter &rewriter) const {
588 const auto &targetModel = AIE::getTargetModel(op);
589 auto *ctx = op->getContext();
590 auto i32ty = IntegerType::get(ctx, 32);
591 auto zero = IntegerAttr::get(i32ty, 0);
592 Location loc = op->getLoc();
593
594 auto dev = op->getParentOfType<AIE::DeviceOp>();
595 if (!dev)
596 return failure();
597 auto infoOp = AIE::ShimDMAAllocationOp::getForSymbol(
598 dev, op.getMetadata().getRootReference());
599 if (!infoOp)
600 return op->emitOpError("couldn't find shim_dma_allocation op.");
601 AIE::TileOp shimTile = infoOp.getTileOp();
602 if (!shimTile)
603 return op->emitOpError(
604 "shim_dma_allocation op must reference a valid TileOp.");
605 auto channelDir = infoOp.getChannelDir();
606 bool isMM2S = channelDir == AIE::DMAChannelDir::MM2S;
607 int tileCol = shimTile.getCol();
608 int tileRow = shimTile.getRow();
609
610 // Packet / token setup (identical to the static path).
611 auto column = IntegerAttr::get(i32ty, tileCol);
612 auto row = IntegerAttr::get(i32ty, tileRow);
613 auto enable_packet = zero, packet_id = zero, packet_type = zero;
614 if (auto packetInfo = op.getPacket()) {
615 enable_packet = IntegerAttr::get(i32ty, 1);
616 packet_type = IntegerAttr::get(i32ty, packetInfo->getPktType());
617 packet_id = IntegerAttr::get(i32ty, packetInfo->getPktId());
618 }
619 auto issue_token = BoolAttr::get(ctx, op.getIssueToken());
620 if (!isMM2S)
621 issue_token = BoolAttr::get(ctx, true);
622
623 // Emit the BD template with zeros in every size/stride word; the write32
624 // overrides below supply the runtime values. valid_bd = 1.
625 NpuWriteBdOp::create(
626 rewriter, loc, column, /*bd_id=*/IntegerAttr::get(i32ty, op.getId()),
627 /*buffer_length=*/zero, /*buffer_offset=*/zero, enable_packet,
628 /*out_of_order_id=*/zero, packet_id, packet_type, /*d0_size=*/zero,
629 /*d0_stride=*/zero, /*d1_size=*/zero, /*d1_stride=*/zero,
630 /*d2_size=*/zero, /*d2_stride=*/zero, /*iteration_current=*/zero,
631 /*iteration_size=*/zero, /*iteration_stride=*/zero, /*next_bd=*/zero,
632 row, /*use_next_bd=*/zero, /*valid_bd=*/IntegerAttr::get(i32ty, 1),
633 /*lock_rel_val=*/zero, /*lock_rel_id=*/zero, /*lock_acq_enable=*/zero,
634 /*lock_acq_val=*/zero, /*lock_acq_id=*/zero, /*d0_zero_before=*/zero,
635 /*d1_zero_before=*/zero, /*d2_zero_before=*/zero,
636 /*d0_zero_after=*/zero,
637 /*d1_zero_after=*/zero, /*d2_zero_after=*/zero,
638 /*burst_length=*/IntegerAttr::get(i32ty, op.getBurstLength()));
639
640 // Emit the runtime size/stride BD-word overrides (shared with dma_task).
641 // buffer_length is the size-product here, so pass no override (null); the
642 // encoder returns the hw repeat_count for the queue push.
643 Value repeatCount;
644 if (failed(emitDynamicShimBdWordOverrides(
645 rewriter, loc, targetModel, tileCol, tileRow, op.getId(),
646 op.getMixedSizes(), op.getMixedStrides(),
647 op.getElementTypeBitwidth(), op.getBurstLength(),
648 /*bufLenOverride=*/Value(), repeatCount)))
649 return failure();
650
651 // Address patch for the buffer pointer (offset is compile-time here).
652 int arg_idx = -1;
653 if (failed(emitBufferAddressPatch(op, adaptor, rewriter, tileCol, tileRow,
654 arg_idx)))
655 return failure();
656
657 NpuPushQueueOp::create(
658 rewriter, loc, column, row, infoOp.getChannelDirAttr(),
659 infoOp.getChannelIndexAttr(), issue_token, repeatCount,
660 createConstantI32(rewriter, loc, op.getId()));
661
662 rewriter.eraseOp(op);
663 return success();
664 }
665};

714struct WriteBdToBlockWritePattern : OpConversionPattern<NpuWriteBdOp> {
715 using OpConversionPattern::OpConversionPattern;
716
717public:
718 WriteBdToBlockWritePattern(
719 MLIRContext *context,
720 llvm::DenseMap<Attribute, memref::GlobalOp> *dedupCache, unsigned *nextId,
721 PatternBenefit benefit = 1)
722 : OpConversionPattern(context, benefit), dedupCache(dedupCache),
723 nextId(nextId) {}
724
725 // Per-pass-run memo + next free name index for blockwrite data globals (see
726 // getOrCreateDataMemref). Owned by the pass; live only during the conversion.
727 llvm::DenseMap<Attribute, memref::GlobalOp> *dedupCache;
728 unsigned *nextId;
729
730 LogicalResult
731 matchAndRewrite(NpuWriteBdOp op, OpAdaptor adaptor,
732 ConversionPatternRewriter &rewriter) const override {
733
734 AIE::DeviceOp dev = op->getParentOfType<AIE::DeviceOp>();
735 const AIE::AIETargetModel &tm = dev.getTargetModel();
736 int col = op.getColumn();
737 int row = op.getRow();
738
739 int num_words = 0;
740 if (isa<AIE::AIE2TargetModel>(tm)) {
741 // Tile DMAs have 6 words, MemTile and Shim have 8 words
742 if (tm.isCoreTile(col, row))
743 num_words = 6;
744 else
745 num_words = 8;
746 } else {
747 llvm_unreachable(
748 "Unsupported AIETargetModel in WriteBdToBlockWritePattern");
749 }
750
751 std::vector<uint32_t> words(num_words, 0);
752
753 uint32_t bd_id = op.getBdId();
754 uint64_t bd_addr = tm.getDmaBdAddress(col, row, bd_id);
755 if (tm.isShimNOCTile(col, row)) {
756 // DMA_BDX_0
757 words[0] = op.getBufferLength();
758
759 // DMA_BDX_1
760 words[1] = op.getBufferOffset();
761
762 // DMA_BDX_2
763 // En Packet , OoO BD ID , Packet ID , Packet Type
764 words[2] |= (op.getEnablePacket() & 0x1) << 30;
765 words[2] |= (op.getOutOfOrderId() & 0x3f) << 24;
766 words[2] |= (op.getPacketId() & 0x1f) << 19;
767 words[2] |= (op.getPacketType() & 0x7) << 16;
768
769 // DMA_BDX_3
770 // TODO: Secure Access
771 words[3] |= (op.getD0Size() & 0x3ff) << 20;
772 words[3] |= op.getD0Stride() & 0xfffff;
773
774 // DMA_BDX_4
775 words[4] = (getShimBurstLengthEncoding(tm, op.getBurstLength()) & 0x3)
776 << 30;
777 words[4] |= (op.getD1Size() & 0x3ff) << 20;
778 words[4] |= op.getD1Stride() & 0xfffff;
779
780 // DMA_BDX_5
781 // TODO: SIMID, AXQoS
782 words[5] |= (2 & 0xf) << 24; // AXCache = 2 to enable upsizing in NoC
783 words[5] |= op.getD2Stride() & 0xfffff;
784
785 // DMA_BDX_6
786 words[6] |= (op.getIterationCurrent() & 0x3f) << 26;
787 words[6] |= (op.getIterationSize() & 0x3f) << 20;
788 words[6] |= op.getIterationStride() & 0xfffff;
789
790 // DMA_BDX_7
791 // TODO: TLAST Suppress
792 words[7] |= (op.getNextBd() & 0xf) << 27;
793 words[7] |= (op.getUseNextBd() & 0x1) << 26;
794 words[7] |= (op.getValidBd() & 0x1) << 25;
795 words[7] |= (op.getLockRelVal() & 0x7f) << 18;
796 words[7] |= (op.getLockRelId() & 0xf) << 13;
797 words[7] |= (op.getLockAcqEnable() & 0x1) << 12;
798 words[7] |= (op.getLockAcqVal() & 0x7f) << 5;
799 words[7] |= op.getLockAcqId() & 0xf;
800 if (op.getD0ZeroBefore() || op.getD1ZeroBefore() ||
801 op.getD2ZeroBefore() || op.getD0ZeroAfter() || op.getD1ZeroAfter() ||
802 op.getD2ZeroAfter()) {
803 op->emitError("Zero padding is only available on MemTile");
804 }
805 } else if (tm.isMemTile(op.getColumn(), op.getRow())) {
806
807 // DMA_BDX_0
808 words[0] |= (op.getEnablePacket() & 0x1) << 31;
809 words[0] |= (op.getPacketType() & 0x7) << 28;
810 words[0] |= (op.getPacketId() & 0x1f) << 23;
811 words[0] |= (op.getOutOfOrderId() & 0x3f) << 17;
812 words[0] |= op.getBufferLength() & 0x1ffff;
813
814 // DMA_BDX_1
815 words[1] |= (op.getD0ZeroBefore() & 0x3F) << 26;
816 words[1] |= (op.getNextBd() & 0x3f) << 20;
817 words[1] |= (op.getUseNextBd() & 0x1) << 19;
818 words[1] |= op.getBufferOffset() & 0x7ffff;
819
820 // DMA_BDX_2
821 words[2] |= (op.getD0Size() & 0x3ff) << 17;
822 words[2] |= op.getD0Stride() & 0x1ffff;
823
824 // DMA_BDX_3
825 // TODO: Secure Access
826 words[3] |= (op.getD1ZeroBefore() & 0x1F) << 27;
827 words[3] |= (op.getD1Size() & 0x3ff) << 17;
828 words[3] |= op.getD1Stride() & 0x1ffff;
829
830 // DMA_BDX_4
831 // TODO: D2Size
832 words[4] |= (op.getD2ZeroBefore() & 0xF) << 27;
833 words[4] |= op.getD2Stride() & 0x1ffff;
834
835 // DMA_BDX_5
836 // ToDO: D3Stride
837 words[5] |= (op.getD2ZeroAfter() & 0xF) << 28;
838 words[5] |= (op.getD1ZeroAfter() & 0x1F) << 23;
839 words[5] |= (op.getD0ZeroAfter() & 0x3F) << 17;
840
841 // DMA_BDX_6
842 words[6] |= (op.getIterationCurrent() & 0x3f) << 23;
843 words[6] |= (op.getIterationSize() & 0x3f) << 17;
844 words[6] |= op.getIterationStride() & 0x1ffff;
845
846 // DMA_BDX_7
847 words[7] |= (op.getValidBd() & 0x1) << 31;
848 words[7] |= (op.getLockRelVal() & 0x7f) << 24;
849 words[7] |= (op.getLockRelId() & 0xff) << 16;
850 words[7] |= (op.getLockAcqEnable() & 0x1) << 15;
851 words[7] |= (op.getLockAcqVal() & 0x7f) << 8;
852 words[7] |= op.getLockAcqId() & 0xff;
853 } else {
854 // AIE2 Tile DMA - 6 words
855 // DMA_BDX_0
856 // Base_Address [27:14], Buffer_Length [13:0]
857 words[0] = ((op.getBufferOffset() / 4) & 0x3fff) << 14;
858 words[0] |= op.getBufferLength() & 0x3fff;
859
860 // DMA_BDX_1
861 // Enable_Compression [31], Enable_Packet [30], Out_Of_Order_BD_ID
862 // [29:24], Packet_ID [23:19], Packet_Type [18:16]
863 words[1] = 0; // Enable_Compression
864 words[1] |= (op.getEnablePacket() & 0x1) << 30;
865 words[1] |= (op.getOutOfOrderId() & 0x3f) << 24;
866 words[1] |= (op.getPacketId() & 0x1f) << 19;
867 words[1] |= (op.getPacketType() & 0x7) << 16;
868
869 // DMA_BDX_2
870 // D1_Stepsize [25:13], D0_Stepsize [12:0]
871 words[2] = (op.getD1Stride() & 0x1fff) << 13;
872 words[2] |= op.getD0Stride() & 0x1fff;
873
874 // DMA_BDX_3
875 // D1_Wrap [28:21], D0_Wrap [20:13], D2_Stepsize [12:0]
876 words[3] = (op.getD1Size() & 0xff) << 21;
877 words[3] |= (op.getD0Size() & 0xff) << 13;
878 words[3] |= op.getD2Stride() & 0x1fff;
879
880 // DMA_BDX_4
881 // Iteration_Current [24:19], Iteration_Wrap [18:13], Iteration_Stepsize
882 // [12:0]
883 words[4] = (op.getIterationCurrent() & 0x3f) << 19;
884 words[4] |= (op.getIterationSize() & 0x3f) << 13;
885 words[4] |= op.getIterationStride() & 0x1fff;
886
887 // DMA_BDX_5
888 // TLAST_Suppress [31], Next_BD [30:27], Use_Next_BD [26], Valid_BD [25],
889 // Lock_Rel_Value [24:18], Lock_Rel_ID [16:13], Lock_Acq_Enable [12],
890 // Lock_Acq_Value [11:5], Lock_Acq_ID [3:0]
891 words[5] = 0; // TLAST_Suppress
892 words[5] |= (op.getNextBd() & 0xf) << 27;
893 words[5] |= (op.getUseNextBd() & 0x1) << 26;
894 words[5] |= (op.getValidBd() & 0x1) << 25;
895 words[5] |= (op.getLockRelVal() & 0x7f) << 18;
896 words[5] |= (op.getLockRelId() & 0xf) << 13;
897 words[5] |= (op.getLockAcqEnable() & 0x1) << 12;
898 words[5] |= (op.getLockAcqVal() & 0x7f) << 5;
899 words[5] |= op.getLockAcqId() & 0xf;
900 }
901
902 memref::GlobalOp global = nullptr;
903 {
904 OpBuilder::InsertionGuard guard(rewriter);
905 rewriter.setInsertionPoint(op->getParentOfType<AIE::RuntimeSequenceOp>());
906 global = getOrCreateDataMemref(rewriter, dev, op.getLoc(), words,
907 dedupCache, nextId);
908 }
909 auto memref = memref::GetGlobalOp::create(
910 rewriter, op.getLoc(), global.getType(), global.getName());
911
912 (void)rewriter.replaceOpWithNewOp<NpuBlockWriteOp>(
913 op, rewriter.getUI32IntegerAttr(bd_addr), memref.getResult(), nullptr,
914 nullptr, nullptr);
915 return success();
916 }
917};

919struct AIEDmaToNpuPass : xilinx::AIEX::impl::AIEDmaToNpuBase<AIEDmaToNpuPass> {
920
921 void runOnOperation() override {
922
923 AIE::DeviceOp device = getOperation();
924
925 ConversionTarget target(getContext());
926 target.addLegalDialect<AIEXDialect>();
927 target.addLegalDialect<memref::MemRefDialect>();
928 target.addLegalDialect<arith::ArithDialect>();
929 target.addLegalOp<AIE::BufferOp>();
930 target.addLegalOp<AIE::ShimDMAAllocationOp>();
931 target.addLegalOp<AIE::TileOp>();
932
933 target.addIllegalOp<NpuDmaMemcpyNdOp>();
934 target.addIllegalOp<NpuDmaWaitOp>();
935 target.addIllegalOp<NpuPushQueueOp>();
936 target.addIllegalOp<NpuWriteRTPOp>();
937 target.addIllegalOp<NpuWriteBdOp>();
938 target.addDynamicallyLegalOp<NpuWrite32Op>(
939 [&](NpuWrite32Op op) { return !op.getBuffer(); });
940 target.addDynamicallyLegalOp<NpuBlockWriteOp>(
941 [&](NpuBlockWriteOp op) { return !op.getBuffer(); });
942 target.addDynamicallyLegalOp<NpuMaskWrite32Op>(
943 [&](NpuMaskWrite32Op op) { return !op.getBuffer(); });
944
945 // Seed, from the device's existing globals, a dedup cache (initial-value ->
946 // global) and the next free "blockwrite_data_<n>" name index (one past the
947 // current max). WriteBdToBlockWritePattern keeps both in sync as it creates
948 // globals, giving O(1) dedup and name-uniquing per BD; seeding the index
949 // past the max keeps names unique by construction. Both are locals holding
950 // raw op handles and must not outlive this conversion.
951 llvm::DenseMap<Attribute, memref::GlobalOp> dataMemrefCache;
952 unsigned nextBlockwriteId = 0;
953 for (auto g : device.getOps<memref::GlobalOp>()) {
954 if (auto initVal = g.getInitialValue())
955 dataMemrefCache.try_emplace(*initVal, g);
956 StringRef suffix = g.getSymName();
957 if (suffix.consume_front("blockwrite_data_")) {
958 unsigned idx;
959 if (!suffix.getAsInteger(10, idx) && idx >= nextBlockwriteId)
960 nextBlockwriteId = idx + 1;
961 }
962 }
963
964 RewritePatternSet patterns(&getContext());
965 patterns.insert<BlockWriteSymToAddr>(&getContext());
966 patterns.insert<DmaToNpuPattern>(&getContext());
967 patterns.insert<DmaWaitToSyncPattern>(&getContext());
968 patterns.insert<MaskWrite32SymToAddr>(&getContext());
969 patterns.insert<PushQueuetoWrite32Pattern>(&getContext());
970 patterns.insert<RtpToWrite32Pattern>(&getContext());
971 patterns.insert<Write32SymToAddr>(&getContext());
972 patterns.insert<WriteBdToBlockWritePattern>(&getContext(), &dataMemrefCache,
973 &nextBlockwriteId);
974
975 if (failed(applyPartialConversion(device, target, std::move(patterns))))
976 signalPassFailure();
977 }
978};
