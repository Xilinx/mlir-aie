7//
8// HRX host backend for the example `test.cpp` files.
9//
10// It exposes the *same* `args`, `parse_args()`, and templated
11// `setup_and_run_aie<...>()` surface the example `test.cpp` files already use,
12// and dispatches through HRX (libhrx + the amdxdna HAL). A test.cpp keeps
13// `#include "xrt_test_wrapper.h"`; defining `TEST_UTILS_USE_HRX` makes that
14// header pull this one in instead, so no example source changes are needed.
15//
16// It consumes the `aiecc` artifacts (`final.xclbin` + `insts.bin`): the raw
17// insts.bin TXN stream is handed straight to libhrx via
18// `hrx_amdxdna_executable_create`, which builds the amdxdna XADX package and
19// derives the (offset, arg_idx, addend) patch table from the transaction's
20// BLOCKWRITE/DDR_PATCH ops internally -- there is no separate XADX helper or
21// host-side patch-table extraction. An ELF input (aiecc --aie-generate-elf) is
22// reduced to its .ctrltext (the TXN verbatim) so libhrx still sees those ops.
23//
24// The bindings list is [in1, in2, out] (2-input form) or [in1, out] (1-input
25// form), so each buffer lands in the binding slot the transaction's DDR-patch
26// ops expect.
27//
28//===----------------------------------------------------------------------===//

56// ---------------------------------------------------------------------------
57// Status check
58// ---------------------------------------------------------------------------
59inline void hrx_check(hrx_status_t status, const char *what) {
60 if (hrx_status_is_ok(status))
61 return;
62 char *msg = nullptr;
63 size_t len = 0;
64 hrx_status_t s2 = hrx_status_to_string(status, &msg, &len);
65 int code = hrx_status_code(status);
66 std::string text = msg ? std::string(msg, len) : std::string("?");
67 if (msg)
68 hrx_status_free_message(msg);
69 hrx_status_ignore(s2);
70 hrx_status_ignore(status);
71 throw std::runtime_error(std::string(what) + " failed (hrx status code " +
72 std::to_string(code) + "): " + text);
73}

96// ---------------------------------------------------------------------------
97// A persistent, host-mapped, device-visible HRX buffer.
98// Coherence is maintained explicitly via flush_range / invalidate_range.
99// ---------------------------------------------------------------------------
100class Buffer {
101public:
102 Buffer(hrx_stream_t stream, size_t nbytes) : nbytes_(nbytes) {
103 size_t alloc = nbytes ? nbytes : 1; // HRX rejects 0-size allocations
104 hrx_check(hrx_buffer_allocate(stream, alloc,
105 HRX_MEMORY_TYPE_HOST_LOCAL |
106 HRX_MEMORY_TYPE_DEVICE_VISIBLE,
107 HRX_BUFFER_USAGE_DEFAULT |
108 HRX_BUFFER_USAGE_MAPPING_PERSISTENT,
109 &handle_),
110 "hrx_buffer_allocate");
111 // PERSISTENT keeps host_ptr_ valid across dispatches (paired with
112 // HRX_BUFFER_USAGE_MAPPING_PERSISTENT above).
113 hrx_check(hrx_buffer_map_with_mode(handle_, HRX_MAPPING_MODE_PERSISTENT,
114 HRX_MAP_READ | HRX_MAP_WRITE, 0, alloc,
115 &host_ptr_),
116 "hrx_buffer_map_with_mode");
117 }
118
119 ~Buffer() {
120 if (handle_)
121 hrx_buffer_release(handle_);
122 }
123
124 Buffer(const Buffer &) = delete;
125 Buffer &operator=(const Buffer &) = delete;
126
127 void flush() {
128 if (nbytes_)
129 hrx_check(hrx_buffer_flush_range(handle_, 0, nbytes_),
130 "hrx_buffer_flush_range");
131 }
132 void invalidate() {
133 if (nbytes_)
134 hrx_check(hrx_buffer_invalidate_range(handle_, 0, nbytes_),
135 "hrx_buffer_invalidate_range");
136 }
137
138 hrx_buffer_t handle() const { return handle_; }
139 void *host_ptr() const { return host_ptr_; }
140 size_t nbytes() const { return nbytes_; }
141
142private:
143 hrx_buffer_t handle_ = nullptr;
144 void *host_ptr_ = nullptr;
145 size_t nbytes_ = 0;
146};

148// ---------------------------------------------------------------------------
149// Extract the XAie transaction words (.ctrltext) from a control ELF (ELF32,
150// little-endian). Used only for an ELF input; .ctrltext is the TXN verbatim.
151// ---------------------------------------------------------------------------
152inline std::vector<uint8_t>
153control_code_from_elf(const std::vector<uint8_t> &d) {
154 auto rd32 = [&](size_t off) -> uint32_t {
155 uint32_t v;
156 std::memcpy(&v, d.data() + off, 4);
157 return v;
158 };
159 auto rd16 = [&](size_t off) -> uint16_t {
160 uint16_t v;
161 std::memcpy(&v, d.data() + off, 2);
162 return v;
163 };
164
165 if (d.size() < 52 || d[0] != 0x7f || d[1] != 'E' || d[2] != 'L' ||
166 d[3] != 'F')
167 throw std::runtime_error("control ELF is not a valid ELF32 file");
168
169 uint32_t e_shoff = rd32(0x20);
170 uint16_t e_shentsize = rd16(0x2E);
171 uint16_t e_shnum = rd16(0x30);
172 uint16_t e_shstrndx = rd16(0x32);
173
174 auto sh = [&](uint32_t i, uint32_t f) -> size_t {
175 return e_shoff + static_cast<size_t>(i) * e_shentsize + f;
176 };
177 uint32_t shstr_off = rd32(sh(e_shstrndx, 0x10));
178
179 auto sname = [&](uint32_t i) -> std::string {
180 uint32_t nm = rd32(sh(i, 0));
181 size_t off = shstr_off + nm, end = off;
182 while (end < d.size() && d[end] != 0)
183 ++end;
184 return std::string(reinterpret_cast<const char *>(d.data() + off),
185 end - off);
186 };
187
188 for (uint32_t i = 0; i < e_shnum; ++i) {
189 if (sname(i) == ".ctrltext") {
190 uint32_t coff = rd32(sh(i, 0x10));
191 uint32_t csize = rd32(sh(i, 0x14));
192 return std::vector<uint8_t>(d.begin() + coff, d.begin() + coff + csize);
193 }
194 }
195 throw std::runtime_error("control ELF has no .ctrltext section");
196}

198// ---------------------------------------------------------------------------
199// Read the XAie transaction libhrx patches from. For a raw insts.bin the
200// transaction is the file verbatim; for an ELF input (aiecc --aie-generate-elf)
201// it is the .ctrltext section (also the TXN verbatim). libhrx derives the patch
202// table from the transaction, so no host-side patch-table extraction is needed.
203// ---------------------------------------------------------------------------
204inline std::vector<uint8_t> read_transaction(const std::string &insts_path) {
205 std::ifstream f(insts_path, std::ios::binary);
206 if (!f)
207 throw std::runtime_error("could not read insts: " + insts_path);
208 std::vector<uint8_t> data((std::istreambuf_iterator<char>(f)),
209 std::istreambuf_iterator<char>());
210
211 if (data.size() >= 4 && data[0] == 0x7f && data[1] == 'E' && data[2] == 'L' &&
212 data[3] == 'F')
213 return control_code_from_elf(data);
214 return data;
215}

217// ---------------------------------------------------------------------------
218// Build + load an HRX executable from the artifacts, returning (exe, ordinal).
219// ---------------------------------------------------------------------------
220struct LoadedKernel {
221 hrx_executable_t exe = nullptr;
222 uint32_t ordinal = 0;
223};
224
225inline LoadedKernel load_kernel(const std::string &xclbin_path,
226 const std::string &insts_path,
227 const std::string &kernel_name) {
228 // Read xclbin bytes.
229 std::ifstream xf(xclbin_path, std::ios::binary);
230 if (!xf)
231 throw std::runtime_error("could not read xclbin: " + xclbin_path);
232 std::vector<uint8_t> xclbin((std::istreambuf_iterator<char>(xf)),
233 std::istreambuf_iterator<char>());
234
235 // Resolve the XAie transaction. For a raw insts.bin this is the TXN verbatim;
236 // for an ELF input it is the .ctrltext (also the TXN verbatim). libhrx
237 // derives the patch table from the transaction internally.
238 std::vector<uint8_t> transaction = read_transaction(insts_path);
239
240 // Describe the executable to libhrx: one xclbin + one CREATE entry point with
241 // a single run carrying the transaction. libhrx builds the amdxdna XADX
242 // package and derives the patch table internally.
243 hrx_amdxdna_executable_run_t run{};
244 run.record_length = sizeof(run);
245 run.abi_version = HRX_AMDXDNA_EXECUTABLE_RUN_ABI_VERSION_0;
246 run.transaction.data = transaction.data();
247 run.transaction.data_length = transaction.size();
248 run.data_payload.data = nullptr;
249 run.data_payload.data_length = 0;
250
251 hrx_amdxdna_executable_entry_point_t entry{};
252 entry.record_length = sizeof(entry);
253 entry.abi_version = HRX_AMDXDNA_EXECUTABLE_ENTRY_POINT_ABI_VERSION_0;
254 entry.name.data = kernel_name.c_str();
255 entry.name.size = kernel_name.size();
256 entry.context_mode = HRX_AMDXDNA_CONTEXT_MODE_CREATE;
257 entry.xclbin_ordinal = 0;
258 entry.pdi_ordinal = 0;
259 entry.source_line = 0;
260 entry.source_file.data = nullptr;
261 entry.source_file.size = 0;
262 entry.runs = &run;
263 entry.run_count = 1;
264
265 hrx_const_byte_span_t xclbin_span{};
266 xclbin_span.data = xclbin.data();
267 xclbin_span.data_length = xclbin.size();
268
269 hrx_amdxdna_executable_create_params_t params{};
270 params.record_length = sizeof(params);
271 params.abi_version = HRX_AMDXDNA_EXECUTABLE_CREATE_PARAMS_ABI_VERSION_0;
272 params.flags = 0;
273 params.reserved = 0;
274 params.xclbins = &xclbin_span;
275 params.xclbin_count = 1;
276 params.entry_points = &entry;
277 params.entry_point_count = 1;
278
279 Context &ctx = Context::get();
280 LoadedKernel lk;
281 hrx_check(hrx_amdxdna_executable_create(ctx.device, &params, &lk.exe),
282 "hrx_amdxdna_executable_create");
283 hrx_check(hrx_executable_lookup_export_by_name(lk.exe, kernel_name.c_str(),
284 &lk.ordinal),
285 "hrx_executable_lookup_export_by_name");
286 return lk;
287}

289// ---------------------------------------------------------------------------
290// One dispatch + synchronize of `lk` with `bindings`. Returns elapsed us.
291// ---------------------------------------------------------------------------
292inline double dispatch_once(const LoadedKernel &lk,
293 const std::vector<Buffer *> &bindings) {
294 Context &ctx = Context::get();
295 std::vector<hrx_buffer_ref_t> refs(bindings.size());
296 for (size_t i = 0; i < bindings.size(); ++i) {
297 refs[i].buffer = bindings[i]->handle();
298 refs[i].offset = 0;
299 refs[i].length = bindings[i]->nbytes();
300 }
301 hrx_dispatch_config_t cfg{};
302 cfg.workgroup_count[0] = cfg.workgroup_count[1] = cfg.workgroup_count[2] = 1;
303 cfg.workgroup_size[0] = cfg.workgroup_size[1] = cfg.workgroup_size[2] = 1;
304 cfg.subgroup_size = 0;
305
306 auto start = std::chrono::high_resolution_clock::now();
307 hrx_check(hrx_stream_dispatch(ctx.stream, lk.exe, lk.ordinal, &cfg, nullptr,
308 0, refs.data(), refs.size(),
309 HRX_DISPATCH_FLAG_NONE),
310 "hrx_stream_dispatch");
311 hrx_check(hrx_stream_synchronize(ctx.stream), "hrx_stream_synchronize");
312 auto stop = std::chrono::high_resolution_clock::now();
313 return std::chrono::duration_cast<std::chrono::microseconds>(stop - start)
314 .count();
315}

317// ---------------------------------------------------------------------------
318// Multi-dispatch chain. Each entry is one kernel dispatch with its own
319// bindings; all are recorded, in order, into a single HRX command buffer.
320// HRX inserts an execution + memory barrier after every dispatch, so a later
321// run observes an earlier run's device writes (producer -> consumer chains
322// work, e.g. one run's output buffer is the next run's input). A single
323// `synchronize` then submits the whole batch as one execution -- the amdxdna
324// HAL lowers a multi-dispatch command buffer into one `ERT_CMD_CHAIN`. Entries
325// may share one LoadedKernel (re-dispatch the same kernel) or use different
326// ones (a true multi-kernel pipeline). Returns elapsed us for the whole chain.
327// ---------------------------------------------------------------------------
328struct ChainRun {
329 const LoadedKernel *lk;
330 std::vector<Buffer *> bindings;
331};
332
333inline double dispatch_chain(const std::vector<ChainRun> &runs) {
334 Context &ctx = Context::get();
335 hrx_dispatch_config_t cfg{};
336 cfg.workgroup_count[0] = cfg.workgroup_count[1] = cfg.workgroup_count[2] = 1;
337 cfg.workgroup_size[0] = cfg.workgroup_size[1] = cfg.workgroup_size[2] = 1;
338 cfg.subgroup_size = 0;
339
340 auto start = std::chrono::high_resolution_clock::now();
341 for (const auto &run : runs) {
342 // hrx_stream_dispatch copies the binding refs into the recorded command, so
343 // a per-iteration local is safe; the batch stays pending until synchronize.
344 std::vector<hrx_buffer_ref_t> refs(run.bindings.size());
345 for (size_t i = 0; i < run.bindings.size(); ++i) {
346 refs[i].buffer = run.bindings[i]->handle();
347 refs[i].offset = 0;
348 refs[i].length = run.bindings[i]->nbytes();
349 }
350 hrx_check(hrx_stream_dispatch(ctx.stream, run.lk->exe, run.lk->ordinal,
351 &cfg, nullptr, 0, refs.data(), refs.size(),
352 HRX_DISPATCH_FLAG_NONE),
353 "hrx_stream_dispatch");
354 }
355 hrx_check(hrx_stream_synchronize(ctx.stream), "hrx_stream_synchronize");
356 auto stop = std::chrono::high_resolution_clock::now();
357 return std::chrono::duration_cast<std::chrono::microseconds>(stop - start)
358 .count();
359}

361// ---------------------------------------------------------------------------
362// Reject features the HRX backend does not implement (trace capture, control
363// packets). Throwing here -- rather than warning and continuing -- means a
364// caller that asked for trace/ctrl-pkts gets a hard error instead of a
365// misleading PASS. Trace/control-packet designs are not supported on HRX.
366// ---------------------------------------------------------------------------
367inline void reject_unsupported_features(int trace_size, bool enable_ctrl_pkts) {
368 if (trace_size > 0)
369 throw std::runtime_error(
370 "trace capture is not supported on the HRX backend "
371 "(--trace_sz/trace_size > 0). Re-run without trace, or use the XRT "
372 "backend for trace-enabled designs.");
373 if (enable_ctrl_pkts)
374 throw std::runtime_error(
375 "control packets are not supported on the HRX backend "
376 "(enable_ctrl_pkts=true). Use the XRT backend for control-packet "
377 "designs.");
378}

424/*
425 ******************************************************************************
426 * Typed buffer descriptors for the variadic HRX test wrapper
427 ******************************************************************************
428 *
429 * Typed buffer descriptors so example test.cpp files that use the variadic form
430 * setup_and_run_aie(verify, std::make_tuple(make_in<T>(...), ...),
431 * make_out<T>(...), myargs)
432 * build unchanged against the HRX backend. Buffer binding order: data buffers
433 * in order, then the output buffer.
434 */
435
436// A typed input buffer: element type, element count, and initializer.
437template <typename T>
438struct InBuf {
439 using elem_type = T;
440 int volume;
441 void (*init)(T *, int);
442};
443
444template <typename T>
445InBuf<T> make_in(int volume, void (*init)(T *, int)) {
446 return InBuf<T>{volume, init};
447}
448
449// A typed output buffer.
450template <typename T>
451struct OutBuf {
452 using elem_type = T;
453 int volume;
454 void (*init)(T *, int);
455};
456
457template <typename T>
458OutBuf<T> make_out(int volume, void (*init)(T *, int)) {
459 return OutBuf<T>{volume, init};
460}

462/*
463 ******************************************************************************
464 * HRX based test wrapper for N inputs and 1 output (variadic)
465 ******************************************************************************
466 *
467 * Inputs are passed as InBuf<T> descriptors (any number, any types). The
468 * verifier is called as verify(in0*, in1*, ..., out*, first_input_volume,
469 * verbosity) and returns an error count -- matching the XRT wrapper contract.
470 */
471template <typename Verify, typename TOut, typename... TIns>
472int setup_and_run_aie(Verify verify_results, std::tuple<InBuf<TIns>...> inputs,
473 OutBuf<TOut> output, struct args myargs,
474 bool enable_ctrl_pkts = false) {
475 using namespace hrx_test;
476 srand(time(NULL));
477
478 try {
479 // The HRX backend does not (yet) implement trace capture or control-packet
480 // configuration. Fail loudly rather than silently ignoring the request:
481 // running without them would emit a misleading "PASS" with no trace / an
482 // unconfigured control path. Trace/control-packet designs must use XRT.
483 reject_unsupported_features(myargs.trace_size, enable_ctrl_pkts);
484
485 Context &ctx = Context::get();
486 LoadedKernel lk = load_kernel(myargs.xclbin, myargs.instr, myargs.kernel);
487
488 constexpr int num_inputs = sizeof...(TIns);
489
490 // Allocate + initialize the input buffers, in order. Buffer is non-movable
491 // (owns an HRX allocation), so hold them via unique_ptr.
492 std::vector<std::unique_ptr<Buffer>> in_bufs;
493 in_bufs.reserve(num_inputs);
494 std::tuple<TIns *...> in_ptrs;
495 // The verifier's SIZE argument is the first input's element count, matching
496 // the historical wrapper contract (verify iterates the primary input).
497 int first_in_volume = 0;
498
499 auto alloc_inputs = [&](auto &...inb) {
500 int idx = 0;
501 auto one = [&](auto &desc, auto &ptr_slot) {
502 using T = typename std::remove_reference_t<decltype(desc)>::elem_type;
503 auto buf = std::make_unique<Buffer>(ctx.stream,
504 (size_t)desc.volume * sizeof(T));
505 T *mapped = reinterpret_cast<T *>(buf->host_ptr());
506 desc.init(mapped, desc.volume);
507 buf->flush();
508 ptr_slot = mapped;
509 if (idx == 0)
510 first_in_volume = desc.volume;
511 in_bufs.push_back(std::move(buf));
512 idx++;
513 };
514 std::apply([&](auto &...slots) { (one(inb, slots), ...); }, in_ptrs);
515 };
516 std::apply(alloc_inputs, inputs);
517
518 // Output buffer follows the inputs.
519 Buffer bo_out(ctx.stream, (size_t)output.volume * sizeof(TOut));
520 TOut *bufOut = reinterpret_cast<TOut *>(bo_out.host_ptr());
521 output.init(bufOut, output.volume);
522 bo_out.flush();
523
524 // Binding order: data buffers in order, then out.
525 std::vector<Buffer *> bindings;
526 bindings.reserve(num_inputs + 1);
527 for (auto &b : in_bufs)
528 bindings.push_back(b.get());
529 bindings.push_back(&bo_out);
530
531 unsigned num_iter = myargs.n_iterations + myargs.n_warmup_iterations;
532 double npu_time_total = 0, npu_time_min = 1e30, npu_time_max = 0;
533 int errors = 0;
534
535 for (unsigned iter = 0; iter < num_iter; iter++) {
536 double us = dispatch_once(lk, bindings);
537 bo_out.invalidate();
538
539 if (iter < (unsigned)myargs.n_warmup_iterations)
540 continue;
541
542 if (myargs.do_verify) {
543 // verify(in0*, in1*, ..., out*, first_input_volume, verbosity)
544 errors += std::apply(
545 [&](auto *...in_ptr) {
546 return verify_results(in_ptr..., bufOut, first_in_volume,
547 myargs.verbosity);
548 },
549 in_ptrs);
550 }
551 npu_time_total += us;
552 npu_time_min = us < npu_time_min ? us : npu_time_min;
553 npu_time_max = us > npu_time_max ? us : npu_time_max;
554 }
555
556 report_timing(npu_time_total, npu_time_min, npu_time_max,
557 myargs.n_iterations);
558
559 if (!errors) {
560 std::cout << "\nPASS!\n\n";
561 return 0;
562 }
563 std::cout << "\nError count: " << errors << "\n\n";
564 std::cout << "\nFailed.\n\n";
565 return 1;
566 } catch (const std::exception &e) {
567 std::cerr << "\nHRX error: " << e.what() << "\n\nFailed.\n\n";
568 return 1;
569 }
570}

572/*
573 ******************************************************************************
574 * HRX based test wrapper for 2 inputs and 1 output (legacy fixed-arity)
575 ******************************************************************************
576 */
577template <typename T1, typename T2, typename T3, void (*init_bufIn1)(T1 *, int),
578 void (*init_bufIn2)(T2 *, int), void (*init_bufOut)(T3 *, int),
579 int (*verify_results)(T1 *, T2 *, T3 *, int, int)>
580int setup_and_run_aie(int IN1_VOLUME, int IN2_VOLUME, int OUT_VOLUME,
581 struct args myargs, bool enable_ctrl_pkts = false) {
582 using namespace hrx_test;
583 srand(time(NULL));
584
585 try {
586 // The HRX backend does not (yet) implement trace capture or control-packet
587 // configuration. Fail loudly rather than silently ignoring the request:
588 // running without them would emit a misleading "PASS" with no trace / an
589 // unconfigured control path. Trace/control-packet designs must use XRT.
590 reject_unsupported_features(myargs.trace_size, enable_ctrl_pkts);
591
592 Context &ctx = Context::get();
593 LoadedKernel lk = load_kernel(myargs.xclbin, myargs.instr, myargs.kernel);
