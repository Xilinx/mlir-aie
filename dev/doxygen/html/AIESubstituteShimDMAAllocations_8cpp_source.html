33struct DMAConfigureTaskForOpPattern
34 : public mlir::OpRewritePattern<DMAConfigureTaskForOp> {
35
36 // Build-speed lever (byte-identical): the canonical path resolves each
37 // task's shim-DMA allocation via AIE::ShimDMAAllocationOp::getForSymbol ->
38 // device.lookupSymbol, a LINEAR symbol-table scan run once per
39 // DMAConfigureTaskForOp. At B=128/L12 there are tens of thousands of these,
40 // so the per-op O(allocations) scan makes the pass O(n^2) (measured
41 // super-linear: L6 40s -> L12 156s = 3.86x for 2x layers). We instead build a
42 // SymbolTable for the device ONCE and look up in O(1). Same fix as the
43 // AIETargetNPU getDataWords symbol-cache. The pattern never erases/creates a
44 // symbol (it rewrites task ops, not ShimDMAAllocationOps), so the prebuilt
45 // table stays valid across the greedy run.
46 const mlir::SymbolTable &symbolTable;
47
48 DMAConfigureTaskForOpPattern(mlir::MLIRContext *ctx,
49 const mlir::SymbolTable &symbolTable)
50 : OpRewritePattern<DMAConfigureTaskForOp>(ctx), symbolTable(symbolTable) {
51 }
52
53 LogicalResult matchAndRewrite(DMAConfigureTaskForOp op,
54 PatternRewriter &rewriter) const override {
55 AIE::ShimDMAAllocationOp alloc_op =
56 symbolTable.lookup<AIE::ShimDMAAllocationOp>(
57 op.getAlloc().getRootReference());
58 if (!alloc_op) {
59 return op.emitOpError("no shim DMA allocation found for symbol");
60 }
61
62 AIE::TileOp tile = alloc_op.getTileOp();
63 if (!tile) {
64 return op.emitOpError(
65 "shim DMA allocation must reference a valid TileOp");
66 }
67
68 DMAConfigureTaskOp new_op = DMAConfigureTaskOp::create(
69 rewriter, op.getLoc(), rewriter.getIndexType(), tile.getResult(),
70 alloc_op.getChannelDir(), (int32_t)alloc_op.getChannelIndex(),
71 op.getIssueToken(), op.getRepeatCount(),
72 alloc_op.getPacket().value_or(nullptr));
73 rewriter.replaceAllUsesWith(op.getResult(), new_op.getResult());
74 rewriter.inlineRegionBefore(op.getBody(), new_op.getBody(),
75 new_op.getBody().begin());
76 rewriter.eraseOp(op);
77 return success();
78 }
79};

81struct AIESubstituteShimDMAAllocationsPass
82 : xilinx::AIEX::impl::AIESubstituteShimDMAAllocationsBase<
83 AIESubstituteShimDMAAllocationsPass> {
84
85 void runOnOperation() override {
86 AIE::DeviceOp device = getOperation();
87
88 // Build the device symbol table ONCE (O(1) per-task allocation lookup
89 // instead of getForSymbol's per-task linear scan -> O(n^2)).
90 // Byte-identical.
91 mlir::SymbolTable symbolTable(device);
92
93 // Convert DMAConfigureTaskForOps that reference shim DMA allocations
94 // to regular DMAConfigureTaskOps
95 RewritePatternSet patterns(&getContext());
96 patterns.insert<DMAConfigureTaskForOpPattern>(&getContext(), symbolTable);
97
98 (void)applyPatternsGreedily(device, std::move(patterns));
99 }
100};
